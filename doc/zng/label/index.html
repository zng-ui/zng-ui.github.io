Label widget and properties.

The Label! widget is a text presenter that is focusable, when it receives focus it can transfer it to another target widget.

use zng::prelude::*;

Container! {
    child_start = zng::label::Label!("Name", "name-field");
    child_spacing = 5;
    child = TextInput! {
        id = "name-field";
        txt = var_from("");
    };
}
