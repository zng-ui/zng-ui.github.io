140fn default_scrollbar() -> WidgetFn<ScrollBarArgs> {
141    wgt_fn!(|args: ScrollBarArgs| {
142        Scrollbar! {
143            thumb = Thumb! {
144                viewport_ratio = args.viewport_ratio();
145                offset = args.offset();
146            };
147            orientation = args.orientation;
148            visibility = args.content_overflows().map_into()
149        }
150    })
151}

473/// Binds the [`horizontal_offset`] scroll var to the property value.
474///
475/// The binding is bidirectional and the scroll variable is assigned on init.
476///
477/// Note that settings the offset directly overrides effects like smooth scrolling, prefer using
478/// the scroll commands to scroll over this property.
479///
480/// [`horizontal_offset`]: super::SCROLL::horizontal_offset
481#[property(EVENT, widget_impl(Scroll))]
482pub fn horizontal_offset(child: impl UiNode, offset: impl IntoVar<Factor>) -> impl UiNode {
483    let offset = offset.into_var();
484    match_node(child, move |_, op| {
485        if let UiNodeOp::Init = op {
486            let scroll_offset = super::SCROLL.horizontal_offset();
487
488            if !offset.capabilities().is_always_static() {

493        }
494    })

598/// Scroll by grabbing and dragging the content with the mouse primary button.
599///
600/// This is not enabled by default. Note that couch pan is always enabled, this property implements
601/// a similar behavior for the mouse pointer.
602#[property(LAYOUT, default(false), widget_impl(Scroll))]
603pub fn mouse_pan(child: impl UiNode, enabled: impl IntoVar<bool>) -> impl UiNode {
604    let enabled = enabled.into_var();
605    let mut mouse_input = EventHandle::dummy();
606
607    struct Dragging {
608        _mouse_move: EventHandle,
609        start: PxPoint,
610        applied_offset: PxVector,
611        factor: Factor,
612    }
613    let mut dragging = None;
614
615    match_node(child, move |c, op| match op {
616        UiNodeOp::Init => {
617            WIDGET.sub_var(&enabled);
618            if enabled.get() {
619                mouse_input = MOUSE_INPUT_EVENT.subscribe(WIDGET.id());
620            }
621        }
622        UiNodeOp::Deinit => {
623            mouse_input = EventHandle::dummy();
624            dragging = None;
625        }
626        UiNodeOp::Update { .. } => {
627            if let Some(enabled) = enabled.get_new() {
628                if enabled && mouse_input.is_dummy() {
629                    mouse_input = MOUSE_INPUT_EVENT.subscribe(WIDGET.id());
630                } else {
631                    mouse_input = EventHandle::dummy();
632                    dragging = None;
633                }
634            }
635        }
636        UiNodeOp::Event { update } => {
637            if enabled.get() {
638                c.event(update);
639
640                if let Some(args) = MOUSE_INPUT_EVENT.on_unhandled(update) {
641                    if args.is_primary() {
642                        if args.is_mouse_down() {
643                            let id = WIDGET.id();
644                            POINTER_CAPTURE.capture_widget(id);
645                            let factor = WINDOW.info().scale_factor();
646                            dragging = Some(Dragging {
647                                _mouse_move: MOUSE_MOVE_EVENT.subscribe(id),
648                                start: args.position.to_px(factor),
649                                applied_offset: PxVector::zero(),
650                                factor,
651                            });
652                        } else {
653                            dragging = None;
654                            SCROLL.clear_vertical_overscroll();
655                            SCROLL.clear_horizontal_overscroll();
656                        }
657                    }
658                } else if let Some(d) = &mut dragging {
659                    if let Some(args) = MOUSE_MOVE_EVENT.on_unhandled(update) {
660                        let offset = d.start - args.position.to_px(d.factor);
661                        let delta = d.applied_offset - offset;
662                        d.applied_offset = offset;
663
664                        if delta.y != Px(0) {
665                            SCROLL.scroll_vertical_touch(-delta.y);
666                        }
667                        if delta.x != Px(0) {
668                            SCROLL.scroll_horizontal_touch(-delta.x);
669                        }
670                    }
671                }
672            }
673        }
674        _ => {}
675    })
676}
