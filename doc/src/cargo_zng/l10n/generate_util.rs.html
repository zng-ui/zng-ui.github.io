7/// Visit all FTL in `dir`, apply `transform` for each entry and write it to a new dir in `to_name`.
8pub fn generate(dir: &str, to_name: &str, file_header: &str, transform: &impl Fn(&str) -> Cow<str>, check: bool, verbose: bool) {
9    let dir = Path::new(dir);
10    let to_dir = dir.with_file_name(to_name);
11
12    for entry in fs::read_dir(dir).unwrap_or_else(|e| fatal!("cannot read `{}`, {e}", dir.display())) {
13        let entry = entry.unwrap_or_else(|e| fatal!("cannot read `{}` entry, {e}", dir.display()));
14        let path = entry.path();
15        if path.is_file() && path.extension().map(|e| e == "ftl").unwrap_or(false) {
16            let _ = util::check_or_create_dir(check, &to_dir);
17            generate_file(
18                &path,
19                &to_dir.join(path.file_name().unwrap()),
20                file_header,
21                transform,
22                check,
23                verbose,
24            );
25        }
26    }
27
28    let unnamed = dir.with_extension("ftl");
29    if unnamed.exists() {
30        generate_file(&unnamed, &to_dir.with_extension("ftl"), file_header, transform, check, verbose);
31    }
32}

34fn generate_file(from: &Path, to: &Path, file_header: &str, transform: &impl Fn(&str) -> Cow<str>, check: bool, verbose: bool) {
35    let source = match fs::read_to_string(from) {
36        Ok(s) => s,
37        Err(e) => {
38            error!("cannot read `{}`, {e}", from.display());
39            return;
40        }
41    };
42
43    let source = match fluent_syntax::parser::parse(source) {
44        Ok(s) => s,
45        Err((s, e)) => {
46            error!(
47                "cannot parse `{}`\n{}",
48                from.display(),
49                e.into_iter().map(|e| format!("    {e}")).collect::<Vec<_>>().join("\n")
50            );
51            s
52        }
53    };
54
55    let mut output = file_header.to_owned();
56
57    for entry in source.body {
58        match entry {
59            Entry::Message(m) => write_entry(&mut output, &m.id, m.value.as_ref(), &m.attributes, transform),
60            Entry::Term(t) => write_entry(&mut output, &t.id, Some(&t.value), &t.attributes, transform),
61            Entry::Comment(_) | Entry::GroupComment(_) | Entry::ResourceComment(_) | Entry::Junk { .. } => {}
62        }
63    }
64
65    if let Err(e) = util::check_or_write(check, to, output.as_bytes(), verbose) {
66        error!("cannot write `{}`, {e}", to.display());
67    } else {
68        println!("  generated {}", to.display());
69    }
70}

72fn write_entry(
73    output: &mut String,
74    id: &Identifier<String>,
75    value: Option<&Pattern<String>>,
76    attributes: &[Attribute<String>],
77    transform: &impl Fn(&str) -> Cow<str>,
78) {
79    write!(output, "\n\n{} = ", id.name).unwrap();
80    if let Some(value) = value {
81        write_pattern(output, value, transform);
82    }
83    for attr in attributes {
84        write!(output, "\n    .{} = ", attr.id.name).unwrap();
85        write_pattern(output, &attr.value, transform);
86    }
87}

89fn write_pattern(output: &mut String, pattern: &Pattern<String>, transform: &impl Fn(&str) -> Cow<str>) {
90    for el in &pattern.elements {
91        match el {
92            PatternElement::TextElement { value } => {
93                let mut prefix = "";
94                for line in value.lines() {
95                    write!(output, "{prefix}{}", transform(line)).unwrap();
96                    prefix = "    ";
97                }
98            }
99            PatternElement::Placeable { expression } => write_expression(output, expression, transform),
100        }
101    }
102}

104fn write_expression(output: &mut String, expr: &Expression<String>, transform: &impl Fn(&str) -> Cow<str>) {
105    match expr {
106        Expression::Select { selector, variants } => {
107            write!(output, "{{").unwrap();
108            write_inline_expression(output, selector, transform);
109            writeln!(output, " ->").unwrap();
110
111            for v in variants {
112                write!(output, "    ").unwrap();
113                if v.default {
114                    write!(output, "*").unwrap();
115                }
116                let key = match &v.key {
117                    VariantKey::Identifier { name } => name,
118                    VariantKey::NumberLiteral { value } => value,
119                };
120                write!(output, "[{key}] ").unwrap();
121
122                write_pattern(output, &v.value, transform);
123            }
124
125            writeln!(output, "}}").unwrap();
126        }
127        Expression::Inline(e) => write_inline_expression(output, e, transform),
128    }
129}
130fn write_inline_expression(output: &mut String, expr: &InlineExpression<String>, transform: &impl Fn(&str) -> Cow<str>) {
131    match expr {
132        InlineExpression::StringLiteral { value } => write!(output, "{{ \"{value}\" }}").unwrap(),
133        InlineExpression::NumberLiteral { value } => write!(output, "{{ {value} }}").unwrap(),
134        InlineExpression::FunctionReference { id, arguments } => {
135            write!(output, "{{ {}", id.name).unwrap();
136            write_arguments(output, arguments, transform);
137            write!(output, " }}").unwrap()
138        }
139        InlineExpression::MessageReference { id, attribute } => {
140            write!(output, "{{ {}", id.name).unwrap();
141            if let Some(a) = attribute {
142                write!(output, ".{}", a.name).unwrap();
143            }
144            write!(output, " }}").unwrap()
145        }
146        InlineExpression::TermReference { id, attribute, arguments } => {
147            write!(output, "{{ -{}", id.name).unwrap();
148            if let Some(a) = attribute {
149                write!(output, ".{}", a.name).unwrap();
150            }
151            if let Some(args) = arguments {
152                write_arguments(output, args, transform);
153            }
154            write!(output, " }}").unwrap()
155        }
156        InlineExpression::VariableReference { id } => write!(output, "{{ ${} }}", id.name).unwrap(),
157        InlineExpression::Placeable { expression } => {
158            write!(output, "{{ ").unwrap();
159            write_expression(output, expression, transform);
160            write!(output, " }}").unwrap();
161        }
162    }
163}

165fn write_arguments(output: &mut String, arguments: &CallArguments<String>, transform: &impl Fn(&str) -> Cow<str>) {
166    write!(output, "(").unwrap();
167    let mut sep = "";
168    for a in &arguments.positional {
169        write!(output, "{sep}").unwrap();
170        write_inline_expression(output, a, transform);
171        sep = ", ";
172    }
173    for a in &arguments.named {
174        write!(output, "{sep}{}", a.name.name).unwrap();
175        write_inline_expression(output, &a.value, transform);
176        sep = ", ";
177    }
178    write!(output, ")").unwrap();
179}
