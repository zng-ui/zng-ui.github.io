14#[derive(Args, Debug)]
15pub struct NewArgs {
16    /// Set template values by position
17    ///
18    /// The first value for all templates is the app name.
19    ///
20    /// EXAMPLE
21    ///
22    /// cargo zng new "My App!" | creates a "my-app" project.
23    ///
24    /// cargo zng new "my_app"  | creates a "my_app" project.
25    #[arg(num_args(0..))]
26    value: Vec<String>,
27
28    /// Zng template
29    ///
30    /// Can be a .git URL or an `owner/repo` for a GitHub repository.
31    /// Can also be an absolute path or `./path` to a local template directory.
32    ///
33    /// Use `#branch` to select a branch, that is `owner/repo#branch`.
34    #[arg(short, long, default_value = "zng-ui/zng-template")]
35    template: String,
36
37    /// Set a template value
38    ///
39    /// Templates have a `.zng-template/keys` file that defines the possible options.
40    ///
41    /// EXAMPLE
42    ///
43    /// -s"key=value" -s"k2=v2"
44    #[arg(short, long, num_args(0..))]
45    set: Vec<String>,
46
47    /// Show all possible values that can be set on the template
48    #[arg(short, long, action)]
49    keys: bool,
50}

52pub fn run(args: NewArgs) {
53    let template = parse_template(args.template);
54
55    if args.keys {
56        return print_keys(template);
57    }
58
59    let arg_keys = match parse_key_values(args.value, args.set) {
60        Ok(arg_keys) => {
61            if arg_keys.is_empty() || (!arg_keys[0].0.is_empty() && arg_keys.iter().all(|(k, _)| k != "app")) {
62                fatal!("missing required key `app`")
63            }
64            arg_keys
65        }
66        Err(e) => fatal!("{e}"),
67    };
68
69    // validate name and init
70    let app = &arg_keys[0].1;
71    let project_name = util::clean_value(app, true)
72        .unwrap_or_else(|e| fatal!("{e}"))
73        .replace(' ', "-")
74        .to_lowercase();
75
76    if Path::new(project_name.as_str()).exists() {
77        let full_path = Path::new(project_name.as_str())
78            .canonicalize()
79            .map(|p| p.display().to_string())
80            .unwrap_or(project_name);
81        fatal!("destination `{full_path}` already exists");
82    }
83
84    if let Err(e) = util::cmd("cargo new --quiet --bin", &[project_name.as_str()], &[]) {
85        let _ = std::fs::remove_dir_all(&project_name);
86        fatal!("cannot init project folder, {e}");
87    }
88
89    if let Err(e) = cleanup_cargo_new(&project_name) {
90        fatal!("failed to cleanup `cargo new` template, {e}");
91    }
92
93    // clone template
94    let template_temp = PathBuf::from(format!("{project_name}.zng_template.tmp"));
95
96    let fatal_cleanup = || {
97        let _ = fs::remove_dir_all(&template_temp);
98        let _ = fs::remove_dir_all(&project_name);
99    };
100
101    let (template_keys, ignore) = template.git_clone(&template_temp, false).unwrap_or_else(|e| {
102        fatal_cleanup();
103        fatal!("failed to clone template, {e}")
104    });
105
106    let cx = Context::new(&template_temp, template_keys, arg_keys, ignore).unwrap_or_else(|e| {
107        fatal_cleanup();
108        fatal!("cannot parse template, {e}")
109    });
110    // generate template
111    if let Err(e) = apply_template(&cx, &project_name) {
112        error!("cannot generate, {e}");
113        fatal_cleanup();
114        util::exit();
115    }
116
117    // cargo zng fmt
118    if Path::new(&project_name).join("Cargo.toml").exists() {
119        if let Err(e) = std::env::set_current_dir(project_name) {
120            fatal!("cannot format generated project, {e}")
121        }
122        crate::fmt::run(crate::fmt::FmtArgs::default());
123    }
124}

126fn parse_key_values(value: Vec<String>, define: Vec<String>) -> io::Result<ArgsKeyMap> {
127    let mut r = Vec::with_capacity(value.len() + define.len());
128
129    for value in value {
130        r.push((String::new(), value));
131    }
132
133    for key_value in define {
134        if let Some((key, value)) = key_value.trim_matches('"').split_once('=') {
135            if !is_key(key) {
136                return Err(io::Error::new(io::ErrorKind::InvalidInput, format!("invalid key `{key}`")));
137            }
138            r.push((key.to_owned(), value.to_owned()));
139        }
140    }
141
142    Ok(r)
143}

145fn print_keys(template: Template) {
146    for i in 0..100 {
147        let template_temp = std::env::temp_dir().join(format!("cargo-zng-template-keys-help-{i}"));
148        if template_temp.exists() {
149            continue;
150        }
151
152        match template.git_clone(&template_temp, true) {
153            Ok((keys, _)) => {
154                println!("TEMPLATE KEYS\n");
155                for kv in keys {
156                    let value = match &kv.value {
157                        Some(dft) => dft.as_str(),
158                        None => cstr!("<bold><y>required</y></bold>"),
159                    };
160                    println!(cstr!("<bold>{}=</bold>{}"), kv.key, value);
161                    if !kv.docs.is_empty() {
162                        for line in kv.docs.lines() {
163                            println!("   {line}");
164                        }
165                        println!();
166                    }
167                }
168            }
169            Err(e) => {
170                error!("failed to clone template, {e}");
171            }
172        }
173        let _ = fs::remove_dir_all(&template_temp);
174        return;
175    }
176    fatal!("failed to clone template, no temp dir available");
177}

179fn parse_template(arg: String) -> Template {
180    let (arg, branch) = arg.rsplit_once('#').unwrap_or((&arg, ""));
181
182    if arg.ends_with(".git") {
183        return Template::Git(arg.to_owned(), branch.to_owned());
184    }
185
186    if arg.starts_with("./") {
187        return Template::Local(PathBuf::from(arg), branch.to_owned());
188    }
189
190    if let Some((owner, repo)) = arg.split_once('/') {
191        if !owner.is_empty() && !repo.is_empty() && !repo.contains('/') {
192            return Template::Git(format!("https://github.com/{owner}/{repo}.git"), branch.to_owned());
193        }
194    }
195
196    let path = PathBuf::from(arg);
197    if path.is_absolute() {
198        return Template::Local(path.to_owned(), branch.to_owned());
199    }
200
201    fatal!("--template must be a `.git` URL, `owner/repo`, `./local` or `/absolute/local`");
202}

204enum Template {
205    Git(String, String),
206    Local(PathBuf, String),
207}
208impl Template {
209    /// Clone repository, if it is a template return the `.zng-template/keys,ignore` files contents.
210    fn git_clone(self, to: &Path, include_docs: bool) -> io::Result<(KeyMap, Vec<glob::Pattern>)> {
211        let (from, branch) = match self {
212            Template::Git(url, b) => (url, b),
213            Template::Local(path, b) => {
214                let path = dunce::canonicalize(path)?;
215                (path.display().to_string(), b)
216            }
217        };
218        let to_str = to.display().to_string();
219        let mut args = vec![from.as_str(), &to_str];
220        if !branch.is_empty() {
221            args.push("--branch");
222            args.push(&branch);
223        }
224        util::cmd_silent("git clone --depth 1", &args, &[])?;
225
226        let keys = match fs::read_to_string(to.join(".zng-template/keys")) {
227            Ok(s) => parse_keys(s, include_docs)?,
228            Err(e) => {
229                if e.kind() == io::ErrorKind::NotFound {
230                    return Err(io::Error::new(
231                        io::ErrorKind::NotFound,
232                        "git repo is not a zng template, missing `.zng-template/keys`",
233                    ));
234                }
235                return Err(e);
236            }
237        };
238
239        let mut ignore = vec![];
240        match fs::read_to_string(to.join(".zng-template/ignore")) {
241            Ok(i) => {
242                for glob in i.lines().map(|l| l.trim()).filter(|l| !l.is_empty() && !l.starts_with('#')) {
243                    let glob = glob::Pattern::new(glob).map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))?;
244                    ignore.push(glob);
245                }
246            }
247            Err(e) => {
248                if e.kind() != io::ErrorKind::NotFound {
249                    return Err(e);
250                }
251            }
252        }
253
254        Ok((keys, ignore))
255    }
256}

258fn cleanup_cargo_new(path: &str) -> io::Result<()> {
259    for entry in fs::read_dir(path)? {
260        let path = entry?.path();
261        if path.components().any(|c| c.as_os_str() == ".git") {
262            continue;
263        }
264        if path.is_dir() {
265            fs::remove_dir_all(path)?;
266        } else if path.is_file() {
267            fs::remove_file(path)?;
268        }
269    }
270    Ok(())
271}

273fn apply_template(cx: &Context, package_name: &str) -> io::Result<()> {
274    let template_temp = &cx.template_root;
275
276    // remove template .git
277    fs::remove_dir_all(template_temp.join(".git"))?;
278
279    // replace keys in post scripts
280    let post = template_temp.join(".zng-template/post");
281    if post.is_dir() {
282        let post_replaced = template_temp.join(".zng-template/post-temp");
283        fs::create_dir_all(&post_replaced)?;
284        apply(cx, true, &post, &post_replaced)?;
285        fs::remove_dir_all(&post)?;
286        fs::rename(&post_replaced, &post)?;
287
288        unsafe {
289            // SAFETY: cargo-zng new is single-threaded
290            std::env::set_var("ZNG_TEMPLATE_POST_DIR", &post);
291        }
292    }
293
294    // rename/rewrite template and move it to new package dir
295    let to = PathBuf::from(package_name);
296    apply(cx, false, template_temp, &to)?;
297
298    let bash = post.join("post.sh");
299    if bash.is_file() {
300        let script = fs::read_to_string(bash)?;
301        crate::res::built_in::sh_run(script, false, Some(&to))?;
302    } else {
303        let manifest = post.join("Cargo.toml");
304        if manifest.exists() {
305            let s = std::process::Command::new("cargo")
306                .arg("run")
307                .arg("--quiet")
308                .arg("--manifest-path")
309                .arg(manifest)
310                .current_dir(to)
311                .status()?;
312            if !s.success() {}
313        } else if post.exists() {
314            return Err(io::Error::new(
315                io::ErrorKind::InvalidData,
316                ".zng-template/post does not contain 'post.sh' nor 'Cargo.toml'",
317            ));
318        }
319    }
320
321    // remove template temp
322    fs::remove_dir_all(template_temp)
323}

325fn apply(cx: &Context, is_post: bool, from: &Path, to: &Path) -> io::Result<()> {
326    for entry in walkdir::WalkDir::new(from).min_depth(1).max_depth(1).sort_by_file_name() {
327        let entry = entry?;
328        let from = entry.path();
329        if cx.ignore(from, is_post) {
330            continue;
331        }
332        if from.is_dir() {
333            let from = cx.rename(from)?;
334            let to = to.join(from.file_name().unwrap());
335            println!("  {}", to.display());
336            fs::create_dir(&to)?;
337            apply(cx, is_post, &from, &to)?;
338        } else if from.is_file() {
339            let from = cx.rename(from)?;
340            let to = to.join(from.file_name().unwrap());
341            cx.rewrite(&from)?;
342            println!("  {}", to.display());
343            fs::rename(from, to).unwrap();
344        }
345    }
346    Ok(())
347}

349struct Context {
350    template_root: PathBuf,
351    replace: ReplaceMap,
352    ignore_workspace: glob::Pattern,
353    ignore: Vec<glob::Pattern>,
354}
355impl Context {
356    fn new(template_root: &Path, mut template_keys: KeyMap, arg_keys: ArgsKeyMap, ignore: Vec<glob::Pattern>) -> io::Result<Self> {
357        for (i, (key, value)) in arg_keys.into_iter().enumerate() {
358            if key.is_empty() {
359                if i >= template_keys.len() {
360                    return Err(io::Error::new(
361                        io::ErrorKind::InvalidInput,
362                        "more positional values them template keys",
363                    ));
364                }
365                template_keys[i].value = Some(value);
366            } else if let Some(kv) = template_keys.iter_mut().find(|kv| kv.key == key) {
367                kv.value = Some(value);
368            } else {
369                return Err(io::Error::new(
370                    io::ErrorKind::InvalidInput,
371                    format!("unknown key `{key}`, not declared by template"),
372                ));
373            }
374        }
375        Ok(Self {
376            template_root: dunce::canonicalize(template_root)?,
377            replace: make_replacements(&template_keys)?,
378            ignore_workspace: glob::Pattern::new(".zng-template").unwrap(),
379            ignore,
380        })
381    }
382
383    fn ignore(&self, template_path: &Path, is_post: bool) -> bool {
384        let template_path = template_path.strip_prefix(&self.template_root).unwrap();
385
386        if !is_post && self.ignore_workspace.matches_path(template_path) {
387            return true;
388        }
389
390        for glob in &self.ignore {
391            if glob.matches_path(template_path) {
392                return true;
393            }
394        }
395        false
396    }
397
398    fn rename(&self, template_path: &Path) -> io::Result<PathBuf> {
399        let mut path = template_path.to_string_lossy().into_owned();
400        for (key, value) in &self.replace {
401            let s_value;
402            let value = if is_sanitized_key(key) {
403                value
404            } else {
405                s_value = sanitise_file_name::sanitize(value);
406                &s_value
407            };
408            path = path.replace(key, value);
409        }
410        let path = PathBuf::from(path);
411        if template_path != path {
412            fs::rename(template_path, &path)?;
413        }
414        Ok(path)
415    }
416
417    fn rewrite(&self, template_path: &Path) -> io::Result<()> {
418        match fs::read_to_string(template_path) {
419            Ok(txt) => {
420                let mut new_txt = txt.clone();
421                for (key, value) in &self.replace {
422                    new_txt = new_txt.replace(key, value);
423                }
424                if new_txt != txt {
425                    fs::write(template_path, new_txt.as_bytes())?;
426                }
427                Ok(())
428            }
429            Err(e) => {
430                if e.kind() == io::ErrorKind::InvalidData {
431                    // not utf-8 text file
432                    Ok(())
433                } else {
434                    Err(e)
435                }
436            }
437        }
438    }
439}

441static PATTERNS: &[(&str, &str, Option<Case>)] = &[
442    ("t-key-t", "kebab-case", Some(Case::Kebab)),
443    ("T-KEY-T", "UPPER-KEBAB-CASE", Some(Case::UpperKebab)),
444    ("t_key_t", "snake_case", Some(Case::Snake)),
445    ("T_KEY_T", "UPPER_SNAKE_CASE", Some(Case::UpperSnake)),
446    ("T-Key-T", "Train-Case", Some(Case::Train)),
447    ("t.key.t", "lower case", Some(Case::Lower)),
448    ("T.KEY.T", "UPPER CASE", Some(Case::Upper)),
449    ("T.Key.T", "Title Case", Some(Case::Title)),
450    ("ttKeyTt", "camelCase", Some(Case::Camel)),
451    ("TtKeyTt", "PascalCase", Some(Case::Pascal)),
452    ("{{key}}", "Unchanged", None),
453    ("f-key-f", "Sanitized", None),
454    ("F-Key-F", "Title Sanitized", None),
455];

468fn is_key(s: &str) -> bool {
469    s.len() >= 3 && s.is_ascii() && s.chars().all(|c| c.is_ascii_alphabetic() && c.is_lowercase())
470}

472fn parse_keys(zng_template_v1: String, include_docs: bool) -> io::Result<KeyMap> {
473    let mut r = vec![];
474
475    let mut docs = String::new();
476
477    for (i, line) in zng_template_v1.lines().enumerate() {
478        let line = line.trim();
479        if line.is_empty() {
480            docs.clear();
481            continue;
482        }
483
484        if line.starts_with('#') {
485            if include_docs {
486                let mut line = line.trim_start_matches('#');
487                if line.starts_with(' ') {
488                    line = &line[1..];
489                }
490                docs.push_str(line);
491                docs.push('\n');
492            }
493            continue;
494        }
495
496        if r.is_empty() && line != "app=" {
497            return Err(io::Error::new(
498                io::ErrorKind::InvalidData,
499                "broken template, first key must be `app=`",
500            ));
501        }
502
503        let docs = mem::take(&mut docs);
504        if let Some((key, val)) = line.split_once('=') {
505            if is_key(key) {
506                if val.is_empty() {
507                    r.push(TemplateKey {
508                        docs,
509                        key: key.to_owned(),
510                        value: None,
511                        required: true,
512                    });
513                    continue;
514                } else if val.starts_with('"') && val.ends_with('"') {
515                    r.push(TemplateKey {
516                        docs,
517                        key: key.to_owned(),
518                        value: Some(val[1..val.len() - 1].to_owned()),
519                        required: false,
520                    });
521                    continue;
522                }
523            }
524        }
525        return Err(io::Error::new(
526            io::ErrorKind::InvalidData,
527            format!("broken template, invalid syntax in `.zng-template:{}`", i + 1),
528        ));
529    }
530
531    Ok(r)
532}
533fn make_replacements(keys: &KeyMap) -> io::Result<ReplaceMap> {
534    let mut r = Vec::with_capacity(keys.len() * PATTERNS.len());
535    for kv in keys {
536        let value = match &kv.value {
537            Some(v) => v,
538            None => {
539                return Err(io::Error::new(
540                    io::ErrorKind::InvalidInput,
541                    format!("missing required key `{}`", kv.key),
542                ));
543            }
544        };
545        let clean_value = util::clean_value(value, kv.required)?;
546
547        for (pattern, _, case) in PATTERNS {
548            let prefix = &pattern[..2];
549            let suffix = &pattern[pattern.len() - 2..];
550            let (key, value) = if let Some(case) = case {
551                let key_case = match case {
552                    Case::Camel => Case::Pascal,
553                    c => *c,
554                };
555                let value = match !pattern.contains('.') && !pattern.contains('{') {
556                    true => &clean_value,
557                    false => value,
558                };
559                (kv.key.to_case(key_case), value.to_case(*case))
560            } else {
561                (kv.key.to_owned(), value.to_owned())
562            };
563            let value = if is_sanitized_key(&key) {
564                sanitise_file_name::sanitize(&value)
565            } else {
566                value
567            };
568            let key = format!("{prefix}{key}{suffix}");
569            r.push((key, value));
570        }
571    }
572    Ok(r)
573}

575fn is_sanitized_key(key: &str) -> bool {
576    key.starts_with('f') || key.starts_with('F')
