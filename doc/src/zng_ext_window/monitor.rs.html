22/// Monitors service.
23///
24/// List monitor screens and configure the PPI of a given monitor.
25///
26/// # Uses
27///
28/// Uses of this service:
29///
30/// #### Start Position
31///
32/// Windows are positioned on a virtual screen that overlaps all monitors, but all position configuration is done relative to
33/// an specific parent monitor, it is important to track the parent monitor as it defines properties that affect the layout of the window.
34/// This service is used to provide information to implement this feature.
35///
36/// #### Fullscreen
37///
38/// To set a window to fullscreen a monitor must be selected, by default it can be the one the window is at but
39/// the users may want to select a monitor. To enter fullscreen exclusive the video mode must also be decided, all video
40/// modes supported by the monitor are available in the [`MonitorInfo`] value.
41///
42/// #### Real-Size Preview
43///
44/// Some apps, like image editors, may implement a feature where the user can preview the *real* dimensions of
45/// the content they are editing, to accurately implement this you must known the real dimensions of the monitor screen,
46/// unfortunately this information is not provided by display drivers. You can ask the user to measure their screen and
47/// set the **pixel-per-inch** ratio for the screen using the [`ppi`] variable, this value is then available in the [`LayoutMetrics`]
48/// for the next layout. If not set, the default is `96.0ppi`.
49///
50/// # Provider
51///
52/// This service is provided by the [`WindowManager`].
53///
54/// [`ppi`]: MonitorInfo::ppi
55/// [`scale_factor`]: MonitorInfo::scale_factor
56/// [`LayoutMetrics`]: zng_layout::context::LayoutMetrics
57/// [`WindowManager`]: crate::WindowManager
58pub struct MONITORS;
59impl MONITORS {
60    /// Get monitor info.
61    ///
62    /// Returns `None` if the monitor was not found or the app is running in headless mode without renderer.
63    pub fn monitor(&self, monitor_id: MonitorId) -> Option<MonitorInfo> {
64        MONITORS_SV.read().monitors.get(&monitor_id).cloned()
65    }
66
67    /// Iterate over all available monitors.
68    ///
69    /// Is empty if no monitor was found or the app is running in headless mode without renderer.
70    pub fn available_monitors(&self) -> Vec<MonitorInfo> {
71        MONITORS_SV.read().monitors.values().cloned().collect()
72    }
73
74    /// Gets the monitor info marked as primary.
75    pub fn primary_monitor(&self) -> Option<MonitorInfo> {
76        MONITORS_SV.read().monitors.values().find(|m| m.is_primary().get()).cloned()
77    }
78}

83impl MonitorsService {
84    fn on_monitors_changed(&mut self, args: &RawMonitorsChangedArgs) {
85        let mut available_monitors: IdMap<_, _> = args.available_monitors.iter().cloned().collect();
86
87        let mut removed = vec![];
88        let mut changed = vec![];
89
90        self.monitors.retain(|key, value| {
91            if let Some(new) = available_monitors.remove(key) {
92                if value.update(new) {
93                    changed.push(*key);
94                }
95                true
96            } else {
97                removed.push(*key);
98                false
99            }
100        });
101
102        let mut added = Vec::with_capacity(available_monitors.len());
103
104        for (id, info) in available_monitors {
105            added.push(id);
106
107            self.monitors.insert(id, MonitorInfo::from_gen(id, info));
108        }
109
110        if !removed.is_empty() || !added.is_empty() || !changed.is_empty() {
111            let args = MonitorsChangedArgs::new(args.timestamp, args.propagation().clone(), removed, added, changed);
112            MONITORS_CHANGED_EVENT.notify(args);
113        }
114    }
115
116    pub(super) fn on_pre_event(update: &EventUpdate) {
117        if let Some(args) = RAW_SCALE_FACTOR_CHANGED_EVENT.on(update) {
118            if let Some(m) = MONITORS_SV.read().monitors.get(&args.monitor_id) {
119                m.scale_factor.set(args.scale_factor);
120            }
121        } else if let Some(args) = RAW_MONITORS_CHANGED_EVENT.on(update) {
122            MONITORS_SV.write().on_monitors_changed(args);
123        } else if let Some(args) = VIEW_PROCESS_INITED_EVENT.on(update) {
124            let args = RawMonitorsChangedArgs::new(args.timestamp, args.propagation().clone(), args.available_monitors.clone());
125            MONITORS_SV.write().on_monitors_changed(&args);
126        }
127    }
128}

130/// "Monitor" configuration used by windows in [headless mode].
131///
132/// [headless mode]: zng_app::window::WindowMode::is_headless
133#[derive(Clone, Copy, PartialEq)]
134#[non_exhaustive]
135pub struct HeadlessMonitor {
136    /// The scale factor used for the headless layout and rendering.
137    ///
138    /// If set to `None`, falls back to the [`parent`] scale-factor, or `1.0` if the headless window has not parent.
139    ///
140    /// `None` by default.
141    ///
142    /// [`parent`]: crate::WindowVars::parent
143    pub scale_factor: Option<Factor>,
144
145    /// Size of the imaginary monitor screen that contains the headless window.
146    ///
147    /// This is used to calculate relative lengths in the window size definition and is defined in
148    /// layout pixels instead of device like in a real monitor info.
149    ///
150    /// `(11608, 8708)` by default.
151    pub size: DipSize,
152
153    /// Pixel-per-inches used for the headless layout and rendering.
154    pub ppi: Ppi,
155}
156impl fmt::Debug for HeadlessMonitor {
157    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
158        if f.alternate() || self.ppi != Ppi::default() {
159            f.debug_struct("HeadlessMonitor")
160                .field("scale_factor", &self.scale_factor)
161                .field("screen_size", &self.size)
162                .field("ppi", &self.ppi)
163                .finish()
164        } else {
165            write!(f, "({:?}, ({}, {}))", self.scale_factor, self.size.width, self.size.height)
166        }
167    }
168}
169impl HeadlessMonitor {
170    /// New with custom size at `None` scale.
171    pub fn new(size: DipSize) -> Self {
172        HeadlessMonitor {
173            scale_factor: None,
174            size,
175            ppi: Ppi::default(),
176        }
177    }
178
179    /// New with custom size and scale.
180    pub fn new_scaled(size: DipSize, scale_factor: Factor) -> Self {
181        HeadlessMonitor {
182            scale_factor: Some(scale_factor),
183            size,
184            ppi: Ppi::default(),
185        }
186    }
187
188    /// New with default size `(11608, 8708)` and custom scale.
189    pub fn new_scale(scale_factor: Factor) -> Self {
190        HeadlessMonitor {
191            scale_factor: Some(scale_factor),
192            ..Self::default()
193        }
194    }
195}
196impl Default for HeadlessMonitor {
197    /// New `(11608, 8708)` at `None` scale.
198    fn default() -> Self {
199        (11608, 8708).into()
200    }
201}

211/// All information about a monitor that [`MONITORS`] can provide.
212#[derive(Clone)]
213pub struct MonitorInfo {
214    id: MonitorId,
215    is_primary: ArcVar<bool>,
216    name: ArcVar<Txt>,
217    position: ArcVar<PxPoint>,
218    size: ArcVar<PxSize>,
219    video_modes: ArcVar<Vec<VideoMode>>,
220    scale_factor: ArcVar<Factor>,
221    ppi: ArcVar<Ppi>,
222}
223impl fmt::Debug for MonitorInfo {
224    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
225        f.debug_struct("MonitorFullInfo").field("id", &self.id).finish_non_exhaustive()
226    }
227}
228impl MonitorInfo {
229    /// New from a [`zng_view_api::MonitorInfo`].
230    fn from_gen(id: MonitorId, info: zng_view_api::window::MonitorInfo) -> Self {
231        MonitorInfo {
232            id,
233            is_primary: var(info.is_primary),
234            name: var(info.name.to_txt()),
235            position: var(info.position),
236            size: var(info.size),
237            scale_factor: var(info.scale_factor),
238            video_modes: var(info.video_modes),
239            ppi: var(Ppi::default()),
240        }
241    }
242
243    /// Update variables from fresh [`zng_view_api::MonitorInfo`],
244    /// returns if any value changed.
245    fn update(&self, info: zng_view_api::window::MonitorInfo) -> bool {
246        fn check_set<T: VarValue + PartialEq>(var: &impl Var<T>, value: T) -> bool {
247            let ne = var.with(|v| v != &value);
248            var.set(value).unwrap();
249            ne
250        }
251
252        check_set(&self.is_primary, info.is_primary)
253            | check_set(&self.name, info.name.to_txt())
254            | check_set(&self.position, info.position)
255            | check_set(&self.size, info.size)
256            | check_set(&self.scale_factor, info.scale_factor)
257            | check_set(&self.video_modes, info.video_modes)
258    }
259
260    /// Unique ID.
261    pub fn id(&self) -> MonitorId {
262        self.id
263    }
264
265    /// If this monitor is the primary screen.
266    pub fn is_primary(&self) -> ReadOnlyArcVar<bool> {
267        self.is_primary.read_only()
268    }
269
270    /// Name of the monitor.
271    pub fn name(&self) -> ReadOnlyArcVar<Txt> {
272        self.name.read_only()
273    }
274    /// Top-left offset of the monitor region in the virtual screen, in pixels.
275    pub fn position(&self) -> ReadOnlyArcVar<PxPoint> {
276        self.position.read_only()
277    }
278    /// Width/height of the monitor region in the virtual screen, in pixels.
279    pub fn size(&self) -> ReadOnlyArcVar<PxSize> {
280        self.size.read_only()
281    }
282
283    /// Exclusive fullscreen video modes.
284    pub fn video_modes(&self) -> ReadOnlyArcVar<Vec<VideoMode>> {
285        self.video_modes.read_only()
286    }
287
288    /// The monitor scale factor.
289    ///
290    /// Can update if the user changes system settings.
291    pub fn scale_factor(&self) -> ReadOnlyArcVar<Factor> {
292        self.scale_factor.read_only()
293    }
294    /// Pixel-per-inch config var.
295    pub fn ppi(&self) -> ArcVar<Ppi> {
296        self.ppi.clone()
297    }
298
299    /// Gets the monitor area in pixels.
300    pub fn px_rect(&self) -> PxRect {
301        let pos = self.position.get();
302        let size = self.size.get();
303
304        PxRect::new(pos, size)
305    }
306
307    /// Gets the monitor area in device independent pixels.
308    pub fn dip_rect(&self) -> DipRect {
309        let pos = self.position.get();
310        let size = self.size.get();
311        let factor = self.scale_factor.get();
312
313        PxRect::new(pos, size).to_dip(factor)
314    }
315
316    /// Bogus metadata for the [`MonitorId::fallback`].
317    ///
318    /// [`MonitorId::fallback`]: crate::monitor::MonitorId::fallback
319    pub fn fallback() -> Self {
320        let defaults = HeadlessMonitor::default();
321        let fct = 1.fct();
322
323        Self {
324            id: MonitorId::fallback(),
325            is_primary: var(false),
326            name: var("<fallback>".into()),
327            position: var(PxPoint::zero()),
328            size: var(defaults.size.to_px(fct)),
329            video_modes: var(vec![]),
330            scale_factor: var(fct),
331            ppi: var(Ppi::default()),
332        }
333    }

336/// A selector that returns a [`MonitorInfo`].
337#[derive(Clone, Default)]
338pub enum MonitorQuery {
339    /// The parent window monitor, or `Primary` if the window has no parent.
340    ///
341    /// Note that the window is not moved automatically if the parent window is moved to another monitor, only
342    /// after the query variable updates.
343    ///
344    /// This is the default value.
345    #[default]
346    ParentOrPrimary,
347
348    /// The primary monitor, if there is any monitor.
349    Primary,
350    /// Custom query closure.
351    ///
352    /// If the closure returns `None` the `ParentOrPrimary` query is used, if there is any.
353    ///
354    /// You can use the [`MONITORS`] service in the query closure to select a monitor.
355    Query(Arc<dyn Fn() -> Option<MonitorInfo> + Send + Sync>),
356}
357impl std::fmt::Debug for MonitorQuery {
358    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
359        if f.alternate() {
360            write!(f, "MonitorQuery::")?;
361        }
362        match self {
363            Self::ParentOrPrimary => write!(f, "ParentOrPrimary"),
364            Self::Primary => write!(f, "Primary"),
365            Self::Query(_) => write!(f, "Query(_)"),
366        }
367    }
368}
369impl MonitorQuery {
370    /// New query.
371    pub fn new(query: impl Fn() -> Option<MonitorInfo> + Send + Sync + 'static) -> Self {
372        Self::Query(Arc::new(query))
373    }
374
375    /// Runs the query.
376    pub fn select(&self) -> Option<MonitorInfo> {
377        self.select_for(WINDOW.id())
378    }
379    fn select_for(&self, win_id: WindowId) -> Option<MonitorInfo> {
380        match self {
381            MonitorQuery::ParentOrPrimary => Self::parent_or_primary_query(win_id),
382            MonitorQuery::Primary => Self::primary_query(),
383            MonitorQuery::Query(q) => q(),
384        }
385    }
386
387    /// Runs the query. Falls back to `Primary`, or the largest or [`MonitorInfo::fallback`].
388    pub fn select_fallback(&self) -> MonitorInfo {
389        match self {
390            MonitorQuery::ParentOrPrimary => Self::parent_or_primary_query(WINDOW.id()),
391            MonitorQuery::Primary => Self::primary_query(),
392            MonitorQuery::Query(q) => q().or_else(Self::primary_query),
393        }
394        .unwrap_or_else(Self::fallback)
395    }
396
397    fn fallback() -> MonitorInfo {
398        let mut best = MonitorInfo::fallback();
399        let mut best_area = Px(0);
400        for m in MONITORS.available_monitors() {
401            let m_area = m.px_rect().area();
402            if m_area > best_area {
403                best = m;
404                best_area = m_area;
405            }
406        }
407        best
408    }
409
410    fn parent_or_primary_query(win_id: WindowId) -> Option<MonitorInfo> {
411        if let Some(parent) = WINDOWS.vars(win_id).unwrap().parent().get() {
412            if let Ok(w) = WINDOWS.vars(parent) {
413                return if let Some(monitor) = w.actual_monitor().get() {
414                    MONITORS.monitor(monitor)
415                } else {
416                    w.monitor().get().select_for(parent)
417                };
418            }
419        }
420        MONITORS.primary_monitor()
421    }
422
423    fn primary_query() -> Option<MonitorInfo> {
424        MONITORS.primary_monitor()
425    }
426}
427impl PartialEq for MonitorQuery {
428    /// Returns `true` only if both are [`MonitorQuery::Primary`].
429    fn eq(&self, other: &Self) -> bool {
430        matches!((self, other), (Self::Primary, Self::Primary))
431    }
432}
