139    fn open_impl(&mut self, id: WindowId, new_window: UiTask<WindowRoot>, force_headless: Option<WindowMode>) -> ResponseVar<WindowId> {
140        let (responder, response) = response_var();
141        let request = OpenWindowRequest {
142            id,

144            force_headless,
145            responder,
146        };

154    fn loading_handle_impl(&mut self, window_id: WindowId, deadline: Deadline) -> Option<WindowLoadingHandle> {
155        let mut handle = None;
156
157        if let Some(info) = self.windows_info.get_mut(&window_id) {
158            // window already opened, check if not loaded
159            if !info.is_loaded {
160                handle = Some(info.loading_handle.new_handle(UPDATES.sender(), deadline))
161            }
162
163            // drop timer to nearest deadline, will recreate in the next update.
164            self.loading_deadline = None;
165        } else if let Some(h) = self.open_loading.get_mut(&window_id) {
166            // window not opened yet
167            handle = Some(h.new_handle(UPDATES.sender(), deadline));
168        }
169
170        handle
171    }

173    fn close_together(
174        &mut self,
175        windows: impl IntoIterator<Item = WindowId>,
176    ) -> Result<ResponseVar<CloseWindowResult>, WindowNotFoundError> {
177        let mut group = IdSet::default();
178
179        for w in windows {
180            if !self.windows_info.contains_key(&w) {
181                return Err(WindowNotFoundError::new(w));
182            }
183            group.insert(w);
184        }
185
186        if group.is_empty() {
187            return Ok(response_done_var(CloseWindowResult::Cancel));
188        }
189
190        let (responder, response) = response_var();
191        self.close_requests.push(CloseWindowRequest { responder, windows: group });
192        UPDATES.update(None);
193
194        Ok(response)
195    }

202    ) -> ImageVar {
203        if let Some(w) = self.windows_info.get(&window_id) {

211                        img.read_only()
212                    }
213                    Err(_) => var(Img::dummy(Some(formatx!("{}", WindowNotFoundError::new(window_id))))).read_only(),
214                }
215            } else {
216                var(Img::dummy(Some(formatx!("window `{window_id}` is headless without renderer")))).read_only()
217            }
218        } else {
219            var(Img::dummy(Some(formatx!("{}", WindowNotFoundError::new(window_id))))).read_only()
220        }

546        }
547        WINDOWS_SV.write().frame_image_impl(window_id, |vr| vr.frame_image_rect(rect, mask))

627    ///
628    /// If `wait_event` is `true` also awaits for the [`WINDOW_LOAD_EVENT`] to finish notifying.
629    ///
630    /// Returns `true` if the window iis open and has no pending loading handles.
631    pub fn wait_loaded(&self, window_id: impl Into<WindowId>, wait_event: bool) -> impl Future<Output = bool> + Send + Sync + 'static {
632        Self::wait_loaded_impl(window_id.into(), wait_event)
633    }
634    async fn wait_loaded_impl(window_id: WindowId, wait_event: bool) -> bool {
635        if Self.is_loaded(window_id) {
636            if wait_event {
637                // unlikely, but it can just have loaded and the event is ongoing.
638                zng_task::yield_now().await;
639            }
640            return true;
641        }
642
643        // start receiving before loading check otherwise could load after check and before receiver creation.
644        let recv = WINDOW_LOAD_EVENT.receiver();
645        while Self.is_loading(window_id) {
646            while let Ok(msg) = zng_task::with_deadline(recv.recv_async(), 1.secs()).await {
647                if let Ok(args) = msg
648                    && args.window_id == window_id
649                {
650                    if wait_event {
651                        zng_task::yield_now().await;
652                    }
653                    return true;
654                }
655            }
656            // deadline, rare case window closes before load
657        }
658
659        if Self.is_loaded(window_id) {
660            if wait_event {
661                zng_task::yield_now().await;
662            }
663            return true;
664        }
665        false
666    }

962    pub(super) fn on_pre_event(update: &EventUpdate) {
963        if let Some(args) = RAW_WINDOW_FOCUS_EVENT.on(update) {
964            let mut wns = WINDOWS_SV.write();
965
966            let mut prev = None;
967            let mut new = None;
968
969            if let Some(prev_focus) = args.prev_focus
970                && let Some(window) = wns.windows_info.get_mut(&prev_focus)
971            {
972                if window.is_focused {
973                    window.is_focused = false;
974                    prev = Some(prev_focus);
975                } else if args.new_focus.is_none()
976                    && let Some(focused) = wns.windows_info.values_mut().find(|w| w.is_focused)
977                    && focused.vars.nest_parent().get().is_some()
978                    && focused.vars.parent().get() == Some(prev_focus)
979                {
980                    // focus is in nested window of the system window that lost app focus.
981                    focused.is_focused = false;
982                    prev = Some(focused.id);
983                }
984            }
985            if let Some(new_focus) = args.new_focus {
986                if prev.is_none()
987                    && let Some((&id, window)) = wns.windows_info.iter_mut().find(|w| w.1.is_focused)
988                    && new_focus != id
989                {
990                    window.is_focused = false;
991                    prev = Some(id);
992                }
993
994                if let Some(window) = wns.windows_info.get_mut(&new_focus)
995                    && !window.is_focused
996                {
997                    window.is_focused = true;
998                    window.vars.focus_indicator().set(None);
999                    new = Some(new_focus);
1000                }
1001            }
1002
1003            if prev.is_some() || new.is_some() {
1004                let args = WindowFocusChangedArgs::new(args.timestamp, args.propagation().clone(), prev, new, false);
1005                WINDOW_FOCUS_CHANGED_EVENT.notify(args);
1006            }
1007        } else if let Some(args) = RAW_WINDOW_CLOSE_REQUESTED_EVENT.on(update) {
1008            let _ = WINDOWS.close(args.window_id);
1009        } else if let Some(args) = RAW_WINDOW_CLOSE_EVENT.on(update) {
1010            if WINDOWS_SV.read().windows.contains_key(&args.window_id) {
1011                tracing::error!("view-process closed window without request");
1012                let mut windows = IdSet::default();
1013                windows.insert(args.window_id);
1014                let args = WindowCloseArgs::new(args.timestamp, args.propagation().clone(), windows);
1015                WINDOW_CLOSE_EVENT.notify(args);
1016            }
1017        } else if let Some(args) = RAW_COLORS_CONFIG_CHANGED_EVENT.on(update) {
1018            WINDOWS_SV.write().latest_colors_cfg = args.config;
1019        } else if let Some(args) = RAW_CHROME_CONFIG_CHANGED_EVENT.on(update) {
1020            WINDOWS_SV.read().system_chrome.set(args.config);
1021        } else if VIEW_PROCESS_INITED_EVENT.has(update) {
1022            // we skipped request fulfillment until this event.
1023            UPDATES.update(None);
1024        } else {
1025            #[cfg(feature = "image")]
1026            if let Some(args) = RAW_IMAGE_LOADED_EVENT.on(update).or_else(|| RAW_IMAGE_LOAD_ERROR_EVENT.on(update)) {
1027                // update ready frame images.
1028                let mut sv = WINDOWS_SV.write();
1029                sv.frame_images.retain(|i| {
1030                    if let Some(i) = i.upgrade() {
1031                        if Some(&args.image) == i.get().view() {
1032                            i.update();
1033                            false
1034                        } else {
1035                            true
1036                        }
1037                    } else {
1038                        false
1039                    }
1040                });
1041            }
1042        }
1043
1044        Self::with_detached_windows(|windows, parallel| {
1045            if windows.len() > 1 && parallel.contains(ParallelWin::EVENT) {
1046                windows.par_iter_mut().with_ctx().for_each(|(_, window)| {
1047                    window.pre_event(update);
1048                });
1049            } else {
1050                for (_, window) in windows.iter_mut() {
1051                    window.pre_event(update);
1052                }
1053            }
1054        })
1055    }

1076    pub(super) fn on_event(update: &mut EventUpdate) {
1077        if let Some(args) = WINDOW_CLOSE_REQUESTED_EVENT.on(update) {
1078            let key = args.windows.iter().next().unwrap();
1079            let mut sv = WINDOWS_SV.write();
1080            if let Some(rsp) = sv.close_responders.remove(key) {
1081                if !args.propagation().is_stopped() {
1082                    // close requested by us and not canceled.
1083                    WINDOW_CLOSE_EVENT.notify(WindowCloseArgs::now(args.windows.clone()));
1084                    for r in rsp {
1085                        r.respond(CloseWindowResult::Closed);
1086                    }
1087                } else {
1088                    for r in rsp {
1089                        r.respond(CloseWindowResult::Cancel);
1090                    }
1091                    // already cancelled exit request
1092                    sv.exit_on_close = false;
1093                }
1094            }
1095        } else if let Some(args) = WINDOW_CLOSE_EVENT.on(update) {
1096            // finish close, this notifies `UiNode::deinit` and drops the window
1097            // causing the ViewWindow to drop and close.
1098
1099            for w in args.windows.iter() {
1100                let w = WINDOWS_SV.write().windows.remove(w);
1101                if let Some(w) = w {
1102                    let id = w.ctx.id();
1103                    w.close();
1104
1105                    let info = WINDOWS_SV.write().windows_info.remove(&id).unwrap();
1106
1107                    info.vars.0.is_open.set(false);
1108
1109                    if info.is_focused {
1110                        let args = WindowFocusChangedArgs::now(Some(info.id), None, true);
1111                        WINDOW_FOCUS_CHANGED_EVENT.notify(args)
1112                    }
1113                }
1114            }
1115
1116            let is_headless_app = zng_app::APP.window_mode().is_headless();
1117            let mut wns = WINDOWS_SV.write();
1118
1119            // if windows closed because of app exit request
1120            // OR
1121            // if set to exit on last headed window close in a headed app,
1122            // AND there is no more open headed window OR request for opening a headed window.
1123            if mem::take(&mut wns.exit_on_close)
1124                || (wns.exit_on_last_close.get()
1125                    && !is_headless_app
1126                    && !wns.windows.values().any(|w| matches!(w.ctx.mode(), WindowMode::Headed))
1127                    && !wns
1128                        .open_requests
1129                        .iter()
1130                        .any(|w| matches!(w.force_headless, None | Some(WindowMode::Headed)))
1131                    && !wns.open_tasks.iter().any(|t| matches!(t.mode, WindowMode::Headed)))
1132            {
1133                // fulfill `exit_on_close` or `exit_on_last_close`
1134                APP.exit();
1135            }
1136        } else if let Some(args) = EXIT_REQUESTED_EVENT.on(update)
1137            && !args.propagation().is_stopped()
1138        {
1139            let mut windows = WINDOWS_SV.write();
1140            if !windows.windows_info.is_empty() {
1141                args.propagation().stop();
1142                windows.exit_on_close = true;
1143                drop(windows);
1144                WINDOWS.close_all();
1145            }
1146        }
1147    }

1174        if VIEW_PROCESS.is_available() && !VIEW_PROCESS.is_connected() {
1175            // wait ViewProcessInitedEvent
1176            return;
1177        }
1178
1179        let ((open, mut open_tasks, close, focus, bring_to_top, view_tasks), colors_cfg) = {
1180            let mut wns = WINDOWS_SV.write();
1181            (wns.take_requests(), wns.latest_colors_cfg)
1182        };
1183
1184        let window_mode = zng_app::APP.window_mode();
1185
1186        // fulfill open requests.
1187        for r in open {
1188            let window_mode = match (window_mode, r.force_headless) {
1189                (WindowMode::Headed | WindowMode::HeadlessWithRenderer, Some(mode)) => {
1190                    debug_assert!(!matches!(mode, WindowMode::Headed));
1191                    mode
1192                }
1193                (mode, _) => mode,
1194            };
1195
1196            let colors_cfg = match window_mode {
1197                WindowMode::Headed => colors_cfg,
1198                WindowMode::Headless | WindowMode::HeadlessWithRenderer => ColorsConfig::default(),
1199            };
1200
1201            let task = AppWindowTask::new(r.id, window_mode, colors_cfg, r.new.into_inner(), r.responder);
1202            open_tasks.push(task);
1203        }
1204
1205        // update open tasks.
1206        let mut any_ready = false;
1207        for task in &mut open_tasks {
1208            let ready = task.update();
1209            any_ready |= ready;
1210        }
1211        if any_ready {
1212            for mut task in open_tasks {
1213                if task.is_ready() {
1214                    let window_id = task.ctx.id();
1215
1216                    let mut wns = WINDOWS_SV.write();
1217                    let loading = wns.open_loading.remove(&window_id).unwrap();
1218                    let mut root_extenders = mem::take(&mut wns.root_extenders);
1219                    let mut open_nested_handlers = mem::take(&mut wns.open_nested_handlers);
1220                    drop(wns);
1221                    let (window, info, responder) =
1222                        task.finish(loading, &mut root_extenders.get_mut()[..], &mut open_nested_handlers.get_mut()[..]);
1223
1224                    let mut wns = WINDOWS_SV.write();
1225                    root_extenders.get_mut().append(wns.root_extenders.get_mut());
1226                    open_nested_handlers.get_mut().append(wns.open_nested_handlers.get_mut());
1227                    wns.root_extenders = root_extenders;
1228                    wns.open_nested_handlers = open_nested_handlers;
1229
1230                    if wns.windows.insert(window_id, window).is_some() {
1231                        // id conflict resolved on request.
1232                        unreachable!();
1233                    }
1234                    wns.windows_info.insert(info.id, info);
1235
1236                    responder.respond(window_id);
1237                    // WINDOW_OPEN_EVENT.notify happens after init, so that handlers
1238                    // on the window itself can subscribe to the event.
1239                } else {
1240                    let mut wns = WINDOWS_SV.write();
1241                    wns.open_tasks.push(task);
1242                }
1243            }
1244        } else {
1245            let mut wns = WINDOWS_SV.write();
1246            debug_assert!(wns.open_tasks.is_empty());
1247            wns.open_tasks = open_tasks;
1248        }
1249
1250        // notify close requests, the request is fulfilled or canceled
1251        // in the `event` handler.
1252
1253        {
1254            let mut wns = WINDOWS_SV.write();
1255            let wns = &mut *wns;
1256
1257            let mut close_wns = IdSet::default();
1258
1259            for r in close {
1260                for w in r.windows {
1261                    if let Some(info) = wns.windows_info.get(&w)
1262                        && close_wns.insert(w)
1263                    {
1264                        wns.close_responders
1265                            .entry(w)
1266                            .or_insert_with(Default::default)
1267                            .push(r.responder.clone());
1268
1269                        info.vars.0.children.with(|c| {
1270                            for &c in c.iter() {
1271                                if wns.windows_info.contains_key(&c) && close_wns.insert(c) {
1272                                    wns.close_responders
1273                                        .entry(c)
1274                                        .or_insert_with(Default::default)
1275                                        .push(r.responder.clone());
1276                                }
1277                            }
1278                        });
1279                    }
1280                }
1281            }
1282            if !close_wns.is_empty() {
1283                let args = WindowCloseRequestedArgs::now(close_wns);
1284                WINDOW_CLOSE_REQUESTED_EVENT.notify(args);
1285            }
1286        }
1287
1288        // fulfill focus request
1289        if let Some(w_id) = focus {
1290            Self::with_detached_windows(|windows, _| {
1291                if let Some(w) = windows.get_mut(&w_id) {
1292                    w.focus();
1293                }
1294            });
1295        }
1296
1297        for w_id in bring_to_top {
1298            Self::with_detached_windows(|windows, _| {
1299                if let Some(w) = windows.get_mut(&w_id) {
1300                    w.bring_to_top();
1301                }
1302            });
1303        }
1304
1305        for view_task in view_tasks {
1306            let task = view_task.task.into_inner();
1307            Self::with_detached_windows(|windows, _| {
1308                if let Some(w) = windows.get_mut(&view_task.window_id) {
1309                    w.view_task(task);
1310                } else {
1311                    task(None);
1312                }
1313            })
1314        }
1315    }

1605        loading: WindowLoading,
1606        extenders: &mut [Box<dyn FnMut(WindowRootExtenderArgs) -> UiNode + Send>],
1607        open_nested_handlers: &mut [Box<dyn FnMut(&mut crate::OpenNestedHandlerArgs) + Send>],
1608    ) -> (AppWindow, AppWindowInfo, ResponderVar<WindowId>) {
1609        let mut window = self.task.into_inner().into_result().unwrap_or_else(|_| panic!());
1610        let mut ctx = self.ctx;

1613            for ext in extenders.iter_mut().rev() {
1614                let root = mem::replace(&mut window.child, UiNode::nil());
1615                window.child = ext(WindowRootExtenderArgs { root });
1616            }
1617            let child = mem::replace(&mut window.child, UiNode::nil());
1618            let vars = WINDOW.vars();
1619            let child = with_context_var(child, ACCENT_COLOR_VAR, vars.actual_accent_color());
1620            let child = with_context_var(child, COLOR_SCHEME_VAR, vars.actual_color_scheme());
1621            let child = with_context_var(child, PARALLEL_VAR, vars.parallel());
1622            window.child = child;
1623        });
1624
1625        let mode = self.mode;
1626        let id = ctx.id();
1627
1628        ctx.set_widget_tree(WidgetInfoTree::wgt(id, window.id));
1629
1630        let vars = ctx.with_state(|s| s.borrow().get_clone(*WINDOW_VARS_ID)).unwrap();
1631
1632        if window.kiosk {
1633            vars.chrome().set(false);
1634            vars.visible().set(true);
1635            if !vars.state().get().is_fullscreen() {
1636                vars.state().set(WindowState::Exclusive);
1637            }
1638        }
1639
1640        let commands = WindowCommands::new(id);
1641
1642        let root_id = window.id;
1643
1644        let mut args = crate::OpenNestedHandlerArgs::new(ctx, vars.clone(), commands, window);
1645        for h in open_nested_handlers.iter_mut().rev() {
1646            h(&mut args);
1647            if args.has_nested() {
1648                break;
1649            }
1650        }
1651
1652        let (ctx, ctrl) = match args.take_normal() {
1653            Ok((ctx, vars, commands, window)) => (ctx, WindowCtrl::new(&vars, commands, mode, window)),
1654            Err((ctx, node)) => (ctx, WindowCtrl::new_nested(node)),
1655        };
1656
1657        let window = AppWindow {
1658            ctrl: Mutex::new(ctrl),
1659            ctx,
1660        };
1661        let info = AppWindowInfo::new(id, root_id, mode, vars, loading);
1662
1663        (window, info, self.responder)
1664    }
1665}

1734struct WindowLoading {
1735    handles: Vec<std::sync::Weak<crate::WindowLoadingHandleData>>,
1736    timer: Option<DeadlineHandle>,
1737}
1738impl WindowLoading {
1739    pub fn new() -> Self {
1740        WindowLoading {
1741            handles: vec![],
1742            timer: None,
1743        }
1744    }
1745
1746    /// Returns `true` if the window can load.
1747    pub fn try_load(&mut self, window_id: WindowId) -> bool {
1748        let mut deadline = Deadline::MAX;
1749        self.handles.retain(|h| match h.upgrade() {
1750            Some(h) => {
1751                if h.deadline.has_elapsed() {
1752                    false
1753                } else {
1754                    deadline = deadline.min(h.deadline);
1755                    true
1756                }
1757            }
1758            None => false,
1759        });
1760
1761        if self.handles.is_empty() {
1762            self.timer = None;
1763            true
1764        } else {
1765            if let Some(t) = &self.timer
1766                && t.deadline() != deadline
1767            {
1768                self.timer = None;
1769            }
1770            if self.timer.is_none() {
1771                let t = TIMERS.on_deadline(
1772                    deadline,
1773                    app_hn_once!(|_| {
1774                        UPDATES.update_window(window_id).layout_window(window_id).render_window(window_id);
1775                    }),
1776                );
1777                self.timer = Some(t);
1778            }
1779
1780            false
1781        }
1782    }
1783
1784    pub fn new_handle(&mut self, update: AppEventSender, deadline: Deadline) -> WindowLoadingHandle {
1785        let h = Arc::new(crate::WindowLoadingHandleData { update, deadline });
