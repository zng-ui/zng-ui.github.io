3//!
4//! Helpers for enabling ANGLE support of the ZNG apps on Windows.
5//!
6//! The default ZNG view-process implementation will attempt to dynamically load ANGLE DLLs if requested,
7//! this crate helps signal the view-process and acquire the DLLs.
8//!
9//! The following is a minimal setup example:
10//!
11//! **First** add the crate dependency, ZNG only supports ANGLE on Windows so the dependency can be conditional.
12//!
13//! ```toml
14//! [target.'cfg(windows)'.dependencies]
15//! zng-view-angle = { version = "0.2.5", features = ["download"] }
16//! ```
17//!
18//! With the `"download"` feature enabled the crate will download the required ANGLE DLLs from [zng-ui/build-angle]
19//! and copy them to the output dir. Its only two files `libEGL.dll` and `libGLESv2.dll`, around 5MB total, you must
20//! ensure these files are packaged with the app installer.
21//!
22//! **Second** signal the view-process to use ANGLE:
23//!
24//! ```
25//! # macro_rules! demo { () => {
26//! use zng::prelude::*;
27//!
28//! fn main() {
29//!     zng::env::init!();
30//!
31//!     APP.defaults().run_window(async {
32//!         #[cfg(windows)]
33//!         {
34//!             zng_view_angle::register_license();
35//!             zng_view_angle::register_root_extender();
36//!         }
37//!
38//!         Window! {
39//!             // ...
40//!         }
41//!     });
42//! }
43//! # } }
44//! ```
45//!
46//! The [`register_root_extender`] will insert a node in all subsequent windows that signals the view-process to
47//! find and use the ANGLE DLLs when render mode uses the GPU. If the DLLs are not found the native OpenGL driver is used instead.
48//!
49//! The [`register_license`] simply adds the ANGLE BSD3-Clause license to the [`LICENSES`] service.
50//!
51//! [zng-ui/build-angle]: https://github.com/zng-ui/build-angle
52//!
53//! # Crate
54//!
55#![doc = include_str!(concat!("../", std::env!("CARGO_PKG_README")))]

65fn init(win_id: WindowId) {
66    debug_assert!(VIEW_PROCESS.is_connected());
67
68    const EXT_NAME: &str = "zng-view.prefer_angle";
69    let ext_id = VIEW_PROCESS
70        .extension_id(EXT_NAME)
71        .ok()
72        .flatten()
73        .unwrap_or(ApiExtensionId::INVALID);
74    if ext_id == ApiExtensionId::INVALID {
75        return tracing::error!("view-process does not support {EXT_NAME:?}");
76    }
77
78    WINDOWS_EXTENSIONS
79        .view_extensions_init(win_id, ext_id, ApiExtensionPayload::serialize(&true).unwrap())
80        .unwrap();
81}

83/// Signal the view-process to try and use EGL for this window when the ANGLE DLLs can be found and the render mode is not software.
84///
85/// Note that the `libEGL.dll` and `libGLESv2.dll` files must be distributed with the executable.
86///
87/// Note that this will enable ANGLE just for this window, use [`register_root_extender`] to enable angle for all windows.
88///
89/// This does nothing when app is not running on Windows.
90#[property(CONTEXT)]
91pub fn prefer_angle_egl(child: impl IntoUiNode, enable: impl IntoValue<bool>) -> UiNode {
92    if enable.into() && cfg!(windows) {
93        match_node(child, move |_c, op| {
94            if let UiNodeOp::Init = op {
95                let win_id = WINDOW.id();
96
97                let mut loading_handle = None;
98                if VIEW_PROCESS.is_connected() {
99                    init(win_id);
100                } else {
101                    loading_handle = WINDOW.loading_handle(5.secs(), "prefer_angle_egl");
102                }
103                let handle = VIEW_PROCESS_INITED_EVENT.hook(move |_| {
104                    let _ = loading_handle.take();
105                    init(win_id);
106                    // retain in case of respawn
107                    true
108                });
109                WIDGET.push_var_handle(handle);
110            }
111        })
112    } else {
113        child.into_node()
114    }
115}

117/// Enable ANGLE for all subsequent open windows.
118///
119/// This uses a [`WINDOWS_EXTENSIONS::register_root_extender`] to set [`prefer_angle_egl`] in all windows.
120///
121/// [`prefer_angle_egl`]: fn@prefer_angle_egl
122pub fn register_root_extender() {
123    WINDOWS_EXTENSIONS.register_root_extender(|a| prefer_angle_egl(a.root, true));
124}

126/// Register the ANGLE license with the [`LICENSES`] service.
127///
128/// The license is a [BSD3-Clause type license](https://github.com/google/angle/blob/fca8ca8a87c387515e0d2901916e4fae6b97e83f/LICENSE).
129pub fn register_license() {
130    fn l() -> Vec<LicenseUsed> {
131        vec![LicenseUsed {
132            license: License::new("BSD3-Clause", r#"BSD 3-Clause "Revised" License"#, LICENSE),
133            used_by: vec![User::new(
134                "ANGLE",
135                "fca8ca8",
136                "https://github.com/zng-ui/build-angle/releases/tag/2026-05-17",
137            )],
138        }]
139    }
140    LICENSES.register(l);
141}
