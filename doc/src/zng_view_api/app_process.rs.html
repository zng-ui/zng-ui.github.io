34/// View Process controller, used in the App Process.
35///
36/// # Exit
37///
38/// The View Process is [killed] when the controller is dropped, if the app is running in same process mode
39/// then the current process [exits] with code 0 on drop.
40///
41/// [killed]: std::process::Child::kill
42/// [exits]: std::process::exit
43#[cfg_attr(not(ipc), allow(unused))]
44pub struct Controller {
45    process: Arc<Mutex<Option<std::process::Child>>>,
46    view_state: ViewState,

50    view_process_env: HashMap<Txt, Txt>,
51    request_sender: ipc::RequestSender,
52    response_receiver: ipc::ResponseReceiver,
53    event_listener: Option<EventListenerJoin>,
54    headless: bool,
55    same_process: bool,
56    last_respawn: Option<Instant>,
57    fast_respawn_count: u8,
58}

64    /// Start with a custom view process.
65    ///
66    /// The `view_process_exe` must be an executable that starts a view server.
67    /// Note that the [`VERSION`] of this crate must match in both executables.
68    ///
69    /// The `view_process_env` can be set to any env var needed to start the view-process. Note that if `view_process_exe`
70    /// is the current executable this most likely need set `zng_env::PROCESS_MAIN`.
71    ///
72    /// The `on_event` closure is called in another thread every time the app receives an event.
73    ///
74    /// # Tests
75    ///
76    /// The [`current_exe`] cannot be used in tests, you should set an external view-process executable. Unfortunately there
77    /// is no way to check if `start` was called in a test so we cannot provide an error message for this.
78    /// If the test is hanging in debug builds or has a timeout error in release builds this is probably the reason.
79    ///
80    /// [`current_exe`]: std::env::current_exe
81    /// [`VERSION`]: crate::VERSION
82    pub fn start<F>(view_process_exe: PathBuf, view_process_env: HashMap<Txt, Txt>, headless: bool, on_event: F) -> Self
83    where
84        F: FnMut(Event) + Send + 'static,
85    {
86        Self::start_impl(view_process_exe, view_process_env, headless, Box::new(on_event))
87    }
88    fn start_impl(
89        view_process_exe: PathBuf,
90        view_process_env: HashMap<Txt, Txt>,
91        headless: bool,
92        on_event: Box<dyn FnMut(Event) + Send>,
93    ) -> Self {
94        if ViewConfig::from_env().is_some() {
95            panic!("cannot start Controller in process configured to be view-process");
96        }
97
98        let (process, request_sender, response_receiver, event_receiver) =
99            Self::spawn_view_process(&view_process_exe, &view_process_env, headless).expect("failed to spawn or connect to view-process");
100        let same_process = process.is_none();
101        let process = Arc::new(Mutex::new(process));
102        let ev = if same_process {
103            Self::spawn_same_process_listener(on_event, event_receiver, ViewProcessGen::first())
104        } else {
105            Self::spawn_other_process_listener(on_event, event_receiver, process.clone(), ViewProcessGen::first())
106        };
107
108        let mut c = Controller {
109            same_process,
110            view_state: ViewState::NotRunning,
111            process,
112            view_process_exe,
113            view_process_env,
114            request_sender,
115            response_receiver,
116            event_listener: Some(ev),
117            headless,
118            generation: ViewProcessGen::first(),
119            is_respawn: false,
120            last_respawn: None,
121            fast_respawn_count: 0,
122        };
123
124        if let Err(ipc::ViewChannelError::Disconnected) = c.try_init() {
125            panic!("respawn on init");
126        }
127
128        c
129    }
130    fn spawn_same_process_listener(
131        mut on_event: Box<dyn FnMut(Event) + Send>,
132        mut event_receiver: EventReceiver,
133        generation: ViewProcessGen,
134    ) -> std::thread::JoinHandle<Box<dyn FnMut(Event) + Send>> {
135        thread::spawn(move || {
136            while let Ok(ev) = event_receiver.recv() {
137                on_event(ev);
138            }
139            on_event(Event::Disconnected(generation));
140
141            // return to reuse in respawn.
142            on_event
143        })
144    }
145    fn spawn_other_process_listener(
146        mut on_event: Box<dyn FnMut(Event) + Send>,
147        mut event_receiver: EventReceiver,
148        process: Arc<Mutex<Option<std::process::Child>>>,
149        generation: ViewProcessGen,
150    ) -> std::thread::JoinHandle<Box<dyn FnMut(Event) + Send>> {
151        // ipc-channel sometimes does not signal disconnect when the view-process dies
152        thread::spawn(move || {
153            let ping_time = Duration::from_secs(1);
154            while let Ok(maybe) = event_receiver.recv_timeout(ping_time) {
155                match maybe {
156                    Some(ev) => on_event(ev),
157                    None => {
158                        if let Some(p) = &mut *process.lock() {
159                            match p.try_wait() {
160                                Ok(c) => {
161                                    if c.is_some() {
162                                        // view-process died
163                                        break;
164                                    }
165                                }
166                                Err(e) => {
167                                    if e.kind() != std::io::ErrorKind::Interrupted {
168                                        // signal disconnected to trigger a respawn
169                                        break;
170                                    }
171                                }
172                            }
173                        } else {
174                            // respawning already
175                            break;
176                        }
177                    }
178                }
179            }
180            on_event(Event::Disconnected(generation));
181
182            // return to reuse in respawn.
183            on_event
184        })
185    }

224    pub(crate) fn talk(&mut self, req: Request) -> VpResult<Response> {
225        debug_assert!(req.expect_response());
226
227        if req.must_be_connected() {
228            self.disconnected_err()?;
229        }
230
231        match self.try_talk(req) {
232            Ok(r) => Ok(r),
233            Err(ipc::ViewChannelError::Disconnected) => {
234                self.handle_disconnect(self.generation);
235                Err(ipc::ViewChannelError::Disconnected)
236            }
237        }
238    }

240    pub(crate) fn command(&mut self, req: Request) -> VpResult<()> {
241        debug_assert!(!req.expect_response());
242
243        if req.must_be_connected() {
244            self.disconnected_err()?;
245        }
246
247        match self.request_sender.send(req) {
248            Ok(_) => Ok(()),
249            Err(ipc::ViewChannelError::Disconnected) => {
250                self.handle_disconnect(self.generation);
251                Err(ipc::ViewChannelError::Disconnected)
252            }
253        }
254    }

256    fn spawn_view_process(
257        view_process_exe: &Path,
258        view_process_env: &HashMap<Txt, Txt>,
259        headless: bool,
260    ) -> AnyResult<(
261        Option<std::process::Child>,
262        ipc::RequestSender,
263        ipc::ResponseReceiver,
264        ipc::EventReceiver,
265    )> {
266        let _span = tracing::trace_span!("spawn_view_process").entered();
267
268        let init = ipc::AppInit::new();
269
270        // create process and spawn it, unless is running in same process mode.
271        let process = if ViewConfig::is_awaiting_same_process() {
272            ViewConfig::set_same_process(ViewConfig {
273                version: crate::VERSION.into(),
274                server_name: Txt::from_str(init.name()),
275                headless,
276            });
277            None
278        } else {
279            #[cfg(not(ipc))]
280            {
281                let _ = (view_process_exe, view_process_env);
282                panic!("expected only same_process mode with `ipc` feature disabled");
283            }
284
285            #[cfg(ipc)]
286            {
287                let mut process = std::process::Command::new(view_process_exe);
288                for (name, val) in view_process_env {
289                    process.env(name, val);
290                }
291                let process = process
292                    .env(VIEW_VERSION, crate::VERSION)
293                    .env(VIEW_SERVER, init.name())
294                    .env(VIEW_MODE, if headless { "headless" } else { "headed" })
295                    .env("RUST_BACKTRACE", "full")
296                    .spawn()?;
297                Some(process)
298            }
299        };
300
301        let (req, rsp, ev) = match init.connect() {
302            Ok(r) => r,
303            Err(e) => {
304                #[cfg(ipc)]
305                if let Some(mut p) = process {
306                    if let Err(ke) = p.kill() {
307                        tracing::error!(
308                            "failed to kill new view-process after failing to connect to it\n connection error: {e:?}\n kill error: {ke:?}",
309                        );
310                    } else {
311                        match p.wait() {
312                            Ok(output) => {
313                                let code = output.code();
314                                if ViewConfig::is_version_err(code, None) {
315                                    let code = code.unwrap_or(1);
316                                    tracing::error!(
317                                        "view-process API version mismatch, the view-process build must use the same exact version as the app-process, \
318                                                will exit app-process with code 0x{code:x}"
319                                    );
320                                    zng_env::exit(code);
321                                } else {
322                                    tracing::error!("view-process exit code: {}", output.code().unwrap_or(1));
323                                }
324                            }
325                            Err(e) => {
326                                tracing::error!("failed to read output status of killed view-process, {e}");
327                            }
328                        }
329                    }
330                } else {
331                    tracing::error!("failed to connect with same process");
332                }
333                return Err(e);
334            }
335        };
336
337        Ok((process, req, rsp, ev))
338    }

340    /// Handle an [`Event::Inited`].
341    ///
342    /// Set the connected flag to `true`.
343    pub fn handle_inited(&mut self, vp_gen: ViewProcessGen) {
344        match self.view_state {
345            ViewState::NotRunning => {
346                if self.generation == vp_gen {
347                    // crash respawn already sets gen
348                    self.view_state = ViewState::RunningAndConnected;
349                }
350            }
351            ViewState::Suspended => {
352                self.generation = vp_gen;
353                self.view_state = ViewState::RunningAndConnected;
354            }
355            ViewState::RunningAndConnected => {}
356        }
357    }

366    /// Handle an [`Event::Disconnected`].
367    ///
368    /// The `gen` parameter is the generation provided by the event. It is used to determinate if the disconnect has
369    /// not been handled already.
370    ///
371    /// Tries to cleanup the old view-process and start a new one, if all is successful an [`Event::Inited`] is send.
372    ///
373    /// The old view-process exit code and std output is logged using the `vp_respawn` target.
374    ///
375    /// Exits the current process with code `1` if the view-process was killed by the user. In Windows this is if
376    /// the view-process exit code is `1`. In Unix if it was killed by SIGKILL, SIGSTOP, SIGINT.
377    ///
378    /// # Panics
379    ///
380    /// If the last five respawns happened all within 500ms of the previous respawn.
381    ///
382    /// If the an error happens three times when trying to spawn the new view-process.
383    ///
384    /// If another disconnect happens during the view-process startup dialog.
385    pub fn handle_disconnect(&mut self, vp_gen: ViewProcessGen) {
386        if vp_gen == self.generation {
387            #[cfg(not(ipc))]
388            {
389                tracing::error!(target: "vp_respawn", "cannot recover in same_process mode (no ipc)");
390            }
391
392            #[cfg(ipc)]
393            {
394                self.respawn_impl(true)
395            }
396        } else {
397            tracing::warn!("disconnected event from previous generation ignored")
398        }
399    }
400
401    /// Reopen the view-process, causing another [`Event::Inited`].
402    ///
403    /// This is similar to [`handle_disconnect`] but the current process does not
404    /// exit depending on the view-process exit code.
405    ///
406    /// [`handle_disconnect`]: Controller::handle_disconnect
407    pub fn respawn(&mut self) {
408        #[cfg(not(ipc))]
409        {
410            tracing::error!(target: "vp_respawn", "cannot recover in same_process mode (no ipc)");
411        }
412
413        #[cfg(ipc)]
414        self.respawn_impl(false);
415    }
416    #[cfg(ipc)]
417    fn respawn_impl(&mut self, is_crash: bool) {
418        use zng_unit::TimeUnits;
419
420        self.view_state = ViewState::NotRunning;
421        self.is_respawn = true;
422
423        let mut process = if let Some(p) = self.process.lock().take() {
424            p
425        } else {
426            if self.same_process {
427                tracing::error!(target: "vp_respawn", "cannot recover in same_process mode");
428            }
429            return;
430        };
431        if is_crash {
432            tracing::error!(target: "vp_respawn", "channel disconnect, will try respawn");
433        }
434
435        if is_crash {
436            let t = Instant::now();
437            if let Some(last_respawn) = self.last_respawn {
438                if t - last_respawn < Duration::from_secs(60) {
439                    self.fast_respawn_count += 1;
440                    if self.fast_respawn_count == 2 {
441                        panic!("disconnect respawn happened 2 times in less than 1 minute");
442                    }
443                } else {
444                    self.fast_respawn_count = 0;
445                }
446            }
447            self.last_respawn = Some(t);
448        } else {
449            self.last_respawn = None;
450        }
451
452        // try exit
453        let mut killed_by_us = false;
454        if !is_crash {
455            let _ = process.kill();
456            killed_by_us = true;
457        } else if !matches!(process.try_wait(), Ok(Some(_))) {
458            // if not exited, give the process 300ms to close with the preferred exit code.
459            thread::sleep(300.ms());
460
461            if !matches!(process.try_wait(), Ok(Some(_))) {
462                // if still not exited, kill it.
463                killed_by_us = true;
464                let _ = process.kill();
465            }
466        }
467
468        let code_and_output = match process.wait() {
469            Ok(c) => Some(c),
470            Err(e) => {
471                tracing::error!(target: "vp_respawn", "view-process could not be killed, will abandon running, {e:?}");
472                None
473            }
474        };
475
476        // try print stdout/err and exit code.
477        if let Some(c) = code_and_output {
478            tracing::info!(target: "vp_respawn", "view-process killed");
479
480            let code = c.code();
481            #[allow(unused_mut)]
482            let mut signal = None::<i32>;
483
484            if !killed_by_us {
485                // check if user killed the view-process, in this case we exit too.
486
487                #[cfg(windows)]
488                if code == Some(1) {
489                    tracing::warn!(target: "vp_respawn", "view-process exit code (1), probably killed by the system, \
490                                        will exit app-process with the same code");
491                    zng_env::exit(1);
492                }
493
494                #[cfg(unix)]
495                if code.is_none() {
496                    use std::os::unix::process::ExitStatusExt as _;
497                    signal = c.signal();
498
499                    if let Some(sig) = signal
500                        && [2, 9, 17, 19, 23].contains(&sig)
501                    {
502                        tracing::warn!(target: "vp_respawn", "view-process exited by signal ({sig}), \
503                                            will exit app-process with code 1");
504                        zng_env::exit(1);
505                    }
506                }
507            }
508
509            if !killed_by_us {
510                let code = code.unwrap_or(0);
511                let signal = signal.unwrap_or(0);
512                tracing::error!(target: "vp_respawn", "view-process exit code: {code:#X}, signal: {signal}");
513            }
514
515            if ViewConfig::is_version_err(code, None) {
516                let code = code.unwrap_or(1);
517                tracing::error!(target: "vp_respawn", "view-process API version mismatch, the view-process build must use the same exact version as the app-process, \
518                                        will exit app-process with code 0x{code:x}");
519                zng_env::exit(code);
520            }
521        } else {
522            tracing::error!(target: "vp_respawn", "failed to kill view-process, will abandon it running and spawn a new one");
523        }
524
525        // recover event listener closure (in a box).
526        let on_event = match self.event_listener.take().unwrap().join() {
527            Ok(fn_) => fn_,
528            Err(p) => panic::resume_unwind(p),
529        };
530
531        // respawn
532        let mut retries = 3;
533        let (new_process, request, response, event_listener) = loop {
534            match Self::spawn_view_process(&self.view_process_exe, &self.view_process_env, self.headless) {
535                Ok(r) => break r,
536                Err(e) => {
537                    tracing::error!(target: "vp_respawn", "failed to respawn, {e:?}");
538                    retries -= 1;
539                    if retries == 0 {
540                        panic!("failed to respawn `view-process` after 3 retries");
541                    }
542                    tracing::info!(target: "vp_respawn", "retrying respawn");
543                }
544            }
545        };
546        debug_assert!(new_process.is_some());
547
548        // update connections
549        self.process = Arc::new(Mutex::new(new_process));
550        self.request_sender = request;
551        self.response_receiver = response;
552
553        let next_id = self.generation.next();
554        self.generation = next_id;
555
556        let ev = Self::spawn_other_process_listener(on_event, event_listener, self.process.clone(), self.generation);
557        self.event_listener = Some(ev);
558
559        if let Err(ipc::ViewChannelError::Disconnected) = self.try_init() {
560            panic!("respawn on respawn startup");
561        }
562    }
563}
564impl Drop for Controller {
565    /// Kills the View Process, unless it is running in the same process.
566    fn drop(&mut self) {
567        let _ = self.exit();
568        #[cfg(ipc)]
569        if let Some(mut process) = self.process.lock().take()
570            && process.try_wait().is_err()
571        {
572            std::thread::sleep(Duration::from_secs(1));
573            if process.try_wait().is_err() {
574                tracing::error!("view-process did not exit after 1s, killing");
575                let _ = process.kill();
576                let _ = process.wait();
577            }
578        }
579    }
