69trait SelectorImpl: Send {
70    fn selection(&self) -> AnyVar;
71    fn thumbs(&self) -> Var<Vec<ThumbValue>>;
72    fn set(&self, nearest: Factor, to: Factor);
73
74    fn to_offset(&self, t: &dyn AnyVarValue) -> Option<Factor>;
75    #[allow(clippy::wrong_self_convention)]
76    fn from_offset(&self, offset: Factor) -> Box<dyn Any>;
77}

79trait OffsetConvert<T>: Send + Sync {
80    fn to(&self, t: &T) -> Factor;
81    fn from(&self, f: Factor) -> T;
82}
83impl<T, Tf: Fn(&T) -> Factor + Send + Sync, Ff: Fn(Factor) -> T + Send + Sync> OffsetConvert<T> for (Tf, Ff) {
84    fn to(&self, t: &T) -> Factor {
85        (self.0)(t)
86    }
87
88    fn from(&self, f: Factor) -> T {
89        (self.1)(f)
90    }
91}

93/// Represents a type that can auto implement a [`Selector`].
94///
95/// # Implementing
96///
97/// This trait is implemented for all primitive type and Zng layout types, if a type does not you
98/// can declare custom conversions using [`Selector::value`].
99pub trait SelectorValue: VarValue {
100    /// Make the selector.
101    fn to_selector(value: Var<Self>, min: Self, max: Self) -> Selector;
102}
103
104/// Defines the values and ranges selected by a slider.
105///
106/// Selectors are set on the [`selector`](fn@selector) property.
107#[derive(Clone)]
108pub struct Selector(Arc<Mutex<dyn SelectorImpl>>);
109impl fmt::Debug for Selector {
110    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
111        write!(f, "Selector(_)")
112    }
113}
114impl PartialEq for Selector {
115    fn eq(&self, other: &Self) -> bool {
116        Arc::ptr_eq(&self.0, &other.0)
117    }
118}
119impl Selector {
120    /// New with single value thumb of type `T` that can be set any value in the `min..=max` range.
121    pub fn value<T: SelectorValue>(selection: impl IntoVar<T>, min: T, max: T) -> Self {
122        T::to_selector(selection.into_var(), min, max)
123    }
124
125    /// New with a single value thumb of type `T`.
126    ///
127    /// The value must convert to a normalized factor `[0.fct()..=1.fct()]` where `0.fct()` is the minimum possible value and `1.fct()` is the maximum
128    /// possible value. If a value outside of this range is returned it is clamped to the range and the `selection` variable is updated back.
129    pub fn value_with<T>(
130        selection: impl IntoVar<T>,
131        to_offset: impl Fn(&T) -> Factor + Send + Sync + 'static,
132        from_offset: impl Fn(Factor) -> T + Send + Sync + 'static,
133    ) -> Self
134    where
135        T: VarValue,
136    {
137        struct SingleImpl<T: VarValue> {
138            selection: Var<T>,
139            thumbs: Var<Vec<ThumbValue>>,
140            to_from: Arc<dyn OffsetConvert<T>>,
141        }
142        impl<T: VarValue> SelectorImpl for SingleImpl<T> {
143            fn selection(&self) -> AnyVar {
144                self.selection.as_any().clone()
145            }
146
147            fn set(&self, _: Factor, to: Factor) {
148                self.selection.set(self.to_from.from(to));
149            }
150
151            fn thumbs(&self) -> Var<Vec<ThumbValue>> {
152                self.thumbs.clone()
153            }
154
155            fn to_offset(&self, t: &dyn AnyVarValue) -> Option<Factor> {
156                let f = self.to_from.to(t.downcast_ref::<T>()?);
157                Some(f)
158            }
159
160            fn from_offset(&self, offset: Factor) -> Box<dyn Any> {
161                Box::new(self.to_from.from(offset))
162            }
163        }
164        let to_from = Arc::new((to_offset, from_offset));
165        let selection = selection.into_var();
166        let thumbs = selection.map(clmv!(to_from, |s| vec![ThumbValue {
167            offset: to_from.to(s),
168            n_of: (0, 1)
169        }]));
170        Self(Arc::new(Mutex::new(SingleImpl {
171            thumbs,
172            selection,
173            to_from,
174        })))
175    }
176
177    /// New with many value thumbs of type `T` that can be set any value in the `min..=max` range.
178    pub fn many<T: SelectorValue>(many: impl IntoVar<Vec<T>>, min: T, max: T) -> Self {
179        // create a selector just to get the conversion closures
180        let convert = T::to_selector(zng_var::const_var(min.clone()), min, max);
181        Self::many_with(many.into_var(), clmv!(convert, |t| convert.to_offset(t).unwrap()), move |f| {
182            convert.from_offset(f).unwrap()
183        })
184    }
185
186    /// New with many value thumbs of type `T`.
187    ///
188    /// The conversion closure have the same constraints as [`value_with`].
189    ///
190    /// [`value_with`]: Self::value_with
191    pub fn many_with<T>(
192        many: impl IntoVar<Vec<T>>,
193        to_offset: impl Fn(&T) -> Factor + Send + Sync + 'static,
194        from_offset: impl Fn(Factor) -> T + Send + Sync + 'static,
195    ) -> Self
196    where
197        T: VarValue,
198    {
199        struct ManyImpl<T: VarValue> {
200            selection: Var<Vec<T>>,
201            thumbs: Var<Vec<ThumbValue>>,
202            to_from: Arc<dyn OffsetConvert<T>>,
203        }
204        impl<T: VarValue> SelectorImpl for ManyImpl<T> {
205            fn selection(&self) -> AnyVar {
206                self.selection.as_any().clone()
207            }
208
209            fn set(&self, from: Factor, to: Factor) {
210                // modify selection to remove nearest and insert to in new sorted position
211                // or just replace it if it is the same position
212
213                let mut selection = self.selection.get();
214                if selection.is_empty() {
215                    return;
216                }
217
218                let to_value = self.to_from.from(to);
219
220                let (remove_i, mut insert_i) = self.thumbs.with(|t| {
221                    let (remove_i, _) = t
222                        .iter()
223                        .enumerate()
224                        .map(|(i, f)| (i, (f.offset - from).abs()))
225                        .reduce(|a, b| if a.1 < b.1 { a } else { b })
226                        .unwrap_or((t.len(), 0.fct()));
227                    let insert_i = t.iter().position(|t| t.offset >= to).unwrap_or(t.len());
228                    (remove_i, insert_i)
229                });
230
231                if remove_i == insert_i {
232                    selection[remove_i] = to_value;
233                } else {
234                    if insert_i > remove_i {
235                        insert_i -= 1;
236                    }
237                    selection.remove(remove_i);
238                    selection.insert(insert_i, to_value)
239                }
240
241                self.selection.set(selection);
242            }
243
244            fn thumbs(&self) -> Var<Vec<ThumbValue>> {
245                self.thumbs.clone()
246            }
247
248            fn to_offset(&self, t: &dyn AnyVarValue) -> Option<Factor> {
249                let f = self.to_from.to(t.downcast_ref::<T>()?);
250                Some(f)
251            }
252
253            fn from_offset(&self, offset: Factor) -> Box<dyn Any> {
254                Box::new(self.to_from.from(offset))
255            }
256        }
257
258        let to_from = Arc::new((to_offset, from_offset));
259        let selection = many.into_var();
260        let thumbs = selection.map(clmv!(to_from, |s| {
261            let len = s.len().min(u16::MAX as _) as u16;
262            s.iter()
263                .enumerate()
264                .take(u16::MAX as _)
265                .map(|(i, s)| ThumbValue {
266                    offset: to_from.to(s),
267                    n_of: (i as u16, len),
268                })
269                .collect()
270        }));
271
272        Self(Arc::new(Mutex::new(ManyImpl {
273            selection,
274            thumbs,
275            to_from,
276        })))
277    }
278
279    /// New with no value thumb.
280    pub fn nil() -> Self {
281        Self::many_with(vec![], |_: &bool| 0.fct(), |_| false)
282    }
283
284    /// Convert the value to a normalized factor.
285    ///
286    /// If `T` is not the same type returns `None`.
287    pub fn to_offset<T: VarValue>(&self, t: &T) -> Option<Factor> {
288        self.0.lock().to_offset(t)
289    }
290
291    /// Convert the normalized factor to a value `T`.
292    ///
293    /// If `T` is not the same type returns `None`.
294    pub fn from_offset<T: VarValue>(&self, offset: impl IntoValue<Factor>) -> Option<T> {
295        let b = self.0.lock().from_offset(offset.into()).downcast().ok()?;
296        Some(*b)
297    }
298
299    /// Move the thumb nearest to `from` to a new offset `to`.
300    ///
301    /// Note that ranges don't invert, this operation may swap the thumb roles.
302    pub fn set(&self, from: impl IntoValue<Factor>, to: impl IntoValue<Factor>) {
303        self.0.lock().set(from.into(), to.into())
304    }
305
306    /// The selection var.
307    ///
308    /// Downcast to `T`' or `Vec<T>` to get and set the value.
309    pub fn selection(&self) -> AnyVar {
310        self.0.lock().selection()
311    }
312
313    /// Read-only variable mapped from the [`selection`].
314    ///
315    /// [`selection`]: Self::selection
316    pub fn thumbs(&self) -> Var<Vec<ThumbValue>> {
317        self.0.lock().thumbs()
318    }
319}

321/// Represents a selector thumb in a slider.
322#[derive(Clone, Debug, PartialEq, Copy)]
323pub struct ThumbValue {
324    offset: Factor,
325    n_of: (u16, u16),
326}
327impl ThumbValue {
328    /// Thumb offset.
329    pub fn offset(&self) -> Factor {
330        self.offset
331    }
332
333    /// Thumb position among others.
334    ///
335    /// In a single value this is `(0, 1)`, in a range this is `(0, 2)` for the start thumb and `(1, 2)` for the end thumb.
336    pub fn n_of(&self) -> (u16, u16) {
337        self.n_of
338    }
339
340    /// Is first thumb (smallest offset).
341    pub fn is_first(&self) -> bool {
342        self.n_of.0 == 0
343    }
344
345    /// Is last thumb (largest offset).
346    pub fn is_last(&self) -> bool {
347        self.n_of.0 == self.n_of.1
348    }
349}

407/// Slider orientation and direction.
408#[derive(Clone, Copy, PartialEq, Eq)]
409pub enum SliderDirection {
410    /// Horizontal. Minimum at start, maximum at end.
411    ///
412    /// Start is left in LTR contexts and right in RTL contexts.
413    StartToEnd,
414    /// Horizontal. Minimum at end, maximum at start.
415    ///
416    /// Start is left in LTR contexts and right in RTL contexts.
417    EndToStart,
418    /// Horizontal. Minimum at left, maximum at right.
419    LeftToRight,
420    /// Horizontal. Minimum at right, maximum at left.
421    RightToLeft,
422    /// Vertical. Minimum at bottom, maximum at top.
423    BottomToTop,
424    /// Vertical. Minimum at top, maximum at bottom.
425    TopToBottom,
426}
427impl SliderDirection {
428    /// Slider track is vertical.
429    pub fn is_vertical(&self) -> bool {
430        matches!(self, Self::BottomToTop | Self::TopToBottom)
431    }
432
433    /// Slider track is horizontal.
434    pub fn is_horizontal(&self) -> bool {
435        !self.is_vertical()
436    }
437
438    /// Convert start/end to left/right in the given `direction` context.
439    pub fn layout(&self, direction: LayoutDirection) -> Self {
440        match *self {
441            SliderDirection::StartToEnd => {
442                if direction.is_ltr() {
443                    Self::LeftToRight
444                } else {
445                    Self::RightToLeft
446                }
447            }
448            SliderDirection::EndToStart => {
449                if direction.is_ltr() {
450                    Self::RightToLeft
451                } else {
452                    Self::LeftToRight
453                }
454            }
455            s => s,
456        }
457    }
458}
459impl fmt::Debug for SliderDirection {
460    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
461        if f.alternate() {
462            write!(f, "SliderDirection::")?;
463        }
464        match self {
465            Self::StartToEnd => write!(f, "StartToEnd"),
466            Self::EndToStart => write!(f, "EndToStart"),
467            Self::LeftToRight => write!(f, "LeftToRight"),
468            Self::RightToLeft => write!(f, "RightToLeft"),
469            Self::BottomToTop => write!(f, "BottomToTop"),
470            Self::TopToBottom => write!(f, "TopToBottom"),
471        }
472    }
473}

506fn slider_track_node() -> UiNode {
507    let mut layout_direction = LayoutDirection::LTR;
508    match_node(ui_vec![], move |thumbs, op| match op {
509        UiNodeOp::Init => {
510            WIDGET
511                .sub_var(&THUMB_FN_VAR)
512                .sub_event(&MOUSE_INPUT_EVENT)
513                .sub_event(&TOUCH_INPUT_EVENT)
514                .sub_event(&MOUSE_MOVE_EVENT);
515
516            let thumb_fn = THUMB_FN_VAR.get();
517
518            let thumbs_var = SELECTOR.get().thumbs();
519            let thumbs_len = thumbs_var.with(|t| t.len());
520            let thumbs_typed = thumbs.node_impl::<UiVec>();
521            thumbs_typed.reserve(thumbs_len);
522            for i in 0..thumbs_len {
523                let thumb_var = thumbs_var.map(move |t| {
524                    t.get(i).copied().unwrap_or(ThumbValue {
525                        offset: 0.fct(),
526                        n_of: (0, 0),
527                    })
528                });
529                thumbs_typed.push(thumb_fn(ThumbArgs { thumb: thumb_var }))
530            }
531
532            thumbs.init();
533        }
534        UiNodeOp::Deinit => {
535            thumbs.deinit();
536            *thumbs.node_impl::<UiVec>() = ui_vec![];
537        }
538        UiNodeOp::Info { info } => {
539            info.flag_meta(*IS_SLIDER_ID);
540            thumbs.info(info);
541        }
542        UiNodeOp::Measure { desired_size, .. } => {
543            thumbs.delegated();
544            *desired_size = LAYOUT.constraints().fill_size();
545        }
546        UiNodeOp::Layout { final_size, wl } => {
547            *final_size = LAYOUT.constraints().fill_size();
548            layout_direction = LAYOUT.direction();
549            let _ = thumbs.layout_list(wl, |_, n, wl| n.layout(wl), |_, _| PxSize::zero());
550        }
551        UiNodeOp::Event { update } => {
552            thumbs.event(update);
553
554            let mut pos = None;
555
556            if let Some(args) = MOUSE_MOVE_EVENT.on_unhandled(update) {
557                if let Some(cap) = &args.capture
558                    && cap.target.contains(WIDGET.id())
559                {
560                    pos = Some(args.position);
561                    args.propagation().stop();
562                }
563            } else if let Some(args) = MOUSE_INPUT_EVENT.on_unhandled(update) {
564                if args.state == ButtonState::Pressed {
565                    pos = Some(args.position);
566                    args.propagation().stop();
567                }
568            } else if let Some(args) = TOUCH_INPUT_EVENT.on_unhandled(update)
569                && args.phase == TouchPhase::Start
570            {
571                pos = Some(args.position);
572                args.propagation().stop();
573            }
574
575            if let Some(pos) = pos {
576                let track_info = WIDGET.info();
577                let track_bounds = track_info.inner_bounds();
578                let track_orientation = SLIDER_DIRECTION_VAR.get();
579
580                let (track_min, track_max) = match track_orientation.layout(layout_direction) {
581                    SliderDirection::LeftToRight => (track_bounds.min_x(), track_bounds.max_x()),
582                    SliderDirection::RightToLeft => (track_bounds.max_x(), track_bounds.min_x()),
583                    SliderDirection::BottomToTop => (track_bounds.max_y(), track_bounds.min_y()),
584                    SliderDirection::TopToBottom => (track_bounds.min_y(), track_bounds.max_y()),
585                    _ => unreachable!(),
586                };
587                let cursor = if track_orientation.is_horizontal() {
588                    pos.x.to_px(track_info.tree().scale_factor())
589                } else {
590                    pos.y.to_px(track_info.tree().scale_factor())
591                };
592                let new_offset = (cursor - track_min).0 as f32 / (track_max - track_min).abs().0 as f32;
593                let new_offset = new_offset.fct().clamp_range();
594
595                let selector = crate::SELECTOR.get();
596                selector.set(new_offset, new_offset);
597            }
598        }
599        UiNodeOp::Update { updates } => {
600            if let Some(thumb_fn) = THUMB_FN_VAR.get_new() {
601                thumbs.deinit();
602
603                let thumbs_vec = thumbs.node_impl::<UiVec>();
604                thumbs_vec.clear();
605
606                let thumbs_var = SELECTOR.get().thumbs();
607                let thumbs_len = thumbs_var.with(|t| t.len());
608                thumbs_vec.reserve(thumbs_len);
609                for i in 0..thumbs_len {
610                    let thumb_var = thumbs_var.map(move |t| {
611                        t.get(i).copied().unwrap_or(ThumbValue {
612                            offset: 0.fct(),
613                            n_of: (0, 0),
614                        })
615                    });
616                    thumbs_vec.push(thumb_fn(ThumbArgs { thumb: thumb_var }))
617                }
618
619                thumbs.init();
620
621                WIDGET.update_info().layout().render();
622            } else {
623                thumbs.update(updates);
624
625                // sync views and vars with updated SELECTOR thumbs
626
627                let thumbs_var = SELECTOR.get().thumbs();
628                let thumbs_len = thumbs_var.with(|t| t.len());
629
630                match thumbs_len.cmp(&thumbs.node().children_len()) {
631                    std::cmp::Ordering::Less => {
632                        // now has less thumbs
633                        for mut drop in thumbs.node_impl::<UiVec>().drain(thumbs_len..) {
634                            drop.deinit();
635                        }
636                    }
637                    std::cmp::Ordering::Greater => {
638                        // now has more thumbs
639                        let thumbs_vec = thumbs.node_impl::<UiVec>();
640                        let thumb_fn = THUMB_FN_VAR.get();
641                        let from_len = thumbs_vec.len();
642                        thumbs_vec.reserve(thumbs_len - from_len);
643                        for i in from_len..thumbs_len {
644                            let thumb_var = thumbs_var.map(move |t| {
645                                t.get(i).copied().unwrap_or(ThumbValue {
646                                    offset: 0.fct(),
647                                    n_of: (0, 0),
648                                })
649                            });
650                            thumbs_vec.push(thumb_fn(ThumbArgs { thumb: thumb_var }))
651                        }
652                    }
653                    std::cmp::Ordering::Equal => {}
654                }
655            }
656        }
657        _ => {}
658    })
659}

661macro_rules! impl_32 {
662    (($to_f32:expr, $from_f32:expr) => $($T:ident),+ $(,)?) => {
663        $(
664            impl SelectorValue for $T {
665                #[allow(clippy::unnecessary_cast)]
666                fn to_selector(value: Var<Self>, min: Self, max: Self) -> Selector {
667                    let to_f32 = $to_f32;
668                    let from_f32 = $from_f32;
669
670                    let min = to_f32(min);
671                    let max = to_f32(max);
672                    if min >= max {
673                        Selector::nil()
674                    } else {
675                        let d = max - min;
676                        Selector::value_with(value, move |i| {
677                            let i = to_f32(i.clone());
678                            ((i - min)  / d).fct()
679                        }, move |f| {
680                            from_f32((f.0 * d + min).round())
681                        })
682                    }
683                }
684            }
685        )+
686    };
687}
688impl_32!((|i| i as f32, |f| f as Self) => u32, i32, u16, i16, u8, i8, f32);
689impl_32!((|p: Self| p.0 as f32, |f| Self(f as _)) => Px, Factor, FactorPercent);
690impl_32!((|p: Dip| p.to_f32(), |f| Dip::new_f32(f)) => Dip);

701                    } else {

717impl SelectorValue for Length {
718    fn to_selector(value: Var<Self>, min: Self, max: Self) -> Selector {
719        let (min_f32, max_f32) = LAYOUT.with_context(LayoutMetrics::new(1.fct(), PxSize::splat(Px(1000)), Px(16)), || {
720            (min.layout_f32(LayoutAxis::X), max.layout_f32(LayoutAxis::X))
721        });
722        if min_f32 >= max_f32 {
723            Selector::nil()
724        } else {
725            let d_f32 = max_f32 - min_f32;
726            let d = max - min.clone();
727            Selector::value_with(
728                value,
729                move |l| {
730                    let l = LAYOUT.with_context(LayoutMetrics::new(1.fct(), PxSize::splat(Px(1000)), Px(16)), || {
731                        l.layout_f32(LayoutAxis::X)
732                    });
733                    Factor((l - min_f32) / d_f32)
734                },
735                move |f| d.clone() * f + min.clone(),
736            )
737        }
738    }
739}
