3//! Keyboard shortcut display widget.
4//!
5//! The [`ShortcutText!`] is composite widget that generates localized and styled shortcut *text*, it
6//! can handle all shortcut variations, multiple shortcuts and partially invalid shortcuts.
7//! Extensive configuration is possible using contextual properties that can override.
8//!
9//! The example below demonstrates a basic *key binding editor* that uses the [`ShortcutText!`] widget in multiple places.
10//!
11//! ```
12//! # use zng::focus::{focus_on_init, focusable};
13//! # use zng::gesture::Shortcuts;
14//! # use zng::keyboard::{Key, KeyInputArgs, on_pre_key_down};
15//! # use zng::layout::{align, min_height};
16//! # use zng::prelude::*;
17//! # use zng::shortcut_text::ShortcutText;
18//! #
19//! pub fn shortcut_input(shortcut: Var<Shortcuts>) -> UiNode {
20//!     Button! {
21//!         // display the shortcut, or the `none_fn` content if there is no shortcut.
22//!         child = ShortcutText! {
23//!             shortcut = shortcut.clone();
24//!             none_fn = wgt_fn!(|_| Text!("no shortcut"));
25//!         };
26//!         on_click = hn!(|_| {
27//!             DIALOG.custom(shortcut_input_dialog(shortcut.clone()));
28//!         });
29//!     }
30//! }
31//! fn shortcut_input_dialog(output: Var<gesture::Shortcuts>) -> UiNode {
32//!     let pressed = var(Shortcuts::new());
33//!     let is_valid = var(true);
34//!     Container! {
35//!         child_top =
36//!             Wrap!(ui_vec![
37//!                 Text!("Press the new shortcut and then press "),
38//!                 ShortcutText!(shortcut!(Enter)), // shortcut text supports inlining
39//!             ]),
40//!             20,
41//!         ;
42//!         // default style is derived from the `font_size` and `font_color` values.
43//!         child = ShortcutText! {
44//!             shortcut = pressed.clone();
45//!             font_size = 3.em();
46//!             align = Align::TOP;
47//!             when !#{is_valid.clone()} {
48//!                 font_color = colors::RED;
49//!             }
50//!         };
51//!
52//!         on_pre_key_down = hn!(|args: &KeyInputArgs| {
53//!             args.propagation().stop();
54//!             match &args.key {
55//!                 Key::Enter => {
56//!                     let shortcut = pressed.get();
57//!                     if shortcut.is_empty() || shortcut[0].is_valid() {
58//!                         is_valid.set(true);
59//!                         output.set(shortcut);
60//!                         DIALOG.respond(dialog::Response::ok());
61//!                     } else {
62//!                         is_valid.set(false);
63//!                     }
64//!                 }
65//!                 Key::Escape => {
66//!                     DIALOG.respond(dialog::Response::cancel());
67//!                 }
68//!                 _ => {
69//!                     is_valid.set(true); // clear
70//!                     pressed.set(args.editing_shortcut().unwrap());
71//!                 }
72//!             }
73//!         });
74//!         align = Align::CENTER;
75//!         min_height = 200;
76//!         focusable = true;
77//!         focus_on_init = true;
78//!     }
79//! }
80//! ```
81//!
82//! [`ShortcutText!`]: struct@ShortcutText
83//!
84//! # Full API
85//!
86//! See [`zng_wgt_shortcut`] for the full widget API.
