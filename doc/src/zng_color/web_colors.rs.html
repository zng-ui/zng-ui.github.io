5macro_rules! rgb {
6    ($r:literal, $g:literal, $b:literal) => {
7        Rgba {
8            red: $r as f32 / 255.,
9            green: $g as f32 / 255.,
10            blue: $b as f32 / 255.,
11            alpha: 1.0,
12        }
13    };
14}
