6/// Collapse adjacent descendant rule lines.
7///
8/// Set this in a panel widget to automatically collapse rule lines that would appear repeated on screen.
9#[property(LAYOUT - 100)]
10pub fn collapse_scope(child: impl IntoUiNode, mode: impl IntoVar<CollapseMode>) -> UiNode {
11    let mode = mode.into_var();
12    let mut scope: Option<Arc<CollapseScope>> = None;
13    match_node(child, move |c, op| match op {
14        UiNodeOp::Init => {
15            WIDGET.sub_var_layout(&mode);
16            scope = Some(Arc::new(CollapseScope::new(WIDGET.id())));
17        }
18        UiNodeOp::Deinit => {
19            scope = None;
20        }
21        UiNodeOp::Measure { wm, desired_size } => {
22            *desired_size = SCOPE.with_context(&mut scope, || c.measure(wm));
23        }
24        UiNodeOp::Layout { wl, final_size } => {
25            *final_size = SCOPE.with_context(&mut scope, || c.layout(wl));
26
27            // update collapsed list
28            let mode = mode.get();
29
30            // try to reuse the current scope
31            let maybe_exclusive = Arc::get_mut(scope.as_mut().unwrap());
32            // if not possible alloc new (this can happen if a child captured context and is keeping it)
33            let is_new = maybe_exclusive.is_none();
34            let mut new = CollapseScope::new(WIDGET.id());
35            let s = maybe_exclusive.unwrap_or(&mut new);
36
37            // tracks changes in reused set, ignore this if is_new
38            let mut changes = UpdateDeliveryList::new_any();
39            if mode.is_empty() {
40                if !s.collapse.is_empty() {
41                    let info = WIDGET.info();
42                    let info = info.tree();
43                    for id in s.collapse.drain() {
44                        if let Some(wgt) = info.get(id) {
45                            changes.insert_wgt(&wgt);
46                        }
47                    }
48                }
49            } else {
50                // does one pass of the info tree descendants collecting collapsable lines,
51                // it is a bit complex to avoid allocating the `IdMap` for most widgets,
52                // flags `changed` if a second layout pass is needed
53
54                let info = WIDGET.info();
55
56                let mut trim_start = mode.contains(CollapseMode::TRIM_START);
57                let mut trim_end_id = None;
58                if mode.contains(CollapseMode::TRIM_END) {
59                    // find trim_end start *i* first, so that we can update `s.collapse` in a single pass
60                    for wgt in info.descendants().tree_rev() {
61                        if wgt.meta().flagged(*COLLAPSABLE_LINE_ID) {
62                            trim_end_id = Some(wgt.id());
63                        } else if wgt.descendants_len() == 0 && !wgt.bounds_info().inner_size().is_empty() {
64                            // only consider leafs that are not collapsed
65                            break;
66                        }
67                    }
68                }
69                let mut trim_end = false;
70                let mut merge = false;
71                for wgt in info.descendants() {
72                    if wgt.meta().flagged(*COLLAPSABLE_LINE_ID) {
73                        if let Some(id) = trim_end_id
74                            && id == wgt.id()
75                        {
76                            trim_end_id = None;
77                            trim_end = true;
78                        }
79                        let changed = if trim_start || merge || trim_end {
80                            s.collapse.insert(wgt.id())
81                        } else {
82                            s.collapse.remove(&wgt.id())
83                        };
84                        if changed && !is_new {
85                            changes.insert_wgt(&wgt);
86                        }
87                    } else if wgt.descendants_len() == 0 && !wgt.bounds_info().inner_size().is_empty() {
88                        // only consider leafs that are not collapsed
89                        trim_start = false;
90                        merge = false;
91                    }
92                }
93            }
94            if is_new {
95                let s = scope.as_mut().unwrap();
96                // previous changed state set assuming it was reusing set, override it
97                let info = WIDGET.info();
98                let info = info.tree();
99                for id in s.collapse.symmetric_difference(&new.collapse) {
100                    if let Some(wgt) = info.get(*id) {
101                        changes.insert_wgt(&wgt);
102                    }
103                }
104                if !changes.widgets().is_empty() {
105                    scope = Some(Arc::new(new));
106                }
107            }
108
109            if !changes.widgets().is_empty() {
110                *final_size = wl.with_layout_updates(Arc::new(LayoutUpdates::new(changes)), |wl| {
111                    SCOPE.with_context(&mut scope, || c.layout(wl))
112                });
113            }
114        }
115        _ => {}
116    })
117}

119bitflags::bitflags! {
120    /// Represents what rule lines are collapsed in a [`collapse_scope`].
121    ///
122    /// [`collapse_scope`]: fn@collapse_scope
123    #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
124    pub struct CollapseMode: u8 {
125        /// Collapse first line(s) if it has no previous visible sibling in the scope.
126        const TRIM_START = 0b0000_0001;
127        /// Collapse the last line(s) if it has no next visible sibling in the scope.
128        const TRIM_END = 0b0000_0010;
129        /// Collapse start and end.
130        const TRIM = Self::TRIM_START.bits() | Self::TRIM_END.bits();
131
132        /// Adjacent lines without intermediary visible siblings are collapsed except the first in sequence.
133        const MERGE = 0b0001_0000;
134    }
135}
136impl_from_and_into_var! {
137    fn from(all: bool) -> CollapseMode {
138        if all { CollapseMode::all() } else { CollapseMode::empty() }
139    }
140}

142/// Contextual service managed by [`collapse_scope`].
143///
144/// Custom line widgets not derived from [`RuleLine!`] can participate in [`collapse_scope`] by setting [`COLLAPSABLE_LINE_ID`]
145/// during info build and using this service during measure and layout.
146///
147/// [`collapse_scope`]: fn@collapse_scope
148/// [`RuleLine!`]: struct@crate::RuleLine
149#[allow(non_camel_case_types)]
150pub struct COLLAPSE_SCOPE;
151impl COLLAPSE_SCOPE {
152    /// Get the parent scope ID.
153    pub fn scope_id(&self) -> Option<WidgetId> {
154        SCOPE.get().scope_id
155    }
156
157    ///Gets if the line widget needs to collapse
158    pub fn collapse(&self, line_id: WidgetId) -> bool {
159        let scope = SCOPE.get();
160        scope.collapse.contains(&line_id)
161    }
162}
