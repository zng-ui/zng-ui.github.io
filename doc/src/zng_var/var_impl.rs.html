193bitflags! {
194    /// Kinds of interactions allowed by a [`Var<T>`] in the current update.
195    ///
196    /// You can get the current capabilities of a var by using the [`AnyVar::capabilities`] method.
197    ///
198    /// [`Var<T>`]: crate::Var
199    /// [`AnyVar::capabilities`]: crate::AnyVar::capabilities
200    #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
201    pub struct VarCapability: u8 {
202        /// Variable value can change.
203        ///
204        /// If this is set the [`AnyVar::is_new`] can be `true` in some updates, a variable can `NEW`
205        /// even if it cannot `MODIFY`, in this case the variable is a read-only wrapper on a read-write variable.
206        ///
207        /// [`AnyVar::is_new`]: crate::AnyVar::is_new
208        const NEW = 0b0000_0010;
209
210        /// Variable can be modified.
211        ///
212        /// If this is set [`Var::try_modify`] always returns `Ok`, if this is set `NEW` is also set.
213        ///
214        /// Note that modify requests from inside overridden animations can still be ignored, see [`AnyVar::modify_importance`].
215        ///
216        /// [`AnyVar::modify_importance`]: crate::AnyVar::modify_importance
217        /// [`Var::try_modify`]: crate::Var::try_modify
218        const MODIFY = 0b0000_0011;
219
220        /// Var represents different inner variables depending on the context it is used.
221        const CONTEXT = 0b1000_0000;
222
223        /// Variable capabilities can change to sometimes have the `MODIFY` capability.
224        const MODIFY_CHANGES = 0b0100_0000;
225        /// Variable capabilities can change to sometimes have the `CONTEXT` capability.
226        const CONTEXT_CHANGES = 0b0010_0000;
227
228        /// Var is an *arc* reference to the value and variable state, cloning the variable only clones a
229        /// reference to the variable, all references modify and notify the same state.
230        const SHARE = 0b0001_0000;
231    }
232}
233impl VarCapability {
234    /// If cannot `NEW` and is not `MODIFY_CHANGES`.
235    pub fn is_const(self) -> bool {
236        !self.contains(Self::NEW) && !self.contains(Self::MODIFY_CHANGES)
237    }
238
239    /// If does not have `MODIFY` capability and is not `MODIFY_CHANGES`.
240    pub fn is_always_read_only(&self) -> bool {
241        !self.contains(Self::MODIFY) && !self.contains(Self::MODIFY_CHANGES)
242    }
243
244    /// If does not have `MODIFY` capability.
245    pub fn is_read_only(self) -> bool {
246        !self.can_modify()
247    }
248
249    /// Has the `MODIFY` capability.
250    pub fn can_modify(self) -> bool {
251        self.contains(Self::MODIFY)
252    }
253
254    /// Has the `CONTEXT` capability.
255    pub fn is_contextual(self) -> bool {
256        self.contains(Self::CONTEXT)
257    }
258
259    /// Has the `CONTEXT` capability and does not have `CONTEXT_CHANGES`.
260    pub fn is_always_contextual(self) -> bool {
261        self.contains(Self::CONTEXT) && !self.contains(Self::CONTEXT_CHANGES)
262    }
263
264    /// Has the `SHARE` capability.
265    pub fn is_share(&self) -> bool {
266        self.contains(Self::SHARE)
267    }
268
269    /// Does not have the `SHARE` capability.
270    ///
271    /// Cloning this variable clones the value.
272    pub fn is_local(&self) -> bool {
273        !self.is_share()
274    }
275}
276impl VarCapability {
277    pub(crate) fn as_always_read_only(self) -> Self {
278        let mut out = self;
279
280        // can be new, but not modify
281        out.remove(Self::MODIFY & !Self::NEW);
282        // never will allow modify
283        out.remove(Self::MODIFY_CHANGES);
284
285        out
286    }
287}

301/// Mutable reference to a variable value.
302///
303/// The variable will notify an update only on `deref_mut`.
304pub struct AnyVarModify<'a> {
305    pub(crate) value: &'a mut BoxAnyVarValue,
306    pub(crate) update: VarModifyUpdate,
307    pub(crate) tags: Vec<BoxAnyVarValue>,
308    pub(crate) custom_importance: Option<usize>,
309}
310impl<'a> AnyVarModify<'a> {
311    /// Replace the value if not equal.
312    ///
313    /// Note that you can also deref_mut to modify the value.
314    pub fn set(&mut self, mut new_value: BoxAnyVarValue) -> bool {
315        if **self.value != *new_value {
316            if !self.value.try_swap(&mut *new_value) {
317                #[cfg(feature = "type_names")]
318                panic!(
319                    "cannot AnyVarModify::set `{}` on variable of type `{}`",
320                    new_value.type_name(),
321                    self.value.type_name()
322                );
323                #[cfg(not(feature = "type_names"))]
324                panic!("cannot modify set, type mismatch");
325            }
326            self.update |= VarModifyUpdate::TOUCHED;
327            true
328        } else {
329            false
330        }
331    }
332
333    /// Notify an update, even if the value does not actually change.
334    pub fn update(&mut self) {
335        self.update |= VarModifyUpdate::REQUESTED;
336    }
337
338    /// Custom tags that will be shared with the var hooks if the value updates.
339    ///
340    /// The tags where set by previous modify closures or this one during this update cycle, so
341    /// tags can also be used to communicate between modify closures.
342    pub fn tags(&self) -> &[BoxAnyVarValue] {
343        &self.tags
344    }
345
346    /// Add a custom tag object that will be shared with the var hooks if the value updates.
347    pub fn push_tag(&mut self, tag: impl AnyVarValue) {
348        self.tags.push(BoxAnyVarValue::new(tag));
349    }
350
351    /// Sets a custom [`AnyVar::modify_importance`] value.
352    ///
353    /// Note that the modify info is already automatically set, using a custom value here
354    /// can easily break all future modify requests for this variable. The importance is set even if the
355    /// variable does not update (no actual value change or update request).
356    ///
357    /// [`AnyVar::modify_importance`]: crate::AnyVar::modify_importance
358    pub fn set_modify_importance(&mut self, importance: usize) {
359        self.custom_importance = Some(importance);
360    }
361
362    /// Strongly typed reference, if it is of the same type.
363    pub fn downcast<'s, T: VarValue>(&'s mut self) -> Option<VarModify<'s, 'a, T>> {
364        if self.value.is::<T>() {
365            Some(VarModify {
366                inner: self,
367                _t: PhantomData,
368            })
369        } else {
370            None
371        }
372    }
373
374    /// Immutable reference to the value.
375    ///
376    /// Note that you can also simply deref to the value.
377    pub fn value(&self) -> &dyn AnyVarValue {
378        &**self
379    }
380
381    /// Mutable reference to the value.
382    ///
383    /// Getting a mutable reference to the value flags the variable to notify update.
384    ///
385    /// Note that you can also simply deref to the value.
386    pub fn value_mut(&mut self) -> &mut dyn AnyVarValue {
387        &mut **self
388    }
389}
390impl<'a> ops::Deref for AnyVarModify<'a> {
391    type Target = dyn AnyVarValue;
392
393    fn deref(&self) -> &Self::Target {
394        &**self.value
395    }
396}
397impl<'a> ops::DerefMut for AnyVarModify<'a> {
398    fn deref_mut(&mut self) -> &mut Self::Target {
399        self.update |= VarModifyUpdate::TOUCHED;
400        self.value.deref_mut()
401    }
402}

488/// Handle to a variable or animation hook.
489///
490/// This can represent a widget subscriber, a var binding, var app handler or animation, dropping the handler stops
491/// the behavior it represents.
492///
493/// Note that the hook closure is not dropped immediately when the handle is dropped, usually it will drop only the next
494/// time it would have been called.
495#[derive(Clone, Default)]
496#[must_use = "var handle stops the behavior it represents on drop"]
497pub struct VarHandle(Option<Arc<AtomicBool>>);
498impl PartialEq for VarHandle {
499    fn eq(&self, other: &Self) -> bool {
500        if let Some(a) = &self.0
501            && let Some(b) = &other.0
502        {
503            Arc::ptr_eq(a, b)
504        } else {
505            false
506        }
507    }
508}
509impl fmt::Debug for VarHandle {
510    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
511        if self.is_dummy() {
512            write!(f, "VarHandle(<dummy>)")
513        } else {
514            f.debug_tuple("VarHandle").finish_non_exhaustive()
515        }
516    }
517}
518impl VarHandle {
519    /// Handle to no variable.
520    pub const fn dummy() -> Self {
521        VarHandle(None)
522    }
523
524    pub(crate) fn new() -> (VarHandlerOwner, Self) {
525        let h = Arc::new(AtomicBool::new(false));
526        (VarHandlerOwner(h.clone()), Self(Some(h)))
527    }
528
529    /// Returns `true` if the handle is a [`dummy`].
530    ///
531    /// [`dummy`]: VarHandle::dummy
532    pub fn is_dummy(&self) -> bool {
533        self.0.is_none()
534    }
535
536    /// Drop the handle without stopping the behavior it represents.
537    ///
538    /// Note that the behavior can still be stopped by dropping the involved variables.
539    pub fn perm(self) {
540        if let Some(c) = &self.0 {
541            c.store(true, std::sync::atomic::Ordering::Relaxed);
542        }
543    }
544
545    /// Create a [`VarHandles`] collection with `self` and `other`.
546    pub fn chain(self, other: Self) -> VarHandles {
547        VarHandles(smallvec::smallvec![self, other])
548    }
549}
550pub(crate) struct VarHandlerOwner(Arc<AtomicBool>);
551impl fmt::Debug for VarHandlerOwner {
552    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
553        write!(f, "{}", Arc::strong_count(&self.0) - 1)?;
554        if self.0.load(std::sync::atomic::Ordering::Relaxed) {
555            write!(f, " perm")
556        } else {
557            Ok(())
558        }
559    }
560}
561impl VarHandlerOwner {
562    pub fn is_alive(&self) -> bool {
563        Arc::strong_count(&self.0) > 1 || self.0.load(std::sync::atomic::Ordering::Relaxed)
564    }
565}

567/// Represents a collection of var handles.
568#[must_use = "var handles stops the behavior they represents on drop"]
569#[derive(Clone, Default)]
570pub struct VarHandles(smallvec::SmallVec<[VarHandle; 2]>);
571impl VarHandles {
572    /// Empty collection.
573    pub const fn dummy() -> Self {
574        VarHandles(smallvec::SmallVec::new_const())
575    }
576
577    /// Returns `true` if empty or all handles are dummy.
578    pub fn is_dummy(&self) -> bool {
579        self.0.is_empty() || self.0.iter().all(VarHandle::is_dummy)
580    }
581
582    /// Drop all handles without stopping their behavior.
583    pub fn perm(self) {
584        for handle in self.0 {
585            handle.perm()
586        }
587    }
588
589    /// Add the `other` handle to the collection, if it is not dummy.
590    pub fn push(&mut self, other: VarHandle) -> &mut Self {
591        if !other.is_dummy() {
592            self.0.push(other);
593        }
594        self
595    }
596
597    /// Drop all handles.
598    pub fn clear(&mut self) {
599        self.0.clear()
600    }
601}
602impl FromIterator<VarHandle> for VarHandles {
603    fn from_iter<T: IntoIterator<Item = VarHandle>>(iter: T) -> Self {
604        VarHandles(iter.into_iter().filter(|h| !h.is_dummy()).collect())
605    }
606}

609        handles.into_iter().collect()
610    }

614        for handle in iter {

617    }
