15type MapFn = SmallBox<dyn FnMut(&dyn AnyVarValue) -> AnyVar + Send + 'static, smallbox::space::S4>;
16
17/// source var is stored as the value of the `write` SharedVar
18struct FlatMapData {
19    source: AnyVar,
20    map: Mutex<MapFn>,
21    current: RwLock<(AnyVar, VarHandle)>,
22    hooks: MutexHooks,
23}
24
25#[derive(Clone)]
26pub(crate) struct FlatMapVar(Arc<FlatMapData>);
27impl fmt::Debug for FlatMapVar {
28    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
29        let mut b = f.debug_struct("FlatMapVar");
30        b.field("source", &self.0.source);
31        if let Some(c) = self.0.current.try_read() {
32            b.field("current", &c.0);
33        } else {
34            b.field("current", &"<locked>");
35        }
36        b.field("hooks", &self.0.hooks);
37        b.finish()
38    }
39}
40impl FlatMapVar {
41    pub(crate) fn new(source: AnyVar, mut map: MapFn) -> Self {
42        let init = source.with(|v| map(v));
43
44        let data = Arc::new(FlatMapData {
45            source,
46            map: Mutex::new(map),
47            current: RwLock::new((init, VarHandle::dummy())),
48            hooks: MutexHooks::default(),
49        });
50
51        hook_inner_var(&data, data.current.write());
52
53        let weak = Arc::downgrade(&data);
54        data.source
55            .hook(move |args| {
56                if let Some(data) = weak.upgrade() {
57                    let new_inner = data.map.lock()(args.value());
58
59                    let mut current = data.current.write();
60                    if !current.0.var_eq(&new_inner) {
61                        current.1 = VarHandle::dummy();
62                        current.0 = new_inner;
63                        hook_inner_var(&data, current);
64
65                        // FlatMapVar will show as new because `source` just updated, so
66                        // notify hooks here to match
67                        data.current.read().0.with(|v| {
68                            data.hooks.notify(&AnyVarHookArgs {
69                                var_instance_tag: VarInstanceTag(Arc::as_ptr(&data) as _),
70                                value: v,
71                                update: args.update,
72                                tags: args.tags,
73                            });
74                        });
75                    }
76                    // retain hook, we are still alive
77                    true
78                } else {
79                    false
80                }
81            })
82            .perm();
83
84        Self(data)
85    }
86}
87
88fn hook_inner_var(data: &Arc<FlatMapData>, mut current: parking_lot::RwLockWriteGuard<(AnyVar, VarHandle)>) {
89    let weak = Arc::downgrade(data);
90    let init_handle = current.0.hook(move |args| {
91        if let Some(data) = weak.upgrade() {
92            data.hooks.notify(&AnyVarHookArgs {
93                var_instance_tag: VarInstanceTag(Arc::as_ptr(&data) as _),
94                value: args.value,
95                update: args.update,
96                tags: args.tags,
97            });
98
99            // retain hook to current inner var
100            true
101        } else {
102            false
103        }
104    });
105    current.1 = init_handle;
106}
