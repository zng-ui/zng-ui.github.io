15/// New read/write shared reference variable.
16pub fn var<T: VarValue>(initial_value: T) -> Var<T> {
17    Var::new_any(any_var(BoxAnyVarValue::new(initial_value)))
18}

20/// New read/write shared reference type-erased variable that has initial value derived from `source`.
21///
22/// This function is useful for creating custom mapping outputs, the new variable
23/// starts with the same [`AnyVar::last_update`] and animation handle as the `source`.
24pub fn var_derived<T: VarValue>(initial_value: T, source: &AnyVar) -> Var<T> {
25    Var::new_any(any_var_derived(BoxAnyVarValue::new(initial_value), source))
26}

49/// Variable for state properties (`is_*`, `has_*`).
50///
51/// State variables are `bool` probes that are set by the property, they are created automatically
52/// by the property default when used in `when` expressions, but can be created manually.
53pub fn var_state() -> Var<bool> {
54    var(false)
55}
56
57/// Variable for getter properties (`get_*`, `actual_*`).
58///
59/// Getter variables are inited with a default value that is overridden by the property on node init and updated
60/// by the property when the internal state they track changes. They are created automatically by the property
61/// default when used in `when` expressions, but can be created manually.
62pub fn var_getter<T: VarValue + Default>() -> Var<T> {
63    var(T::default())
64}

72pub(crate) struct SharedVar(pub(super) Arc<VarData>);
73impl fmt::Debug for SharedVar {
74    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
75        let mut b = f.debug_struct("SharedVar");
76        b.field("var_instance_tag()", &Arc::as_ptr(&self.0));
77        b.field("strong_count()", &self.strong_count());
78
79        if let Some(value) = self.0.value.try_read() {
80            b.field("value", &value.0.detailed_debug());
81            b.field("last_update", &value.1);
82            b.field("modify_info", &value.2);
83        } else {
84            b.field("value", &"<locked>");
85        }
86
87        b.field("hooks", &self.0.hooks);
88
89        b.finish()
90    }
91}

195impl SharedVar {
196    fn modify_impl(&self, value_or_modify: ValueOrModify) {
197        let name = value_type_name(self);
198        let var = self.clone();
199        // not weak ref here because some vars are spawned modified just to notify something and dropped
200        VARS.schedule_update(name, move || {
201            let mut value = var.0.value.write();
202
203            // verify if contextual animation can still set
204            let current_modify = VARS.current_modify();
205            if current_modify.importance() < value.2.importance() {
206                return;
207            }
208            value.2 = current_modify;
209
210            // modify
211            let mut m = AnyVarModify {
212                value: &mut value.0,
213                update: VarModifyUpdate::empty(),
214                tags: vec![],
215                custom_importance: None,
216            };
217            match value_or_modify {
218                ValueOrModify::Value(v) => {
219                    m.set(v);
220                }
221                ValueOrModify::Modify(mut f) => (f)(&mut m),
222            }
223
224            let AnyVarModify {
225                update,
226                tags,
227                custom_importance,
228                ..
229            } = m;
230
231            if let Some(i) = custom_importance {
232                value.2.importance = i;
233            }
234
235            if update.contains(VarModifyUpdate::UPDATE) {
236                value.1 = VARS.update_id();
237
238                let value = parking_lot::RwLockWriteGuard::downgrade(value);
239                let args = AnyVarHookArgs::new(
240                    var.var_instance_tag(),
241                    &*value.0,
242                    update.contains(VarModifyUpdate::REQUESTED),
243                    &tags,
244                );
245                var.0.hooks.notify(&args);
246            }
247        });
248    }
249}
250// both boxes are space::S4, so can't implement `set` as `modify` without alloc
251// this type and `modify_impl` work around that
252enum ValueOrModify {
253    Value(BoxAnyVarValue),
254    Modify(SmallBox<dyn FnMut(&mut AnyVarModify) + Send + 'static, smallbox::space::S4>),
255}

283#[derive(Default)]
284pub(super) struct MutexHooks {
285    h: Mutex<SmallVec<[(HookFn, VarHandlerOwner); 1]>>,
286}
287impl MutexHooks {
288    pub fn push(&self, on_new: HookFn) -> VarHandle {
289        let (owner, handle) = VarHandle::new();
290        self.h.lock().push((on_new, owner));
291        handle
292    }
293
294    pub fn notify(&self, args: &AnyVarHookArgs) {
295        let mut hooks = mem::take(&mut *self.h.lock());
296
297        hooks.retain(|(f, handle)| handle.is_alive() && f(args));
298
299        if !hooks.is_empty() {
300            let mut hs = self.h.lock();
301            if hs.capacity() > hooks.capacity() {
302                hs.append(&mut hooks);
303            } else {
304                hooks.append(&mut *hs);
305                *hs = hooks;
306            }
307        }
308    }
309}
310impl fmt::Debug for MutexHooks {
311    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
312        if let Some(h) = self.h.try_lock() {
313            let mut b = f.debug_list();
314            for (_, h) in h.iter() {
315                b.entry(h);
316            }
317            b.finish()
318        } else {
319            write!(f, "<locked>")
320        }
321    }
322}
