19/// Represents the last time a variable was mutated or the current update cycle.
20#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, bytemuck::NoUninit)]
21#[repr(transparent)]
22pub struct VarUpdateId(pub(crate) u32);
23impl VarUpdateId {
24    /// ID that is never new.
25    pub const fn never() -> Self {
26        VarUpdateId(0)
27    }
28
29    fn next(&mut self) {
30        if self.0 == u32::MAX {
31            self.0 = 1;
32        } else {
33            self.0 += 1;
34        }
35    }
36}
37impl Default for VarUpdateId {
38    fn default() -> Self {
39        Self::never()
40    }
41}

172    /// Adds an `animation` closure that is called every frame to update captured variables, starting after next frame.
173    ///
174    /// This is used to implement all [`Var<T>`] animations, it enables any kind of variable animation,
175    /// including multiple variables.
176    ///
177    /// Returns an [`AnimationHandle`] that can be used to monitor the animation status and to [`stop`] or to
178    /// make the animation [`perm`].
179    ///
180    /// # Variable Control
181    ///
182    /// Animations assume *control* of a variable on the first time they cause its value to be new, after this
183    /// moment the [`AnyVar::is_animating`] value is `true` and [`AnyVar::modify_importance`] is the animation's importance,
184    /// until the animation stops. Only one animation can control a variable at a time, if an animation loses control of a
185    /// variable all attempts to modify it from inside the animation are ignored.
186    ///
187    /// Later started animations steal control from previous animations, update, modify or set calls also remove the variable
188    /// from being affected by a running animation, even if just set to an equal value, that is, not actually updated.
189    ///
190    /// # Nested Animations
191    ///
192    /// Other animations can be started from inside the animation closure, these *nested* animations have the same importance
193    /// as the *parent* animation, the animation handle is different and [`AnyVar::is_animating`] is `false` if the nested animation
194    /// is dropped before the *parent* animation. But because the animations share the same importance the parent animation can
195    /// set the variable again.
196    ///
197    /// # Examples
198    ///
199    /// The example animates a `text` variable from `"Animation at 0%"` to `"Animation at 100%"`, when the animation
200    /// stops the `completed` variable is set to `true`.
201    ///
202    /// ```
203    /// # use zng_var::{*, animation::easing};
204    /// # use zng_txt::*;
205    /// # use zng_unit::*;
206    /// # use zng_clone_move::*;
207    /// #
208    /// fn animate_text(text: &Var<Txt>, completed: &Var<bool>) {
209    ///     let transition = animation::Transition::new(0u8, 100);
210    ///     let mut prev_value = 101;
211    ///     VARS.animate(clmv!(text, completed, |animation| {
212    ///         let step = easing::expo(animation.elapsed_stop(1.secs()));
213    ///         let value = transition.sample(step);
214    ///         if value != prev_value {
215    ///             if value == 100 {
216    ///                 animation.stop();
217    ///                 let _ = completed.set(true);
218    ///             }
219    ///             let _ = text.set(formatx!("Animation at {value}%"));
220    ///             prev_value = value;
221    ///         }
222    ///     }))
223    ///     .perm()
224    /// }
225    /// ```
226    ///
227    /// Note that the animation can be stopped from the inside, the closure parameter is an [`Animation`]. In
228    /// the example this is the only way to stop the animation, because [`perm`] was called. Animations hold a clone
229    /// of the variables they affect and exist for the duration of the app if not stopped, causing the app to wake and call the
230    /// animation closure for every frame.
231    ///
232    /// This method is the most basic animation interface, used to build all other animations, its rare that you
233    /// will need to use it directly, most of the time animation effects can be composted using the [`Var`] easing and mapping
234    /// methods.
235    ///
236    /// ```
237    /// # use zng_var::{*, animation::easing};
238    /// # use zng_txt::*;
239    /// # use zng_unit::*;
240    /// # fn demo() {
241    /// let value = var(0u8);
242    /// let text = value.map(|v| formatx!("Animation at {v}%"));
243    /// value.ease(100, 1.secs(), easing::expo);
244    /// # }
245    /// ```
246    ///
247    /// # Optimization Tips
248    ///
249    /// When no animation is running the app *sleeps* awaiting for an external event, update request or timer elapse, when at least one
250    /// animation is running the app awakes every [`VARS.frame_duration`]. You can use [`Animation::sleep`] to *pause* the animation
251    /// for a duration, if all animations are sleeping the app is also sleeping.
252    ///
253    /// Animations lose control over a variable permanently when a newer animation modifies the var or
254    /// the var is modified directly, but even if the animation can't control any variables **it keeps running**.
255    /// This happens because the system has no insight of all side effects caused by the `animation` closure. You
256    /// can use the [`VARS.current_modify`] and [`AnyVar::modify_importance`] to detect when the animation no longer affects
257    /// any variables and stop the animation to avoid awaking the app for no reason.
258    ///
259    /// These optimizations are already implemented by the animations provided as methods of [`Var<T>`].
260    ///
261    /// # External Controller
262    ///
263    /// The animation can be controlled from the inside using the [`Animation`] reference, it can be stopped using the returned
264    /// [`AnimationHandle`], and it can also be controlled by a registered [`AnimationController`] that can manage multiple
265    /// animations at the same time, see [`with_animation_controller`] for more details.
266    ///
267    /// [`Animation::sleep`]: Animation::sleep
268    /// [`stop`]: AnimationHandle::stop
269    /// [`perm`]: AnimationHandle::perm
270    /// [`with_animation_controller`]: VARS::with_animation_controller
271    /// [`VARS.frame_duration`]: VARS::frame_duration
272    /// [`VARS.current_modify`]: VARS::current_modify
273    pub fn animate<A>(&self, animation: A) -> AnimationHandle
274    where
275        A: FnMut(&Animation) + Send + 'static,
276    {
277        VARS.animate_impl(smallbox!(animation))
278    }

280    /// Calls `animate` while `controller` is registered as the animation controller.
281    ///
282    /// The `controller` is notified of animation events for each animation spawned by `animate` and can affect then with the same
283    /// level of access as [`VARS.animate`]. Only one controller can affect animations at a time.
284    ///
285    /// This can be used to manage multiple animations at the same time, or to get [`VARS.animate`] level of access to an animation
286    /// that is not implemented to allow such access. Note that animation implementers are not required to support the full
287    /// [`Animation`] API, for example, there is no guarantee that a restart requested by the controller will repeat the same animation.
288    ///
289    /// The controller can start new animations, these animations will have the same controller if not overridden, you can
290    /// use this method and the `()` controller to avoid this behavior.
291    ///
292    /// [`Animation`]: crate::animation::Animation
293    /// [`VARS.animate`]: VARS::animate
294    pub fn with_animation_controller<R>(&self, controller: impl AnimationController, animate: impl FnOnce() -> R) -> R {
295        let controller: Box<dyn AnimationController> = Box::new(controller);
296        let mut opt = Some(Arc::new(controller));
297        VARS_ANIMATION_CTRL_CTX.with_context(&mut opt, animate)
298    }

382    fn schedule_update_impl(&self, value_type_name: &'static str, update: VarUpdateFn) {
383        let vars = VARS_SV.read();
384        if let Some(trace) = &vars.modify_trace {
385            trace(value_type_name);
386        }
387        let cur_modify = match VARS_MODIFY_CTX.get_clone() {
388            Some(current) => current, // override set by modify and animation closures.
389            None => vars.ans_current_modify.clone(),
390        };
391
392        if let Some(id) = vars.updating_thread {
393            if std::thread::current().id() == id {
394                // is binding request, enqueue for immediate exec.
395                vars.updates.lock().push((cur_modify, update));
396            } else {
397                // is request from app task thread when we are already updating, enqueue for exec after current update.
398                vars.updates_after.lock().push((cur_modify, update));
399            }
400        } else {
401            // request from any app thread,
402            vars.updates.lock().push((cur_modify, update));
403            vars.wake_app();
404        }
405    }
406
407    fn apply_updates_and_after(&self, depth: u8) {
408        let mut vars = VARS_SV.write();
409
410        match depth {
411            0 => {
412                vars.update_id.next();
413                vars.ans_animation_start_time = None;
414            }
415            10 => {
416                // high-pressure from worker threads, skip
417                return;
418            }
419            _ => {}
420        }
421
422        // updates requested by other threads while was applying updates
423        let mut updates = mem::take(vars.updates_after.get_mut());
424        // normal updates
425        if updates.is_empty() {
426            updates = mem::take(vars.updates.get_mut());
427        } else {
428            updates.append(vars.updates.get_mut());
429        }
430        // apply pending updates
431        if !updates.is_empty() {
432            debug_assert!(vars.updating_thread.is_none());
433            vars.updating_thread = Some(std::thread::current().id());
434
435            drop(vars);
436            update_each_and_bindings(updates, 0);
437
438            vars = VARS_SV.write();
439            vars.updating_thread = None;
440
441            if !vars.updates_after.get_mut().is_empty() {
442                drop(vars);
443                VARS.apply_updates_and_after(depth + 1)
444            }
445        }
446
447        fn update_each_and_bindings(updates: Vec<(ModifyInfo, VarUpdateFn)>, depth: u16) {
448            if depth == 1000 {
449                tracing::error!(
450                    "updated variable bindings 1000 times, probably stuck in an infinite loop\n\
451                    will skip next updates"
452                );
453                return;
454            }
455
456            for (info, mut update) in updates {
457                #[allow(clippy::redundant_closure)] // false positive
458                VARS_MODIFY_CTX.with_context(&mut Some(Arc::new(Some(info))), || (update)());
459
460                let mut vars = VARS_SV.write();
461                let updates = mem::take(vars.updates.get_mut());
462                if !updates.is_empty() {
463                    drop(vars);
464                    update_each_and_bindings(updates, depth + 1);
465                }
466            }
467        }
468    }

470    fn animate_impl(&self, mut animation: SmallBox<dyn FnMut(&Animation) + Send + 'static, smallbox::space::S4>) -> AnimationHandle {
471        let mut vars = VARS_SV.write();
472
473        // # Modify Importance
474        //
475        // Variables only accept modifications from an importance (IMP) >= the previous IM that modified it.
476        //
477        // Direct modifications always overwrite previous animations, so we advance the IMP for each call to
478        // this method **and then** advance the IMP again for all subsequent direct modifications.
479        //
480        // Example sequence of events:
481        //
482        // |IM| Modification  | Accepted
483        // |--|---------------|----------
484        // | 1| Var::set      | YES
485        // | 2| Var::ease     | YES
486        // | 2| ease update   | YES
487        // | 3| Var::set      | YES
488        // | 3| Var::set      | YES
489        // | 2| ease update   | NO
490        // | 4| Var::ease     | YES
491        // | 2| ease update   | NO
492        // | 4| ease update   | YES
493        // | 5| Var::set      | YES
494        // | 2| ease update   | NO
495        // | 4| ease update   | NO
496
497        // ensure that all animations started in this update have the same exact time, we update then with the same `now`
498        // timestamp also, this ensures that synchronized animations match perfectly.
499        let start_time = if let Some(t) = vars.ans_animation_start_time {
500            t
501        } else {
502            let t = INSTANT.now();
503            vars.ans_animation_start_time = Some(t);
504            t
505        };
506
507        let mut anim_imp = None;
508        if let Some(c) = VARS_MODIFY_CTX.get_clone()
509            && c.is_animating()
510        {
511            // nested animation uses parent importance.
512            anim_imp = Some(c.importance);
513        }
514        let anim_imp = match anim_imp {
515            Some(i) => i,
516            None => {
517                // not nested, advance base imp
518                let mut imp = vars.ans_animation_imp.wrapping_add(1);
519                if imp == 0 {
520                    imp = 1;
521                }
522
523                let mut next_imp = imp.wrapping_add(1);
524                if next_imp == 0 {
525                    next_imp = 1;
526                }
527
528                vars.ans_animation_imp = next_imp;
529                vars.ans_current_modify.importance = next_imp;
530
531                imp
532            }
533        };
534
535        let (handle_owner, handle) = AnimationHandle::new();
536        let weak_handle = handle.downgrade();
537
538        let controller = VARS_ANIMATION_CTRL_CTX.get();
539
540        let anim = Animation::new(vars.animations_enabled.get(), start_time, vars.animation_time_scale.get());
541
542        drop(vars);
543
544        controller.on_start(&anim);
545        let mut controller = Some(controller);
546        let mut anim_modify_info = Some(Arc::new(Some(ModifyInfo {
547            handle: Some(weak_handle.clone()),
548            importance: anim_imp,
549        })));
550
551        let mut vars = VARS_SV.write();
552
553        vars.ans_animations.get_mut().push(smallbox!(move |info: AnimationUpdateInfo| {
554            let _handle_owner = &handle_owner; // capture and own the handle owner.
555
556            if weak_handle.upgrade().is_some() {
557                if anim.stop_requested() {
558                    // drop
559                    controller.as_ref().unwrap().on_stop(&anim);
560                    return None;
561                }
562
563                if let Some(sleep) = anim.sleep_deadline() {
564                    if sleep > info.next_frame {
565                        // retain sleep
566                        return Some(sleep);
567                    } else if sleep.0 > info.now {
568                        // sync-up to frame rate after sleep
569                        anim.reset_sleep();
570                        return Some(info.next_frame);
571                    }
572                }
573
574                anim.reset_state(info.animations_enabled, info.now, info.time_scale);
575
576                VARS_ANIMATION_CTRL_CTX.with_context(&mut controller, || {
577                    VARS_MODIFY_CTX.with_context(&mut anim_modify_info, || animation(&anim))
578                });
579
580                // retain until next frame
581                //
582                // stop or sleep may be requested after this (during modify apply),
583                // these updates are applied on the next frame.
584                Some(info.next_frame)
585            } else {
586                // drop
587                controller.as_ref().unwrap().on_stop(&anim);
588                None
589            }
590        }));
591
592        vars.ans_next_frame = Some(Deadline(DInstant::EPOCH));
593
594        vars.wake_app();
595
596        handle
597    }

599    fn update_animations_impl(&self, timer: &mut dyn AnimationTimer) {
600        let mut vars = VARS_SV.write();
601        if let Some(next_frame) = vars.ans_next_frame
602            && timer.elapsed(next_frame)
603        {
604            let mut animations = mem::take(vars.ans_animations.get_mut());
605            debug_assert!(!animations.is_empty());
606
607            let info = AnimationUpdateInfo {
608                animations_enabled: vars.animations_enabled.get(),
609                time_scale: vars.animation_time_scale.get(),
610                now: timer.now(),
611                next_frame: next_frame + vars.frame_duration.get(),
612            };
613
614            let mut min_sleep = Deadline(info.now + Duration::from_secs(60 * 60));
615
616            drop(vars);
617
618            animations.retain_mut(|animate| {
619                if let Some(sleep) = animate(info) {
620                    min_sleep = min_sleep.min(sleep);
621                    true
622                } else {
623                    false
624                }
625            });
626
627            let mut vars = VARS_SV.write();
628
629            let self_animations = vars.ans_animations.get_mut();
630            if !self_animations.is_empty() {
631                min_sleep = Deadline(info.now);
632            }
633            animations.append(self_animations);
634            *self_animations = animations;
635
636            if !self_animations.is_empty() {
637                vars.ans_next_frame = Some(min_sleep);
638                timer.register(min_sleep);
639            } else {
640                vars.ans_next_frame = None;
641            }
642        }
643    }
644
645    fn next_deadline_impl(&self, timer: &mut dyn AnimationTimer) {
646        if let Some(next_frame) = VARS_SV.read().ans_next_frame {
647            timer.register(next_frame);
648        }
649    }
650}
