25/// Simple text editor widget.
26///
27/// If `txt` is set to a variable that can be modified the widget becomes interactive, it implements
28/// the usual *text box* capabilities: keyboard editing of text in a single style, pointer
29/// caret positioning and text selection.
30///
31/// You can also use [`text::cmd`] to edit the text.
32///
33/// [`text::cmd`]: zng_wgt_text::cmd
34///
35/// # Undo/Redo
36///
37/// Undo/redo is enabled by default, the widget is an undo scope and handles undo commands. Note that external
38/// changes to the `txt` variable clears the undo stack, only changes done by the widget can be undone.
39///
40/// # Shorthand
41///
42/// The `TextInput!` macro provides shorthand syntax that sets the `txt` property.
43///
44/// ```
45/// # zng_wgt::enable_widget_macros!();
46/// # use zng_wgt::prelude::*;
47/// # use zng_wgt_text_input::*;
48/// #
49/// # fn main() {
50/// let editable_text = TextInput!(var_from(""));
51/// # }
52/// ```
53#[widget($crate::TextInput {
54    ($txt:expr) => {
55        txt = $txt;
56    };
57})]
58pub struct TextInput(StyleMix<UndoMix<Text>>);
59impl TextInput {
60    fn widget_intrinsic(&mut self) {
61        self.style_intrinsic(STYLE_FN_VAR, property_id!(self::style_fn));
62        widget_set! {
63            self;
64
65            access_role = AccessRole::TextInput;
66            txt_editable = true;
67            txt_selectable = true;
68            capture_pointer = true;
69            txt_align = Align::TOP_START;
70            focusable = true;
71            undo_scope = true;
72            undo_limit = 100;
73            style_base_fn = style_fn!(|_| DefaultStyle!());
74        }
75    }
76}

113/// Context captured for the context menu, set by the [`DefaultStyle!`].
114///
115/// Captures all context vars, except text style vars.
116///
117/// [`DefaultStyle!`]: struct@DefaultStyle
118pub fn default_popup_context_capture() -> popup::ContextCapture {
119    popup::ContextCapture::CaptureBlend {
120        filter: CaptureFilter::ContextVars {
121            exclude: {
122                let mut exclude = ContextValueSet::new();
123                Text::context_vars_set(&mut exclude);
124
125                let mut allow = ContextValueSet::new();
126                LangMix::<()>::context_vars_set(&mut allow);
127                exclude.remove_all(&allow);
128
129                exclude
130            },
131        },
132        over: false,
133    }
134}

247/// Widget shown when the `txt` is empty.
248///
249/// The `placeholder` can be any widget, the `Text!` widget is recommended.
250#[property(CHILD, widget_impl(TextInput))]
251pub fn placeholder(child: impl IntoUiNode, placeholder: impl IntoUiNode) -> UiNode {
252    let mut txt_is_empty = None;
253    zng_wgt_container::child_under(
254        child,
255        match_widget(placeholder, move |c, op| match op {
256            UiNodeOp::Init => {
257                let is_empty = zng_wgt_text::node::TEXT.resolved().txt.map(|t| t.is_empty());
258                WIDGET.sub_var_render(&is_empty);
259                txt_is_empty = Some(is_empty);
260            }
261            UiNodeOp::Deinit => {
262                txt_is_empty = None;
263            }
264            UiNodeOp::Render { frame } => {
265                c.delegated();
266                if txt_is_empty.as_ref().unwrap().get() {
267                    c.render(frame);
268                } else {
269                    frame.hide(|frame| c.render(frame));
270                }
271            }
272            UiNodeOp::RenderUpdate { update } => {
273                c.delegated();
274                if txt_is_empty.as_ref().unwrap().get() {
275                    c.render_update(update);
276                } else {
277                    update.hidden(|update| c.render_update(update));
278                }
279            }
280            _ => {}
281        }),
282    )
283}

285/// Text input style that shows data notes, info, warn and error.
286///
287/// You can also set the [`field_help`] property in text inputs with this style to set a text that
288/// shows in place of the data notes when there are none.
289///
290/// [`field_help`]: fn@field_help
291#[widget($crate::FieldStyle)]
292pub struct FieldStyle(DefaultStyle);
293impl FieldStyle {
294    fn widget_intrinsic(&mut self) {
295        let top_notes = var(DataNotes::default());
296
297        let top_level_and_txt = top_notes.map(|ns| {
298            if let Some(n) = ns.first() {
299                return (n.level(), formatx!("{ns}"));
300            }
301            (DataNoteLevel::INFO, "".into())
302        });
303        let top_txt = top_level_and_txt.map(|(_, t)| t.clone());
304        let top_color = DATA.note_color(top_level_and_txt.map(|(l, _)| *l));
305
306        let highlight = top_level_and_txt.map(|(l, _)| *l >= DataNoteLevel::WARN);
307        let adorn = merge_var!(top_txt.clone(), FIELD_HELP_VAR, |t, h| (t.is_empty(), h.is_empty()));
308
309        let chars_count = var(0usize);
310        let has_max_count = MAX_CHARS_COUNT_VAR.map(|&c| c > 0);
311
312        widget_set! {
313            self;
314            zng_wgt_data::get_data_notes_top = top_notes.clone();
315            get_chars_count = chars_count.clone();
316            auto_selection = true;
317
318            foreground_highlight = {
319                offsets: -2, // -1 border plus -1 to be outside
320                widths: 1,
321                sides: merge_var!(highlight, top_color.clone(), |&h, &c| if h {
322                    c.into()
323                } else {
324                    BorderSides::hidden()
325                }),
326            };
327            data_notes_adorner_fn = adorn.map(move |&(top_txt_empty, help_empty)| {
328                if !top_txt_empty {
329                    wgt_fn!(top_txt, top_color, |_| Text! {
330                        focusable = false;
331                        txt_editable = false;
332                        txt_selectable = false;
333                        txt = top_txt.clone();
334                        font_color = top_color.clone();
335                        font_size = 0.8.em();
336                        align = Align::BOTTOM_START;
337                        margin = (0, 4);
338                        y = 2.dip() + 100.pct();
339                    })
340                } else if !help_empty {
341                    wgt_fn!(|_| Text! {
342                        focusable = false;
343                        txt_editable = false;
344                        txt_selectable = false;
345                        txt = FIELD_HELP_VAR;
346                        font_size = 0.8.em();
347                        font_color = text::FONT_COLOR_VAR.map(|c| colors::GRAY.with_alpha(10.pct()).mix_normal(*c));
348                        align = Align::BOTTOM_START;
349                        margin = (0, 4);
350                        y = 2.dip() + 100.pct();
351                    })
352                } else {
353                    WidgetFn::nil()
354                }
355            });
356
357            max_chars_count_adorner_fn = has_max_count.map(move |&has| {
358                if has {
359                    wgt_fn!(chars_count, |_| Text! {
360                        focusable = false;
361                        txt_editable = false;
362                        txt_selectable = false;
363                        txt = merge_var!(chars_count.clone(), text::MAX_CHARS_COUNT_VAR, |c, m| formatx!("{c}/{m}"));
364                        font_color = text::FONT_COLOR_VAR.map(|c| colors::GRAY.with_alpha(10.pct()).mix_normal(*c));
365                        font_size = 0.8.em();
366                        align = Align::BOTTOM_END;
367                        offset = (-4, 2.dip() + 100.pct());
368                    })
369                } else {
370                    WidgetFn::nil()
371                }
372            });
373
374            margin = (0, 0, 1.2.em(), 0);
375        }
376    }
377}
