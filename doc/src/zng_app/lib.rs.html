183/// Desired next step of app main loop.
184#[derive(Copy, Clone, Debug, PartialEq, Eq)]
185#[must_use = "methods that return `AppControlFlow` expect to be inside a controlled loop"]
186pub enum AppControlFlow {
187    /// Immediately try to receive more app events.
188    Poll,
189    /// Sleep until an app event is received.
190    ///
191    /// Note that a deadline might be set in case a timer is running.
192    Wait,
193    /// Exit the loop and drop the app.
194    Exit,
195}
196impl AppControlFlow {
197    /// Assert that the value is [`AppControlFlow::Wait`].
198    #[track_caller]
199    pub fn assert_wait(self) {
200        assert_eq!(AppControlFlow::Wait, self)
201    }
202
203    /// Assert that the value is [`AppControlFlow::Exit`].
204    #[track_caller]
205    pub fn assert_exit(self) {
206        assert_eq!(AppControlFlow::Exit, self)
207    }
208}

210/// A headless app controller.
211///
212/// Headless apps don't cause external side-effects like visible windows and don't listen to system events.
213/// They can be used for creating apps like a command line app that renders widgets, or for creating integration tests.
214///
215/// You can start a headless app using [`AppBuilder::run_headless`].
216pub struct HeadlessApp {
217    app: RunningApp,
218}
219impl HeadlessApp {
220    /// If headless rendering is enabled wait until view-process is connected.
221    ///
222    /// When enabled windows are still not visible but frames will be rendered and the frame
223    /// image can be requested.
224    ///
225    /// Note that [`UiNode::render`] is still called when a renderer is disabled and you can still
226    /// query the latest frame from `WINDOWS.widget_tree`. The only thing that
227    /// is disabled is the actual renderer that converts display lists to pixels.
228    ///
229    /// [`UiNode::render`]: crate::widget::node::UiNode::render
230    pub fn renderer_enabled(&mut self) -> bool {
231        if VIEW_PROCESS.is_available() {
232            self.run_task(async {
233                let args = crate::view_process::VIEW_PROCESS_INITED_EVENT.var();
234                args.wait_match(|a| !a.is_empty()).await;
235            });
236            true
237        } else {
238            false
239        }
240    }
241
242    /// Does updates.
243    ///
244    /// If `wait_app_event` is `true` the thread sleeps until at least one app event is received or a timer elapses,
245    /// if it is `false` only responds to app events already in the buffer.
246    pub fn update(&mut self, mut wait_app_event: bool) -> AppControlFlow {
247        if self.app.has_exited() {
248            return AppControlFlow::Exit;
249        }
250
251        loop {
252            match self.app.poll(wait_app_event) {
253                AppControlFlow::Poll => {
254                    wait_app_event = false;
255                    continue;
256                }
257                flow => return flow,
258            }
259        }
260    }
261
262    /// Does updates and calls `on_pre_update` on the first update.
263    pub fn update_observe(&mut self, on_pre_update: impl FnOnce() + Send + 'static) -> bool {
264        let u = Arc::new(AtomicBool::new(false));
265        UPDATES
266            .on_pre_update(hn_once!(u, |_| {
267                u.store(true, std::sync::atomic::Ordering::Relaxed);
268                on_pre_update();
269            }))
270            .perm();
271        let _ = self.update(false);
272        u.load(std::sync::atomic::Ordering::Relaxed)
273    }
274
275    /// Execute the async `task` in the UI thread, updating the app until it finishes or the app shuts-down.
276    ///
277    /// Returns the task result if the app has not shutdown.
278    pub fn run_task<R, T>(&mut self, task: impl IntoFuture<IntoFuture = T>) -> Option<R>
279    where
280        R: Send + 'static,
281        T: Future<Output = R> + Send + 'static,
282    {
283        let task = task.into_future();
284
285        if self.app.has_exited() {
286            return None;
287        }
288
289        let r = Arc::new(Mutex::new(None::<R>));
290        UPDATES
291            .run(async_clmv!(r, {
292                let fr = task.await;
293                *r.lock() = Some(fr);
294            }))
295            .perm();
296
297        loop {
298            match self.app.poll(true) {
299                AppControlFlow::Exit => return None,
300                _ => {
301                    let mut r = r.lock();
302                    if r.is_some() {
303                        return r.take();
304                    }
305                }
306            }
307        }
308    }
309
310    /// Does [`run_task`] with a `deadline`.
311    ///
312    /// Returns the task result if the app has not shutdown and the `deadline` is not reached.
313    ///
314    /// If the `deadline` is reached an error is logged. Note that you can use [`with_deadline`] to create
315    /// a future with timeout and handle the timeout error.
316    ///
317    /// [`run_task`]: Self::run_task
318    /// [`with_deadline`]: zng_task::with_deadline
319    pub fn run_task_deadline<R, T>(&mut self, task: impl IntoFuture<IntoFuture = T>, deadline: impl Into<Deadline>) -> Option<R>
320    where
321        R: Send + 'static,
322        T: Future<Output = R> + Send + 'static,
323    {
324        let task = task.into_future();
325        let task = zng_task::with_deadline(task, deadline.into());
326        match self.run_task(task)? {
327            Ok(r) => Some(r),
328            Err(e) => {
329                tracing::error!("run_task reached deadline, {e}");
330                None
331            }
332        }
333    }
334
335    /// Does [`run_task`] with a deadline, panics on timeout.
336    ///
337    /// [`run_task`]: Self::run_task
338    #[cfg(any(test, feature = "test_util"))]
339    pub fn run_test<R, T>(&mut self, task: impl IntoFuture<IntoFuture = T>) -> Option<R>
340    where
341        R: Send + 'static,
342        T: Future<Output = R> + Send + 'static,
343    {
344        use std::time::Duration;
345
346        thread_local! {
347            static TIMEOUT: Duration = {
348                let t = std::env::var("ZNG_APP_RUN_TEST_TIMEOUT").unwrap_or_else(|_| "60".to_string());
349                let t: u64 = match t.parse() {
350                    Ok(0) => 60,
351                    Ok(t) => t,
352                    Err(_) => 60,
353                };
354                std::time::Duration::from_secs(t)
355            }
356        }
357        let task = task.into_future();
358        let task = zng_task::with_deadline(task, TIMEOUT.with(|t| *t));
359        match self.run_task(task)? {
360            Ok(r) => Some(r),
361            Err(e) => {
362                panic!("run_test {e}")
363            }
364        }
365    }

367    /// Requests and wait for app exit.
368    ///
369    /// Forces deinit if exit is cancelled.
370    pub fn exit(mut self) {
371        let req = APP.exit();
372        while req.is_waiting() {
373            if let AppControlFlow::Exit = self.update(true) {
374                break;
375            }
376        }
377    }
378    /// If the app has exited.
379    ///
380    /// Exited apps cannot update anymore. The app should be dropped to unload the app scope.
381    pub fn has_exited(&self) -> bool {
382        self.app.has_exited()
383    }

390    ///
391    /// If `true` multiple apps can run in the same process, but only one app per thread at a time.

396    /// If an app started building or is running in the current thread.
397    ///
398    /// This is `true` as soon as `APP.minimal()` or `APP.defaults()` is called.
399    ///
400    /// You can use [`app_local!`] to create *static* resources that live for the app lifetime, these statics can be used
401    /// as soon as this is `true`.
402    ///
403    /// [`app_local!`]: zng_app_context::app_local
404    pub fn is_started(&self) -> bool {
405        LocalContext::current_app().is_some()
406    }
407
408    /// If an app is running in the current thread.
409    ///
410    /// Apps are *running* as soon as [`run`], [`run_headless`] or `run_window` are called.
411    /// This will remain `true` until run returns or the [`HeadlessApp`] is dropped.
412    ///
413    /// [`run`]: AppBuilder::run
414    /// [`run_headless`]: AppBuilder::run_headless
415    pub fn is_running(&self) -> bool {
416        self.is_started() && !APP_PROCESS_SV.read().exit
417    }
418
419    /// Gets the unique ID of the current app.
420    ///
421    /// This ID usually does not change as most apps only run once per process, but it can change often during tests.
422    /// Resources that interact with [`app_local!`] values can use this ID to ensure that they are still operating in the same
423    /// app.
424    ///
425    /// [`app_local!`]: zng_app_context::app_local
426    pub fn id(&self) -> Option<AppId> {
427        LocalContext::current_app()
428    }

430    #[cfg(not(feature = "multi_app"))]
431    fn assert_can_run_single() {
432        use std::sync::atomic::*;
433        static CAN_RUN: AtomicBool = AtomicBool::new(true);
434
435        if !CAN_RUN.swap(false, Ordering::SeqCst) {
436            panic!("only one app is allowed per process")
437        }
438    }
439
440    fn assert_can_run() {
441        #[cfg(not(feature = "multi_app"))]
442        Self::assert_can_run_single();
443        if APP.is_running() {
444            panic!("only one app is allowed per thread")
445        }
446    }

487impl APP {
488    /// Starts building an application with only the minimum required config and resources.
489    ///
490    /// This is the recommended builder for tests, it signal init handlers to only load required resources.
491    pub fn minimal(&self) -> AppBuilder {
492        zng_env::init_process_name("app-process");
493
494        #[cfg(debug_assertions)]
495        print_tracing(tracing::Level::INFO, false, |_| true);
496        assert_not_view_process();
497        Self::assert_can_run();
498        spawn_deadlock_detection();
499
500        let _ = INSTANT.now();
501        let scope = LocalContext::start_app(AppId::new_unique());
502        AppBuilder {
503            view_process_exe: None,
504            view_process_env: HashMap::new(),
505            with_defaults: false,
506            _cleanup: scope,
507        }
508    }
509
510    /// Starts building an application with all compiled config and resources.
511    ///
512    /// This is the recommended builder for apps, it signals init handlers to setup all resources upfront, on run, for example, register icon sets,
513    /// default settings views and more. Note that you can still define a lean app by managing the compile time feature flags, and you can also
514    /// override any default resource on run.
515    pub fn defaults(&self) -> AppBuilder {
516        let mut app = self.minimal();
517        app.with_defaults = true;
518        app
519    }

522/// Application builder.
523///
524/// You can use `APP` to start building the app.
525pub struct AppBuilder {
526    view_process_exe: Option<PathBuf>,
527    view_process_env: HashMap<Txt, Txt>,
528    with_defaults: bool,
529
530    // cleanup on drop.
531    _cleanup: AppScope,
532}
533impl AppBuilder {
534    fn run_impl(self, start: std::pin::Pin<Box<dyn Future<Output = ()> + Send + 'static>>) {
535        let app = RunningApp::start(
536            self._cleanup,
537            true,
538            true,
539            self.view_process_exe,
540            self.view_process_env,
541            !self.with_defaults,
542        );
543
544        UPDATES.run(start).perm();
545
546        app.run_headed();
547    }
548
549    fn run_headless_impl(self, with_renderer: bool) -> HeadlessApp {
550        if with_renderer {
551            // disable ping timeout, headless apps manually update so don't ping on a schedule.
552            unsafe {
553                std::env::set_var("ZNG_VIEW_TIMEOUT", "false");
554            }
555        }
556
557        let app = RunningApp::start(
558            self._cleanup,
559            false,
560            with_renderer,
561            self.view_process_exe,
562            self.view_process_env,
563            !self.with_defaults,
564        );
565
566        HeadlessApp { app }
567    }
568}
569impl AppBuilder {
570    /// Set the path to the executable for the *View Process*.
571    ///
572    /// By the default the current executable is started again as a *View Process*, you can use
573    /// two executables instead, by setting this value.
574    ///
575    /// Note that the `view_process_exe` must start a view server and both
576    /// executables must be build using the same exact [`VERSION`].
577    ///
578    /// [`VERSION`]: zng_view_api::VERSION  
579    pub fn view_process_exe(mut self, view_process_exe: impl Into<PathBuf>) -> Self {
580        self.view_process_exe = Some(view_process_exe.into());
581        self
582    }
583
584    /// Set an env variable for the view-process.
585    pub fn view_process_env(mut self, name: impl Into<Txt>, value: impl Into<Txt>) -> Self {
586        self.view_process_env.insert(name.into(), value.into());
587        self
588    }
589
590    /// Starts the app, then starts polling `start` to run.
591    ///
592    /// This method only returns when the app has exited.
593    ///
594    /// The `start` task runs in the app context, note that the future only needs to start the app, usually
595    /// by opening a window, the app will keep running after `start` is finished.
596    pub fn run<F: Future<Output = ()> + Send + 'static>(self, start: impl IntoFuture<IntoFuture = F>) {
597        let start = start.into_future();
598        self.run_impl(Box::pin(start))
599    }
600
601    /// Initializes extensions in headless mode and returns an [`HeadlessApp`].
602    ///
603    /// If `with_renderer` is `true` spawns a renderer process for headless rendering. See [`HeadlessApp::renderer_enabled`]
604    /// for more details.
605    pub fn run_headless(self, with_renderer: bool) -> HeadlessApp {
606        self.run_headless_impl(with_renderer)
607    }
608}

620/// Enables [`tracing`] events printing if a subscriber is not already set.
621///
622/// All non-fatal errors in the Zng project are logged using tracing, printing these errors is essential for debugging.
623/// In debug builds this is enabled by default in the app-process on app init with `INFO` level and no span events.
624///
625/// If `span_events` are enabled `tracing::span!` enter and exit are also printed as events.
626///
627/// In `"wasm32"` builds logs to the browser console.
628///
629/// In `"android"` builds logs to logcat.
630///
631/// See also [`test_log`] to enable panicking on error log.
632///
633/// See also [`print_tracing_filter`] for the filter used by this.
634///
635/// # Examples
636///
637/// In the example below this function is called before `init!`, enabling it in all app processes.
638///
639/// ```
640/// # macro_rules! demo { () => {
641/// fn main() {
642///     zng::app::print_tracing(tracing::Level::INFO, false, |_| true);
643///     zng::env::init!();
644/// }
645/// # }}
646/// ```
647///
648/// [`tracing`]: https://docs.rs/tracing
649pub fn print_tracing(max: tracing::Level, span_events: bool, filter: impl Fn(&tracing::Metadata) -> bool + Send + Sync + 'static) -> bool {
650    print_tracing_impl(max, span_events, Box::new(filter))
651}
652fn print_tracing_impl(
653    max: tracing::Level,
654    span_events: bool,
655    filter: Box<dyn Fn(&tracing::Metadata) -> bool + Send + Sync + 'static>,
656) -> bool {
657    use tracing_subscriber::prelude::*;
658
659    let layers = tracing_subscriber::registry().with(FilterLayer(max, filter));
660
661    #[cfg(target_os = "android")]
662    let layers = layers.with(tracing_android::layer(&zng_env::about().pkg_name).unwrap());
663    #[cfg(target_os = "android")]
664    let _ = span_events;
665
666    #[cfg(not(target_os = "android"))]
667    let layers = {
668        let mut fmt_layer = tracing_subscriber::fmt::layer().without_time();
669        if span_events {
670            fmt_layer = fmt_layer.with_span_events(tracing_subscriber::fmt::format::FmtSpan::ACTIVE);
671        }
672
673        #[cfg(target_arch = "wasm32")]
674        let fmt_layer = fmt_layer.with_ansi(false).with_writer(tracing_web::MakeWebConsoleWriter::new());
675
676        layers.with(fmt_layer)
677    };
678
679    layers.try_init().is_ok()
680}
681struct FilterLayer(tracing::Level, Box<dyn Fn(&tracing::Metadata) -> bool + Send + Sync>);
682impl<S: tracing::Subscriber> tracing_subscriber::Layer<S> for FilterLayer {
683    fn enabled(&self, metadata: &tracing::Metadata<'_>, _: tracing_subscriber::layer::Context<'_, S>) -> bool {
684        print_tracing_filter(&self.0, metadata, &self.1)
685    }
686
687    fn max_level_hint(&self) -> Option<tracing::metadata::LevelFilter> {
688        Some(self.0.into())
689    }
690
691    #[cfg(any(test, feature = "test_util"))]
692    fn on_event(&self, event: &tracing::Event<'_>, _ctx: tracing_subscriber::layer::Context<'_, S>) {
693        if event.metadata().level() == &tracing::Level::ERROR && APP.is_running() && TEST_LOG.get() {
694            struct MsgCollector<'a>(&'a mut String);
695            impl tracing::field::Visit for MsgCollector<'_> {
696                fn record_debug(&mut self, field: &tracing::field::Field, value: &dyn std::fmt::Debug) {
697                    use std::fmt::Write;
698                    write!(self.0, "\n  {} = {:?}", field.name(), value).unwrap();
699                }
700            }
701
702            let meta = event.metadata();
703            let file = meta.file().unwrap_or("");
704            let line = meta.line().unwrap_or(0);
705
706            let mut msg = format!("[{file}:{line}]");
707            event.record(&mut MsgCollector(&mut msg));
708
709            panic!("[LOG-ERROR]{msg}")
710        }
711    }
712}
713/// Filter used by [`print_tracing`], removes some log noise from dependencies.
714///
715/// Use `tracing_subscriber::filter::FilterFn` plug this filter into a tracing setup.
716pub fn print_tracing_filter(level: &tracing::Level, metadata: &tracing::Metadata, filter: &dyn Fn(&tracing::Metadata) -> bool) -> bool {
717    if metadata.level() > level {
718        return false;
719    }
720
721    if metadata.level() == &tracing::Level::INFO && level < &tracing::Level::TRACE {
722        // suppress large info about texture cache
723        if metadata.target() == "zng_webrender::device::gl" {
724            return false;
725        }
726        // suppress config dump
727        if metadata.target() == "zng_webrender::renderer::init" {
728            return false;
729        }
730    } else if metadata.level() == &tracing::Level::WARN && level < &tracing::Level::DEBUG {
731        // suppress webrender warnings:
732        //
733        if metadata.target() == "zng_webrender::device::gl" {
734            // Suppress "Cropping texture upload Box2D((0, 0), (0, 1)) to None"
735            // This happens when an empty frame is rendered.
736            if metadata.line() == Some(4647) {
737                return false;
738            }
739        }
740
741        // suppress font-kit warnings:
742        //
743        if metadata.target() == "font_kit::loaders::freetype" {
744            // Suppress "$fn(): found invalid platform ID $n"
745            // This does not look fully implemented and generates a lot of warns
746            // with the default Ubuntu font set all with valid platform IDs.
747            if metadata.line() == Some(734) {
748                return false;
749            }
750        }
751    }
752
753    filter(metadata)
754}

767#[doc(hidden)]
768pub fn name_from_pkg_name(name: &'static str) -> Txt {
769    let mut n = String::new();
770    let mut sep = "";
771    for part in name.split(&['-', '_']) {
772        n.push_str(sep);
773        let mut chars = part.char_indices();
774        let (_, c) = chars.next().unwrap();
775        c.to_uppercase().for_each(|c| n.push(c));
776        if let Some((i, _)) = chars.next() {
777            n.push_str(&part[i..]);
778        }
779        sep = " ";
780    }
781    n.into()
782}
