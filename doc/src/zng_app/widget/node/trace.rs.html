18pub(super) struct TraceNode {
19    node: UiNode,
20    trace: Box<dyn FnMut(&mut dyn UiNodeImpl, UiNodeMethod, &mut dyn FnMut(&mut dyn UiNodeImpl)) + Send + 'static>,
21}
22impl TraceNode {
23    pub(super) fn new<S>(node: UiNode, mut enter_mtd: impl FnMut(UiNodeMethod) -> S + Send + 'static) -> Self {
24        Self {
25            node,
26            trace: Box::new(move |node, op, call| {
27                let _span = if let Some(w) = node.as_widget() {
28                    let mut s = None;
29                    w.with_context(WidgetUpdateMode::Bubble, &mut || s = Some(enter_mtd(op)));
30                    match s {
31                        Some(s) => s,
32                        None => enter_mtd(op),
33                    }
34                } else {
35                    enter_mtd(op)
36                };
37
38                call(node);
39
40                if let Some(w) = node.as_widget() {
41                    let mut _span = Some(_span);
42                    w.with_context(WidgetUpdateMode::Bubble, &mut || drop(_span.take()));
43                }
44            }),
45        }
46    }
47}
