37    pub fn best(
38        font_name: &FontName,
39        style: FontStyle,
40        weight: FontWeight,
41        stretch: FontStretch,
42    ) -> Result<Option<(FontDataRef, u32)>, FontLoadingError> {
43        if font_name == "Ubuntu"
44            && let Ok(Some(h)) = workaround_ubuntu(style, weight, stretch)
45        {
46            return Ok(Some(h));
47        }
48
49        let family_name = font_kit::family_name::FamilyName::from(font_name.clone());
50        match font_kit::source::SystemSource::new().select_best_match(
51            &[family_name],
52            &font_kit::properties::Properties {
53                style: style.into(),
54                weight: weight.into(),
55                stretch: stretch.into(),
56            },
57        ) {
58            Ok(handle) => {
59                let r = load_handle(&handle)?;
60                Ok(Some(r))
61            }
62            Err(font_kit::error::SelectionError::NotFound) => {
63                tracing::debug!(target: "font_loading", "system font not found\nquery: {:?}", (font_name, style, weight, stretch));
64                Ok(None)
65            }
66            Err(font_kit::error::SelectionError::CannotAccessSource { reason }) => {
67                Err(FontLoadingError::Io(Arc::new(std::io::Error::other(reason.unwrap_or_default()))))
68            }
69        }
70    }

73    fn workaround_ubuntu(
74        style: FontStyle,
75        weight: FontWeight,
76        stretch: FontStretch,
77    ) -> Result<Option<(FontDataRef, u32)>, FontLoadingError> {
78        let source = font_kit::source::SystemSource::new();
79        let ubuntu = match source.select_family_by_name("Ubuntu") {
80            Ok(u) => u,
81            Err(e) => {
82                return match e {
83                    font_kit::error::SelectionError::NotFound => Ok(None),
84                    font_kit::error::SelectionError::CannotAccessSource { reason } => {
85                        Err(FontLoadingError::Io(Arc::new(std::io::Error::other(reason.unwrap_or_default()))))
86                    }
87                };
88            }
89        };
90        for handle in ubuntu.fonts() {
91            let font = handle.load()?;
92            let name = match font.postscript_name() {
93                Some(n) => n,
94                None => continue,
95            };
96
97            // Ubuntu-ExtraBold
98            // Ubuntu-Condensed
99            // Ubuntu-CondensedLight
100            // Ubuntu-CondensedBold
101            // Ubuntu-CondensedMedium
102            // Ubuntu-CondensedExtraBold
103            // UbuntuItalic-CondensedLightItalic
104            // UbuntuItalic-CondensedItalic
105            // UbuntuItalic-CondensedMediumItalic
106            // UbuntuItalic-CondensedBoldItalic
107            // UbuntuItalic-CondensedExtraBoldItalic
108            // Ubuntu-Italic
109            // UbuntuItalic-ThinItalic
110            // UbuntuItalic-LightItalic
111            // UbuntuItalic-Italic
112            // UbuntuItalic-MediumItalic
113            // UbuntuItalic-BoldItalic
114            // UbuntuItalic-ExtraBoldItalic
115            // UbuntuItalic-CondensedThinItalic
116            // Ubuntu-Thin
117            // Ubuntu-Regular
118            // Ubuntu-Light
119            // Ubuntu-Bold
120            // Ubuntu-Medium
121            // Ubuntu-CondensedThin
122
123            if (style == FontStyle::Italic) != name.contains("Italic") {
124                continue;
125            }
126
127            if (FontWeight::MEDIUM..FontWeight::SEMIBOLD).contains(&weight) != name.contains("Medium") {
128                continue;
129            }
130            if (weight >= FontWeight::EXTRA_BOLD) != name.contains("ExtraBold") {
131                continue;
132            }
133            if (FontWeight::SEMIBOLD..FontWeight::EXTRA_BOLD).contains(&weight) != name.contains("Bold") {
134                continue;
135            }
136
137            if (FontWeight::EXTRA_LIGHT..FontWeight::LIGHT).contains(&weight) != name.contains("Light") {
138                continue;
139            }
140            if (weight < FontWeight::EXTRA_LIGHT) != name.contains("Thin") {
141                continue;
142            }
143
144            if (stretch <= FontStretch::CONDENSED) != name.contains("Condensed") {
145                continue;
146            }
147
148            return Ok(Some(load_handle(handle)?));
149        }
150        Ok(None)
151    }

153    fn load_handle(handle: &font_kit::handle::Handle) -> Result<(FontDataRef, u32), FontLoadingError> {
154        match handle {
155            font_kit::handle::Handle::Path { path, font_index } => {
156                let mut path = Cow::Borrowed(path);
157                // try replacing type1 fonts with OpenType
158                // RustyBuzz does not support type1 (neither does Harfbuzz, it is obsolete)
159                //
160                // Example case from default Ubuntu fonts:
161                // /usr/share/fonts/type1/urw-base35/Z003-MediumItalic.t1
162                // /usr/share/fonts/opentype/urw-base35/Z003-MediumItalic.otf
163                if let Ok(base) = path.strip_prefix("/usr/share/fonts/type1/")
164                    && let Some(name) = base.file_name()
165                    && let Some(name) = name.to_str()
166                    && name.ends_with(".t1")
167                {
168                    let rep = Path::new("/usr/share/fonts/opentype/").join(base.with_extension("otf"));
169                    if rep.exists() {
170                        tracing::debug!("replaced `{name}` with .otf of same name");
171                        path = Cow::Owned(rep);
172                    }
173                }
174
175                for (k, data) in DATA_CACHE.lock().iter() {
176                    if *k == *path
177                        && let Some(data) = data.upgrade()
178                    {
179                        return Ok((FontDataRef(data), *font_index));
180                    }
181                }
182
183                let bytes = std::fs::read(&*path)?;
184                tracing::debug!("read font `{}:{}`, using {}", path.display(), font_index, bytes.capacity().bytes());
185
186                let data = Arc::new(bytes);
187                let mut cache = DATA_CACHE.lock();
188                cache.retain(|(_, v)| v.strong_count() > 0);
189                cache.push((path.to_path_buf(), Arc::downgrade(&data)));
190
191                Ok((FontDataRef(data), *font_index))
192            }
193            font_kit::handle::Handle::Memory { bytes, font_index } => Ok((FontDataRef(bytes.clone()), *font_index)),
194        }
195    }

197    impl From<font_kit::error::FontLoadingError> for FontLoadingError {
198        fn from(ve: font_kit::error::FontLoadingError) -> Self {
199            match ve {
200                font_kit::error::FontLoadingError::UnknownFormat => Self::UnknownFormat,
201                font_kit::error::FontLoadingError::NoSuchFontInCollection => Self::NoSuchFontInCollection,
202                font_kit::error::FontLoadingError::Parse => Self::Parse(ttf_parser::FaceParsingError::MalformedFont),
203                font_kit::error::FontLoadingError::NoFilesystem => Self::NoFilesystem,
204                font_kit::error::FontLoadingError::Io(e) => Self::Io(Arc::new(e)),
205            }
206        }

289    pub fn system_all() -> ResponseVar<Vec<FontName>> {
290        zng_task::wait_respond(|| {
291            let mut prev = None;
292            cached_system_all()
293                .iter()
294                .flat_map(|(k, _)| {
295                    if prev == Some(k) {
296                        None
297                    } else {
298                        prev = Some(k);
299                        Some(k)
300                    }
301                })
302                .cloned()
303                .collect()
304        })
305    }

307    fn cached_system_all() -> parking_lot::MappedRwLockReadGuard<'static, Vec<(FontName, PathBuf)>> {
308        let lock = SYSTEM_ALL.read();
309        if !lock.is_empty() {
310            return lock;
311        }
312
313        drop(lock);
314        let mut lock = SYSTEM_ALL.write();
315        if lock.is_empty() {
316            for entry in ["/system/fonts/", "/system/font/", "/data/fonts/", "/system/product/fonts/"]
317                .iter()
318                .flat_map(std::fs::read_dir)
319                .flatten()
320                .flatten()
321            {
322                let entry = entry.path();
323                let ext = entry.extension().and_then(|e| e.to_str()).unwrap_or_default().to_ascii_lowercase();
324                if ["ttf", "otf"].contains(&ext.as_str()) && entry.is_file() {
325                    if let Ok(bytes) = std::fs::read(&entry) {
326                        match crate::FontFace::load(FontDataRef(Arc::new(bytes)), 0) {
327                            Ok(f) => {
328                                lock.push((f.family_name().clone(), entry));
329                            }
330                            Err(e) => tracing::error!("error parsing '{}', {e}", entry.display()),
331                        }
332                    }
333                }
334            }
335            lock.sort_by(|a, b| a.0.cmp(&b.0));
336        }
337        drop(lock);
338        SYSTEM_ALL.read()
339    }

341    pub fn best(
342        font_name: &FontName,
343        style: FontStyle,
344        weight: FontWeight,
345        stretch: FontStretch,
346    ) -> Result<Option<(FontDataRef, u32)>, FontLoadingError> {
347        let lock = cached_system_all();
348        let lock = &*lock;
349
350        // special names
351        // source: https://android.googlesource.com/platform/frameworks/base/+/master/data/fonts/fonts.xml
352        let font_name = match font_name.name() {
353            "sans-serif" => Cow::Owned(FontName::new("Roboto")),
354            "serif" | "fantasy" => Cow::Owned(FontName::new("Noto Serif")),
355            "cursive" => Cow::Owned(FontName::new("Dancing Script")),
356            "monospace" => Cow::Owned(FontName::new("Droid Sans Mono")),
357            _ => Cow::Borrowed(font_name),
358        };
359        let font_name = &*font_name;
360
361        let mut start_i = match lock.binary_search_by(|a| a.0.cmp(font_name)) {
362            Ok(i) => i,
363            Err(_) => {
364                tracing::debug!(target: "font_loading", "system font not found\nquery: {:?}", (font_name, style, weight, stretch));
365                return Ok(None);
366            }
367        };
368        while start_i > 0 && &lock[start_i - 1].0 == font_name {
369            start_i -= 1
370        }
371        let mut end_i = start_i;
372        while end_i + 1 < lock.len() && &lock[end_i + 1].0 == font_name {
373            end_i += 1
374        }
375
376        let family_len = end_i - start_i;
377        let mut options = Vec::with_capacity(family_len);
378        let mut candidates = Vec::with_capacity(family_len);
379
380        for (_, path) in &lock[start_i..=end_i] {
381            if let Ok(bytes) = std::fs::read(path) {
382                if let Ok(f) = ttf_parser::Face::parse(&bytes, 0) {
383                    candidates.push(matching::Properties {
384                        style: f.style(),
385                        weight: f.weight(),
386                        stretch: f.width(),
387                    });
388                    options.push(bytes);
389                }
390            }
391        }
392
393        match matching::find_best_match(
394            &candidates,
395            &matching::Properties {
396                style: style.into(),
397                weight: weight.into(),
398                stretch: stretch.into(),
399            },
400        ) {
401            Ok(i) => {
402                let bytes = options.swap_remove(i);
403                Ok(Some((FontDataRef(Arc::new(bytes)), 0)))
404            }
405            Err(FontLoadingError::NoSuchFontInCollection) => {
406                tracing::debug!(target: "font_loading", "system font not found\nquery: {:?}", (font_name, style, weight, stretch));
407                Ok(None)
408            }
409            Err(e) => Err(FontLoadingError::Io(Arc::new(std::io::Error::other(e)))),
410        }
411    }

428        //! Determines the closest font matching a description per the CSS Fonts Level 3 specification.
429
430        use ttf_parser::{Style, Weight, Width as Stretch};
431
432        use crate::FontLoadingError;
433
434        pub struct Properties {
435            pub style: Style,
436            pub weight: Weight,
437            pub stretch: Stretch,
438        }
439
440        /// This follows CSS Fonts Level 3 § 5.2 [1].
441        ///
442        /// https://drafts.csswg.org/css-fonts-3/#font-style-matching
443        pub fn find_best_match(candidates: &[Properties], query: &Properties) -> Result<usize, FontLoadingError> {
444            // Step 4.
445            let mut matching_set: Vec<usize> = (0..candidates.len()).collect();
446            if matching_set.is_empty() {
447                return Err(FontLoadingError::NoSuchFontInCollection);
448            }
449
450            // Step 4a (`font-stretch`).
451            let matching_stretch = if matching_set.iter().any(|&index| candidates[index].stretch == query.stretch) {
452                // Exact match.
453                query.stretch
454            } else if query.stretch <= Stretch::Normal {
455                // Closest width, first checking narrower values and then wider values.
456                match matching_set
457                    .iter()
458                    .filter(|&&index| candidates[index].stretch < query.stretch)
459                    .min_by_key(|&&index| query.stretch.to_number() - candidates[index].stretch.to_number())
460                {
461                    Some(&matching_index) => candidates[matching_index].stretch,
462                    None => {
463                        let matching_index = *matching_set
464                            .iter()
465                            .min_by_key(|&&index| candidates[index].stretch.to_number() - query.stretch.to_number())
466                            .unwrap();
467                        candidates[matching_index].stretch
468                    }
469                }
470            } else {
471                // Closest width, first checking wider values and then narrower values.
472                match matching_set
473                    .iter()
474                    .filter(|&&index| candidates[index].stretch > query.stretch)
475                    .min_by_key(|&&index| candidates[index].stretch.to_number() - query.stretch.to_number())
476                {
477                    Some(&matching_index) => candidates[matching_index].stretch,
478                    None => {
479                        let matching_index = *matching_set
480                            .iter()
481                            .min_by_key(|&&index| query.stretch.to_number() - candidates[index].stretch.to_number())
482                            .unwrap();
483                        candidates[matching_index].stretch
484                    }
485                }
486            };
487            matching_set.retain(|&index| candidates[index].stretch == matching_stretch);
488
489            // Step 4b (`font-style`).
490            let style_preference = match query.style {
491                Style::Italic => [Style::Italic, Style::Oblique, Style::Normal],
492                Style::Oblique => [Style::Oblique, Style::Italic, Style::Normal],
493                Style::Normal => [Style::Normal, Style::Oblique, Style::Italic],
494            };
495            let matching_style = *style_preference
496                .iter()
497                .find(|&query_style| matching_set.iter().any(|&index| candidates[index].style == *query_style))
498                .unwrap();
499            matching_set.retain(|&index| candidates[index].style == matching_style);
500
501            // Step 4c (`font-weight`).
502            //
503            // The spec doesn't say what to do if the weight is between 400 and 500 exclusive, so we
504            // just use 450 as the cutoff.
505            let matching_weight = if matching_set.iter().any(|&index| candidates[index].weight == query.weight) {
506                query.weight
507            } else if query.weight.to_number() >= 400
508                && query.weight.to_number() < 450
509                && matching_set.iter().any(|&index| candidates[index].weight == Weight::from(500))
510            {
511                // Check 500 first.
512                Weight::from(500)
513            } else if query.weight.to_number() >= 450
514                && query.weight.to_number() <= 500
515                && matching_set.iter().any(|&index| candidates[index].weight.to_number() == 400)
516            {
517                // Check 400 first.
518                Weight::from(400)
519            } else if query.weight.to_number() <= 500 {
520                // Closest weight, first checking thinner values and then fatter ones.
521                match matching_set
522                    .iter()
523                    .filter(|&&index| candidates[index].weight.to_number() <= query.weight.to_number())
524                    .min_by_key(|&&index| query.weight.to_number() - candidates[index].weight.to_number())
525                {
526                    Some(&matching_index) => candidates[matching_index].weight,
527                    None => {
528                        let matching_index = *matching_set
529                            .iter()
530                            .min_by_key(|&&index| candidates[index].weight.to_number() - query.weight.to_number())
531                            .unwrap();
532                        candidates[matching_index].weight
533                    }
534                }
535            } else {
536                // Closest weight, first checking fatter values and then thinner ones.
537                match matching_set
538                    .iter()
539                    .filter(|&&index| candidates[index].weight.to_number() >= query.weight.to_number())
540                    .min_by_key(|&&index| candidates[index].weight.to_number() - query.weight.to_number())
541                {
542                    Some(&matching_index) => candidates[matching_index].weight,
543                    None => {
544                        let matching_index = *matching_set
545                            .iter()
546                            .min_by_key(|&&index| query.weight.to_number() - candidates[index].weight.to_number())
547                            .unwrap();
548                        candidates[matching_index].weight
549                    }
550                }
551            };
552            matching_set.retain(|&index| candidates[index].weight == matching_weight);
553
554            // Step 4d concerns `font-size`, but fonts in `font-kit` are unsized, so we ignore that.
555
556            // Return the result.
557            matching_set.into_iter().next().ok_or(FontLoadingError::NoSuchFontInCollection)
558        }
559    }
