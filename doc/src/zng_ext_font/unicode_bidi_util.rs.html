10pub(super) fn visual_runs(
11    mut levels: Vec<unicode_bidi::Level>,
12    line_classes: Vec<unicode_bidi::BidiClass>,
13    para_level: unicode_bidi::Level,
14) -> (Vec<unicode_bidi::Level>, Vec<unicode_bidi::LevelRun>) {
15    use unicode_bidi::BidiClass::*;
16
17    let line_levels = &mut levels;
18
19    // Reset some whitespace chars to paragraph level.
20    // <http://www.unicode.org/reports/tr9/#L1>
21    let mut reset_from: Option<usize> = Some(0);
22    let mut reset_to: Option<usize> = None;
23    let mut prev_level = para_level;
24    for i in 0..line_classes.len() {
25        match line_classes[i] {
26            // Segment separator, Paragraph separator
27            B | S => {
28                assert_eq!(reset_to, None);
29                reset_to = Some(i + 1);
30                if reset_from.is_none() {
31                    reset_from = Some(i);
32                }
33            }
34            // Whitespace, isolate formatting
35            WS | FSI | LRI | RLI | PDI => {
36                if reset_from.is_none() {
37                    reset_from = Some(i);
38                }
39            }
40            // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
41            // same as above + set the level
42            RLE | LRE | RLO | LRO | PDF | BN => {
43                if reset_from.is_none() {
44                    reset_from = Some(i);
45                }
46                // also set the level to previous
47                line_levels[i] = prev_level;
48            }
49            _ => {
50                reset_from = None;
51            }
52        }
53        if let Some(from) = reset_from
54            && let Some(to) = reset_to
55        {
56            for level in &mut line_levels[from..to] {
57                *level = para_level;
58            }
59            reset_from = None;
60            reset_to = None;
61        }
62        prev_level = line_levels[i];
63    }
64    if let Some(from) = reset_from {
65        for level in &mut line_levels[from..] {
66            *level = para_level;
67        }
68    }
69
70    // Find consecutive level runs.
71    let mut runs = Vec::new();
72    let mut start = 0;
73    let mut run_level = levels[start];
74    let mut min_level = run_level;
75    let mut max_level = run_level;
76
77    for (i, &new_level) in levels.iter().enumerate().skip(1) {
78        if new_level != run_level {
79            // End of the previous run, start of a new one.
80            runs.push(start..i);
81            start = i;
82            run_level = new_level;
83            min_level = std::cmp::min(run_level, min_level);
84            max_level = std::cmp::max(run_level, max_level);
85        }
86    }
87    runs.push(start..line_classes.len());
88
89    let run_count = runs.len();
90
91    // Re-order the odd runs.
92    // <http://www.unicode.org/reports/tr9/#L2>
93
94    // Stop at the lowest *odd* level.
95    min_level = min_level.new_lowest_ge_rtl().expect("Level error");
96
97    while max_level >= min_level {
98        // Look for the start of a sequence of consecutive runs of max_level or higher.
99        let mut seq_start = 0;
100        while seq_start < run_count {
101            if levels[runs[seq_start].start] < max_level {
102                seq_start += 1;
103                continue;
104            }
105
106            // Found the start of a sequence. Now find the end.
107            let mut seq_end = seq_start + 1;
108            while seq_end < run_count {
109                if levels[runs[seq_end].start] < max_level {
110                    break;
111                }
112                seq_end += 1;
113            }
114
115            // Reverse the runs within this sequence.
116            runs[seq_start..seq_end].reverse();
117
118            seq_start = seq_end;
119        }
120        max_level.lower(1).expect("Lowering embedding level below zero");
121    }
122
123    (levels, runs)
124}

126pub(super) fn explicit_compute(
127    para_level: Level,
128    original_classes: &[BidiClass],
129    levels: &mut [Level],
130    processing_classes: &mut [BidiClass],
131) {
132    // <http://www.unicode.org/reports/tr9/#X1>
133    let mut stack = DirectionalStatusStack::new();
134    stack.push(para_level, OverrideStatus::Neutral);
135
136    let mut overflow_isolate_count = 0u32;
137    let mut overflow_embedding_count = 0u32;
138    let mut valid_isolate_count = 0u32;
139
140    for i in 0..original_classes.len() {
141        use BidiClass::*;
142        match original_classes[i] {
143            // Rules X2-X5c
144            RLE | LRE | RLO | LRO | RLI | LRI | FSI => {
145                let last_level = stack.last().level;
146
147                // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
148                levels[i] = last_level;
149
150                // X5a-X5c: Isolate initiators get the level of the last entry on the stack.
151                let is_isolate = matches!(original_classes[i], RLI | LRI | FSI);
152                if is_isolate {
153                    // Redundant due to "Retaining explicit formatting characters" step.
154                    // levels[i] = last_level;
155                    match stack.last().status {
156                        OverrideStatus::RTL => processing_classes[i] = R,
157                        OverrideStatus::LTR => processing_classes[i] = L,
158                        _ => {}
159                    }
160                }
161
162                let new_level = if is_rtl(original_classes[i]) {
163                    last_level.new_explicit_next_rtl()
164                } else {
165                    last_level.new_explicit_next_ltr()
166                };
167
168                if let Ok(new_level) = new_level
169                    && overflow_isolate_count == 0
170                    && overflow_embedding_count == 0
171                {
172                    stack.push(
173                        new_level,
174                        match original_classes[i] {
175                            RLO => OverrideStatus::RTL,
176                            LRO => OverrideStatus::LTR,
177                            RLI | LRI | FSI => OverrideStatus::Isolate,
178                            _ => OverrideStatus::Neutral,
179                        },
180                    );
181                    if is_isolate {
182                        valid_isolate_count += 1;
183                    } else {
184                        // The spec doesn't explicitly mention this step, but it is necessary.
185                        // See the reference implementations for comparison.
186                        levels[i] = new_level;
187                    }
188                } else if is_isolate {
189                    overflow_isolate_count += 1;
190                } else if overflow_isolate_count == 0 {
191                    overflow_embedding_count += 1;
192                }
193
194                if !is_isolate {
195                    // X9 +
196                    // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
197                    // (PDF handled below)
198                    processing_classes[i] = BN;
199                }
200            }
201
202            // <http://www.unicode.org/reports/tr9/#X6a>
203            PDI => {
204                if overflow_isolate_count > 0 {
205                    overflow_isolate_count -= 1;
206                } else if valid_isolate_count > 0 {
207                    overflow_embedding_count = 0;
208                    loop {
209                        // Pop everything up to and including the last Isolate status.
210                        match stack.vec.pop() {
211                            None
212                            | Some(Status {
213                                status: OverrideStatus::Isolate,
214                                ..
215                            }) => break,
216                            _ => continue,
217                        }
218                    }
219                    valid_isolate_count -= 1;
220                }
221                let last = stack.last();
222                levels[i] = last.level;
223                match last.status {
224                    OverrideStatus::RTL => processing_classes[i] = R,
225                    OverrideStatus::LTR => processing_classes[i] = L,
226                    _ => {}
227                }
228            }
229
230            // <http://www.unicode.org/reports/tr9/#X7>
231            PDF => {
232                if overflow_isolate_count > 0 {
233                    // do nothing
234                } else if overflow_embedding_count > 0 {
235                    overflow_embedding_count -= 1;
236                } else if stack.last().status != OverrideStatus::Isolate && stack.vec.len() >= 2 {
237                    stack.vec.pop();
238                }
239                // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
240                levels[i] = stack.last().level;
241                // X9 part of retaining explicit formatting characters.
242                processing_classes[i] = BN;
243            }
244
245            // Nothing.
246            // BN case moved down to X6, see <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
247            B => {}
248
249            // <http://www.unicode.org/reports/tr9/#X6>
250            _ => {
251                let last = stack.last();
252                levels[i] = last.level;
253                // This condition is not in the spec, but I am pretty sure that is a spec bug.
254                // https://www.unicode.org/L2/L2023/23014-amd-to-uax9.pdf
255                if original_classes[i] != BN {
256                    match last.status {
257                        OverrideStatus::RTL => processing_classes[i] = R,
258                        OverrideStatus::LTR => processing_classes[i] = L,
259                        _ => {}
260                    }
261                }
262            }
263        }
264    }
265}

267pub(super) fn prepare_isolating_run_sequences(
268    para_level: Level,
269    original_classes: &[BidiClass],
270    levels: &[Level],
271) -> Vec<IsolatingRunSequence> {
272    let runs = level_runs(levels, original_classes);
273
274    // Compute the set of isolating run sequences.
275    // <http://www.unicode.org/reports/tr9/#BD13>
276    let mut sequences = Vec::with_capacity(runs.len());
277
278    // When we encounter an isolate initiator, we push the current sequence onto the
279    // stack so we can resume it after the matching PDI.
280    let mut stack = vec![Vec::new()];
281
282    use BidiClass::*;
283
284    for run in runs {
285        assert!(!run.is_empty());
286        assert!(!stack.is_empty());
287
288        let start_class = original_classes[run.start];
289        let end_class = original_classes[run.end - 1];
290
291        let mut sequence = if start_class == PDI && stack.len() > 1 {
292            // Continue a previous sequence interrupted by an isolate.
293            stack.pop().unwrap()
294        } else {
295            // Start a new sequence.
296            Vec::new()
297        };
298
299        sequence.push(run);
300
301        if let RLI | LRI | FSI = end_class {
302            // Resume this sequence after the isolate.
303            stack.push(sequence);
304        } else {
305            // This sequence is finished.
306            sequences.push(sequence);
307        }
308    }
309    // Pop any remaining sequences off the stack.
310    sequences.extend(stack.into_iter().rev().filter(|seq| !seq.is_empty()));
311
312    // Determine the `sos` and `eos` class for each sequence.
313    // <http://www.unicode.org/reports/tr9/#X10>
314    sequences
315        .into_iter()
316        .map(|sequence: Vec<LevelRun>| {
317            assert!(!sequence.is_empty());
318
319            let mut result = IsolatingRunSequence {
320                runs: sequence,
321                sos: L,
322                eos: L,
323            };
324
325            let start_of_seq = result.runs[0].start;
326            let runs_len = result.runs.len();
327            let end_of_seq = result.runs[runs_len - 1].end;
328
329            // > (not counting characters removed by X9)
330            let seq_level = result
331                .iter_forwards_from(start_of_seq, 0)
332                .filter(|i| not_removed_by_x9(&original_classes[*i]))
333                .map(|i| levels[i])
334                .next()
335                .unwrap_or(levels[start_of_seq]);
336
337            let end_level = result
338                .iter_backwards_from(end_of_seq, runs_len - 1)
339                .filter(|i| not_removed_by_x9(&original_classes[*i]))
340                .map(|i| levels[i])
341                .next()
342                .unwrap_or(levels[end_of_seq - 1]);
343
344            #[cfg(test)]
345            for run in result.runs.clone() {
346                for idx in run {
347                    if not_removed_by_x9(&original_classes[idx]) {
348                        assert_eq!(seq_level, levels[idx]);
349                    }
350                }
351            }
352
353            // Get the level of the last non-removed char before the runs.
354            let pred_level = match original_classes[..start_of_seq].iter().rposition(not_removed_by_x9) {
355                Some(idx) => levels[idx],
356                None => para_level,
357            };
358
359            // Get the last non-removed character to check if it is an isolate initiator.
360            // The spec calls for an unmatched one, but matched isolate initiators
361            // will never be at the end of a level run (otherwise there would be more to the run).
362            // We unwrap_or(BN) because BN marks removed classes and it won't matter for the check.
363            let last_non_removed = original_classes[..end_of_seq]
364                .iter()
365                .copied()
366                .rev()
367                .find(not_removed_by_x9)
368                .unwrap_or(BN);
369
370            // Get the level of the next non-removed char after the runs.
371            let succ_level = if let RLI | LRI | FSI = last_non_removed {
372                para_level
373            } else {
374                match original_classes[end_of_seq..].iter().position(not_removed_by_x9) {
375                    Some(idx) => levels[end_of_seq + idx],
376                    None => para_level,
377                }
378            };
379
380            result.sos = std::cmp::max(seq_level, pred_level).bidi_class();
381            result.eos = std::cmp::max(end_level, succ_level).bidi_class();
382            result
383        })
384        .collect()
385}

387/// Entries in the directional status stack:
388struct Status {
389    level: Level,
390    status: OverrideStatus,
391}
392
393#[derive(PartialEq)]
394#[expect(clippy::upper_case_acronyms)]
395enum OverrideStatus {
396    Neutral,
397    RTL,
398    LTR,
399    Isolate,
400}
401
402struct DirectionalStatusStack {
403    vec: Vec<Status>,
404}
405
406impl DirectionalStatusStack {
407    fn new() -> Self {
408        DirectionalStatusStack {
409            vec: Vec::with_capacity(Level::max_explicit_depth() as usize + 2),
410        }
411    }
412
413    fn push(&mut self, level: Level, status: OverrideStatus) {
414        self.vec.push(Status { level, status });
415    }
416
417    fn last(&self) -> &Status {
418        self.vec.last().unwrap()
419    }
420}

435/// Should this character be ignored in steps after X9?
436///
437/// <http://www.unicode.org/reports/tr9/#X9>
438fn removed_by_x9(class: BidiClass) -> bool {
439    use BidiClass::*;
440    matches!(class, RLE | LRE | RLO | LRO | PDF | BN)
441}
442
443// For use as a predicate for `position` / `r_position`
444fn not_removed_by_x9(class: &BidiClass) -> bool {
445    !removed_by_x9(*class)
446}
447
448impl IsolatingRunSequence {
449    /// Given a text-relative position `pos` and an index of the level run it is in,
450    /// produce an iterator of all characters after and pos (`pos..`) that are in this
451    /// run sequence
452    pub(crate) fn iter_forwards_from(&self, pos: usize, level_run_index: usize) -> impl Iterator<Item = usize> + '_ {
453        let runs = &self.runs[level_run_index..];
454
455        // Check that it is in range
456        // (we can't use contains() since we want an inclusive range)
457        debug_assert!(runs[0].start <= pos && pos <= runs[0].end);
458
459        (pos..runs[0].end).chain(runs[1..].iter().flat_map(Clone::clone))
460    }
461
462    /// Given a text-relative position `pos` and an index of the level run it is in,
463    /// produce an iterator of all characters before and excluding pos (`..pos`) that are in this
464    /// run sequence
465    pub(crate) fn iter_backwards_from(&self, pos: usize, level_run_index: usize) -> impl Iterator<Item = usize> + '_ {
466        let prev_runs = &self.runs[..level_run_index];
467        let current = &self.runs[level_run_index];
468
469        // Check that it is in range
470        // (we can't use contains() since we want an inclusive range)
471        debug_assert!(current.start <= pos && pos <= current.end);
472
473        (current.start..pos).rev().chain(prev_runs.iter().rev().flat_map(Clone::clone))
474    }
475}

477/// Finds the level runs in a paragraph.
478///
479/// <http://www.unicode.org/reports/tr9/#BD7>
480fn level_runs(levels: &[Level], original_classes: &[BidiClass]) -> Vec<LevelRun> {
481    assert_eq!(levels.len(), original_classes.len());
482
483    let mut runs = Vec::new();
484    if levels.is_empty() {
485        return runs;
486    }
487
488    let mut current_run_level = levels[0];
489    let mut current_run_start = 0;
490    for i in 1..levels.len() {
491        if !removed_by_x9(original_classes[i]) && levels[i] != current_run_level {
492            // End the last run and start a new one.
493            runs.push(current_run_start..i);
494            current_run_level = levels[i];
495            current_run_start = i;
496        }
497    }
498    runs.push(current_run_start..levels.len());
499
500    runs
501}

503/// 3.3.4 Resolving Weak Types
504///
505/// <http://www.unicode.org/reports/tr9/#Resolving_Weak_Types>
506pub(super) fn implicit_resolve_weak(sequence: &IsolatingRunSequence, processing_classes: &mut [BidiClass]) {
507    use BidiClass::*;
508
509    // Note: The spec treats these steps as individual passes that are applied one after the other
510    // on the entire IsolatingRunSequence at once. We instead collapse it into a single iteration,
511    // which is straightforward for rules that are based on the state of the current character, but not
512    // for rules that care about surrounding characters. To deal with them, we retain additional state
513    // about previous character classes that may have since been changed by later rules.
514
515    // The previous class for the purposes of rule W4/W6, not tracking changes made after or during W4.
516    let mut prev_class_before_w4 = sequence.sos;
517    // The previous class for the purposes of rule W5.
518    let mut prev_class_before_w5 = sequence.sos;
519    // The previous class for the purposes of rule W1, not tracking changes from any other rules.
520    let mut prev_class_before_w1 = sequence.sos;
521    let mut last_strong_is_al = false;
522    let mut et_run_indices = Vec::new(); // for W5
523    let mut bn_run_indices = Vec::new(); // for W5 + <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
524
525    for (run_index, level_run) in sequence.runs.iter().enumerate() {
526        for i in &mut level_run.clone() {
527            if processing_classes[i] == BN {
528                // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
529                // Keeps track of BN runs for W5 in case we see an ET.
530                bn_run_indices.push(i);
531                // BNs aren't real, skip over them.
532                continue;
533            }
534
535            // Store the processing class of all rules before W2/W1.
536            // Used to keep track of the last strong character for W2. W3 is able to insert new strong
537            // characters, so we don't want to be misled by it.
538            let mut w2_processing_class = processing_classes[i];
539
540            // <http://www.unicode.org/reports/tr9/#W1>
541            //
542
543            if processing_classes[i] == NSM {
544                processing_classes[i] = match prev_class_before_w1 {
545                    RLI | LRI | FSI | PDI => ON,
546                    _ => prev_class_before_w1,
547                };
548                // W1 occurs before W2, update this.
549                w2_processing_class = processing_classes[i];
550            }
551
552            prev_class_before_w1 = processing_classes[i];
553
554            // <http://www.unicode.org/reports/tr9/#W2>
555            // <http://www.unicode.org/reports/tr9/#W3>
556            //
557            match processing_classes[i] {
558                EN => {
559                    if last_strong_is_al {
560                        // W2. If previous strong char was AL, change EN to AN.
561                        processing_classes[i] = AN;
562                    }
563                }
564                // W3.
565                AL => processing_classes[i] = R,
566                _ => {}
567            }
568
569            // update last_strong_is_al.
570            match w2_processing_class {
571                L | R => {
572                    last_strong_is_al = false;
573                }
574                AL => {
575                    last_strong_is_al = true;
576                }
577                _ => {}
578            }
579
580            let class_before_w456 = processing_classes[i];
581
582            // <http://www.unicode.org/reports/tr9/#W4>
583            // <http://www.unicode.org/reports/tr9/#W5>
584            // <http://www.unicode.org/reports/tr9/#W6> (separators only)
585            // (see below for W6 terminator code)
586            //
587            match processing_classes[i] {
588                // <http://www.unicode.org/reports/tr9/#W6>
589                EN => {
590                    // W5. If a run of ETs is adjacent to an EN, change the ETs to EN.
591                    for j in &et_run_indices {
592                        processing_classes[*j] = EN;
593                    }
594                    et_run_indices.clear();
595                }
596
597                // <http://www.unicode.org/reports/tr9/#W4>
598                // <http://www.unicode.org/reports/tr9/#W6>
599                ES | CS => {
600                    let mut next_class = sequence
601                        .iter_forwards_from(i, run_index)
602                        .map(|j| processing_classes[j])
603                        // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
604                        .find(not_removed_by_x9)
605                        .unwrap_or(sequence.eos);
606                    if next_class == EN && last_strong_is_al {
607                        // Apply W2 to next_class. We know that last_strong_is_al
608                        // has no chance of changing on this character so we can still assume its value
609                        // will be the same by the time we get to it.
610                        next_class = AN;
611                    }
612                    processing_classes[i] = match (prev_class_before_w4, processing_classes[i], next_class) {
613                        // W4
614                        (EN, ES, EN) | (EN, CS, EN) => EN,
615                        // W4
616                        (AN, CS, AN) => AN,
617                        // W6 (separators only)
618                        (_, _, _) => ON,
619                    };
620
621                    // W6 + <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
622                    // We have to do this before W5 gets its grubby hands on these characters and thinks
623                    // they're part of an ET run.
624                    // We check for ON to ensure that we had hit the W6 branch above, since this `ES | CS` match
625                    // arm handles both W4 and W6.
626                    if processing_classes[i] == ON {
627                        for idx in sequence.iter_backwards_from(i, run_index) {
628                            let class = &mut processing_classes[idx];
629                            if *class != BN {
630                                break;
631                            }
632                            *class = ON;
633                        }
634                        for idx in sequence.iter_forwards_from(i, run_index) {
635                            let class = &mut processing_classes[idx];
636                            if *class != BN {
637                                break;
638                            }
639                            *class = ON;
640                        }
641                    }
642                }
643                // <http://www.unicode.org/reports/tr9/#W5>
644                ET => {
645                    match prev_class_before_w5 {
646                        EN => processing_classes[i] = EN,
647                        _ => {
648                            // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
649                            // If there was a BN run before this, that's now a part of this ET run.
650                            et_run_indices.extend(&bn_run_indices);
651
652                            // In case this is followed by an EN.
653                            et_run_indices.push(i);
654                        }
655                    }
656                }
657                _ => {}
658            }
659
660            // Common loop iteration code
661            //
662
663            // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
664            // BN runs would have already continued the loop, clear them before we get to the next one.
665            bn_run_indices.clear();
666
667            // W6 above only deals with separators, so it doesn't change anything W5 cares about,
668            // so we still can update this after running that part of W6.
669            prev_class_before_w5 = processing_classes[i];
670
671            // <http://www.unicode.org/reports/tr9/#W6> (terminators only)
672            // (see above for W6 separator code)
673            //
674            if prev_class_before_w5 != ET {
675                // W6. If we didn't find an adjacent EN, turn any ETs into ON instead.
676                for j in &et_run_indices {
677                    processing_classes[*j] = ON;
678                }
679                et_run_indices.clear();
680            }
681
682            // We stashed this before W4/5/6 could get their grubby hands on it, and it's not
683            // used in the W6 terminator code below so we can update it now.
684            prev_class_before_w4 = class_before_w456;
685        }
686    }
687    // Rerun this check in case we ended with a sequence of BNs (i.e., we'd never
688    // hit the end of the for loop above).
689    // W6. If we didn't find an adjacent EN, turn any ETs into ON instead.
690    for j in &et_run_indices {
691        processing_classes[*j] = ON;
692    }
693    et_run_indices.clear();
694
695    // W7. If the previous strong char was L, change EN to L.
696    let mut last_strong_is_l = sequence.sos == L;
697    for run in &sequence.runs {
698        for i in run.clone() {
699            match processing_classes[i] {
700                EN if last_strong_is_l => {
701                    processing_classes[i] = L;
702                }
703                L => {
704                    last_strong_is_l = true;
705                }
706                R | AL => {
707                    last_strong_is_l = false;
708                }
709                // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
710                // Already scanning past BN here.
711                _ => {}
712            }
713        }
714    }
715}

717/// 3.3.5 Resolving Neutral Types
718///
719/// <http://www.unicode.org/reports/tr9/#Resolving_Neutral_Types>
720pub(super) fn implicit_resolve_neutral(
721    sequence: &IsolatingRunSequence,
722    levels: &[Level],
723    original_classes: &[BidiClass],
724    processing_classes: &mut [BidiClass],
725    brackets: &HashMap<usize, char>,
726) {
727    use BidiClass::*;
728
729    // e = embedding direction
730    let e: BidiClass = levels[sequence.runs[0].start].bidi_class();
731    let not_e = if e == BidiClass::L { BidiClass::R } else { BidiClass::L };
732    // N0. Process bracket pairs.
733
734    // > Identify the bracket pairs in the current isolating run sequence according to BD16.
735    // We use processing_classes, not original_classes, due to BD14/BD15
736    let bracket_pairs = identify_bracket_pairs(sequence, processing_classes, brackets);
737
738    // > For each bracket-pair element in the list of pairs of text positions
739    //
740    // Note: Rust ranges are interpreted as [start..end), be careful using `pair` directly
741    // for indexing as it will include the opening bracket pair but not the closing one.
742    for pair in bracket_pairs {
743        debug_assert!(
744            pair.start < processing_classes.len(),
745            "identify_bracket_pairs returned a range that is out of bounds!"
746        );
747        debug_assert!(
748            pair.end < processing_classes.len(),
749            "identify_bracket_pairs returned a range that is out of bounds!"
750        );
751        let mut found_e = false;
752        let mut found_not_e = false;
753        let mut class_to_set = None;
754
755        // > Inspect the bidirectional types of the characters enclosed within the bracket pair.
756        //
757        // `pair` is [start, end) so we will end up processing the opening character but not the closing one.
758        //
759        for enclosed_i in sequence.iter_forwards_from(pair.start + 1, pair.start_run) {
760            if enclosed_i >= pair.end {
761                debug_assert!(enclosed_i == pair.end, "If we skipped past this, the iterator is broken");
762                break;
763            }
764            let class = processing_classes[enclosed_i];
765            if class == e {
766                found_e = true;
767            } else if class == not_e {
768                found_not_e = true;
769            } else if class == BidiClass::EN || class == BidiClass::AN {
770                // > Within this scope, bidirectional types EN and AN are treated as R.
771                if e == BidiClass::L {
772                    found_not_e = true;
773                } else {
774                    found_e = true;
775                }
776            }
777
778            // If we have found a character with the class of the embedding direction
779            // we can bail early.
780            if found_e {
781                break;
782            }
783        }
784        // > If any strong type (either L or R) matching the embedding direction is found
785        if found_e {
786            // > .. set the type for both brackets in the pair to match the embedding direction
787            class_to_set = Some(e);
788        // > Otherwise, if there is a strong type it must be opposite the embedding direction
789        } else if found_not_e {
790            // > Therefore, test for an established context with a preceding strong type by
791            // > checking backwards before the opening paired bracket
792            // > until the first strong type (L, R, or sos) is found.
793            // (see note above about processing_classes and character boundaries)
794            let mut previous_strong = sequence
795                .iter_backwards_from(pair.start, pair.start_run)
796                .map(|i| processing_classes[i])
797                .find(|class| *class == BidiClass::L || *class == BidiClass::R || *class == BidiClass::EN || *class == BidiClass::AN)
798                .unwrap_or(sequence.sos);
799
800            // > Within this scope, bidirectional types EN and AN are treated as R.
801            if previous_strong == BidiClass::EN || previous_strong == BidiClass::AN {
802                previous_strong = BidiClass::R;
803            }
804
805            // > If the preceding strong type is also opposite the embedding direction,
806            // > context is established,
807            // > so set the type for both brackets in the pair to that direction.
808            // AND
809            // > Otherwise set the type for both brackets in the pair to the embedding direction.
810            // > Either way it gets set to previous_strong
811            //
812            // Both branches amount to setting the type to the strong type.
813            class_to_set = Some(previous_strong);
814        }
815
816        if let Some(class_to_set) = class_to_set {
817            // Update all processing classes corresponding to the start and end elements, as requested.
818            // We should include all bytes of the character, not the first one.
819            for class in &mut processing_classes[pair.start..pair.start + 1] {
820                *class = class_to_set;
821            }
822            for class in &mut processing_classes[pair.end..pair.end + 1] {
823                *class = class_to_set;
824            }
825            // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
826            for idx in sequence.iter_backwards_from(pair.start, pair.start_run) {
827                let class = &mut processing_classes[idx];
828                if *class != BN {
829                    break;
830                }
831                *class = class_to_set;
832            }
833            // > Any number of characters that had original bidirectional character type NSM prior to the application of
834            // > W1 that immediately follow a paired bracket which changed to L or R under N0 should change to match the type of their preceding bracket.
835
836            // This rule deals with sequences of NSMs, so we can just update them all at once, we don't need to worry
837            // about character boundaries. We do need to be careful to skip the full set of bytes for the parentheses characters.
838            let nsm_start = pair.start + 1;
839            for idx in sequence.iter_forwards_from(nsm_start, pair.start_run) {
840                let class = original_classes[idx];
841                if class == BidiClass::NSM || processing_classes[idx] == BN {
842                    processing_classes[idx] = class_to_set;
843                } else {
844                    break;
845                }
846            }
847            let nsm_end = pair.end + 1;
848            for idx in sequence.iter_forwards_from(nsm_end, pair.end_run) {
849                let class = original_classes[idx];
850                if class == BidiClass::NSM || processing_classes[idx] == BN {
851                    processing_classes[idx] = class_to_set;
852                } else {
853                    break;
854                }
855            }
856        }
857        // > Otherwise, there are no strong types within the bracket pair
858        // > Therefore, do not set the type for that bracket pair
859    }
860
861    // N1 and N2.
862    // Indices of every byte in this isolating run sequence
863    let mut indices = sequence.runs.iter().flat_map(Clone::clone);
864    let mut prev_class = sequence.sos;
865    while let Some(mut i) = indices.next() {
866        // Process sequences of NI characters.
867        let mut ni_run = Vec::new();
868        // The BN is for <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
869        if is_NI(processing_classes[i]) || processing_classes[i] == BN {
870            // Consume a run of consecutive NI characters.
871            ni_run.push(i);
872            let mut next_class;
873            loop {
874                match indices.next() {
875                    Some(j) => {
876                        i = j;
877                        next_class = processing_classes[j];
878                        // The BN is for <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters>
879                        if is_NI(next_class) || next_class == BN {
880                            ni_run.push(i);
881                        } else {
882                            break;
883                        }
884                    }
885                    None => {
886                        next_class = sequence.eos;
887                        break;
888                    }
889                };
890            }
891            // N1-N2.
892            //
893            // <http://www.unicode.org/reports/tr9/#N1>
894            // <http://www.unicode.org/reports/tr9/#N2>
895            let new_class = match (prev_class, next_class) {
896                (L, L) => L,
897                (R, R) | (R, AN) | (R, EN) | (AN, R) | (AN, AN) | (AN, EN) | (EN, R) | (EN, AN) | (EN, EN) => R,
898                (_, _) => e,
899            };
900            for j in &ni_run {
901                processing_classes[*j] = new_class;
902            }
903            ni_run.clear();
904        }
905        prev_class = processing_classes[i];
906    }
907}

909struct BracketPair {
910    /// The text-relative index of the opening bracket.
911    start: usize,
912    /// The text-relative index of the closing bracket.
913    end: usize,
914    /// The index of the run (in the run sequence) that the opening bracket is in.
915    start_run: usize,
916    /// The index of the run (in the run sequence) that the closing bracket is in.
917    end_run: usize,
918}
919/// 3.1.3 Identifying Bracket Pairs
920///
921/// Returns all paired brackets in the source, as indices into the
922/// text source.
923///
924/// <https://www.unicode.org/reports/tr9/#BD16>
925fn identify_bracket_pairs(
926    run_sequence: &IsolatingRunSequence,
927    original_classes: &[BidiClass],
928    brackets: &HashMap<usize, char>,
929) -> Vec<BracketPair> {
930    let data_source = &unicode_bidi::HardcodedBidiData;
931
932    let mut ret = vec![];
933    let mut stack = vec![];
934
935    for (run_index, level_run) in run_sequence.runs.iter().enumerate() {
936        for i in 0..level_run.len() {
937            let actual_index = level_run.start + i;
938            // All bracket characters are ON.
939            // From BidiBrackets.txt:
940            // > The Unicode property value stability policy guarantees that characters
941            // > which have bpt=o or bpt=c also have bc=ON and Bidi_M=Y
942            if original_classes[actual_index] != BidiClass::ON {
943                continue;
944            }
945
946            if let Some(matched) = brackets
947                .get(&actual_index)
948                .and_then(|c| data_source.bidi_matched_opening_bracket(*c))
949            {
950                if matched.is_open {
951                    // > If an opening paired bracket is found ...
952
953                    // > ... and there is no room in the stack,
954                    // > stop processing BD16 for the remainder of the isolating run sequence.
955                    if stack.len() >= 63 {
956                        break;
957                    }
958                    // > ... push its Bidi_Paired_Bracket property value and its text position onto the stack
959                    stack.push((matched.opening, actual_index, run_index))
960                } else {
961                    // > If a closing paired bracket is found, do the following
962
963                    // > Declare a variable that holds a reference to the current stack element
964                    // > and initialize it with the top element of the stack.
965                    // AND
966                    // > Else, if the current stack element is not at the bottom of the stack
967                    for (stack_index, element) in stack.iter().enumerate().rev() {
968                        // > Compare the closing paired bracket being inspected or its canonical
969                        // > equivalent to the bracket in the current stack element.
970                        if element.0 == matched.opening {
971                            // > If the values match, meaning the two characters form a bracket pair, then
972
973                            // > Append the text position in the current stack element together with the
974                            // > text position of the closing paired bracket to the list.
975                            let pair = BracketPair {
976                                start: element.1,
977                                end: actual_index,
978                                start_run: element.2,
979                                end_run: run_index,
980                            };
981                            ret.push(pair);
982
983                            // > Pop the stack through the current stack element inclusively.
984                            stack.truncate(stack_index);
985                            break;
986                        }
987                    }
988                }
989            }
990        }
991    }
992    // > Sort the list of pairs of text positions in ascending order based on
993    // > the text position of the opening paired bracket.
994    ret.sort_by_key(|r| r.start);
995    ret
996}

998/// Neutral or Isolate formatting character (B, S, WS, ON, FSI, LRI, RLI, PDI)
999///
1000/// <http://www.unicode.org/reports/tr9/#NI>
1001#[expect(non_snake_case)]
1002fn is_NI(class: BidiClass) -> bool {
1003    use BidiClass::*;
1004    matches!(class, B | S | WS | ON | FSI | LRI | RLI | PDI)
1005}

1007/// 3.3.6 Resolving Implicit Levels
1008///
1009/// Returns the maximum embedding level in the paragraph.
1010///
1011/// <http://www.unicode.org/reports/tr9/#Resolving_Implicit_Levels>
1012pub(super) fn implicit_resolve_levels(original_classes: &[BidiClass], levels: &mut [Level]) -> Level {
1013    use BidiClass::*;
1014
1015    let mut max_level = Level::ltr();
1016    assert_eq!(original_classes.len(), levels.len());
1017    for i in 0..levels.len() {
1018        match (levels[i].is_rtl(), original_classes[i]) {
1019            (false, AN) | (false, EN) => levels[i].raise(2).expect("Level number error"),
1020            (false, R) | (true, L) | (true, EN) | (true, AN) => levels[i].raise(1).expect("Level number error"),
1021            // <https://www.unicode.org/reports/tr9/#Retaining_Explicit_Formatting_Characters> handled here
1022            (_, _) => {}
1023        }
1024        max_level = std::cmp::max(max_level, levels[i]);
1025    }
1026
1027    max_level
1028}

1030/// Assign levels to characters removed by rule X9.
1031///
1032/// The levels assigned to these characters are not specified by the algorithm. This function
1033/// assigns each one the level of the previous character, to avoid breaking level runs.
1034pub(super) fn assign_levels_to_removed_chars(para_level: Level, classes: &[BidiClass], levels: &mut [Level]) {
1035    for i in 0..levels.len() {
1036        if removed_by_x9(classes[i]) {
1037            levels[i] = if i > 0 { levels[i - 1] } else { para_level };
1038        }
1039    }
