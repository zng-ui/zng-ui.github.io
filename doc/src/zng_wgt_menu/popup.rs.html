20/// Sub-menu popup.
21#[widget($crate::popup::SubMenuPopup)]
22pub struct SubMenuPopup(zng_wgt_layer::popup::Popup);
23impl SubMenuPopup {
24    fn widget_intrinsic(&mut self) {
25        self.style_intrinsic(STYLE_FN_VAR, property_id!(self::style_fn));
26        widget_set! {
27            self;
28            style_base_fn = style_fn!(|_| DefaultStyle!());
29
30            // Supports press-and-drag to click gesture:
31            //
32            // - Sub-menu is `capture_pointer = true`.
33            // - Menu items set`click_mode = release`.
34            //
35            // So the user can press to open the menu, then drag over an item and release to click it.
36            capture_pointer_on_init = CaptureMode::Subtree;
37        }
38
39        self.widget_builder().push_build_action(|wgt| {
40            let id = wgt.capture_value::<WidgetId>(property_id!(Self::parent_id));
41            let children = wgt
42                .capture_property(property_id!(Self::children))
43                .map(|p| p.args.ui_node(0).clone())
44                .unwrap_or_else(|| ArcNode::new(ui_vec![]));
45
46            wgt.set_child(sub_menu_popup_node(children, id));
47        });
48    }
49}
50impl_style_fn!(SubMenuPopup);

97/// Default sub-menu popup panel view.
98///
99/// See [`PANEL_FN_VAR`] for more details.
100pub fn default_panel_fn(args: zng_wgt_panel::PanelArgs) -> UiNode {
101    // remove arrow key shortcuts, they are used to navigate focus.
102    let scroll_id = WidgetId::new_unique();
103    zng_wgt_scroll::cmd::SCROLL_UP_CMD
104        .scoped(scroll_id)
105        .shortcut()
106        .set(Shortcuts::new());
107    zng_wgt_scroll::cmd::SCROLL_DOWN_CMD
108        .scoped(scroll_id)
109        .shortcut()
110        .set(Shortcuts::new());
111
112    zng_wgt_scroll::Scroll! {
113        id = scroll_id;
114        focusable = false;
115        child_align = Align::FILL;
116        child = Stack! {
117            children_align = Align::FILL;
118            children = args.children;
119            direction = zng_wgt_stack::StackDirection::top_to_bottom();
120        };
121        mode = zng_wgt_scroll::ScrollMode::VERTICAL;
122    }
123}

125/// Sub-menu popup implementation.
126pub fn sub_menu_popup_node(children: ArcNode, parent: Option<WidgetId>) -> UiNode {
127    let child = zng_wgt_panel::node(
128        children,
129        if parent.is_none() {
130            super::context::PANEL_FN_VAR
131        } else {
132            PANEL_FN_VAR
133        },
134    );
135    let mut close_timer = None;
136    match_node(child, move |c, op| match op {
137        UiNodeOp::Init => {
138            WIDGET
139                .sub_event(&KEY_INPUT_EVENT)
140                .sub_event(&POPUP_CLOSE_REQUESTED_EVENT)
141                .sub_event(&FOCUS_CHANGED_EVENT);
142        }
143        UiNodeOp::Deinit => {
144            close_timer = None;
145        }
146        UiNodeOp::Info { info } => {
147            // sub-menus set the popup as parent in context menu.
148            let parent_ctx = Some(parent.unwrap_or_else(|| WIDGET.id()));
149            super::sub::SUB_MENU_PARENT_CTX.with_context(&mut Some(Arc::new(parent_ctx)), || c.info(info));
150            info.set_meta(*super::sub::SUB_MENU_POPUP_ID, super::sub::SubMenuPopupInfo { parent });
151        }
152        UiNodeOp::Event { update } => {
153            c.event(update);
154
155            if let Some(args) = KEY_INPUT_EVENT.on_unhandled(update) {
156                if let KeyState::Pressed = args.state {
157                    match &args.key {
158                        Key::Escape => {
159                            let info = WIDGET.info();
160                            if let Some(m) = info.submenu_parent() {
161                                args.propagation().stop();
162
163                                FOCUS.focus_widget(m.id(), true);
164                                POPUP.force_close_id(info.id());
165                            }
166                        }
167                        Key::ArrowLeft | Key::ArrowRight => {
168                            if let Some(info) = WINDOW.info().get(args.target.widget_id()) {
169                                let info = info.into_focus_info(true, true);
170                                if info.focusable_left().is_none() && info.focusable_right().is_none() {
171                                    // escape to parent or change root.
172                                    if let Some(m) = info.info().submenu_parent() {
173                                        let mut escape = false;
174                                        if m.submenu_parent().is_some()
175                                            && let Some(o) = m.orientation_from(info.info().center())
176                                        {
177                                            escape = match o {
178                                                Orientation2D::Left => args.key == Key::ArrowLeft,
179                                                Orientation2D::Right => args.key == Key::ArrowRight,
180                                                Orientation2D::Below | Orientation2D::Above => false,
181                                            };
182                                        }
183
184                                        if escape {
185                                            args.propagation().stop();
186                                            // escape
187
188                                            FOCUS.focus_widget(m.id(), true);
189                                            POPUP.force_close_id(WIDGET.id());
190                                        } else if let Some(m) = info.info().submenu_root() {
191                                            args.propagation().stop();
192                                            // change root
193
194                                            let m = m.into_focus_info(true, true);
195                                            let next_root = match &args.key {
196                                                Key::ArrowLeft => m.next_left(),
197                                                Key::ArrowRight => m.next_right(),
198                                                _ => unreachable!(),
199                                            };
200                                            if let Some(n) = next_root {
201                                                FOCUS.focus_widget(n.info().id(), true);
202                                            }
203                                        }
204                                    }
205                                }
206                            }
207                        }
208                        _ => {}
209                    }
210                }
211            } else if let Some(args) = POPUP_CLOSE_REQUESTED_EVENT.on_unhandled(update) {
212                let sub_self = if parent.is_some() {
213                    WIDGET.info().submenu_parent()
214                } else {
215                    // is context menu
216                    Some(WIDGET.info())
217                };
218                if let Some(sub_self) = sub_self {
219                    let mut close_ancestors = Some(None);
220
221                    if let Some(focused) = FOCUS.focused().get()
222                        && let Some(focused) = sub_self.tree().get(focused.widget_id())
223                        && let Some(sub_focused) = focused.submenu_parent()
224                    {
225                        if sub_focused.submenu_ancestors().any(|a| a.id() == sub_self.id()) {
226                            // keep open, focused child.
227                            args.propagation().stop();
228                            close_ancestors = None;
229                        } else if sub_self.submenu_ancestors().any(|a| a.id() == sub_focused.id()) {
230                            if Some(sub_focused.id()) == sub_self.submenu_parent().map(|s| s.id()) {
231                                // keep open, focused parent.
232                                args.propagation().stop();
233                                close_ancestors = None;
234                            } else {
235                                close_ancestors = Some(Some(sub_focused.id()));
236                            }
237                        }
238                    }
239
240                    if let Some(sub_parent_focused) = close_ancestors {
241                        // close any parent sub-menu that is not focused.
242                        for a in sub_self.submenu_ancestors() {
243                            if Some(a.id()) == sub_parent_focused {
244                                break;
245                            }
246
247                            if let Some(v) = a.is_submenu_open() {
248                                if v.get() {
249                                    // request ancestor close the popup.
250                                    POPUP_CLOSE_CMD.scoped(a.id()).notify();
251                                }
252                            } else if a.menu().is_none() {
253                                // request context menu popup close
254                                POPUP_CLOSE_CMD.scoped(a.id()).notify_param(PopupCloseMode::Force);
255                            }
256                        }
257                    }
258                }
259            } else if let Some(args) = FOCUS_CHANGED_EVENT.on(update)
260                && args.is_focus_leave(WIDGET.id())
261                && let Some(f) = &args.new_focus
262            {
263                let info = WIDGET.info();
264                let sub_self = if parent.is_some() {
265                    info.submenu_parent()
266                } else {
267                    // is context menu
268                    Some(info.clone())
269                };
270                if let Some(sub_menu) = sub_self
271                    && let Some(f) = info.tree().get(f.widget_id())
272                    && !f.submenu_self_and_ancestors().any(|s| s.id() == sub_menu.id())
273                {
274                    // Focus did not move to child sub-menu nor parent,
275                    // close after delay.
276                    //
277                    // This covers the case of focus moving to a widget that is not
278                    // a child sub-menu and is not the parent sub-menu,
279                    // `sub_menu_node` covers the case of focus moving to the parent sub-menu and out.
280                    let t = TIMERS.deadline(HOVER_OPEN_DELAY_VAR.get());
281                    t.subscribe(UpdateOp::Update, info.id()).perm();
282                    close_timer = Some(t);
283                }
284            }
285        }
286        UiNodeOp::Update { .. } => {
287            if let Some(t) = &close_timer
288                && t.get().has_elapsed()
289            {
290                close_timer = None;
291                POPUP.force_close_id(WIDGET.id());
292            }
293        }
294        _ => {}
295    })
