26/// Menu root panel.
27#[widget($crate::Menu { ($children:expr) => { children = $children; } })]
28pub struct Menu(StyleMix<zng_wgt_panel::Panel>);
29impl Menu {
30    fn widget_intrinsic(&mut self) {
31        self.style_intrinsic(STYLE_FN_VAR, property_id!(self::style_fn));
32        widget_set! {
33            self;
34            alt_focus_scope = true;
35            zng_wgt_panel::panel_fn = PANEL_FN_VAR;
36            style_base_fn = style_fn!(|_| DefaultStyle!());
37            access_role = AccessRole::Menu;
38            zng_wgt_rule_line::collapse_scope = true;
39        }
40    }
41}
42impl_style_fn!(Menu);
43
44context_var! {
45    /// Defines the layout widget for [`Menu!`].
46    ///
47    /// Is a [`Wrap!`] panel by default.
48    ///
49    /// [`Menu!`]: struct@Menu
50    /// [`Wrap!`]: struct@zng_wgt_wrap::Wrap
51    pub static PANEL_FN_VAR: WidgetFn<zng_wgt_panel::PanelArgs> = wgt_fn!(|a: zng_wgt_panel::PanelArgs| {
52        zng_wgt_wrap::Wrap! {
53            children = a.children;
54        }
55    });
56
57    /// Minimum space between a menu item child and the [`shortcut_txt`] child.
58    ///
59    /// The spacing is applied only if the shortcut child is set to a non-zero sized widget and
60    /// there is no other wider menu item.
61    ///
62    /// Is `20` by default.
63    ///
64    /// [`shortcut_txt`]: fn@shortcut_txt
65    pub static SHORTCUT_SPACING_VAR: Length = 20;
66}

68/// Widget function that generates the menu layout.
69///
70/// This property can be set in any widget to affect all [`Menu!`] descendants.
71///
72/// This property sets [`PANEL_FN_VAR`].
73///
74/// [`Menu!`]: struct@Menu
75#[property(CONTEXT, default(PANEL_FN_VAR), widget_impl(Menu))]
76pub fn panel_fn(child: impl IntoUiNode, panel: impl IntoVar<WidgetFn<zng_wgt_panel::PanelArgs>>) -> UiNode {
77    with_context_var(child, PANEL_FN_VAR, panel)
78}

105/// Style applied to all [`Button!`] widgets inside [`Menu!`] root.
106///
107/// Gives the button a *toolbar-item* look.
108///
109/// See also [`sub::ButtonStyle!`] for the style of buttons inside the sub-menus.
110///
111/// [`Button!`]: struct@zng_wgt_button::Button
112/// [`Menu!`]: struct@Menu
113/// [`sub::ButtonStyle!`]: struct@sub::ButtonStyle
114#[widget($crate::ButtonStyle)]
115pub struct ButtonStyle(zng_wgt_button::LightStyle);
116impl ButtonStyle {
117    fn widget_intrinsic(&mut self) {
118        widget_set! {
119            self;
120
121            padding = 4;
122
123            access_role = AccessRole::MenuItem;
124            focus_click_behavior = FocusClickBehavior::ExitEnabled;
125
126            zng_wgt_container::child_start =
127                BUTTON
128                    .cmd()
129                    .flat_map(|c| c.as_ref().map(|c| c.icon()).unwrap_or_else(|| const_var(WidgetFn::nil())))
130                    .present_data(()),
131                0,
132            ;
133        }
134    }
135}

137/// Alternate style for buttons inside a menu.
138///
139/// If the button has a command, show the icon as child, if the command has no icon shows the name.
140///
141/// [`Button!`]: struct@zng_wgt_button::Button
142/// [`Menu!`]: struct@Menu
143/// [`sub::ButtonStyle!`]: struct@sub::ButtonStyle
144#[widget($crate::IconButtonStyle)]
145pub struct IconButtonStyle(zng_wgt_button::LightStyle);
146impl IconButtonStyle {
147    fn widget_intrinsic(&mut self) {
148        widget_set! {
149            self;
150
151            padding = 4;
152
153            access_role = AccessRole::MenuItem;
154            focus_click_behavior = FocusClickBehavior::ExitEnabled;
155
156            zng_wgt_container::child =
157                BUTTON
158                    .cmd()
159                    .flat_map(|c| match c {
160                        Some(c) => expr_var! {
161                            let icon = #{c.icon()};
162                            let name = #{c.name()};
163                            wgt_fn!(icon, name, |args| {
164                                let icon = icon(args);
165                                if icon.is_nil() { zng_wgt_text::Text!(name.clone()) } else { icon }
166                            })
167                        },
168                        None => const_var(WidgetFn::nil()),
169                    })
170                    .present_data(()),
171            ;
172
173            zng_wgt_button::cmd_tooltip_fn = wgt_fn!(|args: zng_wgt_button::CmdTooltipArgs| {
174                let name = args.cmd.name();
175                let info = args.cmd.info();
176                let shortcut = args.cmd.shortcut();
177                zng_wgt_tooltip::Tip!(zng_wgt_stack::Stack! {
178                    direction = zng_wgt_stack::StackDirection::top_to_bottom();
179                    spacing = 5;
180                    children = ui_vec![
181                        zng_wgt_text::Text!(name),
182                        zng_wgt_text::Text! {
183                            zng_wgt::visibility = info.map(|s| (!s.is_empty()).into());
184                            txt = info;
185                        },
186                        zng_wgt_shortcut::ShortcutText!(shortcut)
187                    ];
188                })
189            });
190        }
191    }
192}

241/// Menu item shortcut text.
242///
243/// Set this on a [`Button!`] inside a sub-menu to define the shortcut text.
244///
245/// Note that this does not define the click shortcut, just the display of it. The [`ShortcutText!`]
246/// widget is recommended.
247///
248/// [`Button!`]: struct@zng_wgt_button::Button
249/// [`ShortcutText!`]: struct@zng_wgt_shortcut::ShortcutText
250#[property(CHILD_CONTEXT)]
251pub fn shortcut_txt(child: impl IntoUiNode, shortcut: impl IntoUiNode) -> UiNode {
252    let shortcut = margin(shortcut, sub::END_COLUMN_WIDTH_VAR.map(|w| SideOffsets::new(0, w.clone(), 0, 0)));
253    child_end(child, shortcut, SHORTCUT_SPACING_VAR)
254}
