8/// (pkg_name, file, id, attribute)
9type Key = [Txt; 4];
10
11#[derive(Default)]
12struct UsageRecorder {
13    used: Mutex<HashSet<Key>>,
14}
15impl UsageRecorder {
16    pub(crate) fn insert(&self, key: Key) {
17        let mut used = self.used.lock();
18        if !used.contains(&key) {
19            used.insert(key);
20        }
21    }
22}
23
24static USED: LazyLock<UsageRecorder> = LazyLock::new(Default::default);
25
26pub(crate) fn insert(file: &LangFilePath, id: &Txt, attribute: &Txt) {
27    if !file.is_current_app() {
28        USED.insert([file.pkg_name.clone(), file.file.clone(), id.clone(), attribute.clone()]);
29    }
30}
31
32pub(crate) fn on_init() {
33    // assert profile path is valid
34    let _ = env_file();
35
36    zng_app::APP.on_deinit(|_| save());
37}

39fn collect() -> String {
40    use std::fmt::Write as _;
41
42    let mut s = String::new();
43
44    writeln!(&mut s, "# Auto generated by zng-ext-10n usage recorder, don't edit").unwrap();
45    writeln!(&mut s, "# Add manual selections to file `./{{name}}.subset`, without `.rec.`").unwrap();
46    writeln!(&mut s).unwrap();
47
48    let mut used: Vec<_> = USED.used.lock().drain().collect();
49    used.sort();
50    for [pkg_name, file, id, attribute] in used {
51        write!(&mut s, "{pkg_name}//").unwrap();
52        if !file.is_empty() && file != "_" {
53            write!(&mut s, "{file}/").unwrap();
54        }
55        write!(&mut s, "{id}").unwrap();
56        if !attribute.is_empty() {
57            write!(&mut s, ".{attribute}").unwrap();
58        }
59        writeln!(&mut s).unwrap();
60    }
61    s
62}

64fn env_file() -> Option<String> {
65    let p = std::env::var("ZNG_L10N_PROFILE_FILE").ok()?;
66    if !p.ends_with(".rec.subset") {
67        panic!("ZNG_L10N_PROFILE_FILE must be set to a file with extension `.rec.subset`")
68    }
69    Some(p)
70}
71fn save() {
72    fn write(profile: &str) -> io::Result<()> {
73        let path = env_file().unwrap_or_else(|| "res/optimization-profiles/zng-ext-10n.rec.subset".to_owned());
74        let path = PathBuf::from(path);
75        if let Some(dir) = path.parent() {
76            fs::create_dir_all(dir)?;
77        }
78        fs::write(path, profile.as_bytes())
79    }
80    let profile = collect();
81
82    if let Err(e) = write(&profile) {
83        panic!("failed to save l10n usage profile\n{e}\n\n```zng-ext-10n.rec.subset\n{profile}\n```")
84    }
85}
