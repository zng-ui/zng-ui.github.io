55/// Represents the caret position at the [`ResolvedText`] context.
56#[derive(Clone)]
57#[non_exhaustive]
58pub struct CaretInfo {
59    /// Caret opacity.
60    ///
61    /// This variable is replaced often, the text resolver subscribes to it for
62    /// [`UpdateOp::RenderUpdate`] automatically.
63    ///
64    /// [`UpdateOp::RenderUpdate`]: zng_wgt::prelude::UpdateOp::RenderUpdate
65    pub opacity: ReadOnlyArcVar<Factor>,
66
67    /// Caret byte offset in the text string.
68    ///
69    /// This is the insertion offset on the text, it can be the text length.
70    pub index: Option<CaretIndex>,
71
72    /// Second index that defines the start or end of a selection range.
73    pub selection_index: Option<CaretIndex>,
74
75    /// Selection by word or line sets this value, selection extend by word or line
76    /// grows from this central selection. The value is `(selection, is_word)`.
77    pub initial_selection: Option<(ops::Range<CaretIndex>, bool)>,
78
79    /// Value incremented by one every time the `index` is set.
80    ///
81    /// This is used to signal interaction with the `index` value by [`TextEditOp`]
82    /// even if the interaction only sets-it to the index same value.
83    ///
84    /// [`TextEditOp`]: crate::cmd::TextEditOp
85    pub index_version: Wrapping<u8>,
86
87    /// If the index was set by using the [`caret_retained_x`].
88    ///
89    /// [`caret_retained_x`]: LaidoutText::caret_retained_x
90    pub used_retained_x: bool,
91
92    /// Don't scroll to new caret position on the next update.
93    ///
94    /// If this is set to `true` the next time `index` or `index_version` changes auto-scroll is skipped once.
95    pub skip_next_scroll: bool,
96}
97impl fmt::Debug for CaretInfo {
98    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
99        f.debug_struct("CaretInfo")
100            .field("opacity", &self.opacity.debug())
101            .field("index", &self.index)
102            .field("index_version", &self.index_version)
103            .field("used_retained_x", &self.used_retained_x)
104            .finish()
105    }

107impl CaretInfo {
108    /// Set the index and update the index version.
109    pub fn set_index(&mut self, index: CaretIndex) {
110        self.index = Some(index);
111        self.index_version += 1;
112    }
113
114    /// Sets the selection start, end and update the index version.
115    ///
116    /// The `end` is the caret position.
117    pub fn set_selection(&mut self, start: CaretIndex, end: CaretIndex) {
118        self.selection_index = Some(start);
119        self.set_index(end);
120    }
121
122    /// Clears selection.
123    pub fn clear_selection(&mut self) {
124        self.selection_index = None;
125        self.initial_selection = None;
126        self.index_version += 1;
127    }
128
129    /// Set the char byte index and update the index version.
130    ///
131    /// The caret line is always snapped when the caret changes, so the line value will be updated.
132    pub fn set_char_index(&mut self, index: usize) {
133        if let Some(i) = &mut self.index {
134            i.index = index;
135        } else {
136            self.index = Some(CaretIndex { index, line: 0 });
137        }
138        self.index_version += 1;
139    }
140
141    /// Set the char byte index of the selection start, end and update the index version.
142    ///
143    /// The `end` is the caret position.
144    ///
145    /// The caret and selection lines are always snapped when the caret changes, so the line values will be updated.
146    pub fn set_char_selection(&mut self, start: usize, end: usize) {
147        if let Some(s) = &mut self.selection_index {
148            s.index = start;
149        } else {
150            self.selection_index = Some(CaretIndex { index: start, line: 0 });
151        }
152        self.set_char_index(end);
153    }
154
155    /// Gets the selection range if both [`index`] and [`selection_index`] are set.
156    ///
157    /// [`index`]: Self::index
158    /// [`selection_index`]: Self::selection_index
159    pub fn selection_range(&self) -> Option<ops::Range<CaretIndex>> {
160        let a = self.index?;
161        let b = self.selection_index?;
162
163        use std::cmp::Ordering;
164        match a.index.cmp(&b.index) {
165            Ordering::Less => Some(a..b),
166            Ordering::Equal => None,
167            Ordering::Greater => Some(b..a),
168        }
169    }
170
171    /// Gets the character range of the selection if both [`index`] and [`selection_index`] are set.
172    ///
173    /// [`index`]: Self::index
174    /// [`selection_index`]: Self::selection_index
175    pub fn selection_char_range(&self) -> Option<ops::Range<usize>> {
176        self.selection_range().map(|r| r.start.index..r.end.index)
177    }
178}

577pub(super) fn get_caret_index(child: impl UiNode, index: impl IntoVar<Option<CaretIndex>>) -> impl UiNode {
578    let index = index.into_var();
579    match_node(child, move |c, op| {
580        let mut u = false;
581        match op {
582            UiNodeOp::Init => {
583                c.init();
584                let _ = index.set(TEXT.resolved().caret.index);
585            }
586            UiNodeOp::Deinit => {
587                let _ = index.set(None);
588            }
589            UiNodeOp::Event { update } => {
590                c.event(update);
591                u = true;
592            }
593            UiNodeOp::Update { updates } => {
594                c.update(updates);
595                u = true;
596            }
597            _ => {}
598        }
599        if u {
600            let t = TEXT.resolved();
601            let idx = t.caret.index;
602            if !t.pending_edit && index.get() != idx {
603                let _ = index.set(idx);
604            }
605        }
606    })
607}

609pub(super) fn get_caret_status(child: impl UiNode, status: impl IntoVar<CaretStatus>) -> impl UiNode {
610    let status = status.into_var();
611    match_node(child, move |c, op| {
612        let mut u = false;
613        match op {
614            UiNodeOp::Init => {
615                c.init();
616                let t = TEXT.resolved();
617                let _ = status.set(match t.caret.index {
618                    None => CaretStatus::none(),
619                    Some(i) => CaretStatus::new(i.index, &t.segmented_text),
620                });
621            }
622            UiNodeOp::Deinit => {
623                let _ = status.set(CaretStatus::none());
624            }
625            UiNodeOp::Event { update } => {
626                c.event(update);
627                u = true;
628            }
629            UiNodeOp::Update { updates } => {
630                c.update(updates);
631                u = true;
632            }
633            _ => {}
634        }
635        if u {
636            let t = TEXT.resolved();
637            let idx = t.caret.index;
638            if !t.pending_edit && status.get().index() != idx.map(|ci| ci.index) {
639                let _ = status.set(match idx {
640                    None => CaretStatus::none(),
641                    Some(i) => CaretStatus::new(i.index, &t.segmented_text),
642                });
643            }
644        }
645    })
646}

648pub(super) fn get_lines_len(child: impl UiNode, len: impl IntoVar<usize>) -> impl UiNode {
649    let len = len.into_var();
650    match_node(child, move |c, op| match op {
651        UiNodeOp::Deinit => {
652            let _ = len.set(0usize);
653        }
654        UiNodeOp::Layout { wl, final_size } => {
655            *final_size = c.layout(wl);
656            let t = TEXT.laidout();
657            let l = t.shaped_text.lines_len();
658            if l != len.get() {

686// Returns `Some(_)` if the current wrap count changed from `prev`. Only allocates if new count has wrapped lines.
687fn lines_wrap_count(prev: &super::LinesWrapCount, txt: &ShapedText) -> Option<super::LinesWrapCount> {
688    match prev {
689        super::LinesWrapCount::NoWrap(len) => {
690            let mut counter = lines_wrap_counter(txt);
691            let mut l = 0;
692            for c in &mut counter {
693                if c != 1 {
694                    // at least one line wraps now
695                    let mut wrap = vec![1; l];
696                    wrap.push(c);
697                    wrap.extend(&mut counter);
698                    return Some(super::LinesWrapCount::Wrap(wrap));
699                }
700                l += 1;
701            }
702            if l != *len {
703                // no line wraps, but changed line count.
704                Some(super::LinesWrapCount::NoWrap(l))
705            } else {
706                None
707            }
708        }
709        super::LinesWrapCount::Wrap(counts) => {
710            // find `counts[i]` that diverges from counts, OR
711            // find if all new counts is now NoWrap
712            let mut prev_counts = counts.iter();
713            let mut new_counts = lines_wrap_counter(txt);
714            let mut eq_l = 0;
715            let mut eq_wrap = false;
716            for c in &mut new_counts {
717                if prev_counts.next() == Some(&c) {
718                    eq_l += 1;
719                    eq_wrap |= c != 1;
720                } else if eq_wrap || c != 1 {
721                    // not eq, and already found a wrap line
722                    let mut wrap = counts[..eq_l].to_vec();
723                    wrap.push(c);
724                    wrap.extend(&mut new_counts);
725                    return Some(super::LinesWrapCount::Wrap(wrap));
726                } else {
727                    // not eq, but maybe no wrap
728                    let mut l = eq_l + 1; // +1 is +c
729                    for c in &mut new_counts {
730                        if c != 1 {
731                            // nope, found a line wrap
732                            let mut wrap = vec![1; l];
733                            wrap.push(c);
734                            wrap.extend(&mut new_counts);
735                            return Some(super::LinesWrapCount::Wrap(wrap));
736                        }
737                        l += 1;
738                    }
739                    // changed to no wrap
740                    return Some(super::LinesWrapCount::NoWrap(l));
741                }
742            }
743            if prev_counts.next().is_some() {
744                Some(super::LinesWrapCount::Wrap(counts[..eq_l].to_vec()))
745            } else {
746                None
747            }
748        }
749    }
750}
751fn lines_wrap_counter(txt: &ShapedText) -> impl Iterator<Item = u32> + '_ {
752    struct Counter<I> {
753        lines: I,
754        count: u32,
755    }
756    impl<'a, I: Iterator<Item = ShapedLine<'a>>> Iterator for Counter<I> {
757        type Item = u32;
758
759        fn next(&mut self) -> Option<u32> {
760            loop {
761                let line = self.lines.next()?;
762                if line.ended_by_wrap() {
763                    self.count += 1;
764                    continue;
765                }
766
767                let c = self.count;
768                self.count = 1;
769
770                return Some(c);
771            }
772        }
773    }
774    Counter {
775        lines: txt.lines(),
776        count: 1,
777    }
778}

780pub(super) fn parse_text<T>(child: impl UiNode, value: impl IntoVar<T>) -> impl UiNode
781where
782    T: super::TxtParseValue,
783{
784    let value = value.into_var();
785
786    let error = var(Txt::from_static(""));
787    let mut _error_note = DataNoteHandle::dummy();
788
789    #[derive(Clone, Copy, bytemuck::NoUninit)]
790    #[repr(u8)]
791    enum State {
792        Sync,
793        Requested,
794        Pending,
795    }
796    let state = Arc::new(Atomic::new(State::Sync));
797
798    match_node(child, move |_, op| match op {
799        UiNodeOp::Init => {
800            let ctx = TEXT.resolved();
801
802            // initial T -> Txt sync
803            let _ = ctx.txt.set_from_map(&value, |val| val.to_txt());
804
805            // bind `TXT_PARSE_LIVE_VAR` <-> `value` using `bind_filter_map_bidi`:
806            // - in case of parse error, it is set in `error` variable, that is held by the binding.
807            // - on error update the DATA note is updated.
808            // - in case parse is not live, ignores updates (Txt -> None), sets `state` to `Pending`.
809            // - in case of Pending and `PARSE_CMD` state is set to `Requested` and `TXT_PARSE_LIVE_VAR.update()`.
810            // - the pending state is also tracked in `TXT_PARSE_PENDING_VAR` and the `PARSE_CMD` handle.
811
812            let live = TXT_PARSE_LIVE_VAR.actual_var();
813            let is_pending = TXT_PARSE_PENDING_VAR.actual_var();
814            let cmd_handle = Arc::new(super::cmd::PARSE_CMD.scoped(WIDGET.id()).subscribe(false));
815
816            let binding = ctx.txt.bind_filter_map_bidi(
817                &value,
818                clmv!(state, error, is_pending, cmd_handle, |txt| {
819                    if live.get() || matches!(state.load(Ordering::Relaxed), State::Requested) {
820                        // can try parse
821
822                        if !matches!(state.swap(State::Sync, Ordering::Relaxed), State::Sync) {
823                            // exit pending state, even if it parse fails
824                            let _ = is_pending.set(false);
825                            cmd_handle.set_enabled(false);
826                        }
827
828                        // try parse
829                        match T::from_txt(txt) {
830                            Ok(val) => {
831                                error.set(Txt::from_static(""));
832                                Some(val)
833                            }
834                            Err(e) => {
835                                error.set(e);
836                                None
837                            }
838                        }
839                    } else {
840                        // cannot try parse
841
842                        if !matches!(state.swap(State::Pending, Ordering::Relaxed), State::Pending) {
843                            // enter pending state
844                            let _ = is_pending.set(true);
845                            cmd_handle.set_enabled(true);
846                        }
847
848                        // does not update the value
849                        None
850                    }
851                }),
852                clmv!(state, error, |val| {
853                    // value updated externally, exit error, exit pending.
854
855                    error.set(Txt::from_static(""));
856
857                    if !matches!(state.swap(State::Sync, Ordering::Relaxed), State::Sync) {
858                        let _ = is_pending.set(false);
859                        cmd_handle.set_enabled(false);
860                    }
861
862                    Some(val.to_txt())
863                }),
864            );
865
866            // cmd_handle is held by the binding
867
868            WIDGET.sub_var(&TXT_PARSE_LIVE_VAR).sub_var(&error).push_var_handles(binding);
869        }
870        UiNodeOp::Deinit => {
871            _error_note = DataNoteHandle::dummy();
872        }
873        UiNodeOp::Event { update } => {
874            if let Some(args) = super::cmd::PARSE_CMD.scoped(WIDGET.id()).on_unhandled(update) {
875                if matches!(state.load(Ordering::Relaxed), State::Pending) {
876                    // requested parse and parse is pending
877
878                    state.store(State::Requested, Ordering::Relaxed);
879                    let _ = TEXT.resolved().txt.update();
880                    args.propagation().stop();
881                }
882            }
883        }
884        UiNodeOp::Update { .. } => {
885            if let Some(true) = TXT_PARSE_LIVE_VAR.get_new() {
886                if matches!(state.load(Ordering::Relaxed), State::Pending) {
887                    // enabled live parse and parse is pending
888
889                    let _ = TEXT.resolved().txt.update();
890                }
891            }
892
893            if let Some(error) = error.get_new() {
894                // remove or replace the error
895
896                _error_note = if error.is_empty() {
897                    DataNoteHandle::dummy()
898                } else {
899                    DATA.invalidate(error)
900                };
901            }
902        }
903        _ => {}
904    })
905}

907pub(super) fn on_change_stop(child: impl UiNode, mut handler: impl WidgetHandler<ChangeStopArgs>) -> impl UiNode {
908    let mut pending = None;
909    match_node(child, move |c, op| match op {
910        UiNodeOp::Event { update } => {
911            if pending.is_none() {
912                return;
913            }
914
915            if let Some(args) = KEY_INPUT_EVENT.on_unhandled(update) {
916                if let (KeyState::Pressed, Key::Enter) = (args.state, &args.key) {
917                    if !ACCEPTS_ENTER_VAR.get() {
918                        pending = None;
919                        handler.event(&ChangeStopArgs {
920                            cause: ChangeStopCause::Enter,
921                        });
922                    }
923                }
924            } else if let Some(args) = FOCUS_CHANGED_EVENT.on(update) {
925                let target = WIDGET.id();
926                if args.is_blur(target) {
927                    pending = None;
928                    handler.event(&ChangeStopArgs {
929                        cause: ChangeStopCause::Blur,
930                    });
931                }
932            }
933        }
934        UiNodeOp::Update { updates } => {
935            if TEXT.resolved().txt.is_new() {
936                let deadline = TIMERS.deadline(CHANGE_STOP_DELAY_VAR.get());
937                deadline.subscribe(UpdateOp::Update, WIDGET.id()).perm();
938                pending = Some(deadline);
939            } else if let Some(p) = &pending {
940                if p.get().has_elapsed() {
941                    pending = None;
942
943                    handler.event(&ChangeStopArgs {
944                        cause: ChangeStopCause::DelayElapsed,
945                    });
946                }
947            }
948
949            c.update(updates);
950            handler.update();
951        }
952        _ => {}
953    })
954}

956/// Implements the selection toolbar.
957pub fn selection_toolbar_node(child: impl UiNode) -> impl UiNode {
958    use super::node::*;
959
960    let mut selection_range = None;
961    let mut popup_state = None::<ReadOnlyArcVar<PopupState>>;
962    match_node(child, move |c, op| {
963        let mut open = false;
964        let mut open_long_press = false;
965        let mut close = false;
966        match op {
967            UiNodeOp::Init => {
968                WIDGET.sub_var(&SELECTION_TOOLBAR_FN_VAR);
969            }
970            UiNodeOp::Deinit => {
971                close = true;
972            }
973            UiNodeOp::Event { update } => {
974                c.event(update);
975
976                let open_id = || {
977                    if let Some(popup_state) = &popup_state {
978                        if let PopupState::Open(id) = popup_state.get() {
979                            return Some(id);
980                        }
981                    }
982                    None
983                };
984
985                if let Some(args) = MOUSE_INPUT_EVENT.on(update) {
986                    if open_id().map(|id| !args.target.contains(id)).unwrap_or(false) {
987                        close = true;
988                    }
989                    if args.state == ButtonState::Released {
990                        open = true;
991                    }
992                } else if TOUCH_LONG_PRESS_EVENT.has(update) {
993                    open = true;
994                    open_long_press = true;
995                } else if KEY_INPUT_EVENT.has(update) {
996                    close = true;
997                } else if let Some(args) = FOCUS_CHANGED_EVENT.on(update) {
998                    if args.is_blur(WIDGET.id())
999                        && open_id()
1000                            .map(|id| args.new_focus.as_ref().map(|p| !p.contains(id)).unwrap_or(true))
1001                            .unwrap_or(false)
1002                    {
1003                        close = true;
1004                    }
1005                } else if let Some(args) = TOUCH_INPUT_EVENT.on(update) {
1006                    if matches!(args.phase, TouchPhase::Start | TouchPhase::Move)
1007                        && open_id().map(|id| !args.target.contains(id)).unwrap_or(false)
1008                    {
1009                        close = true;
1010                    }
1011                }
1012
1013                if popup_state.is_some() {
1014                    let r_txt = TEXT.resolved();
1015                    if selection_range != r_txt.caret.selection_range() {
1016                        close = true;
1017                    }
1018                }
1019            }
1020            UiNodeOp::Update { .. } => {
1021                if SELECTION_TOOLBAR_FN_VAR.is_new() {
1022                    close = true;
1023                }
1024                if let Some(state) = popup_state.as_ref().and_then(|s| s.get_new()) {
1025                    let is_open = !matches!(state, PopupState::Closed);
1026                    let mut r_txt = TEXT.resolve();
1027                    if r_txt.selection_toolbar_is_open != is_open {
1028                        r_txt.selection_toolbar_is_open = is_open;
1029                        WIDGET.layout().render();
1030
1031                        if !is_open {
1032                            popup_state = None;
1033                        }
1034                    }
1035                }
1036            }
1037            _ => {}
1038        }
1039        if close {
1040            if let Some(state) = &popup_state.take() {
1041                selection_range = None;
1042                POPUP.close(state);
1043                TEXT.resolve().selection_toolbar_is_open = false;
1044                WIDGET.layout().render();
1045            }
1046        }
1047        if open {
1048            let r_txt = TEXT.resolved();
1049
1050            let range = r_txt.caret.selection_range();
1051            if open_long_press || range.is_some() {
1052                selection_range = range;
1053
1054                let toolbar_fn = SELECTION_TOOLBAR_FN_VAR.get();
1055                if let Some(node) = toolbar_fn.call_checked(SelectionToolbarArgs {
1056                    anchor_id: WIDGET.id(),
1057                    is_touch: matches!(r_txt.selection_by, SelectionBy::Touch),
1058                }) {
1059                    let (node, _) = node.init_widget();
1060
1061                    let mut translate = PxVector::zero();
1062                    let transform_key = FrameValueKey::new_unique();
1063                    let node = match_widget(node, move |c, op| match op {
1064                        UiNodeOp::Init => {
1065                            c.init();
1066                            c.with_context(WidgetUpdateMode::Bubble, || WIDGET.sub_var_layout(&SELECTION_TOOLBAR_ANCHOR_VAR));
1067                        }
1068                        UiNodeOp::Layout { wl, final_size } => {
1069                            let r_txt = TEXT.resolved();
1070
1071                            let range = if open_long_press {
1072                                Some(r_txt.caret.selection_range().unwrap_or_else(|| {
1073                                    let i = r_txt.caret.index.unwrap_or(CaretIndex::ZERO);
1074                                    i..i
1075                                }))
1076                            } else {
1077                                r_txt.caret.selection_range()
1078                            };
1079
1080                            if let Some(range) = range {
1081                                let l_txt = TEXT.laidout();
1082                                let r_txt = r_txt.segmented_text.text();
1083
1084                                let mut bounds = PxBox::new(PxPoint::splat(Px::MAX), PxPoint::splat(Px::MIN));
1085                                for line_rect in l_txt.shaped_text.highlight_rects(range, r_txt) {
1086                                    let line_box = line_rect.to_box2d();
1087                                    bounds.min = bounds.min.min(line_box.min);
1088                                    bounds.max = bounds.max.max(line_box.max);
1089                                }
1090                                let selection_bounds = bounds.to_rect();
1091
1092                                *final_size = c.layout(wl);
1093
1094                                let offset = SELECTION_TOOLBAR_ANCHOR_VAR.get();
1095
1096                                fn layout_offset(size: PxSize, point: Point) -> PxVector {
1097                                    LAYOUT
1098                                        .with_constraints(PxConstraints2d::new_exact_size(size), || point.layout())
1099                                        .to_vector()
1100                                }
1101                                let place = layout_offset(selection_bounds.size, offset.place);
1102                                let origin = layout_offset(*final_size, offset.origin);
1103
1104                                translate = selection_bounds.origin.to_vector() + place - origin;
1105                            } else {
1106                                // no selection, must be closing
1107                                wl.collapse();
1108                                *final_size = PxSize::zero();
1109                            };
1110                        }
1111                        UiNodeOp::Render { frame } => {
1112                            let l_txt = TEXT.laidout();
1113                            let transform = l_txt.render_info.transform.then_translate(translate.cast());
1114                            let transform = adjust_viewport_bound(transform, c);
1115
1116                            frame.push_reference_frame(transform_key.into(), FrameValue::Value(transform), true, false, |frame| {
1117                                c.render(frame)
1118                            });
1119                        }
1120                        UiNodeOp::RenderUpdate { update } => {
1121                            let l_txt = TEXT.laidout();
1122                            let transform = l_txt.render_info.transform.then_translate(translate.cast());
1123                            let transform = adjust_viewport_bound(transform, c);
1124
1125                            update.with_transform(transform_key.update(transform, true), false, |update| c.render_update(update));
1126                        }
1127                        _ => {}
1128                    });
1129
1130                    // capture all context including LaidoutText, exclude text style properties.
1131                    let capture = ContextCapture::CaptureBlend {
1132                        filter: CaptureFilter::Exclude({
1133                            let mut exclude = ContextValueSet::new();
1134                            super::Text::context_vars_set(&mut exclude);
1135
1136                            let mut allow = ContextValueSet::new();
1137                            super::LangMix::<()>::context_vars_set(&mut allow);
1138                            exclude.remove_all(&allow);
1139                            exclude.remove(&SELECTION_TOOLBAR_ANCHOR_VAR);
1140
1141                            exclude
1142                        }),
1143                        over: false,
1144                    };
1145
1146                    let mut anchor_mode = AnchorMode::tooltip();
1147                    anchor_mode.transform = AnchorTransform::None;
1148                    let state = POPUP.open_config(node, anchor_mode, capture);
1149                    state.subscribe(UpdateOp::Update, WIDGET.id()).perm();
1150                    popup_state = Some(state);
1151                    drop(r_txt);
1152                    TEXT.resolve().selection_toolbar_is_open = true;
1153                    WIDGET.layout().render();
1154                }
1155            };
1156        }
1157    })
1158}
1159fn adjust_viewport_bound(transform: PxTransform, widget: &mut impl UiNode) -> PxTransform {
1160    let window_bounds = WINDOW.vars().actual_size_px().get();
1161    let wgt_bounds = PxBox::from(
1162        widget
1163            .with_context(WidgetUpdateMode::Ignore, || WIDGET.bounds().outer_size())
1164            .unwrap_or_else(PxSize::zero),
1165    );
1166    let wgt_bounds = transform.outer_transformed(wgt_bounds).unwrap_or_default();
1167
1168    let x_underflow = -wgt_bounds.min.x.min(Px(0));
1169    let x_overflow = (wgt_bounds.max.x - window_bounds.width).max(Px(0));
1170    let y_underflow = -wgt_bounds.min.y.min(Px(0));
1171    let y_overflow = (wgt_bounds.max.y - window_bounds.height).max(Px(0));
1172
1173    let x = x_underflow - x_overflow;
1174    let y = y_underflow - y_overflow;
1175
1176    let correction = PxVector::new(x, y);
1177
1178    transform.then_translate(correction.cast())
1179}
