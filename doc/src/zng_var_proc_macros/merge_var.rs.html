31pub fn expand(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
32    let Input { vars_mod, mut inputs, .. } = parse_macro_input!(input as Input);
33
34    if inputs.len() < 3 {
35        abort_call_site!("expected at least two input vars and one merge closure")
36    }
37
38    let merge = inputs.pop().unwrap();
39
40    let type_idents: Vec<_> = (0..inputs.len()).map(|i| ident!("T{i}")).collect();
41    let input_idents: Vec<_> = (0..inputs.len()).map(|i| ident!("var{i}")).collect();
42    let idx: Vec<_> = (0..inputs.len())
43        .map(|i| syn::Index {
44            index: i as _,
45            span: proc_macro2::Span::call_site(),
46        })
47        .collect();
48
49    let mut merge_with = quote! {
50        output = Some(#vars_mod::merge_var_output(
51            merge(
52                #(#input_idents.downcast_ref().unwrap(),)*
53            )
54        ));
55    };
56    for (idx, input_ident) in idx.iter().zip(&input_idents).rev() {
57        merge_with = quote! {
58            #vars_mod::merge_var_with(&inputs[#idx], &mut |#input_ident| {
59                #merge_with
60            });
61        }
62    }
63
64    let out = quote! {
65        {
66            #[inline(always)]
67            fn merge_var__<
68                #(#type_idents: #vars_mod::VarValue,)*
69                O: #vars_mod::VarValue,
70                F: FnMut(
71                    #(&#type_idents,)*
72                ) -> O + Send + 'static
73            >(
74                #(#input_idents: impl #vars_mod::MergeInput<#type_idents>,)*
75                mut merge: F
76            ) -> #vars_mod::Var<O> {
77                #vars_mod::merge_var(Box::new([
78                    #(#vars_mod::merge_var_input(#input_idents),)*
79                ]), move |inputs| {
80                    let mut output = None;
81                    #merge_with
82                    output.unwrap()
83                })
84            }
85            merge_var__(#inputs #merge)
86        }
87    };
88
89    out.into()
90}

92struct Input {
93    vars_mod: Path,
94    _comma: Token![,],
95    inputs: Punctuated<Expr, Token![,]>,
96}
97impl Parse for Input {
98    fn parse(input: ParseStream) -> syn::Result<Self> {
99        Ok(Input {
100            vars_mod: input.parse()?,
101            _comma: input.parse()?,
102            inputs: Punctuated::parse_terminated(input)?,
103        })
104    }
