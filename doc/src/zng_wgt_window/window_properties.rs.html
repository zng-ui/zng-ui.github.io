20fn bind_window_var<T>(child: impl IntoUiNode, user_var: impl IntoVar<T>, select: impl Fn(&WindowVars) -> Var<T> + Send + 'static) -> UiNode
21where
22    T: VarValue + PartialEq,
23{
24    bind_window_var_impl(child.into_node(), user_var.into_var().into(), move |vars| select(vars).into())
25}
26fn bind_window_var_impl(child: UiNode, user_var: AnyVar, select: impl Fn(&WindowVars) -> AnyVar + Send + 'static) -> UiNode {
27    match_node(child, move |_, op| {
28        if let UiNodeOp::Init = op {
29            let window_var = select(&WINDOW.vars());
30            if !user_var.capabilities().is_const() {
31                let binding = user_var.bind_bidi(&window_var);
32                WIDGET.push_var_handles(binding);
33            }
34            window_var.set_from(&user_var);
35        }
36    })
37}

122/// Window clear color.
123///
124/// Color used to clear the previous frame pixels before rendering a new frame.
125/// It is visible if window content does not completely fill the content area, this
126/// can happen if you do not set a background or the background is semi-transparent, also
127/// can happen during very fast resizes.
128#[property(LAYOUT-1, default(colors::WHITE), widget_impl(Window))]
129pub fn clear_color(child: impl IntoUiNode, color: impl IntoVar<Rgba>) -> UiNode {
130    let clear_color = color.into_var();
131    match_node(child, move |_, op| match op {
132        UiNodeOp::Init => {
133            WIDGET.sub_var_render_update(&clear_color);
134        }
135        UiNodeOp::Render { frame } => {
136            frame.set_clear_color(clear_color.get());
137        }
138        UiNodeOp::RenderUpdate { update } => {
139            update.set_clear_color(clear_color.get());
140        }
141        _ => {}
142    })
143}

145/// Window or widget persistence config.
146///
147/// See the [`save_state_node`] for more details.
148///
149/// [`save_state`]: fn@save_state
150#[derive(Clone, Debug, PartialEq, serde::Serialize, serde::Deserialize)]
151pub enum SaveState {
152    /// Save and restore state.
153    Enabled {
154        /// Config key that identifies the window or widget.
155        ///
156        /// If `None` a key is generated from the widget ID and window ID name, see [`enabled_key`] for
157        /// details about how key generation.
158        ///
159        /// [`enabled_key`]: Self::enabled_key
160        key: Option<ConfigKey>,
161    },
162    /// Don't save nor restore state.
163    Disabled,
164}
165impl Default for SaveState {
166    /// Enabled, no key, delay 1s.
167    fn default() -> Self {
168        Self::enabled()
169    }
170}
171impl SaveState {
172    /// Default, enabled, no key.
173    pub const fn enabled() -> Self {
174        Self::Enabled { key: None }
175    }
176
177    /// Gets the config key if is enabled and can enable on the context.
178    ///
179    /// If is enabled without a key, the key is generated from the widget or window name:
180    ///
181    /// * If the widget ID has a name the key is `"wgt-{name}-state"`.
182    /// * If the context is the window root or just a window and the window ID has a name the key is `"win-{name}-state"`.
183    pub fn enabled_key(&self) -> Option<ConfigKey> {
184        match self {
185            Self::Enabled { key } => {
186                if key.is_some() {
187                    return key.clone();
188                }
189                let mut try_win = true;
190                if let Some(wgt) = WIDGET.try_id() {
191                    let name = wgt.name();
192                    if !name.is_empty() {
193                        return Some(formatx!("wgt-{name}"));
194                    }
195                    try_win = WIDGET.parent_id().is_none();
196                }
197                if try_win && let Some(win) = WINDOW.try_id() {
198                    let name = win.name();
199                    if !name.is_empty() {
200                        return Some(formatx!("win-{name}"));
201                    }
202                }
203                None
204            }
205            Self::Disabled => None,
206        }
207    }
208}
209impl_from_and_into_var! {
210    /// Convert `true` to default config and `false` to `None`.
211    fn from(persist: bool) -> SaveState {
212        if persist { SaveState::default() } else { SaveState::Disabled }
213    }
214}

216/// Helper node for implementing widgets save.
217///
218/// The `on_load_restore` closure is called on window load or on init if the window is already loaded. The argument
219/// is the saved state from a previous instance.
220///
221/// The `on_update_save` closure is called every update after the window loads, if it returns a value the config is updated.
222/// If the argument is `true` the closure must return a value, this value is used as the CONFIG fallback value that is required
223/// by some config backends even when the config is already present.
224pub fn save_state_node<S: ConfigValue>(
225    child: impl IntoUiNode,
226    enabled: impl IntoValue<SaveState>,
227    mut on_load_restore: impl FnMut(Option<S>) + Send + 'static,
228    mut on_update_save: impl FnMut(bool) -> Option<S> + Send + 'static,
229) -> UiNode {
230    let enabled = enabled.into();
231    enum State<S: ConfigValue> {
232        Disabled,
233        AwaitingLoad,
234        Loaded,
235        LoadedWithCfg(Var<S>),
236    }
237    let mut state = State::Disabled;
238    match_node(child, move |_, op| match op {
239        UiNodeOp::Init => {
240            if let Some(key) = enabled.enabled_key() {
241                if WINDOW.is_loaded() {
242                    if CONFIG.contains_key(key.clone()).get() {
243                        let cfg = CONFIG.get(key, on_update_save(true).unwrap());
244                        on_load_restore(Some(cfg.get()));
245                        state = State::LoadedWithCfg(cfg);
246                    } else {
247                        on_load_restore(None);
248                        state = State::Loaded;
249                    }
250                } else {
251                    WIDGET.sub_event(&WINDOW_LOAD_EVENT);
252                    state = State::AwaitingLoad;
253                }
254            } else {
255                state = State::Disabled;
256            }
257        }
258        UiNodeOp::Deinit => {
259            state = State::Disabled;
260        }
261        UiNodeOp::Event { update } => {
262            if matches!(&state, State::AwaitingLoad) && WINDOW_LOAD_EVENT.has(update) {
263                if let Some(key) = enabled.enabled_key() {
264                    if CONFIG.contains_key(key.clone()).get() {
265                        let cfg = CONFIG.get(key, on_update_save(true).unwrap());
266                        on_load_restore(Some(cfg.get()));
267                        state = State::LoadedWithCfg(cfg);
268                    } else {
269                        on_load_restore(None);
270                        state = State::Loaded;
271                    }
272                } else {
273                    // this can happen if the parent widget node is not properly implemented (changed context)
274                    state = State::Disabled;
275                }
276            }
277        }
278        UiNodeOp::Update { .. } => match &mut state {
279            State::LoadedWithCfg(cfg) => {
280                if let Some(new) = on_update_save(false) {
281                    cfg.set(new);
282                }
283            }
284            State::Loaded => {
285                if let Some(new) = on_update_save(false) {
286                    if let Some(key) = enabled.enabled_key() {
287                        let cfg = CONFIG.insert(key, new.clone());
288                        state = State::LoadedWithCfg(cfg);
289                    } else {
290                        state = State::Disabled;
291                    }
292                }
293            }
294            _ => {}
295        },
296        _ => {}
297    })
298}

300/// Save and restore the window state.
301///
302/// If enabled a config entry is created for the window state in [`CONFIG`], and if a config backend is set
303/// the window state is persisted on change and restored when the app reopens.
304///
305/// This property is enabled by default in the `Window!` widget, without a key. Note that without a config key
306/// the state only actually enables if the window root widget ID or the window ID have a name.
307///
308/// [`CONFIG`]: zng_ext_config::CONFIG
309#[property(CONTEXT, default(SaveState::Disabled), widget_impl(Window))]
310pub fn save_state(child: impl IntoUiNode, enabled: impl IntoValue<SaveState>) -> UiNode {
311    #[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
312    struct WindowStateCfg {
313        state: WindowState,
314        restore_rect: euclid::Rect<f32, Dip>,
315    }
316    save_state_node::<WindowStateCfg>(
317        child,
318        enabled,
319        |cfg| {
320            let vars = WINDOW.vars();
321            let state = vars.state();
322            WIDGET.sub_var(&state).sub_var(&vars.restore_rect());
323
324            if let Some(cfg) = cfg {
325                // restore state
326                state.set(cfg.state);
327
328                // restore normal position if it is valid (visible in a monitor)
329                let restore_rect: DipRect = cfg.restore_rect.cast();
330                let visible = MONITORS
331                    .available_monitors()
332                    .with(|w| w.iter().any(|m| m.dip_rect().intersects(&restore_rect)));
333                if visible {
334                    vars.position().set(restore_rect.origin);
335                }
336                vars.size().set(restore_rect.size);
337            }
338        },
339        |required| {
340            let vars = WINDOW.vars();
341            let state = vars.state();
342            let rect = vars.restore_rect();
343            if required || state.is_new() || rect.is_new() {
344                Some(WindowStateCfg {
345                    state: state.get(),
346                    restore_rect: rect.get().cast(),
347                })
348            } else {
349                None
350            }
351        },
352    )
353}

355/// Defines if a widget load affects the parent window load.
356///
357/// Widgets that support this behavior have a `block_window_load` property.
358#[derive(Clone, Copy, Debug, PartialEq, Eq)]
359pub enum BlockWindowLoad {
360    /// Widget requests a [`WindowLoadingHandle`] and retains it until the widget is loaded.
361    ///
362    /// [`WindowLoadingHandle`]: zng_ext_window::WindowLoadingHandle
363    Enabled {
364        /// Handle expiration deadline, if the widget takes longer than this deadline the window loads anyway.
365        deadline: Deadline,
366    },
367    /// Widget does not hold back window load.
368    Disabled,
369}
370impl BlockWindowLoad {
371    /// Enabled value.
372    pub fn enabled(deadline: impl Into<Deadline>) -> BlockWindowLoad {
373        BlockWindowLoad::Enabled { deadline: deadline.into() }
374    }
375
376    /// Returns `true` if it is enabled.
377    pub fn is_enabled(self) -> bool {
378        matches!(self, Self::Enabled { .. })
379    }
380
381    /// Returns `true` if it is disabled.
382    pub fn is_disabled(self) -> bool {
383        matches!(self, Self::Disabled)
384    }
385
386    /// Returns the block deadline if it is enabled and the deadline has not expired.
387    pub fn deadline(self) -> Option<Deadline> {
388        match self {
389            BlockWindowLoad::Enabled { deadline } => {
390                if deadline.has_elapsed() {
391                    None
392                } else {
393                    Some(deadline)
394                }
395            }
396            BlockWindowLoad::Disabled => None,
397        }
398    }
399}
400impl_from_and_into_var! {
401    /// Converts `true` to `BlockWindowLoad::enabled(1.secs())` and `false` to `BlockWindowLoad::Disabled`.
402    fn from(enabled: bool) -> BlockWindowLoad {
403        if enabled {
404            BlockWindowLoad::enabled(1.secs())
405        } else {
406            BlockWindowLoad::Disabled
407        }
408    }
409
410    /// Converts to enabled with the duration timeout.
411    fn from(enabled_timeout: Duration) -> BlockWindowLoad {
412        BlockWindowLoad::enabled(enabled_timeout)
413    }
414}

416/// Block window load until [`CONFIG.status`] is idle.
417///
418/// This property is enabled by default in the `Window!` widget.
419///
420/// [`CONFIG.status`]: CONFIG::status
421#[property(CONTEXT, default(false), widget_impl(Window))]
422pub fn config_block_window_load(child: impl IntoUiNode, enabled: impl IntoValue<BlockWindowLoad>) -> UiNode {
423    let enabled = enabled.into();
424
425    enum State {
426        Allow,
427        Block {
428            _handle: WindowLoadingHandle,
429            cfg: Var<ConfigStatus>,
430        },
431    }
432    let mut state = State::Allow;
433
434    match_node(child, move |_, op| match op {
435        UiNodeOp::Init => {
436            if let Some(delay) = enabled.deadline() {
437                let cfg = CONFIG.status();
438                if !cfg.get().is_idle()
439                    && let Some(_handle) = WINDOW.loading_handle(delay)
440                {
441                    WIDGET.sub_var(&cfg);
442                    state = State::Block { _handle, cfg };
443                }
444            }
445        }
446        UiNodeOp::Deinit => {
447            state = State::Allow;
448        }
449        UiNodeOp::Update { .. } => {
450            if let State::Block { cfg, .. } = &state
451                && cfg.get().is_idle()
452            {
453                state = State::Allow;
454            }
455        }
456        _ => {}
457    })
458}

460/// Gets if is not headless, [`chrome`] is `true`, [`state`] is not fullscreen but [`WINDOWS.system_chrome`]
461/// reports the system does not provide window decorations.
462///
463/// [`chrome`]: fn@chrome
464/// [`state`]: fn@state
465/// [`WINDOWS.system_chrome`]: WINDOWS::system_chrome
466#[property(EVENT, default(var_state()), widget_impl(Window))]
467pub fn needs_fallback_chrome(child: impl IntoUiNode, needs: impl IntoVar<bool>) -> UiNode {
468    zng_wgt::node::bind_state_init(
469        child,
470        || {
471            if WINDOW.mode().is_headless() {
472                const_var(false)
473            } else {
474                let vars = WINDOW.vars();
475                expr_var! {
476                    *#{vars.chrome()} && #{WINDOWS.system_chrome()}.needs_custom() && !#{vars.state()}.is_fullscreen()
477                }
478            }
479        },
480        needs,
481    )
482}
