36impl Wrap {
37    fn widget_intrinsic(&mut self) {
38        self.widget_builder().push_build_action(|wgt| {
39            let child = node(
40                wgt.capture_ui_node_list_or_empty(property_id!(Self::children)),
41                wgt.capture_var_or_else(property_id!(Self::spacing), || {
42                    LINE_SPACING_VAR.map(|s| GridSpacing {
43                        column: Length::zero(),
44                        row: s.clone(),
45                    })
46                }),
47                wgt.capture_var_or_else(property_id!(Self::children_align), || TEXT_ALIGN_VAR),
48            );
49            wgt.set_child(child);
50        });
51    }

89/// Wrap node.
90///
91/// Can be used directly to inline widgets without declaring a wrap widget info. This node is the child
92/// of the `Wrap!` widget.
93pub fn node(children: impl UiNodeList, spacing: impl IntoVar<GridSpacing>, children_align: impl IntoVar<Align>) -> impl UiNode {
94    let children = PanelList::new(children).track_info_range(*PANEL_LIST_ID);
95    let spacing = spacing.into_var();
96    let children_align = children_align.into_var();
97    let mut layout = InlineLayout::default();
98
99    match_node_list(children, move |children, op| match op {
100        UiNodeOp::Init => {
101            WIDGET.sub_var_layout(&spacing).sub_var_layout(&children_align);
102        }
103        UiNodeOp::Update { updates } => {
104            let mut any = false;
105            children.update_all(updates, &mut any);
106
107            if any {
108                WIDGET.layout();
109            }
110        }
111        UiNodeOp::Measure { wm, desired_size } => {
112            let spacing = spacing.layout();
113            children.delegated();
114            *desired_size = layout.measure(wm, children.children(), children_align.get(), spacing);
115        }
116        UiNodeOp::Layout { wl, final_size } => {
117            let spacing = spacing.layout();
118            children.delegated();
119            // rust-analyzer does not find `layout` here if called with dot.
120            *final_size = InlineLayout::layout(&mut layout, wl, children.children(), children_align.get(), spacing);
121        }
122        _ => {}
123    })
124}

126/// Create a node that estimates the size of wrap panel children.
127///
128/// The estimation assumes that all items have a size of `child_size`.
129pub fn lazy_size(children_len: impl IntoVar<usize>, spacing: impl IntoVar<GridSpacing>, child_size: impl IntoVar<Size>) -> impl UiNode {
130    // we don't use `properties::size(NilUiNode, child_size)` because that size disables inlining.
131    let size = child_size.into_var();
132    let sample = match_node_leaf(move |op| match op {
133        UiNodeOp::Init => {
134            WIDGET.sub_var_layout(&size);
135        }
136        UiNodeOp::Measure { desired_size, .. } => {
137            *desired_size = size.layout();
138        }
139        UiNodeOp::Layout { final_size, .. } => {
140            *final_size = size.layout();
141        }
142        _ => {}
143    });
144
145    lazy_sample(children_len, spacing, sample)
146}

148/// Create a node that estimates the size of wrap panel children.
149///
150/// The estimation assumes that all items have the size of `child_sample`.
151pub fn lazy_sample(children_len: impl IntoVar<usize>, spacing: impl IntoVar<GridSpacing>, child_sample: impl UiNode) -> impl UiNode {
152    let children_len = children_len.into_var();
153    let spacing = spacing.into_var();
154
155    match_node(child_sample, move |sample, op| match op {
156        UiNodeOp::Init => {
157            WIDGET.sub_var_layout(&children_len).sub_var_layout(&spacing);
158        }
159        UiNodeOp::Measure { wm, desired_size } => {
160            let child_size = sample.measure(wm);
161            *desired_size = InlineLayout::estimate_measure(wm, children_len.get(), child_size, spacing.layout());
162        }
163        UiNodeOp::Layout { wl, final_size } => {
164            let child_size = sample.layout(wl);
165            *final_size = InlineLayout::estimate_layout(wl, children_len.get(), child_size, spacing.layout());
166        }
167        _ => {}
168    })
169}

171/// Info about segments of a widget in a row.
172#[derive(Debug, Clone)]
173enum ItemSegsInfo {
174    Block(Px),
175    Built {
176        measure: Arc<Vec<InlineSegment>>,
177        layout: Arc<Vec<InlineSegmentPos>>,
178        x: f32,
179        width: f32,
180    },
181}
182impl ItemSegsInfo {
183    pub fn new_collapsed() -> Self {
184        Self::Block(Px(0))
185    }
186
187    pub fn new_block(width: Px) -> Self {
188        Self::Block(width)
189    }
190
191    pub fn new_inlined(measure: Arc<Vec<InlineSegment>>) -> Self {
192        Self::Built {
193            measure,
194            layout: Arc::new(vec![]),
195            x: 0.0,
196            width: 0.0,
197        }
198    }
199
200    pub fn measure(&self) -> &[InlineSegment] {
201        match self {
202            ItemSegsInfo::Built { measure, .. } => measure,
203            _ => &[],
204        }
205    }
206
207    pub fn layout_mut(&mut self) -> &mut Vec<InlineSegmentPos> {
208        self.build();
209        match self {
210            ItemSegsInfo::Built { measure, layout, .. } => {
211                // Borrow checker limitation does not allow `if let Some(l) = Arc::get_mut(..) { l } else { <insert-return> }`
212
213                if Arc::get_mut(layout).is_none() {
214                    *layout = Arc::new(vec![]);
215                }
216
217                let r = Arc::get_mut(layout).unwrap();
218                r.resize(measure.len(), InlineSegmentPos::new(0.0));
219
220                r
221            }
222            _ => unreachable!(),
223        }
224    }
225
226    pub fn iter_mut(&mut self) -> impl Iterator<Item = (&InlineSegment, &mut InlineSegmentPos)> {
227        self.build();
228        match self {
229            ItemSegsInfo::Built { measure, layout, .. } => {
230                if Arc::get_mut(layout).is_none() {
231                    *layout = Arc::new(vec![]);
232                }
233
234                let r = Arc::get_mut(layout).unwrap();
235                r.resize(measure.len(), InlineSegmentPos::new(0.0));
236
237                measure.iter().zip(r)
238            }
239            _ => unreachable!(),
240        }
241    }
242
243    /// Only valid if has bidi items and is up-to-date.
244    pub fn x_width_segs(&self) -> (Px, Px, Arc<Vec<InlineSegmentPos>>) {
245        match self {
246            ItemSegsInfo::Built { layout, x, width, .. } => (Px(x.floor() as i32), Px(width.ceil() as i32), layout.clone()),
247            _ => unreachable!(),
248        }
249    }
250
251    #[cfg(debug_assertions)]
252    pub fn measure_width(&self) -> f32 {
253        match self {
254            ItemSegsInfo::Block(w) => w.0 as f32,
255            ItemSegsInfo::Built { measure, .. } => measure.iter().map(|s| s.width).sum(),
256        }
257    }
258
259    fn build(&mut self) {
260        match self {
261            ItemSegsInfo::Block(width) => {
262                let width = width.0 as f32;
263                *self = ItemSegsInfo::Built {
264                    measure: Arc::new(vec![InlineSegment::new(width, TextSegmentKind::OtherNeutral)]),
265                    layout: Arc::new(Vec::with_capacity(1)),
266                    x: 0.0,
267                    width,
268                }
269            }
270            ItemSegsInfo::Built { .. } => {}
271        }
272    }
273
274    fn set_x_width(&mut self, new_x: f32, new_width: f32) {
275        match self {
276            ItemSegsInfo::Built { x, width, .. } => {
277                *x = new_x;
278                *width = new_width;
279            }
280            _ => unreachable!(),
281        }
282    }
283}

301struct InlineLayout {
302    first_wrapped: bool,
303    rows: RecycleVec<RowInfo>,
304    desired_size: PxSize,

306    // has segments in the opposite direction, requires bidi sorting and positioning.
307    has_bidi_inline: bool,

314impl InlineLayout {
315    pub fn estimate_measure(wm: &mut WidgetMeasure, children_len: usize, child_size: PxSize, spacing: PxGridSpacing) -> PxSize {
316        if children_len == 0 {
317            return PxSize::zero();
318        }
319
320        let metrics = LAYOUT.metrics();
321        let constraints = metrics.constraints();
322
323        if metrics.inline_constraints().is_none()
324            && let Some(known) = constraints.fill_or_exact()
325        {
326            return known;
327        }
328
329        let max_x = constraints.x.max().unwrap_or(Px::MAX).max(child_size.width);
330
331        if let Some(inline) = wm.inline() {
332            let inline_constraints = metrics.inline_constraints().unwrap().measure();
333
334            inline.first_wrapped = inline_constraints.first_max < child_size.width;
335
336            let mut first_max_x = max_x;
337            if !inline.first_wrapped {
338                first_max_x = inline_constraints.first_max;
339            }
340
341            inline.first.height = child_size.height.max(inline_constraints.mid_clear_min);
342
343            let column_len = (first_max_x - child_size.width) / (child_size.width + spacing.column) + Px(1);
344            inline.first.width = (column_len - Px(1)) * (child_size.width + spacing.column) + child_size.width;
345
346            let children_len = Px(children_len as _) - column_len;
347            inline.last_wrapped = children_len.0 > 0;
348
349            let mut size = inline.first;
350
351            if inline.last_wrapped {
352                let column_len = (max_x - child_size.width) / (child_size.width + spacing.column) + Px(1);
353
354                size.width = size
355                    .width
356                    .max((column_len - Px(1)) * (child_size.width + spacing.column) + child_size.width);
357
358                let mid_len = children_len / column_len;
359                if mid_len.0 > 0 {
360                    size.height += (spacing.row + child_size.height) * mid_len;
361                }
362
363                let last_len = children_len % column_len;
364                inline.last.height = child_size.height;
365                if last_len.0 > 0 {
366                    inline.last.width = (last_len - Px(1)) * (child_size.width + spacing.column) + child_size.width;
367
368                    size.height += spacing.row + child_size.height;
369                } else {
370                    inline.last.width = max_x;
371                }
372            } else {
373                inline.last = inline.first;
374            }
375
376            debug_assert_eq!(inline.first.is_empty(), inline.last.is_empty());
377
378            size
379        } else {
380            let column_len = (max_x - child_size.width) / (child_size.width + spacing.column) + Px(1);
381            let row_len = (Px(children_len as i32) / column_len).max(Px(1));
382
383            // spacing in between means space available to divide for pairs (width + column) has 1 less item.
384            let desired_size = PxSize::new(
385                (column_len - Px(1)) * (child_size.width + spacing.column) + child_size.width,
386                (row_len - Px(1)) * (child_size.height + spacing.row) + child_size.height,
387            );
388            constraints.clamp_size(desired_size)
389        }
390    }

392    pub fn measure(&mut self, wm: &mut WidgetMeasure, children: &mut PanelList, child_align: Align, spacing: PxGridSpacing) -> PxSize {
393        let metrics = LAYOUT.metrics();
394        let constraints = metrics.constraints();
395
396        if metrics.inline_constraints().is_none()
397            && let Some(known) = constraints.fill_or_exact()
398        {
399            return known;
400        }
401
402        self.measure_rows(wm, &metrics, children, child_align, spacing);
403
404        if let Some(inline) = wm.inline() {
405            inline.first_wrapped = self.first_wrapped;
406            inline.last_wrapped = self.rows.len() > 1;
407
408            if let Some(first) = self.rows.first() {
409                inline.first = first.size;
410                inline.with_first_segs(|i| {
411                    i.extend(first.item_segs.iter().flat_map(|i| i.measure().iter().copied()));
412                });
413            } else {
414                inline.first = PxSize::zero();
415                inline.with_first_segs(|i| i.clear());
416            }
417            if let Some(last) = self.rows.last() {
418                inline.last = last.size;
419                inline.with_last_segs(|i| {
420                    i.extend(last.item_segs.iter().flat_map(|i| i.measure().iter().copied()));
421                })
422            } else {
423                inline.last = PxSize::zero();
424                inline.with_last_segs(|i| i.clear());
425            }
426        }
427
428        constraints.clamp_size(self.desired_size)
429    }

431    pub fn estimate_layout(wl: &mut WidgetLayout, children_len: usize, child_size: PxSize, spacing: PxGridSpacing) -> PxSize {
432        let is_inline = wl.inline().is_some();
433        let mut wm = wl.to_measure(if is_inline { Some(Default::default()) } else { None });
434        let size = if let Some(inline) = wl.inline() {
435            let mut size = Self::estimate_measure(&mut wm, children_len, child_size, spacing);
436            if let Some(m_inline) = wm.inline() {
437                inline.invalidate_negative_space();
438                inline.inner_size = size;
439
440                let inline_constraints = LAYOUT.inline_constraints().unwrap().layout();
441
442                let mut mid_height = size.height;
443
444                if !m_inline.first_wrapped {
445                    inline.rows.push(inline_constraints.first);
446                    mid_height -= child_size.height + spacing.row;
447                }
448                if m_inline.last_wrapped {
449                    mid_height -= spacing.row + child_size.height;
450                    inline.rows.push(PxRect::new(
451                        PxPoint::new(Px(0), spacing.row + child_size.height),
452                        PxSize::new(size.width, mid_height),
453                    ));
454                    inline.rows.push(inline_constraints.last);
455                }
456
457                size.height = inline_constraints.last.origin.y + inline_constraints.last.size.height;
458            }
459            size
460        } else {
461            Self::estimate_measure(&mut wm, children_len, child_size, spacing)
462        };
463
464        let width = LAYOUT.constraints().x.fill_or(size.width);
465        PxSize::new(width, size.height)
466    }

468    pub fn layout(&mut self, wl: &mut WidgetLayout, children: &mut PanelList, child_align: Align, spacing: PxGridSpacing) -> PxSize {
469        let metrics = LAYOUT.metrics();
470        let inline_constraints = metrics.inline_constraints();
471        let direction = metrics.direction();
472
473        if inline_constraints.is_none() {
474            // if not already measured by parent inline
475            self.measure_rows(&mut wl.to_measure(None), &metrics, children, child_align, spacing);
476        }
477        if self.has_bidi_inline && !self.bidi_layout_fresh {
478            self.layout_bidi(inline_constraints.clone(), direction, spacing.column);
479        }
480
481        let constraints = metrics.constraints();
482        let child_align_x = child_align.x(direction);
483        let child_align_y = child_align.y();
484
485        let panel_width = constraints.x.fill_or(self.desired_size.width);
486
487        let (first, mid, last) = if let Some(s) = inline_constraints.map(|c| c.layout()) {
488            (s.first, s.mid_clear, s.last)
489        } else {
490            // define our own first and last
491            let mut first = PxRect::from_size(self.rows[0].size);
492            let mut last = PxRect::from_size(self.rows.last().unwrap().size);
493
494            #[cfg(debug_assertions)]
495            if self.has_bidi_inline {
496                let segs_max = self.rows[0]
497                    .item_segs
498                    .iter()
499                    .map(|s| {
500                        let (x, width, _) = s.x_width_segs();
501                        x + width
502                    })
503                    .max()
504                    .unwrap_or_default();
505
506                if (first.width() - segs_max).abs() > Px(10) {
507                    tracing::error!("align error, used width: {:?}, but segs max is: {:?}", first.width(), segs_max);
508                }
509            }
510
511            first.origin.x = (panel_width - first.size.width) * child_align_x;
512            last.origin.x = (panel_width - last.size.width) * child_align_x;
513            last.origin.y = self.desired_size.height - last.size.height;
514
515            if let Some(y) = constraints.y.fill_or_exact() {
516                let align_y = (y - self.desired_size.height) * child_align_y;
517                first.origin.y += align_y;
518                last.origin.y += align_y;
519            }
520
521            (first, Px(0), last)
522        };
523        let panel_height = constraints.y.fill_or(last.origin.y - first.origin.y + last.size.height);
524
525        let child_constraints = PxConstraints2d::new_unbounded().with_fill_x(true).with_max_x(panel_width);
526
527        if let Some(inline) = wl.inline() {
528            inline.rows.clear();
529        }
530
531        let fill_width = if child_align.is_fill_x() {
532            Some(panel_width.0 as f32)
533        } else {
534            None
535        };
536
537        LAYOUT.with_constraints(child_constraints, || {
538            let mut row = first;
539            let mut row_segs = &self.rows[0].item_segs;
540            let mut row_advance = Px(0);
541            let mut next_row_i = 1;
542            let mut row_segs_i_start = 0;
543
544            let mut fill_scale = None;
545            if let Some(mut f) = fill_width {
546                if wl.is_inline() {
547                    f = first.width().0 as f32;
548                }
549                fill_scale = Some(f / self.rows[0].size.width.0 as f32);
550            }
551
552            children.for_each(|i, child, o| {
553                if next_row_i < self.rows.len() && self.rows[next_row_i].first_child == i {
554                    // new row
555                    if let Some(inline) = wl.inline() {
556                        inline.rows.push(row);
557                    }
558                    if next_row_i == self.rows.len() - 1 {
559                        row = last;
560                    } else {
561                        row.origin.y += row.size.height + spacing.row;
562                        if next_row_i == 1 {
563                            // clear first row
564                            row.origin.y += mid;
565                        }
566
567                        row.size = self.rows[next_row_i].size;
568                        row.origin.x = (panel_width - row.size.width) * child_align_x;
569                    }
570                    row_segs = &self.rows[next_row_i].item_segs;
571                    row_segs_i_start = self.rows[next_row_i].first_child;
572                    next_row_i += 1;
573                    row_advance = Px(0);
574
575                    fill_scale = None;
576                    if let Some(mut f) = fill_width {
577                        if wl.is_inline() || next_row_i < self.rows.len() {
578                            if wl.is_inline() && next_row_i == self.rows.len() {
579                                f = last.width().0 as f32;
580                            }
581                            // fill row, if it is not the last in a block layout
582                            fill_scale = Some(f / self.rows[next_row_i - 1].size.width.0 as f32);
583                        }
584                    }
585                }
586
587                let (bidi_x, bidi_width, bidi_segs) = if self.has_bidi_inline {
588                    row_segs[i - row_segs_i_start].x_width_segs()
589                } else {
590                    (Px(0), Px(0), self.bidi_default_segs.clone())
591                };
592
593                let child_inline = child
594                    .with_context(WidgetUpdateMode::Ignore, || WIDGET.bounds().measure_inline())
595                    .flatten();
596                if let Some(child_inline) = child_inline {
597                    let child_desired_size = child
598                        .with_context(WidgetUpdateMode::Ignore, || WIDGET.bounds().measure_outer_size())
599                        .unwrap_or_default();
600                    if child_desired_size.is_empty() {
601                        // collapsed, continue.
602                        wl.collapse_child(i);
603                        return;
604                    }
605
606                    let mut child_first = PxRect::from_size(child_inline.first);
607                    let mut child_mid = Px(0);
608                    let mut child_last = PxRect::from_size(child_inline.last);
609
610                    if child_inline.last_wrapped {
611                        // child wraps
612                        debug_assert_eq!(self.rows[next_row_i].first_child, i + 1);
613
614                        child_first.origin.x = row.origin.x + row_advance;
615                        if let LayoutDirection::RTL = direction {
616                            child_first.origin.x -= row_advance;
617                        }
618                        child_first.origin.y += (row.size.height - child_first.size.height) * child_align_y;
619                        child_mid = (row.size.height - child_first.size.height).max(Px(0));
620                        child_last.origin.y = child_desired_size.height - child_last.size.height;
621
622                        if self.has_bidi_inline {
623                            child_first.origin.x = row.origin.x + bidi_x;
624                            child_first.size.width = bidi_width;
625                        }
626
627                        let next_row = if next_row_i == self.rows.len() - 1 {
628                            last
629                        } else {
630                            let mut r = row;
631                            r.origin.y += child_last.origin.y;
632                            r.size = self.rows[next_row_i].size;
633                            r.origin.x = (panel_width - r.size.width) * child_align_x;
634                            r
635                        };
636                        child_last.origin.x = next_row.origin.x;
637                        if let LayoutDirection::RTL = direction {
638                            child_last.origin.x += next_row.size.width - child_last.size.width;
639                        }
640                        child_last.origin.y += (next_row.size.height - child_last.size.height) * child_align_y;
641
642                        let (last_bidi_x, last_bidi_width, last_bidi_segs) = if self.has_bidi_inline {
643                            self.rows[next_row_i].item_segs[0].x_width_segs()
644                        } else {
645                            (Px(0), Px(0), self.bidi_default_segs.clone())
646                        };
647
648                        if self.has_bidi_inline {
649                            child_last.origin.x = next_row.origin.x + last_bidi_x;
650                            child_last.size.width = last_bidi_width;
651                        }
652
653                        if let Some(s) = fill_scale {
654                            child_first.size.width *= s;
655
656                            // child wraps, so last is different row
657                            if wl.is_inline() || next_row_i < self.rows.len() - 1 {
658                                let mut f = fill_width.unwrap();
659                                if wl.is_inline() && next_row_i == self.rows.len() - 1 {
660                                    f = last.width().0 as f32;
661                                }
662                                fill_scale = Some(f / next_row.size.width.0 as f32);
663                                let s = fill_scale.unwrap();
664                                child_last.size.width *= s;
665                            } else {
666                                // only fill last row if Wrap! is nested/inlined
667                                fill_scale = None;
668                            }
669                        }
670
671                        let (_, define_ref_frame) =
672                            wl.with_child(|wl| wl.layout_inline(child_first, child_mid, child_last, bidi_segs, last_bidi_segs, child));
673                        o.child_offset = PxVector::new(Px(0), row.origin.y);
674                        o.define_reference_frame = define_ref_frame;
675
676                        // new row
677                        if let Some(inline) = wl.inline() {
678                            inline.rows.push(row);
679                            child.with_context(WidgetUpdateMode::Ignore, || {
680                                if let Some(inner) = WIDGET.bounds().inline() {
681                                    if inner.rows.len() >= 3 {
682                                        inline.rows.extend(inner.rows[1..inner.rows.len() - 1].iter().map(|r| {
683                                            let mut r = *r;
684                                            r.origin.y += row.origin.y;
685                                            r
686                                        }));
687                                    }
688                                } else {
689                                    tracing::error!("child inlined in measure, but not in layout")
690                                }
691                            });
692                        }
693                        row = next_row;
694                        row_advance = child_last.size.width + spacing.column;
695                        row_segs = &self.rows[next_row_i].item_segs;
696                        row_segs_i_start = self.rows[next_row_i].first_child - 1; // next row first item is also this widget
697                        debug_assert_eq!(row_segs_i_start, i);
698                        next_row_i += 1;
699                    } else {
700                        // child inlined, but fits in the row
701
702                        let mut offset = PxVector::new(row_advance, Px(0));
703                        if let LayoutDirection::RTL = direction {
704                            offset.x = row.size.width - child_last.size.width - offset.x;
705                        }
706                        offset.y = (row.size.height - child_inline.first.height) * child_align_y;
707
708                        let mut max_size = child_inline.first;
709
710                        if self.has_bidi_inline {
711                            max_size.width = bidi_width;
712                            child_first.size.width = bidi_width;
713                            child_last.size.width = bidi_width;
714                        }
715
716                        if let Some(s) = fill_scale {
717                            child_first.size.width *= s;
718                            child_last.size.width *= s;
719                            max_size.width *= s;
720                        }
721
722                        let (_, define_ref_frame) = wl.with_child(|wl| {
723                            LAYOUT.with_constraints(child_constraints.with_fill(false, false).with_max_size(max_size), || {
724                                wl.layout_inline(child_first, child_mid, child_last, bidi_segs.clone(), bidi_segs, child)
725                            })
726                        });
727                        o.child_offset = row.origin.to_vector() + offset;
728                        if self.has_bidi_inline {
729                            o.child_offset.x = row.origin.x + bidi_x;
730                        }
731                        o.define_reference_frame = define_ref_frame;
732
733                        row_advance += child_last.size.width + spacing.column;
734                    }
735                } else {
736                    // inline block
737                    let max_width = if self.has_bidi_inline {
738                        bidi_width
739                    } else {
740                        row.size.width - row_advance
741                    };
742                    let (size, define_ref_frame) = LAYOUT.with_constraints(
743                        child_constraints.with_fill(false, false).with_max(max_width, row.size.height),
744                        || wl.with_child(|wl| wl.layout_block(child)),
745                    );
746                    if size.is_empty() {
747                        // collapsed, continue.
748                        o.child_offset = PxVector::zero();
749                        o.define_reference_frame = false;
750                        return;
751                    }
752
753                    let mut offset = PxVector::new(row_advance, Px(0));
754                    if let LayoutDirection::RTL = direction {
755                        offset.x = row.size.width - size.width - offset.x;
756                    }
757                    offset.y = (row.size.height - size.height) * child_align_y;
758                    o.child_offset = row.origin.to_vector() + offset;
759                    if self.has_bidi_inline {
760                        o.child_offset.x = row.origin.x + bidi_x;
761                    }
762                    o.define_reference_frame = define_ref_frame;
763                    row_advance += size.width + spacing.column;
764                }
765            });
766
767            if let Some(inline) = wl.inline() {
768                // last row
769                inline.rows.push(row);
770            }
771        });
772
773        children.commit_data().request_render();
774
775        constraints.clamp_size(PxSize::new(panel_width, panel_height))
776    }

778    fn measure_rows(
779        &mut self,
780        wm: &mut WidgetMeasure,
781        metrics: &LayoutMetrics,
782        children: &mut PanelList,
783        child_align: Align,
784        spacing: PxGridSpacing,
785    ) {
786        self.rows.begin_reuse();
787        self.bidi_layout_fresh = false;
788
789        self.first_wrapped = false;
790        self.desired_size = PxSize::zero();
791        self.has_bidi_inline = false;
792
793        let direction = metrics.direction();
794        let constraints = metrics.constraints();
795        let inline_constraints = metrics.inline_constraints();
796        let child_inline_constrain = constraints.x.max_or(Px::MAX);
797        let child_constraints = PxConstraints2d::new_unbounded()
798            .with_fill_x(child_align.is_fill_x())
799            .with_max_x(child_inline_constrain);
800        let mut row = self.rows.new_item();
801        LAYOUT.with_constraints(child_constraints, || {
802            children.for_each(|i, child, _| {
803                let mut inline_constrain = child_inline_constrain;
804                let mut wrap_clear_min = Px(0);
805                if self.rows.is_empty() && !self.first_wrapped {
806                    if let Some(InlineConstraints::Measure(InlineConstraintsMeasure {
807                        first_max, mid_clear_min, ..
808                    })) = inline_constraints
809                    {
810                        inline_constrain = first_max;
811                        wrap_clear_min = mid_clear_min;
812                    }
813                }
814                if inline_constrain < Px::MAX {
815                    inline_constrain -= row.size.width;
816                }
817
818                let (inline, size) = wm.measure_inline(inline_constrain, row.size.height - spacing.row, child);
819
820                let can_collapse = size.is_empty()
821                    && match &inline {
822                        Some(i) => i.first_segs.is_empty(),
823                        None => true,
824                    };
825                if can_collapse {
826                    row.item_segs.push(ItemSegsInfo::new_collapsed());
827                    // collapsed, continue.
828                    return;
829                }
830
831                if let Some(inline) = inline {
832                    if !self.has_bidi_inline {
833                        self.has_bidi_inline =
834                            inline
835                                .first_segs
836                                .iter()
837                                .chain(inline.last_segs.iter())
838                                .any(|s| match s.kind.strong_direction() {
839                                    Some(d) => d != direction,
840                                    None => false,
841                                });
842                    }
843
844                    // item mid-rows can be wider
845                    self.desired_size.width = self.desired_size.width.max(size.width);
846
847                    if inline.first_wrapped {
848                        // wrap by us, detected by child
849                        if row.size.is_empty() {
850                            debug_assert!(self.rows.is_empty());
851                            self.first_wrapped = true;
852                        } else {
853                            row.size.width -= spacing.column;
854                            row.size.width = row.size.width.max(Px(0));
855                            self.desired_size.width = self.desired_size.width.max(row.size.width);
856                            self.desired_size.height += row.size.height + spacing.row;
857
858                            self.rows.push_renew(&mut row);
859                        }
860
861                        row.size = inline.first;
862                        row.first_child = i;
863                    } else {
864                        row.size.width += inline.first.width;
865                        row.size.height = row.size.height.max(inline.first.height);
866                    }
867                    row.item_segs.push(ItemSegsInfo::new_inlined(inline.first_segs.clone()));
868
869                    if inline.last_wrapped {
870                        // wrap by child
871                        self.desired_size.width = self.desired_size.width.max(row.size.width);
872                        self.desired_size.height += size.height - inline.first.height;
873
874                        self.rows.push_renew(&mut row);
875                        row.size = inline.last;
876                        row.size.width += spacing.column;
877                        row.first_child = i + 1;
878                        row.item_segs.push(ItemSegsInfo::new_inlined(inline.last_segs));
879                    } else {
880                        // child inlined, but fit in row
881                        row.size.width += spacing.column;
882                    }
883                } else if size.width <= inline_constrain {
884                    row.size.width += size.width + spacing.column;
885                    row.size.height = row.size.height.max(size.height);
886                    row.item_segs.push(ItemSegsInfo::new_block(size.width));
887                } else {
888                    // wrap by us
889                    if row.size.is_empty() {
890                        debug_assert!(self.rows.is_empty());
891                        self.first_wrapped = true;
892                    } else {
893                        row.size.width -= spacing.column;
894                        row.size.width = row.size.width.max(Px(0));
895                        self.desired_size.width = self.desired_size.width.max(row.size.width);
896                        self.desired_size.height += row.size.height.max(wrap_clear_min) + spacing.row;
897                        self.rows.push_renew(&mut row);
898                    }
899
900                    row.size = size;
901                    row.size.width += spacing.column;
902                    row.first_child = i;
903                    row.item_segs.push(ItemSegsInfo::new_block(size.width));
904                }
905            });
906        });
907
908        // last row
909        row.size.width -= spacing.column;
910        row.size.width = row.size.width.max(Px(0));
911        self.desired_size.width = self.desired_size.width.max(row.size.width);
912        self.desired_size.height += row.size.height; // no spacing because it's single line or already added in [^wrap by us]
913        self.rows.push(row);
914
915        self.rows.commit_reuse();
916
917        #[cfg(debug_assertions)]
918        for (i, row) in self.rows.iter().enumerate() {
919            let width = row.size.width;
920            let sum_width = row.item_segs.iter().map(|s| Px(s.measure_width() as i32)).sum::<Px>()
921                + spacing.column * Px(row.item_segs.len().saturating_sub(1) as _);
922
923            if (sum_width - width) > Px(1) {
924                if metrics.inline_constraints().is_some() && (i == 0 || i == self.rows.len() - 1) {
925                    tracing::error!(
926                        "Wrap![{}] panel row {i} inline width is {width}, but sum of segs is {sum_width}",
927                        WIDGET.id()
928                    );
929                    continue;
930                }
931
932                tracing::error!(
933                    "Wrap![{}] panel row {i} computed width {width}, but sum of segs is {sum_width}",
934                    WIDGET.id()
935                );
936            }
937        }
938    }

940    fn layout_bidi(&mut self, constraints: Option<InlineConstraints>, direction: LayoutDirection, spacing_x: Px) {
941        let spacing_x = spacing_x.0 as f32;
942        let mut our_rows = 0..self.rows.len();
943
944        if let Some(l) = constraints {
945            let l = l.layout();
946            our_rows = 0..0;
947
948            if !self.rows.is_empty() {
949                if l.first_segs.len() != self.rows[0].item_segs.iter().map(|s| s.measure().len()).sum::<usize>() {
950                    // parent set first_segs empty (not sorted), or wrong
951                    let mut x = 0.0;
952                    for s in self.rows[0].item_segs.iter_mut() {
953                        let mut spacing_x = spacing_x;
954                        for (seg, pos) in s.iter_mut() {
955                            pos.x = x;
956                            x += seg.width + spacing_x;
957                            spacing_x = 0.0;
958                        }
959                    }
960                } else {
961                    // parent set first_segs
962                    for (pos, (_seg, seg_pos)) in l
963                        .first_segs
964                        .iter()
965                        .zip(self.rows[0].item_segs.iter_mut().flat_map(|s| s.iter_mut()))
966                    {
967                        seg_pos.x = pos.x;
968                    }
969                }
970
971                if self.rows.len() > 1 {
972                    // last row not the same as first
973                    let last_i = self.rows.len() - 1;
974                    let last = &mut self.rows[last_i];
975                    if l.last_segs.len() != last.item_segs.iter().map(|s| s.measure().len()).sum::<usize>() {
976                        // parent set last_segs empty (not sorted), or wrong
977                        let mut x = 0.0;
978                        for s in last.item_segs.iter_mut() {
979                            let mut spacing_x = spacing_x;
980                            for (seg, pos) in s.iter_mut() {
981                                pos.x = x;
982                                x += seg.width + spacing_x;
983                                spacing_x = 0.0;
984                            }
985                        }
986                    } else {
987                        // parent set last_segs
988                        for (pos, (_seg, seg_pos)) in l.last_segs.iter().zip(last.item_segs.iter_mut().flat_map(|s| s.iter_mut())) {
989                            seg_pos.x = pos.x;
990                        }
991                    }
992
993                    if self.rows.len() > 2 {
994                        our_rows = 1..self.rows.len() - 1;
995                    }
996                }
997            }
998        }
999
1000        for row in &mut self.rows[our_rows] {
1001            // rows we sort and set x
1002
1003            unicode_bidi_levels(
1004                direction,
1005                row.item_segs.iter().flat_map(|i| i.measure().iter().map(|i| i.kind)),
1006                &mut self.bidi_levels,
1007            );
1008
1009            unicode_bidi_sort(
1010                direction,
1011                row.item_segs
1012                    .iter()
1013                    .flat_map(|i| i.measure().iter().map(|i| i.kind))
1014                    .zip(self.bidi_levels.iter().copied()),
1015                0,
1016                &mut self.bidi_sorted,
1017            );
1018
1019            let mut x = 0.0;
1020
1021            let mut spacing_count = row.item_segs.len().saturating_sub(1);
1022
1023            let mut last_item_i = usize::MAX;
1024            for &new_i in self.bidi_sorted.iter() {
1025                let mut segs_offset = 0;
1026
1027                // `bidi_sorted` is flatten of `row.segs`
1028                for (i, s) in row.item_segs.iter_mut().enumerate() {
1029                    if segs_offset + s.measure().len() <= new_i {
1030                        segs_offset += s.measure().len();
1031                    } else {
1032                        let new_i = new_i - segs_offset;
1033
1034                        if last_item_i != i {
1035                            last_item_i = i;
1036                            if x > 0.0 && spacing_count > 0 {
1037                                x += spacing_x;
1038                                spacing_count -= 1;
1039                            }
1040                        }
1041
1042                        s.layout_mut()[new_i].x = x;
1043                        x += s.measure()[new_i].width;
1044                        break;
1045                    }
1046                }
1047            }
1048        }
1049
1050        for row in self.rows.iter_mut() {
1051            // update seg.x and seg.width
1052            for seg in &mut row.item_segs {
1053                if seg.measure().is_empty() {
1054                    continue;
1055                }
1056
1057                let mut seg_min = f32::MAX;
1058                let mut seg_max = f32::MIN;
1059                for (m, l) in seg.iter_mut() {
1060                    seg_min = seg_min.min(l.x);
1061                    seg_max = seg_max.max(l.x + m.width);
1062                }
1063                seg.set_x_width(seg_min, seg_max - seg_min);
1064
1065                for (_, l) in seg.iter_mut() {
1066                    l.x -= seg_min;
1067                }
1068            }
1069        }
1070    }
