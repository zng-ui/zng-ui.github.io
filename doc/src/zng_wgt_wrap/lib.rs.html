29/// Wrapping inline layout.

32impl Wrap {
33    fn widget_intrinsic(&mut self) {
34        self.widget_builder().push_build_action(|wgt| {
35            let child = node(
36                wgt.capture_ui_node_or_nil(property_id!(Self::children)),
37                wgt.capture_var_or_else(property_id!(Self::spacing), || {
38                    LINE_SPACING_VAR.map(|s| GridSpacing {
39                        column: Length::zero(),
40                        row: s.clone(),
41                    })
42                }),
43                wgt.capture_var_or_else(property_id!(Self::children_align), || TEXT_ALIGN_VAR),
44            );
45            wgt.set_child(child);
46        });
47    }

85/// Wrap node.
86///
87/// Can be used directly to inline widgets without declaring a wrap widget info. This node is the child
88/// of the `Wrap!` widget.
89pub fn node(children: impl IntoUiNode, spacing: impl IntoVar<GridSpacing>, children_align: impl IntoVar<Align>) -> UiNode {
90    let children = PanelList::new(children).track_info_range(*PANEL_LIST_ID);
91    let spacing = spacing.into_var();
92    let children_align = children_align.into_var();
93    let mut layout = InlineLayout::default();
94
95    match_node(children, move |children, op| match op {
96        UiNodeOp::Init => {
97            WIDGET.sub_var_layout(&spacing).sub_var_layout(&children_align);
98        }
99        UiNodeOp::Update { updates } => {
100            let mut any = false;
101            children.update_list(updates, &mut any);
102
103            if any {
104                WIDGET.layout();
105            }
106        }
107        UiNodeOp::Measure { wm, desired_size } => {
108            let spacing = spacing.layout();
109            children.delegated();
110            *desired_size = layout.measure(wm, children.node_impl::<PanelList>(), children_align.get(), spacing);
111        }
112        UiNodeOp::Layout { wl, final_size } => {
113            let spacing = spacing.layout();
114            children.delegated();
115            // rust-analyzer does not find `layout` here if called with dot.
116            *final_size = InlineLayout::layout(&mut layout, wl, children.node_impl::<PanelList>(), children_align.get(), spacing);
117        }
118        _ => {}
119    })
120}

122/// Create a node that estimates the size of wrap panel children.
123///
124/// The estimation assumes that all items have a size of `child_size`.
125pub fn lazy_size(children_len: impl IntoVar<usize>, spacing: impl IntoVar<GridSpacing>, child_size: impl IntoVar<Size>) -> UiNode {
126    // we don't use `properties::size(UiNode::nil(), child_size)` because that size disables inlining.
127    let size = child_size.into_var();
128    let sample = match_node_leaf(move |op| match op {
129        UiNodeOp::Init => {
130            WIDGET.sub_var_layout(&size);
131        }
132        UiNodeOp::Measure { desired_size, .. } => {
133            *desired_size = size.layout();
134        }
135        UiNodeOp::Layout { final_size, .. } => {
136            *final_size = size.layout();
137        }
138        _ => {}
139    });
140
141    lazy_sample(children_len, spacing, sample)
142}

144/// Create a node that estimates the size of wrap panel children.
145///
146/// The estimation assumes that all items have the size of `child_sample`.
147pub fn lazy_sample(children_len: impl IntoVar<usize>, spacing: impl IntoVar<GridSpacing>, child_sample: impl IntoUiNode) -> UiNode {
148    let children_len = children_len.into_var();
149    let spacing = spacing.into_var();
150
151    match_node(child_sample, move |sample, op| match op {
152        UiNodeOp::Init => {
153            WIDGET.sub_var_layout(&children_len).sub_var_layout(&spacing);
154        }
155        UiNodeOp::Measure { wm, desired_size } => {
156            let child_size = sample.measure(wm);
157            *desired_size = InlineLayout::estimate_measure(wm, children_len.get(), child_size, spacing.layout());
158        }
159        UiNodeOp::Layout { wl, final_size } => {
160            let child_size = sample.layout(wl);
161            *final_size = InlineLayout::estimate_layout(wl, children_len.get(), child_size, spacing.layout());
162        }
163        _ => {}
164    })
165}

167/// Info about segments of a widget in a row.
168#[derive(Debug, Clone)]
169enum ItemSegsInfo {
170    Block(Px),
171    Built {
172        measure: Arc<Vec<InlineSegment>>,
173        layout: Arc<Vec<InlineSegmentPos>>,
174        x: f32,
175        width: f32,
176    },
177}
178impl ItemSegsInfo {
179    pub fn new_collapsed() -> Self {
180        Self::Block(Px(0))
181    }
182
183    pub fn new_block(width: Px) -> Self {
184        Self::Block(width)
185    }
186
187    pub fn new_inlined(measure: Arc<Vec<InlineSegment>>) -> Self {
188        Self::Built {
189            measure,
190            layout: Arc::new(vec![]),
191            x: 0.0,
192            width: 0.0,
193        }
194    }
195
196    pub fn measure(&self) -> &[InlineSegment] {
197        match self {
198            ItemSegsInfo::Built { measure, .. } => measure,
199            _ => &[],
200        }
201    }
202
203    pub fn layout_mut(&mut self) -> &mut Vec<InlineSegmentPos> {
204        self.build();
205        match self {
206            ItemSegsInfo::Built { measure, layout, .. } => {
207                // Borrow checker limitation does not allow `if let Some(l) = Arc::get_mut(..) { l } else { <insert-return> }`
208
209                if Arc::get_mut(layout).is_none() {
210                    *layout = Arc::new(vec![]);
211                }
212
213                let r = Arc::get_mut(layout).unwrap();
214                r.resize(measure.len(), InlineSegmentPos::new(0.0));
215
216                r
217            }
218            _ => unreachable!(),
219        }
220    }
221
222    pub fn iter_mut(&mut self) -> impl Iterator<Item = (&InlineSegment, &mut InlineSegmentPos)> {
223        self.build();
224        match self {
225            ItemSegsInfo::Built { measure, layout, .. } => {
226                if Arc::get_mut(layout).is_none() {
227                    *layout = Arc::new(vec![]);
228                }
229
230                let r = Arc::get_mut(layout).unwrap();
231                r.resize(measure.len(), InlineSegmentPos::new(0.0));
232
233                measure.iter().zip(r)
234            }
235            _ => unreachable!(),
236        }
237    }
238
239    /// Only valid if has bidi items and is up-to-date.
240    pub fn x_width_segs(&self) -> (Px, Px, Arc<Vec<InlineSegmentPos>>) {
241        match self {
242            ItemSegsInfo::Built { layout, x, width, .. } => (Px(x.floor() as i32), Px(width.ceil() as i32), layout.clone()),
243            _ => unreachable!(),
244        }
245    }
246
247    #[cfg(debug_assertions)]
248    pub fn measure_width(&self) -> f32 {
249        match self {
250            ItemSegsInfo::Block(w) => w.0 as f32,
251            ItemSegsInfo::Built { measure, .. } => measure.iter().map(|s| s.width).sum(),
252        }
253    }
254
255    fn build(&mut self) {
256        match self {
257            ItemSegsInfo::Block(width) => {
258                let width = width.0 as f32;
259                *self = ItemSegsInfo::Built {
260                    measure: Arc::new(vec![InlineSegment::new(width, TextSegmentKind::OtherNeutral)]),
261                    layout: Arc::new(Vec::with_capacity(1)),
262                    x: 0.0,
263                    width,
264                }
265            }
266            ItemSegsInfo::Built { .. } => {}
267        }
268    }
269
270    fn set_x_width(&mut self, new_x: f32, new_width: f32) {
271        match self {
272            ItemSegsInfo::Built { x, width, .. } => {
273                *x = new_x;
274                *width = new_width;
275            }
276            _ => unreachable!(),
277        }
278    }
279}

296#[derive(Default)]
297struct InlineLayout {
298    first_wrapped: bool,
299    rows: RecycleVec<RowInfo>,
300    desired_size: PxSize,
301
302    // has segments in the opposite direction, requires bidi sorting and positioning.
303    has_bidi_inline: bool,
304    bidi_layout_fresh: bool,
305    // reused heap alloc

310impl InlineLayout {
311    pub fn estimate_measure(wm: &mut WidgetMeasure, children_len: usize, child_size: PxSize, spacing: PxGridSpacing) -> PxSize {
312        if children_len == 0 {
313            return PxSize::zero();
314        }
315
316        let metrics = LAYOUT.metrics();
317        let constraints = metrics.constraints();
318
319        if metrics.inline_constraints().is_none()
320            && let Some(known) = constraints.fill_or_exact()
321        {
322            return known;
323        }
324
325        let max_x = constraints.x.max().unwrap_or(Px::MAX).max(child_size.width);
326
327        if let Some(inline) = wm.inline() {
328            let inline_constraints = metrics.inline_constraints().unwrap().measure();
329
330            inline.first_wrapped = inline_constraints.first_max < child_size.width;
331
332            let mut first_max_x = max_x;
333            if !inline.first_wrapped {
334                first_max_x = inline_constraints.first_max;
335            }
336
337            inline.first.height = child_size.height.max(inline_constraints.mid_clear_min);
338
339            let column_len = (first_max_x - child_size.width) / (child_size.width + spacing.column) + Px(1);
340            inline.first.width = (column_len - Px(1)) * (child_size.width + spacing.column) + child_size.width;
341
342            let children_len = Px(children_len as _) - column_len;
343            inline.last_wrapped = children_len.0 > 0;
344
345            let mut size = inline.first;
346
347            if inline.last_wrapped {
348                let column_len = (max_x - child_size.width) / (child_size.width + spacing.column) + Px(1);
349
350                size.width = size
351                    .width
352                    .max((column_len - Px(1)) * (child_size.width + spacing.column) + child_size.width);
353
354                let mid_len = children_len / column_len;
355                if mid_len.0 > 0 {
356                    size.height += (spacing.row + child_size.height) * mid_len;
357                }
358
359                let last_len = children_len % column_len;
360                inline.last.height = child_size.height;
361                if last_len.0 > 0 {
362                    inline.last.width = (last_len - Px(1)) * (child_size.width + spacing.column) + child_size.width;
363
364                    size.height += spacing.row + child_size.height;
365                } else {
366                    inline.last.width = max_x;
367                }
368            } else {
369                inline.last = inline.first;
370            }
371
372            debug_assert_eq!(inline.first.is_empty(), inline.last.is_empty());
373
374            size
375        } else {
376            let column_len = (max_x - child_size.width) / (child_size.width + spacing.column) + Px(1);
377            let row_len = (Px(children_len as i32) / column_len).max(Px(1));
378
379            // spacing in between means space available to divide for pairs (width + column) has 1 less item.
380            let desired_size = PxSize::new(
381                (column_len - Px(1)) * (child_size.width + spacing.column) + child_size.width,
382                (row_len - Px(1)) * (child_size.height + spacing.row) + child_size.height,
383            );
384            constraints.clamp_size(desired_size)
385        }
386    }

388    pub fn measure(&mut self, wm: &mut WidgetMeasure, children: &mut PanelList, child_align: Align, spacing: PxGridSpacing) -> PxSize {
389        let metrics = LAYOUT.metrics();
390        let constraints = metrics.constraints();
391
392        if metrics.inline_constraints().is_none()
393            && let Some(known) = constraints.fill_or_exact()
394        {
395            return known;
396        }
397
398        self.measure_rows(wm, &metrics, children, child_align, spacing);
399
400        if let Some(inline) = wm.inline() {
401            inline.first_wrapped = self.first_wrapped;
402            inline.last_wrapped = self.rows.len() > 1;
403
404            if let Some(first) = self.rows.first() {
405                inline.first = first.size;
406                inline.with_first_segs(|i| {
407                    i.extend(first.item_segs.iter().flat_map(|i| i.measure().iter().copied()));
408                });
409            } else {
410                inline.first = PxSize::zero();
411                inline.with_first_segs(|i| i.clear());
412            }
413            if let Some(last) = self.rows.last() {
414                inline.last = last.size;
415                inline.with_last_segs(|i| {
416                    i.extend(last.item_segs.iter().flat_map(|i| i.measure().iter().copied()));
417                })
418            } else {
419                inline.last = PxSize::zero();
420                inline.with_last_segs(|i| i.clear());
421            }
422        }
423
424        constraints.clamp_size(self.desired_size)
425    }

427    pub fn estimate_layout(wl: &mut WidgetLayout, children_len: usize, child_size: PxSize, spacing: PxGridSpacing) -> PxSize {
428        let is_inline = wl.inline().is_some();
429        let mut wm = wl.to_measure(if is_inline { Some(Default::default()) } else { None });
430        let size = if let Some(inline) = wl.inline() {
431            let mut size = Self::estimate_measure(&mut wm, children_len, child_size, spacing);
432            if let Some(m_inline) = wm.inline() {
433                inline.invalidate_negative_space();
434                inline.inner_size = size;
435
436                let inline_constraints = LAYOUT.inline_constraints().unwrap().layout();
437
438                let mut mid_height = size.height;
439
440                if !m_inline.first_wrapped {
441                    inline.rows.push(inline_constraints.first);
442                    mid_height -= child_size.height + spacing.row;
443                }
444                if m_inline.last_wrapped {
445                    mid_height -= spacing.row + child_size.height;
446                    inline.rows.push(PxRect::new(
447                        PxPoint::new(Px(0), spacing.row + child_size.height),
448                        PxSize::new(size.width, mid_height),
449                    ));
450                    inline.rows.push(inline_constraints.last);
451                }
452
453                size.height = inline_constraints.last.origin.y + inline_constraints.last.size.height;
454            }
455            size
456        } else {
457            Self::estimate_measure(&mut wm, children_len, child_size, spacing)
458        };
459
460        let width = LAYOUT.constraints().x.fill_or(size.width);
461        PxSize::new(width, size.height)
462    }

464    pub fn layout(&mut self, wl: &mut WidgetLayout, children: &mut PanelList, child_align: Align, spacing: PxGridSpacing) -> PxSize {
465        let metrics = LAYOUT.metrics();
466        let inline_constraints = metrics.inline_constraints();
467        let direction = metrics.direction();
468
469        if inline_constraints.is_none() {
470            // if not already measured by parent inline
471            self.measure_rows(&mut wl.to_measure(None), &metrics, children, child_align, spacing);
472        }
473        if self.has_bidi_inline && !self.bidi_layout_fresh {
474            self.layout_bidi(inline_constraints.clone(), direction, spacing.column);
475        }
476
477        let constraints = metrics.constraints();
478        let child_align_x = child_align.x(direction);
479        let child_align_y = child_align.y();
480
481        let panel_width = constraints.x.fill_or(self.desired_size.width);
482
483        let (first, mid, last) = if let Some(s) = inline_constraints.map(|c| c.layout()) {
484            (s.first, s.mid_clear, s.last)
485        } else {
486            // define our own first and last
487            let mut first = PxRect::from_size(self.rows[0].size);
488            let mut last = PxRect::from_size(self.rows.last().unwrap().size);
489
490            #[cfg(debug_assertions)]
491            if self.has_bidi_inline {
492                let segs_max = self.rows[0]
493                    .item_segs
494                    .iter()
495                    .map(|s| {
496                        let (x, width, _) = s.x_width_segs();
497                        x + width
498                    })
499                    .max()
500                    .unwrap_or_default();
501
502                if (first.width() - segs_max).abs() > Px(10) {
503                    tracing::error!("align error, used width: {:?}, but segs max is: {:?}", first.width(), segs_max);
504                }
505            }
506
507            first.origin.x = (panel_width - first.size.width) * child_align_x;
508            last.origin.x = (panel_width - last.size.width) * child_align_x;
509            last.origin.y = self.desired_size.height - last.size.height;
510
511            if let Some(y) = constraints.y.fill_or_exact() {
512                let align_y = (y - self.desired_size.height) * child_align_y;
513                first.origin.y += align_y;
514                last.origin.y += align_y;
515            }
516
517            (first, Px(0), last)
518        };
519        let panel_height = constraints.y.fill_or(last.origin.y - first.origin.y + last.size.height);
520
521        let child_constraints = PxConstraints2d::new_unbounded().with_fill_x(true).with_max_x(panel_width);
522
523        if let Some(inline) = wl.inline() {
524            inline.rows.clear();
525        }
526
527        let fill_width = if child_align.is_fill_x() {
528            Some(panel_width.0 as f32)
529        } else {
530            None
531        };
532
533        LAYOUT.with_constraints(child_constraints, || {
534            let mut row = first;
535            let mut row_segs = &self.rows[0].item_segs;
536            let mut row_advance = Px(0);
537            let mut next_row_i = 1;
538            let mut row_segs_i_start = 0;
539
540            let mut fill_scale = None;
541            if let Some(mut f) = fill_width {
542                if wl.is_inline() {
543                    f = first.width().0 as f32;
544                }
545                fill_scale = Some(f / self.rows[0].size.width.0 as f32);
546            }
547
548            children.for_each_child(|i, child, o| {
549                if next_row_i < self.rows.len() && self.rows[next_row_i].first_child == i {
550                    // new row
551                    if let Some(inline) = wl.inline() {
552                        inline.rows.push(row);
553                    }
554                    if next_row_i == self.rows.len() - 1 {
555                        row = last;
556                    } else {
557                        row.origin.y += row.size.height + spacing.row;
558                        if next_row_i == 1 {
559                            // clear first row
560                            row.origin.y += mid;
561                        }
562
563                        row.size = self.rows[next_row_i].size;
564                        row.origin.x = (panel_width - row.size.width) * child_align_x;
565                    }
566                    row_segs = &self.rows[next_row_i].item_segs;
567                    row_segs_i_start = self.rows[next_row_i].first_child;
568                    next_row_i += 1;
569                    row_advance = Px(0);
570
571                    fill_scale = None;
572                    if let Some(mut f) = fill_width
573                        && (wl.is_inline() || next_row_i < self.rows.len())
574                    {
575                        if wl.is_inline() && next_row_i == self.rows.len() {
576                            f = last.width().0 as f32;
577                        }
578                        // fill row, if it is not the last in a block layout
579                        fill_scale = Some(f / self.rows[next_row_i - 1].size.width.0 as f32);
580                    }
581                }
582
583                let (bidi_x, bidi_width, bidi_segs) = if self.has_bidi_inline {
584                    row_segs[i - row_segs_i_start].x_width_segs()
585                } else {
586                    (Px(0), Px(0), self.bidi_default_segs.clone())
587                };
588
589                if let Some(mut w) = child.as_widget()
590                    && let Some(child_inline) = w.with_context(WidgetUpdateMode::Ignore, || WIDGET.bounds().measure_inline())
591                {
592                    let child_desired_size = w.with_context(WidgetUpdateMode::Ignore, || WIDGET.bounds().measure_outer_size());
593                    if child_desired_size.is_empty() {
594                        // collapsed, continue.
595                        wl.collapse_child(i);
596                        return;
597                    }
598
599                    let mut child_first = PxRect::from_size(child_inline.first);
600                    let mut child_mid = Px(0);
601                    let mut child_last = PxRect::from_size(child_inline.last);
602
603                    if child_inline.last_wrapped {
604                        // child wraps
605                        debug_assert_eq!(self.rows[next_row_i].first_child, i + 1);
606
607                        child_first.origin.x = row.origin.x + row_advance;
608                        if let LayoutDirection::RTL = direction {
609                            child_first.origin.x -= row_advance;
610                        }
611                        child_first.origin.y += (row.size.height - child_first.size.height) * child_align_y;
612                        child_mid = (row.size.height - child_first.size.height).max(Px(0));
613                        child_last.origin.y = child_desired_size.height - child_last.size.height;
614
615                        if self.has_bidi_inline {
616                            child_first.origin.x = row.origin.x + bidi_x;
617                            child_first.size.width = bidi_width;
618                        }
619
620                        let next_row = if next_row_i == self.rows.len() - 1 {
621                            last
622                        } else {
623                            let mut r = row;
624                            r.origin.y += child_last.origin.y;
625                            r.size = self.rows[next_row_i].size;
626                            r.origin.x = (panel_width - r.size.width) * child_align_x;
627                            r
628                        };
629                        child_last.origin.x = next_row.origin.x;
630                        if let LayoutDirection::RTL = direction {
631                            child_last.origin.x += next_row.size.width - child_last.size.width;
632                        }
633                        child_last.origin.y += (next_row.size.height - child_last.size.height) * child_align_y;
634
635                        let (last_bidi_x, last_bidi_width, last_bidi_segs) = if self.has_bidi_inline {
636                            self.rows[next_row_i].item_segs[0].x_width_segs()
637                        } else {
638                            (Px(0), Px(0), self.bidi_default_segs.clone())
639                        };
640
641                        if self.has_bidi_inline {
642                            child_last.origin.x = next_row.origin.x + last_bidi_x;
643                            child_last.size.width = last_bidi_width;
644                        }
645
646                        if let Some(s) = fill_scale {
647                            child_first.size.width *= s;
648
649                            // child wraps, so last is different row
650                            if wl.is_inline() || next_row_i < self.rows.len() - 1 {
651                                let mut f = fill_width.unwrap();
652                                if wl.is_inline() && next_row_i == self.rows.len() - 1 {
653                                    f = last.width().0 as f32;
654                                }
655                                fill_scale = Some(f / next_row.size.width.0 as f32);
656                                let s = fill_scale.unwrap();
657                                child_last.size.width *= s;
658                            } else {
659                                // only fill last row if Wrap! is nested/inlined
660                                fill_scale = None;
661                            }
662                        }
663
664                        let (_, define_ref_frame) =
665                            wl.with_child(|wl| wl.layout_inline(child_first, child_mid, child_last, bidi_segs, last_bidi_segs, child));
666                        o.child_offset = PxVector::new(Px(0), row.origin.y);
667                        o.define_reference_frame = define_ref_frame;
668
669                        // new row
670                        if let Some(inline) = wl.inline() {
671                            inline.rows.push(row);
672                            if let Some(mut w) = child.as_widget() {
673                                w.with_context(WidgetUpdateMode::Ignore, || {
674                                    if let Some(inner) = WIDGET.bounds().inline() {
675                                        if inner.rows.len() >= 3 {
676                                            inline.rows.extend(inner.rows[1..inner.rows.len() - 1].iter().map(|r| {
677                                                let mut r = *r;
678                                                r.origin.y += row.origin.y;
679                                                r
680                                            }));
681                                        }
682                                    } else {
683                                        tracing::error!("child inlined in measure, but not in layout")
684                                    }
685                                });
686                            }
687                        }
688                        row = next_row;
689                        row_advance = child_last.size.width + spacing.column;
690                        row_segs = &self.rows[next_row_i].item_segs;
691                        row_segs_i_start = self.rows[next_row_i].first_child - 1; // next row first item is also this widget
692                        debug_assert_eq!(row_segs_i_start, i);
693                        next_row_i += 1;
694                    } else {
695                        // child inlined, but fits in the row
696
697                        let mut offset = PxVector::new(row_advance, Px(0));
698                        if let LayoutDirection::RTL = direction {
699                            offset.x = row.size.width - child_last.size.width - offset.x;
700                        }
701                        offset.y = (row.size.height - child_inline.first.height) * child_align_y;
702
703                        let mut max_size = child_inline.first;
704
705                        if self.has_bidi_inline {
706                            max_size.width = bidi_width;
707                            child_first.size.width = bidi_width;
708                            child_last.size.width = bidi_width;
709                        }
710
711                        if let Some(s) = fill_scale {
712                            child_first.size.width *= s;
713                            child_last.size.width *= s;
714                            max_size.width *= s;
715                        }
716
717                        let (_, define_ref_frame) = wl.with_child(|wl| {
718                            LAYOUT.with_constraints(child_constraints.with_fill(false, false).with_max_size(max_size), || {
719                                wl.layout_inline(child_first, child_mid, child_last, bidi_segs.clone(), bidi_segs, child)
720                            })
721                        });
722                        o.child_offset = row.origin.to_vector() + offset;
723                        if self.has_bidi_inline {
724                            o.child_offset.x = row.origin.x + bidi_x;
725                        }
726                        o.define_reference_frame = define_ref_frame;
727
728                        row_advance += child_last.size.width + spacing.column;
729                    }
730                } else {
731                    // inline block
732                    let max_width = if self.has_bidi_inline {
733                        bidi_width
734                    } else {
735                        row.size.width - row_advance
736                    };
737                    let (size, define_ref_frame) = LAYOUT.with_constraints(
738                        child_constraints.with_fill(false, false).with_max(max_width, row.size.height),
739                        || wl.with_child(|wl| wl.layout_block(child)),
740                    );
741                    if size.is_empty() {
742                        // collapsed, continue.
743                        o.child_offset = PxVector::zero();
744                        o.define_reference_frame = false;
745                        return;
746                    }
747
748                    let mut offset = PxVector::new(row_advance, Px(0));
749                    if let LayoutDirection::RTL = direction {
750                        offset.x = row.size.width - size.width - offset.x;
751                    }
752                    offset.y = (row.size.height - size.height) * child_align_y;
753                    o.child_offset = row.origin.to_vector() + offset;
754                    if self.has_bidi_inline {
755                        o.child_offset.x = row.origin.x + bidi_x;
756                    }
757                    o.define_reference_frame = define_ref_frame;
758                    row_advance += size.width + spacing.column;
759                }
760            });
761
762            if let Some(inline) = wl.inline() {
763                // last row
764                inline.rows.push(row);
765            }
766        });
767
768        children.commit_data().request_render();
769
770        constraints.clamp_size(PxSize::new(panel_width, panel_height))
771    }

773    fn measure_rows(
774        &mut self,
775        wm: &mut WidgetMeasure,
776        metrics: &LayoutMetrics,
777        children: &mut PanelList,
778        child_align: Align,
779        spacing: PxGridSpacing,
780    ) {
781        self.rows.begin_reuse();
782        self.bidi_layout_fresh = false;
783
784        self.first_wrapped = false;
785        self.desired_size = PxSize::zero();
786        self.has_bidi_inline = false;
787
788        let direction = metrics.direction();
789        let constraints = metrics.constraints();
790        let inline_constraints = metrics.inline_constraints();
791        let child_inline_constrain = constraints.x.max_or(Px::MAX);
792        let child_constraints = PxConstraints2d::new_unbounded()
793            .with_fill_x(child_align.is_fill_x())
794            .with_max_x(child_inline_constrain);
795        let mut row = self.rows.new_item();
796        LAYOUT.with_constraints(child_constraints, || {
797            children.for_each_child(|i, child, _| {
798                let mut inline_constrain = child_inline_constrain;
799                let mut wrap_clear_min = Px(0);
800                if self.rows.is_empty()
801                    && !self.first_wrapped
802                    && let Some(InlineConstraints::Measure(InlineConstraintsMeasure {
803                        first_max, mid_clear_min, ..
804                    })) = inline_constraints
805                {
806                    inline_constrain = first_max;
807                    wrap_clear_min = mid_clear_min;
808                }
809                if inline_constrain < Px::MAX {
810                    inline_constrain -= row.size.width;
811                }
812
813                let (inline, size) = wm.measure_inline(inline_constrain, row.size.height - spacing.row, child);
814
815                let can_collapse = size.is_empty()
816                    && match &inline {
817                        Some(i) => i.first_segs.is_empty(),
818                        None => true,
819                    };
820                if can_collapse {
821                    row.item_segs.push(ItemSegsInfo::new_collapsed());
822                    // collapsed, continue.
823                    return;
824                }
825
826                if let Some(inline) = inline {
827                    if !self.has_bidi_inline {
828                        self.has_bidi_inline =
829                            inline
830                                .first_segs
831                                .iter()
832                                .chain(inline.last_segs.iter())
833                                .any(|s| match s.kind.strong_direction() {
834                                    Some(d) => d != direction,
835                                    None => false,
836                                });
837                    }
838
839                    // item mid-rows can be wider
840                    self.desired_size.width = self.desired_size.width.max(size.width);
841
842                    if inline.first_wrapped {
843                        // wrap by us, detected by child
844                        if row.size.is_empty() {
845                            debug_assert!(self.rows.is_empty());
846                            self.first_wrapped = true;
847                        } else {
848                            row.size.width -= spacing.column;
849                            row.size.width = row.size.width.max(Px(0));
850                            self.desired_size.width = self.desired_size.width.max(row.size.width);
851                            self.desired_size.height += row.size.height + spacing.row;
852
853                            self.rows.push_renew(&mut row);
854                        }
855
856                        row.size = inline.first;
857                        row.first_child = i;
858                    } else {
859                        row.size.width += inline.first.width;
860                        row.size.height = row.size.height.max(inline.first.height);
861                    }
862                    row.item_segs.push(ItemSegsInfo::new_inlined(inline.first_segs.clone()));
863
864                    if inline.last_wrapped {
865                        // wrap by child
866                        self.desired_size.width = self.desired_size.width.max(row.size.width);
867                        self.desired_size.height += size.height - inline.first.height;
868
869                        self.rows.push_renew(&mut row);
870                        row.size = inline.last;
871                        row.size.width += spacing.column;
872                        row.first_child = i + 1;
873                        row.item_segs.push(ItemSegsInfo::new_inlined(inline.last_segs));
874                    } else {
875                        // child inlined, but fit in row
876                        row.size.width += spacing.column;
877                    }
878                } else if size.width <= inline_constrain {
879                    row.size.width += size.width + spacing.column;
880                    row.size.height = row.size.height.max(size.height);
881                    row.item_segs.push(ItemSegsInfo::new_block(size.width));
882                } else {
883                    // wrap by us
884                    if row.size.is_empty() {
885                        debug_assert!(self.rows.is_empty());
886                        self.first_wrapped = true;
887                    } else {
888                        row.size.width -= spacing.column;
889                        row.size.width = row.size.width.max(Px(0));
890                        self.desired_size.width = self.desired_size.width.max(row.size.width);
891                        self.desired_size.height += row.size.height.max(wrap_clear_min) + spacing.row;
892                        self.rows.push_renew(&mut row);
893                    }
894
895                    row.size = size;
896                    row.size.width += spacing.column;
897                    row.first_child = i;
898                    row.item_segs.push(ItemSegsInfo::new_block(size.width));
899                }
900            });
901        });
902
903        // last row
904        row.size.width -= spacing.column;
905        row.size.width = row.size.width.max(Px(0));
906        self.desired_size.width = self.desired_size.width.max(row.size.width);
907        self.desired_size.height += row.size.height; // no spacing because it's single line or already added in [^wrap by us]
908        self.rows.push(row);
909
910        self.rows.commit_reuse();
911
912        #[cfg(debug_assertions)]
913        for (i, row) in self.rows.iter().enumerate() {
914            let width = row.size.width;
915            let sum_width = row.item_segs.iter().map(|s| Px(s.measure_width() as i32)).sum::<Px>()
916                + spacing.column * Px(row.item_segs.len().saturating_sub(1) as _);
917
918            if (sum_width - width) > Px(1) {
919                if metrics.inline_constraints().is_some() && (i == 0 || i == self.rows.len() - 1) {
920                    tracing::error!(
921                        "Wrap![{}] panel row {i} inline width is {width}, but sum of segs is {sum_width}",
922                        WIDGET.id()
923                    );
924                    continue;
925                }
926
927                tracing::error!(
928                    "Wrap![{}] panel row {i} computed width {width}, but sum of segs is {sum_width}",
929                    WIDGET.id()
930                );
931            }
932        }
933    }

935    fn layout_bidi(&mut self, constraints: Option<InlineConstraints>, direction: LayoutDirection, spacing_x: Px) {
936        let spacing_x = spacing_x.0 as f32;
937        let mut our_rows = 0..self.rows.len();
938
939        if let Some(l) = constraints {
940            let l = l.layout();
941            our_rows = 0..0;
942
943            if !self.rows.is_empty() {
944                if l.first_segs.len() != self.rows[0].item_segs.iter().map(|s| s.measure().len()).sum::<usize>() {
945                    // parent set first_segs empty (not sorted), or wrong
946                    let mut x = 0.0;
947                    for s in self.rows[0].item_segs.iter_mut() {
948                        let mut spacing_x = spacing_x;
949                        for (seg, pos) in s.iter_mut() {
950                            pos.x = x;
951                            x += seg.width + spacing_x;
952                            spacing_x = 0.0;
953                        }
954                    }
955                } else {
956                    // parent set first_segs
957                    for (pos, (_seg, seg_pos)) in l
958                        .first_segs
959                        .iter()
960                        .zip(self.rows[0].item_segs.iter_mut().flat_map(|s| s.iter_mut()))
961                    {
962                        seg_pos.x = pos.x;
963                    }
964                }
965
966                if self.rows.len() > 1 {
967                    // last row not the same as first
968                    let last_i = self.rows.len() - 1;
969                    let last = &mut self.rows[last_i];
970                    if l.last_segs.len() != last.item_segs.iter().map(|s| s.measure().len()).sum::<usize>() {
971                        // parent set last_segs empty (not sorted), or wrong
972                        let mut x = 0.0;
973                        for s in last.item_segs.iter_mut() {
974                            let mut spacing_x = spacing_x;
975                            for (seg, pos) in s.iter_mut() {
976                                pos.x = x;
977                                x += seg.width + spacing_x;
978                                spacing_x = 0.0;
979                            }
980                        }
981                    } else {
982                        // parent set last_segs
983                        for (pos, (_seg, seg_pos)) in l.last_segs.iter().zip(last.item_segs.iter_mut().flat_map(|s| s.iter_mut())) {
984                            seg_pos.x = pos.x;
985                        }
986                    }
987
988                    if self.rows.len() > 2 {
989                        our_rows = 1..self.rows.len() - 1;
990                    }
991                }
992            }
993        }
994
995        for row in &mut self.rows[our_rows] {
996            // rows we sort and set x
997
998            unicode_bidi_levels(
999                direction,
1000                row.item_segs.iter().flat_map(|i| i.measure().iter().map(|i| i.kind)),
1001                &mut self.bidi_levels,
1002            );
1003
1004            unicode_bidi_sort(
1005                direction,
1006                row.item_segs
1007                    .iter()
1008                    .flat_map(|i| i.measure().iter().map(|i| i.kind))
1009                    .zip(self.bidi_levels.iter().copied()),
1010                0,
1011                &mut self.bidi_sorted,
1012            );
1013
1014            let mut x = 0.0;
1015
1016            let mut spacing_count = row.item_segs.len().saturating_sub(1);
1017
1018            let mut last_item_i = usize::MAX;
1019            for &new_i in self.bidi_sorted.iter() {
1020                let mut segs_offset = 0;
1021
1022                // `bidi_sorted` is flatten of `row.segs`
1023                for (i, s) in row.item_segs.iter_mut().enumerate() {
1024                    if segs_offset + s.measure().len() <= new_i {
1025                        segs_offset += s.measure().len();
1026                    } else {
1027                        let new_i = new_i - segs_offset;
1028
1029                        if last_item_i != i {
1030                            last_item_i = i;
1031                            if x > 0.0 && spacing_count > 0 {
1032                                x += spacing_x;
1033                                spacing_count -= 1;
1034                            }
1035                        }
1036
1037                        s.layout_mut()[new_i].x = x;
1038                        x += s.measure()[new_i].width;
1039                        break;
1040                    }
1041                }
1042            }
1043        }
1044
1045        for row in self.rows.iter_mut() {
1046            // update seg.x and seg.width
1047            for seg in &mut row.item_segs {
1048                if seg.measure().is_empty() {
1049                    continue;
1050                }
1051
1052                let mut seg_min = f32::MAX;
1053                let mut seg_max = f32::MIN;
1054                for (m, l) in seg.iter_mut() {
1055                    seg_min = seg_min.min(l.x);
1056                    seg_max = seg_max.max(l.x + m.width);
1057                }
1058                seg.set_x_width(seg_min, seg_max - seg_min);
1059
1060                for (_, l) in seg.iter_mut() {
1061                    l.x -= seg_min;
1062                }
1063            }
1064        }
1065    }
