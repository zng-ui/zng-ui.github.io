30/// Clipboard app extension.
31///
32/// # Services
33///
34/// Services provided by this extension.
35///
36/// * [`CLIPBOARD`]
37#[derive(Default)]
38#[non_exhaustive]
39pub struct ClipboardManager {}
40
41impl AppExtension for ClipboardManager {
42    fn update(&mut self) {
43        let mut clipboard = CLIPBOARD_SV.write();
44        clipboard.text.update(|v, txt| v.write_text(txt));
45        clipboard.image.map_update(
46            |img| {
47                if let Some(img) = img.view() {
48                    Ok(img.clone())
49                } else {
50                    Err(ClipboardError::ImageNotLoaded)
51                }
52            },
53            |v, img| v.write_image(&img),
54        );
55        clipboard.file_list.update(|v, list| v.write_file_list(list));
56        clipboard.ext.update(|v, (data_type, data)| v.write_extension(data_type, data))
57    }
58}

65#[derive(Default)]
66struct ClipboardService {
67    text: ClipboardData<Txt, Txt>,
68    image: ClipboardData<ImageVar, Img>,
69    file_list: ClipboardData<Vec<PathBuf>, Vec<PathBuf>>,
70    ext: ClipboardData<IpcBytes, (Txt, IpcBytes)>,
71}
72struct ClipboardData<O: 'static, I: 'static> {
73    latest: Option<Result<Option<O>, ClipboardError>>,
74    request: Option<(I, ResponderVar<Result<bool, ClipboardError>>)>,
75}
76impl<O: 'static, I: 'static> Default for ClipboardData<O, I> {
77    fn default() -> Self {
78        Self {
79            latest: None,
80            request: None,
81        }
82    }
83}
84impl<O: Clone + 'static, I: 'static> ClipboardData<O, I> {
85    pub fn get(&mut self, getter: impl FnOnce(&dyn ActualClipboard) -> ActualClipboardResult<O>) -> Result<Option<O>, ClipboardError> {
86        self.latest
87            .get_or_insert_with(|| {
88                let r = match getter(CLIPBOARD.actual()) {
89                    Ok(r) => match r {
90                        Ok(r) => Ok(Some(r)),
91                        Err(e) => match e {
92                            clipboard_api::ClipboardError::NotFound => Ok(None),
93                            clipboard_api::ClipboardError::NotSupported => Err(ClipboardError::NotSupported),
94                            clipboard_api::ClipboardError::Other(e) => Err(ClipboardError::Other(e)),
95                            e => Err(ClipboardError::Other(e.to_txt())),
96                        },
97                    },
98                    Err(_) => Err(ClipboardError::Disconnected),
99                };
100                if let Err(e) = &r {
101                    tracing::error!("clipboard get error, {e:?}");
102                }
103                r
104            })
105            .clone()
106    }
107
108    pub fn request(&mut self, r: I) -> ResponseVar<Result<bool, ClipboardError>> {
109        let (responder, response) = response_var();
110
111        if let Some((_, r)) = self.request.replace((r, responder)) {
112            r.respond(Ok(false));
113        }
114
115        response
116    }
117
118    pub fn update(&mut self, setter: impl FnOnce(&dyn ActualClipboard, I) -> ActualClipboardResult<()>) {
119        self.map_update(Ok, setter)
120    }
121
122    pub fn map_update<VI>(
123        &mut self,
124        to_view: impl FnOnce(I) -> Result<VI, ClipboardError>,
125        setter: impl FnOnce(&dyn ActualClipboard, VI) -> ActualClipboardResult<()>,
126    ) {
127        self.latest = None;
128        if let Some((i, rsp)) = self.request.take() {
129            let vi = match to_view(i) {
130                Ok(vi) => vi,
131                Err(e) => {
132                    tracing::error!("clipboard set error, {e:?}");
133                    rsp.respond(Err(e));
134                    return;
135                }
136            };
137            let r = match setter(CLIPBOARD.actual(), vi) {
138                Ok(r) => match r {
139                    Ok(()) => Ok(true),
140                    Err(e) => match e {
141                        clipboard_api::ClipboardError::NotFound => {
142                            Err(ClipboardError::Other(Txt::from_static("not found error in set operation")))
143                        }
144                        clipboard_api::ClipboardError::NotSupported => Err(ClipboardError::NotSupported),
145                        clipboard_api::ClipboardError::Other(e) => Err(ClipboardError::Other(e)),
146                        e => Err(ClipboardError::Other(e.to_txt())),
147                    },
148                },
149                Err(_) => Err(ClipboardError::Disconnected),
150            };
151            if let Err(e) = &r {
152                tracing::error!("clipboard set error, {e:?}");
153            }
154            rsp.respond(r);
155        }
156    }
157}

159/// Error getting or setting the clipboard.
160///
161/// The [`CLIPBOARD`] service already logs the error.
162#[derive(Debug, Clone, PartialEq)]
163#[non_exhaustive]
164pub enum ClipboardError {
165    /// No view-process available to process the request.
166    ///
167    /// Note that this error only happens if the view-process is respawning. For headless apps (without renderer)
168    /// a in memory "clipboard" is used and this error does not return.
169    Disconnected,
170    /// View-process or operating system does not support the data type.
171    NotSupported,
172    /// Cannot set image in clipboard because it has not finished loading or loaded with error.
173    ImageNotLoaded,
174    /// Other error.
175    ///
176    /// The string can be a debug description of the error, only suitable for logging.
177    Other(Txt),
178}
179impl std::error::Error for ClipboardError {}
180impl fmt::Display for ClipboardError {
181    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
182        match self {
183            ClipboardError::Disconnected => write!(f, "no view-process available to process the request"),
184            ClipboardError::NotSupported => write!(f, "view-process or operating system does not support the data type"),
185            ClipboardError::ImageNotLoaded => write!(
186                f,
187                "cannot set image in clipboard because it has not finished loading or loaded with error"
188            ),
189            ClipboardError::Other(e) => write!(f, "{e}"),
190        }
191    }
192}

194/// Clipboard service.
195///
196/// This service synchronizes with the UI update cycle, the getter methods provide the same data for all requests in the
197/// same update pass, even if the system clipboard happens to change mid update, the setter methods only set the system clipboard
198/// at the end of the update pass.
199///
200/// This service needs a running view-process to actually interact with the system clipboard, in a headless app
201/// without renderer (no view-process) the service will always return [`ClipboardError::Disconnected`].
202pub struct CLIPBOARD;
203impl CLIPBOARD {
204    /// Gets a text string from the clipboard.
205    pub fn text(&self) -> Result<Option<Txt>, ClipboardError> {
206        CLIPBOARD_SV
207            .write()
208            .text
209            .get(|v| v.read_text())
210            .map(|s| s.map(|s| Txt::from_str(&s)))
211    }
212    /// Sets the text string on the clipboard after the current update.
213    ///
214    /// Returns a response var that updates to `Ok(true)` is the text is put on the clipboard,
215    /// `Ok(false)` if another request made on the same update pass replaces this one or `Err(ClipboardError)`.
216    pub fn set_text(&self, txt: impl Into<Txt>) -> ResponseVar<Result<bool, ClipboardError>> {
217        CLIPBOARD_SV.write().text.request(txt.into())
218    }
219
220    /// Gets an image from the clipboard.
221    ///
222    /// The image is loaded in parallel and cached by the [`IMAGES`] service.
223    ///
224    /// [`IMAGES`]: zng_ext_image::IMAGES
225    pub fn image(&self) -> Result<Option<ImageVar>, ClipboardError> {
226        CLIPBOARD_SV.write().image.get(|v| {
227            let img = v.read_image()?;
228            match img {
229                Ok(img) => {
230                    let mut hash = ImageHasher::new();
231                    hash.update("zng_ext_clipboard::CLIPBOARD");
232                    hash.update(img.id().unwrap().get().to_be_bytes());
233                    match IMAGES.register(hash.finish(), img) {
234                        Ok(r) => Ok(Ok(r)),
235                        Err((_, r)) => Ok(Ok(r)),
236                    }
237                }
238                Err(e) => Ok(Err(e)),
239            }
240        })
241    }
242
243    /// Set the image on the clipboard after the current update, if it is loaded.
244    ///
245    /// Returns a response var that updates to `Ok(true)` is the text is put on the clipboard,
246    /// `Ok(false)` if another request made on the same update pass replaces this one or `Err(ClipboardError)`.
247    pub fn set_image(&self, img: Img) -> ResponseVar<Result<bool, ClipboardError>> {
248        CLIPBOARD_SV.write().image.request(img)
249    }
250
251    /// Gets a file list from the clipboard.
252    pub fn file_list(&self) -> Result<Option<Vec<PathBuf>>, ClipboardError> {
253        CLIPBOARD_SV.write().file_list.get(|v| v.read_file_list())
254    }
255
256    /// Sets the file list on the clipboard after the current update.
257    ///
258    /// Returns a response var that updates to `Ok(true)` is the text is put on the clipboard,
259    /// `Ok(false)` if another request made on the same update pass replaces this one or `Err(ClipboardError)`.
260    pub fn set_file_list(&self, list: impl Into<Vec<PathBuf>>) -> ResponseVar<Result<bool, ClipboardError>> {
261        CLIPBOARD_SV.write().file_list.request(list.into())
262    }
263
264    /// Gets custom data from the clipboard.
265    ///
266    /// The current view-process must support `data_type`.
267    pub fn extension(&self, data_type: impl Into<Txt>) -> Result<Option<IpcBytes>, ClipboardError> {
268        CLIPBOARD_SV.write().ext.get(|v| v.read_extension(data_type.into()))
269    }
270
271    /// Set a custom data on the clipboard.
272    ///
273    /// The current view-process must support `data_type` after the current update.
274    ///
275    /// Returns a response var that updates to `Ok(true)` is the text is put on the clipboard,
276    /// `Ok(false)` if another request made on the same update pass replaces this one or `Err(ClipboardError)`.
277    pub fn set_extension(&self, data_type: impl Into<Txt>, data: IpcBytes) -> ResponseVar<Result<bool, ClipboardError>> {
278        CLIPBOARD_SV.write().ext.request((data_type.into(), data))
279    }
280}

282type ActualClipboardResult<T> = Result<Result<T, clipboard_api::ClipboardError>, ViewChannelError>;
283impl CLIPBOARD {
284    fn actual(&self) -> &dyn ActualClipboard {
285        if VIEW_PROCESS.is_available() {
286            match VIEW_PROCESS.clipboard() {
287                Ok(c) => c,
288                Err(_) => {
289                    if !APP.window_mode().has_renderer() {
290                        &CLIPBOARD
291                    } else {
292                        &ViewChannelError::Disconnected
293                    }
294                }
295            }
296        } else if !APP.window_mode().has_renderer() {
297            &CLIPBOARD
298        } else {
299            &ViewChannelError::Disconnected
300        }
301    }
302}

345impl CLIPBOARD {
346    fn headless_clipboard_get<T: Any + Clone>(&self) -> ActualClipboardResult<T> {
347        let sv = HEADLESS_CLIPBOARD.read();
348        Ok(match &*sv {
349            Some(v) => match v.downcast_ref::<T>() {
350                Some(v) => Ok(v.clone()),
351                None => Err(clipboard_api::ClipboardError::NotFound),
352            },
353            None => Err(clipboard_api::ClipboardError::NotFound),
354        })
355    }
356
357    fn headless_clipboard_set(&self, t: impl Any + Send + Sync) -> ActualClipboardResult<()> {
358        *HEADLESS_CLIPBOARD.write() = Some(Box::new(t));
359        Ok(Ok(()))
360    }
361}
362impl ActualClipboard for CLIPBOARD {
363    fn read_text(&self) -> ActualClipboardResult<Txt> {
364        self.headless_clipboard_get()
365    }
366    fn write_text(&self, txt: Txt) -> ActualClipboardResult<()> {
367        self.headless_clipboard_set(txt)
368    }
369    fn read_image(&self) -> ActualClipboardResult<ViewImage> {
370        self.headless_clipboard_get()
371    }
372    fn write_image(&self, img: &ViewImage) -> ActualClipboardResult<()> {
373        self.headless_clipboard_set(img.clone())
374    }
375
376    fn read_file_list(&self) -> ActualClipboardResult<Vec<PathBuf>> {
377        self.headless_clipboard_get()
378    }
379    fn write_file_list(&self, list: Vec<PathBuf>) -> ActualClipboardResult<()> {
380        self.headless_clipboard_set(list)
381    }
382
383    fn read_extension(&self, data_type: Txt) -> ActualClipboardResult<IpcBytes> {
384        match self.headless_clipboard_get::<(Txt, IpcBytes)>()? {
385            Ok((t, b)) => {
386                if t == data_type {
387                    Ok(Ok(b))
388                } else {
389                    Ok(Err(clipboard_api::ClipboardError::NotFound))
390                }
391            }
392            Err(e) => Ok(Err(e)),
393        }
394    }
395    fn write_extension(&self, data_type: Txt, data: IpcBytes) -> ActualClipboardResult<()> {
396        self.headless_clipboard_set((data_type, data))
397    }
