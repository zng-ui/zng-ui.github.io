10/// Image layout mode.
11///
12/// This layout mode can be set to all images inside a widget using [`img_fit`], the [`image_presenter`] uses this value
13/// to calculate the image final size.
14///
15/// The image desired size is its original size, either in pixels or DIPs after cropping and scaling.
16///
17/// [`img_fit`]: fn@img_fit
18/// [`image_presenter`]: crate::node::image_presenter
19#[derive(Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
20pub enum ImageFit {
21    /// The image original size is preserved, the image is clipped if larger then the final size.
22    None,
23    /// The image is resized to fill the final size, the aspect-ratio is not preserved.
24    Fill,
25    /// The image is resized to fit the final size, preserving the aspect-ratio.
26    Contain,
27    /// The image is resized to fill the final size while preserving the aspect-ratio.
28    /// If the aspect ratio of the final size differs from the image, it is clipped.
29    Cover,
30    /// If the image is smaller then the final size applies the [`None`] layout, if its larger applies the [`Contain`] layout.
31    ///
32    /// [`None`]: ImageFit::None
33    /// [`Contain`]: ImageFit::Contain
34    ScaleDown,
35}
36impl fmt::Debug for ImageFit {
37    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
38        if f.alternate() {
39            write!(f, "ImageFit::")?
40        }
41        match self {
42            Self::None => write!(f, "None"),
43            Self::Fill => write!(f, "Fill"),
44            Self::Contain => write!(f, "Contain"),
45            Self::Cover => write!(f, "Cover"),
46            Self::ScaleDown => write!(f, "ScaleDown"),
47        }
48    }
49}

51/// Image repeat mode.
52///
53/// After the image is fit, aligned, offset and clipped the final image can be repeated
54/// to fill any blank space by enabling [`img_repeat`] with one of these options.
55///
56/// [`img_repeat`]: fn@img_repeat
57#[derive(Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
58pub enum ImageRepeat {
59    /// The image is only rendered once.
60    None,
61    /// The image is repeated to fill empty space, border copies are clipped.
62    Repeat,
63}
64impl fmt::Debug for ImageRepeat {
65    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
66        if f.alternate() {
67            write!(f, "ImageRepeat::")?
68        }
69        match self {
70            Self::None => write!(f, "None"),
71            Self::Repeat => write!(f, "Repeat"),
72        }
73    }
74}
75impl_from_and_into_var! {
76    fn from(repeat: bool) -> ImageRepeat {
77        if repeat { ImageRepeat::Repeat } else { ImageRepeat::None }
78    }
79}

427/// Image load or decode error event.
428///
429/// This property calls `handler` every time the [`CONTEXT_IMAGE_VAR`] updates with a different error or on the first update
430/// after init if the image is already in error on init.
431///
432/// # Handlers
433///
434/// This property accepts any [`WidgetHandler`], including the async handlers. Use one of the handler macros, [`hn!`],
435/// [`hn_once!`], [`async_hn!`] or [`async_hn_once!`], to declare a handler closure.
436///
437/// # Route
438///
439/// This property is not routed, it works only inside a widget that loads images. There is also no *preview* event.
440///
441/// [`WidgetHandler`]: zng_wgt::prelude::WidgetHandler
442/// [`hn!`]: zng_wgt::prelude::hn!
443/// [`hn_once!`]: zng_wgt::prelude::hn_once!
444/// [`async_hn!`]: zng_wgt::prelude::async_hn!
445/// [`async_hn_once!`]: zng_wgt::prelude::async_hn_once!
446#[property(EVENT, widget_impl(Image))]
447pub fn on_error(child: impl IntoUiNode, handler: impl WidgetHandler<ImgErrorArgs>) -> UiNode {
448    let mut handler = handler;
449    let mut error = Txt::from_str("");
450    let mut first_update = false;
451
452    match_node(child, move |_, op| match op {
453        UiNodeOp::Init => {
454            WIDGET.sub_var(&CONTEXT_IMAGE_VAR);
455
456            if CONTEXT_IMAGE_VAR.with(Img::is_error) {
457                first_update = true;
458                WIDGET.update();
459            }
460        }
461        UiNodeOp::Update { .. } => {
462            if let Some(new_img) = CONTEXT_IMAGE_VAR.get_new() {
463                first_update = false;
464                if let Some(e) = new_img.error() {
465                    if error != e {
466                        error = e;
467                        handler.event(&ImgErrorArgs { error: error.clone() });
468                    }
469                } else {
470                    error = "".into();
471                }
472            } else if std::mem::take(&mut first_update) {
473                CONTEXT_IMAGE_VAR.with(|i| {
474                    if let Some(e) = i.error() {
475                        error = e;
476                        handler.event(&ImgErrorArgs { error: error.clone() });
477                    }
478                });
479            }
480
481            handler.update();
482        }
483        _ => {}
484    })
485}

487/// Image loaded event.
488///
489/// This property calls `handler` every time the [`CONTEXT_IMAGE_VAR`] updates with a successfully loaded image or on the first
490/// update after init if the image is already loaded on init.
491///
492/// # Handlers
493///
494/// This property accepts any [`WidgetHandler`], including the async handlers. Use one of the handler macros, [`hn!`],
495/// [`hn_once!`], [`async_hn!`] or [`async_hn_once!`], to declare a handler closure.
496///
497/// # Route
498///
499/// This property is not routed, it works only inside a widget that loads images. There is also no *preview* event.
500///
501/// [`WidgetHandler`]: zng_wgt::prelude::WidgetHandler
502/// [`hn!`]: zng_wgt::prelude::hn!
503/// [`hn_once!`]: zng_wgt::prelude::hn_once!
504/// [`async_hn!`]: zng_wgt::prelude::async_hn!
505/// [`async_hn_once!`]: zng_wgt::prelude::async_hn_once!
506#[property(EVENT, widget_impl(Image))]
507pub fn on_load(child: impl IntoUiNode, handler: impl WidgetHandler<ImgLoadArgs>) -> UiNode {
508    let mut handler = handler;
509    let mut first_update = false;
510
511    match_node(child, move |_, op| match op {
512        UiNodeOp::Init => {
513            WIDGET.sub_var(&CONTEXT_IMAGE_VAR);
514
515            if CONTEXT_IMAGE_VAR.with(Img::is_loaded) {
516                first_update = true;
517                WIDGET.update();
518            }
519        }
520        UiNodeOp::Update { .. } => {
521            if let Some(new_img) = CONTEXT_IMAGE_VAR.get_new() {
522                first_update = false;
523                if new_img.is_loaded() {
524                    handler.event(&ImgLoadArgs {});
525                }
526            } else if std::mem::take(&mut first_update) && CONTEXT_IMAGE_VAR.with(Img::is_loaded) {
527                handler.event(&ImgLoadArgs {});
528            }
529
530            handler.update();
531        }
532        _ => {}
533    })
534}

536/// Block window load until image is loaded.
537///
538/// If the image widget is in the initial window content a [`WindowLoadingHandle`] is used to delay the window
539/// visually opening until the source loads, fails to load or a timeout elapses. By default `true` sets the timeout to 1 second.
540///
541/// [`WindowLoadingHandle`]: zng_ext_window::WindowLoadingHandle
542#[property(LAYOUT, default(false), widget_impl(Image))]
543pub fn img_block_window_load(child: impl IntoUiNode, enabled: impl IntoValue<BlockWindowLoad>) -> UiNode {
544    let enabled = enabled.into();
545    let mut block = None;
546
547    match_node(child, move |_, op| match op {
548        UiNodeOp::Init => {
549            WIDGET.sub_var(&CONTEXT_IMAGE_VAR);
550
551            if let Some(delay) = enabled.deadline() {
552                block = WINDOW.loading_handle(delay);
553            }
554        }
555        UiNodeOp::Update { .. } => {
556            if block.is_some() && !CONTEXT_IMAGE_VAR.with(Img::is_loading) {
557                block = None;
558            }
559        }
560        _ => {}
561    })
