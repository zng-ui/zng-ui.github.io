14/// 1D length units.
15///
16/// See [`LengthUnits`] for more details.
17///
18/// # Equality
19///
20/// Two lengths are equal if they are of the same variant and if:
21///
22/// * `Dip` and `px` lengths uses [`Dip`] and [`Px`] equality.
23/// * `Relative`, `Em`, `RootEm` lengths use the [`Factor`] equality.
24/// * Viewport lengths uses [`about_eq`] with `0.00001` granularity.
25#[derive(Clone, serde::Serialize, serde::Deserialize)]
26pub enum Length {
27    /// The default (initial) value.
28    Default,
29    /// The exact length in device independent units.
30    Dip(Dip),
31    /// The exact length in device pixel units.
32    Px(Px),
33    /// The exact length in font points.
34    Pt(f32),
35    /// Relative to the fill length.
36    Factor(Factor),
37    /// Relative to the leftover fill length.
38    Leftover(Factor),
39    /// Relative to the font-size of the widget.
40    Em(Factor),
41    /// Relative to the font-size of the root widget.
42    RootEm(Factor),
43    /// Relative to the width of the nearest viewport ancestor.
44    ViewportWidth(Factor),
45    /// Relative to the height of the nearest viewport ancestor.
46    ViewportHeight(Factor),
47    /// Relative to the smallest of the nearest viewport ancestor's dimensions.
48    ViewportMin(Factor),
49    /// Relative to the smallest of the nearest viewport ancestor's dimensions.
50    ViewportMax(Factor),
51
52    /// The exact length in device independent units, defined using a `f32` value.
53    ///
54    /// This value will be rounded to the nearest pixel after layout,
55    /// but it will be used as is in the evaluation of length expressions.
56    DipF32(f32),
57    /// The exact length in device pixel units, defined using a `f32` value.
58    ///
59    /// This value will be rounded to the nearest pixel after layout,
60    /// but it will be used as is in the evaluation of length expressions.
61    PxF32(f32),
62
63    /// Expression.
64    Expr(Box<LengthExpr>),
65}
66impl<L: Into<Length>> ops::Add<L> for Length {
67    type Output = Length;
68
69    fn add(self, rhs: L) -> Self::Output {
70        use Length::*;
71
72        let rhs = rhs.into();
73
74        if self.is_zero() == Some(true) {
75            return rhs; // 0 + rhs
76        } else if rhs.is_zero() == Some(true) {
77            return self; // self + 0
78        }
79
80        match (self, rhs) {
81            (Dip(a), Dip(b)) => Dip(a + b),
82            (Px(a), Px(b)) => Px(a + b),
83            (Pt(a), Pt(b)) => Pt(a + b),
84            (Factor(a), Factor(b)) => Factor(a + b),
85            (Leftover(a), Leftover(b)) => Leftover(a + b),
86            (Em(a), Em(b)) => Em(a + b),
87            (RootEm(a), RootEm(b)) => RootEm(a + b),
88            (ViewportWidth(a), ViewportWidth(b)) => ViewportWidth(a + b),
89            (ViewportHeight(a), ViewportHeight(b)) => ViewportHeight(a + b),
90            (ViewportMin(a), ViewportMin(b)) => ViewportMin(a + b),
91            (ViewportMax(a), ViewportMax(b)) => ViewportMax(a + b),
92            (PxF32(a), PxF32(b)) => PxF32(a + b),
93            (DipF32(a), DipF32(b)) => DipF32(a + b),
94            (Px(a), PxF32(b)) | (PxF32(b), Px(a)) => PxF32(a.0 as f32 + b),
95            (Dip(a), DipF32(b)) | (DipF32(b), Dip(a)) => DipF32(a.to_f32() + b),
96            (a, b) => LengthExpr::Add(a, b).to_length_checked(),
97        }
98    }

148impl<F: Into<Factor>> ops::Mul<F> for Length {
149    type Output = Length;
150
151    fn mul(self, rhs: F) -> Self::Output {
152        use Length::*;
153        let rhs = rhs.into();
154
155        if self.is_zero() == Some(true) || rhs == 1.fct() {
156            return self; // 0 * fct || len * 1.0
157        } else if rhs == 0.fct() {
158            return Self::zero(); // len * 0.0
159        }
160
161        match self {
162            Dip(e) => DipF32(e.to_f32() * rhs.0),
163            Px(e) => PxF32(e.0 as f32 * rhs.0),
164            Pt(e) => Pt(e * rhs.0),

166            Leftover(r) => Leftover(r * rhs),
167            Em(e) => Em(e * rhs),
168            RootEm(e) => RootEm(e * rhs),
169            ViewportWidth(w) => ViewportWidth(w * rhs),
170            ViewportHeight(h) => ViewportHeight(h * rhs),
171            ViewportMin(m) => ViewportMin(m * rhs),
172            ViewportMax(m) => ViewportMax(m * rhs),
173            DipF32(e) => DipF32(e * rhs.0),
174            PxF32(e) => PxF32(e * rhs.0),

222    fn lerp(self, to: &Self, step: EasingStep) -> Self {
223        use Length::*;
224
225        if step == 0.fct() {
226            return self;
227        }
228        if step == 1.fct() {

283impl PartialEq for Length {
284    fn eq(&self, other: &Self) -> bool {
285        use Length::*;
286        match (self, other) {
287            (Default, Default) => true,
288
289            (Dip(a), Dip(b)) => a == b,
290            (Px(a), Px(b)) => a == b,
291            (Pt(a), Pt(b)) => about_eq(*a, *b, EQ_GRANULARITY_100),
292
293            (DipF32(a), DipF32(b)) | (PxF32(a), PxF32(b)) => about_eq(*a, *b, EQ_GRANULARITY_100),
294
295            (Factor(a), Factor(b))
296            | (Em(a), Em(b))
297            | (RootEm(a), RootEm(b))
298            | (Leftover(a), Leftover(b))
299            | (ViewportWidth(a), ViewportWidth(b))
300            | (ViewportHeight(a), ViewportHeight(b))
301            | (ViewportMin(a), ViewportMin(b))
302            | (ViewportMax(a), ViewportMax(b)) => a == b,
303
304            (Expr(a), Expr(b)) => a == b,
305
306            (Dip(a), DipF32(b)) | (DipF32(b), Dip(a)) => about_eq(a.to_f32(), *b, EQ_GRANULARITY_100),
307            (Px(a), PxF32(b)) | (PxF32(b), Px(a)) => about_eq(a.0 as f32, *b, EQ_GRANULARITY_100),
308
309            (a, b) => {
310                debug_assert_ne!(std::mem::discriminant(a), std::mem::discriminant(b));
311                false
312            }
313        }
314    }
315}

519            Expr(e) => LengthExpr::Abs(Length::Expr(e.clone())).to_length_checked(),
520        }
521    }
522
523    /// If this length is zero in any finite layout context.
524    ///
525    /// Returns `None` if the value depends on the default value or is an expression.
526    pub fn is_zero(&self) -> Option<bool> {
527        use Length::*;
528        match self {
529            Default => None,
530            Dip(l) => Some(*l == self::Dip::new(0)),
531            Px(l) => Some(*l == self::Px(0)),
532            Pt(l) => Some(about_eq(*l, 0.0, EQ_GRANULARITY)),
533            Factor(f) | Leftover(f) | Em(f) | RootEm(f) | ViewportWidth(f) | ViewportHeight(f) | ViewportMin(f) | ViewportMax(f) => {
534                Some(*f == 0.fct())
535            }
536            DipF32(l) => Some(about_eq(*l, 0.0, EQ_GRANULARITY_100)),
537            PxF32(l) => Some(about_eq(*l, 0.0, EQ_GRANULARITY_100)),
538            Expr(_) => None,
539        }
540    }
541
542    /// Convert a `pt` unit value to [`Px`] given a `scale_factor`.
543    pub fn pt_to_px(pt: f32, scale_factor: Factor) -> Px {
544        let px = Self::pt_to_px_f32(pt, scale_factor);
545        Px(px.round() as i32)
546    }
547
548    /// Same operation as [`pt_to_px`] but without rounding to nearest pixel.
549    ///
550    /// [`pt_to_px`]: Self::pt_to_px
551    pub fn pt_to_px_f32(pt: f32, scale_factor: Factor) -> f32 {
552        pt * Self::PT_TO_DIP * scale_factor.0
553    }
554
555    /// Convert a [`Px`] unit value to a `Pt` value given a `scale_factor`.
556    pub fn px_to_pt(px: Px, scale_factor: Factor) -> f32 {
557        let dip = px.0 as f32 / scale_factor.0;
558        dip / Self::PT_TO_DIP
559    }

561    /// If is [`Default`].
562    ///
563    /// [`Default`]: Length::Default
564    pub fn is_default(&self) -> bool {
565        matches!(self, Length::Default)
566    }
567
568    /// If is [`Default`] or is [`Expr`] that contains defaults.
569    ///
570    /// [`Default`]: Length::Default
571    /// [`Expr`]: Length::Expr
572    pub fn has_default(&self) -> bool {
573        match self {
574            Length::Default => true,
575            Length::Expr(e) => e.has_default(),
576            _ => false,
577        }
578    }
579
580    /// Replaces `self` with `overwrite` if `self` is [`Default`] or replace all defaults in [`Expr`].
581    ///
582    /// [`Default`]: Length::Default
583    /// [`Expr`]: Length::Expr
584    pub fn replace_default(&mut self, overwrite: &Length) {
585        match self {
586            Length::Default => *self = overwrite.clone(),
587            Length::Expr(e) => e.replace_default(overwrite),
588            _ => {}
589        }
590    }
591
592    /// Convert [`PxF32`] to [`Px`] and [`DipF32`] to [`Dip`].
593    ///
594    /// [`PxF32`]: Self::PxF32
595    /// [`Px`]: Self::Px
596    /// [`DipF32`]: Self::DipF32
597    /// [`Dip`]: Self::Dip
598    pub fn round_exact(&mut self) {
599        match self {
600            Length::PxF32(l) => *self = Length::Px(Px(l.round() as i32)),
601            Length::DipF32(l) => *self = Length::Dip(Dip::new_f32(*l)),
602            Length::Expr(e) => e.round_exact(),
603            _ => {}
604        }
605    }

607    /// Gets the total memory allocated by this length.
608    ///
609    /// This includes the sum of all nested [`Length::Expr`] heap memory.
610    pub fn memory_used(&self) -> ByteLength {
611        std::mem::size_of::<Length>().bytes() + self.heap_memory_used()
612    }
613
614    /// Sum total memory used in nested [`Length::Expr`] heap memory.
615    pub fn heap_memory_used(&self) -> ByteLength {
616        if let Length::Expr(e) = self { e.memory_used() } else { 0.bytes() }
617    }

619    /// 96.0 / 72.0
620    const PT_TO_DIP: f32 = 96.0 / 72.0; // 1.3333..;

622impl super::Layout1d for Length {
623    fn layout_dft(&self, axis: LayoutAxis, default: Px) -> Px {
624        use Length::*;
625        match self {
626            Default => default,
627            Dip(l) => l.to_px(LAYOUT.scale_factor()),
628            Px(l) => *l,
629            Pt(l) => Self::pt_to_px(*l, LAYOUT.scale_factor()),
630            Factor(f) => LAYOUT.constraints_for(axis).fill() * f.0,
631            Leftover(f) => {
632                if let Some(l) = LAYOUT.leftover_for(axis) {
633                    l
634                } else {
635                    let fill = LAYOUT.constraints_for(axis).fill();
636                    (fill * f.0).clamp(self::Px(0), fill)
637                }
638            }
639            Em(f) => LAYOUT.font_size() * f.0,
640            RootEm(f) => LAYOUT.root_font_size() * f.0,
641            ViewportWidth(p) => LAYOUT.viewport().width * *p,
642            ViewportHeight(p) => LAYOUT.viewport().height * *p,
643            ViewportMin(p) => LAYOUT.viewport_min() * *p,
644            ViewportMax(p) => LAYOUT.viewport_max() * *p,
645            DipF32(l) => self::Px((l * LAYOUT.scale_factor().0).round() as i32),
646            PxF32(l) => self::Px(l.round() as i32),
647            Expr(e) => e.layout_dft(axis, default),
648        }
649    }
650
651    fn layout_f32_dft(&self, axis: LayoutAxis, default: f32) -> f32 {
652        use Length::*;
653        match self {
654            Default => default,
655            Dip(l) => l.to_f32() * LAYOUT.scale_factor().0,
656            Px(l) => l.0 as f32,
657            Pt(l) => Self::pt_to_px_f32(*l, LAYOUT.scale_factor()),
658            Factor(f) => LAYOUT.constraints_for(axis).fill().0 as f32 * f.0,
659            Leftover(f) => {
660                if let Some(l) = LAYOUT.leftover_for(axis) {
661                    l.0 as f32
662                } else {
663                    let fill = LAYOUT.constraints_for(axis).fill().0 as f32;
664                    (fill * f.0).clamp(0.0, fill)
665                }
666            }
667            Em(f) => LAYOUT.font_size().0 as f32 * f.0,
668            RootEm(f) => LAYOUT.root_font_size().0 as f32 * f.0,
669            ViewportWidth(p) => LAYOUT.viewport().width.0 as f32 * *p,
670            ViewportHeight(p) => LAYOUT.viewport().height.0 as f32 * *p,
671            ViewportMin(p) => LAYOUT.viewport_min().0 as f32 * *p,
672            ViewportMax(p) => LAYOUT.viewport_max().0 as f32 * *p,
673            DipF32(l) => *l * LAYOUT.scale_factor().0,
674            PxF32(l) => *l,
675            Expr(e) => e.layout_f32_dft(axis, default),
676        }
677    }
678
679    fn affect_mask(&self) -> LayoutMask {
680        use Length::*;
681        match self {
682            Default => LayoutMask::DEFAULT_VALUE,
683            Dip(_) => LayoutMask::SCALE_FACTOR,
684            Px(_) => LayoutMask::empty(),
685            Pt(_) => LayoutMask::SCALE_FACTOR,
686            Factor(_) => LayoutMask::CONSTRAINTS,
687            Leftover(_) => LayoutMask::LEFTOVER,
688            Em(_) => LayoutMask::FONT_SIZE,
689            RootEm(_) => LayoutMask::ROOT_FONT_SIZE,
690            ViewportWidth(_) => LayoutMask::VIEWPORT,
691            ViewportHeight(_) => LayoutMask::VIEWPORT,
692            ViewportMin(_) => LayoutMask::VIEWPORT,
693            ViewportMax(_) => LayoutMask::VIEWPORT,
694            DipF32(_) => LayoutMask::SCALE_FACTOR,
695            PxF32(_) => LayoutMask::empty(),
696            Expr(e) => e.affect_mask(),
697        }
698    }

701/// Represents an unresolved [`Length`] expression.
702#[derive(Clone, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
703pub enum LengthExpr {
704    /// Sums the both layout length.
705    Add(Length, Length),
706    /// Subtracts the first layout length from the second.
707    Sub(Length, Length),
708    /// Multiplies the layout length by the factor.
709    Mul(Length, Factor),
710    /// Divide the layout length by the factor.
711    Div(Length, Factor),
712    /// Maximum layout length.
713    Max(Length, Length),
714    /// Minimum layout length.
715    Min(Length, Length),
716    /// Computes the absolute layout length.
717    Abs(Length),
718    /// Negate the layout length.
719    Neg(Length),
720    /// Linear interpolate between lengths by factor.
721    Lerp(Length, Length, Factor),
722}
723impl LengthExpr {
724    /// Gets the total memory allocated by this length expression.
725    ///
726    /// This includes the sum of all nested [`Length::Expr`] heap memory.
727    pub fn memory_used(&self) -> ByteLength {
728        use LengthExpr::*;
729        std::mem::size_of::<LengthExpr>().bytes()
730            + match self {
731                Add(a, b) => a.heap_memory_used() + b.heap_memory_used(),
732                Sub(a, b) => a.heap_memory_used() + b.heap_memory_used(),
733                Mul(a, _) => a.heap_memory_used(),
734                Div(a, _) => a.heap_memory_used(),
735                Max(a, b) => a.heap_memory_used() + b.heap_memory_used(),
736                Min(a, b) => a.heap_memory_used() + b.heap_memory_used(),
737                Abs(a) => a.heap_memory_used(),
738                Neg(a) => a.heap_memory_used(),
739                Lerp(a, b, _) => a.heap_memory_used() + b.heap_memory_used(),
740            }
741    }
742
743    /// Convert to [`Length::Expr`], logs warning for memory use above 1kB, logs error for use > 20kB and collapses to [`Length::zero`].
744    ///
745    /// Every length expression created using the [`std::ops`] uses this method to check the constructed expression. Some operations
746    /// like iterator fold can cause an *expression explosion* where two lengths of different units that cannot
747    /// be evaluated immediately start an expression that subsequently is wrapped in a new expression for each operation done on it.
748    pub fn to_length_checked(self) -> Length {
749        let bytes = self.memory_used();
750        if bytes > 20.kibibytes() {
751            tracing::error!(target: "to_length_checked", "length alloc > 20kB, replaced with zero");
752            return Length::zero();
753        }
754        Length::Expr(Box::new(self))
755    }
756
757    /// If contains a [`Length::Default`] value.
758    pub fn has_default(&self) -> bool {
759        match self {
760            LengthExpr::Add(a, b) | LengthExpr::Sub(a, b) | LengthExpr::Max(a, b) | LengthExpr::Min(a, b) | LengthExpr::Lerp(a, b, _) => {
761                a.has_default() || b.has_default()
762            }
763            LengthExpr::Mul(a, _) | LengthExpr::Div(a, _) | LengthExpr::Abs(a) | LengthExpr::Neg(a) => a.has_default(),
764        }
765    }
766
767    /// Replace all [`Length::Default`] values with `overwrite`.
768    pub fn replace_default(&mut self, overwrite: &Length) {
769        match self {
770            LengthExpr::Add(a, b) | LengthExpr::Sub(a, b) | LengthExpr::Max(a, b) | LengthExpr::Min(a, b) | LengthExpr::Lerp(a, b, _) => {
771                a.replace_default(overwrite);
772                b.replace_default(overwrite);
773            }
774            LengthExpr::Mul(a, _) | LengthExpr::Div(a, _) | LengthExpr::Abs(a) | LengthExpr::Neg(a) => a.replace_default(overwrite),
775        }
776    }
777
778    /// Convert [`PxF32`] to [`Px`] and [`DipF32`] to [`Dip`].
779    ///
780    /// [`PxF32`]: Length::PxF32
781    /// [`Px`]: Length::Px
782    /// [`DipF32`]: Length::DipF32
783    /// [`Dip`]: Length::Dip
784    pub fn round_exact(&mut self) {
785        match self {
786            LengthExpr::Add(a, b) | LengthExpr::Sub(a, b) | LengthExpr::Max(a, b) | LengthExpr::Min(a, b) | LengthExpr::Lerp(a, b, _) => {
787                a.round_exact();
788                b.round_exact();
789            }
790            LengthExpr::Mul(a, _) | LengthExpr::Div(a, _) | LengthExpr::Abs(a) | LengthExpr::Neg(a) => a.round_exact(),
791        }
792    }
793}
794impl super::Layout1d for LengthExpr {
795    fn layout_dft(&self, axis: LayoutAxis, default: Px) -> Px {
796        let l = self.layout_f32_dft(axis, default.0 as f32);
797        Px(l.round() as i32)
798    }
799
800    fn layout_f32_dft(&self, axis: LayoutAxis, default: f32) -> f32 {
801        use LengthExpr::*;
802        match self {
803            Add(a, b) => a.layout_f32_dft(axis, default) + b.layout_f32_dft(axis, default),
804            Sub(a, b) => a.layout_f32_dft(axis, default) - b.layout_f32_dft(axis, default),
805            Mul(l, s) => l.layout_f32_dft(axis, default) * s.0,
806            Div(l, s) => l.layout_f32_dft(axis, default) / s.0,
807            Max(a, b) => {
808                let a = a.layout_f32_dft(axis, default);
809                let b = b.layout_f32_dft(axis, default);
810                a.max(b)
811            }
812            Min(a, b) => {
813                let a = a.layout_f32_dft(axis, default);
814                let b = b.layout_f32_dft(axis, default);
815                a.min(b)
816            }
817            Abs(e) => e.layout_f32_dft(axis, default).abs(),
818            Neg(e) => -e.layout_f32_dft(axis, default),
819            Lerp(a, b, f) => a.layout_f32_dft(axis, default).lerp(&b.layout_f32_dft(axis, default), *f),
820        }
821    }
822
823    fn affect_mask(&self) -> LayoutMask {
824        use LengthExpr::*;
825        match self {
826            Add(a, b) => a.affect_mask() | b.affect_mask(),
827            Sub(a, b) => a.affect_mask() | b.affect_mask(),
828            Mul(a, _) => a.affect_mask(),
829            Div(a, _) => a.affect_mask(),
830            Max(a, b) => a.affect_mask() | b.affect_mask(),
831            Min(a, b) => a.affect_mask() | b.affect_mask(),
832            Abs(a) => a.affect_mask(),
833            Neg(a) => a.affect_mask(),
834            Lerp(a, b, _) => a.affect_mask() | b.affect_mask(),
835        }
836    }

885/// Extension methods for initializing [`Length`] units.
886///
887/// This trait is implemented for [`f32`] and [`u32`] allowing initialization of length units using the `<number>.<unit>()` syntax.
888///
889/// # Examples
890///
891/// ```
892/// # use zng_layout::unit::*;
893/// let font_size = 1.em();
894/// let root_font_size = 1.rem();
895/// let viewport_width = 100.vw();
896/// let viewport_height = 100.vh();
897/// let viewport_min = 100.vmin(); // min(width, height)
898/// let viewport_max = 100.vmax(); // max(width, height)
899///
900/// // other length units not provided by `LengthUnits`:
901///
902/// let exact_size: Length = 500.into();
903/// let relative_size: Length = 100.pct().into(); // FactorUnits
904/// let relative_size: Length = 1.0.fct().into(); // FactorUnits
905/// ```
906pub trait LengthUnits {
907    /// Exact size in device independent pixels.
908    ///
909    /// Returns [`Length::Dip`].
910    fn dip(self) -> Length;
911
912    /// Exact size in device pixels.
913    ///
914    /// Returns [`Length::Px`].
915    fn px(self) -> Length;
916
917    /// Exact size in font units.
918    ///
919    /// Returns [`Length::Pt`].
920    fn pt(self) -> Length;
921
922    /// Factor of the fill length.
923    ///
924    /// This is the same as [`FactorUnits::fct`], but produces a [`Length`] directly. This might be needed
925    /// in places that don't automatically convert [`Factor`] to [`Length`].
926    ///
927    /// Returns [`Length::Factor`].
928    fn fct_l(self) -> Length;
929
930    /// Percentage of the fill length.
931    ///
932    /// This is the same as [`FactorUnits::pct`], but produces a [`Length`] directly. This might be needed
933    /// in places that don't automatically convert [`FactorPercent`] to [`Length`].
934    ///
935    /// Returns [`Length::Factor`].
936    fn pct_l(self) -> Length;
937
938    /// Factor of the font-size of the widget.
939    ///
940    /// Returns [`Length::Em`].
941    fn em(self) -> Length;
942
943    /// Percentage of the font-size of the widget.
944    ///
945    /// Returns [`Length::Em`].
946    fn em_pct(self) -> Length;
947
948    /// Factor of the font-size of the root widget.
949    ///
950    /// Returns [`Length::RootEm`].
951    fn rem(self) -> Length;
952
953    /// Percentage of the font-size of the root widget.
954    ///
955    /// Returns [`Length::RootEm`].
956    fn rem_pct(self) -> Length;
957
958    /// Factor of the width of the nearest viewport ancestor.
959    ///
960    /// Returns [`Length::ViewportWidth`].
961    fn vw(self) -> Length;
962
963    /// Percentage of the width of the nearest viewport ancestor.
964    ///
965    /// Returns [`Length::ViewportWidth`].
966    fn vw_pct(self) -> Length;
967
968    /// Factor of the height of the nearest viewport ancestor.
969    ///
970    /// Returns [`Length::ViewportHeight`].
971    fn vh(self) -> Length;
972
973    /// Percentage of the height of the nearest viewport ancestor.
974    ///
975    /// Returns [`Length::ViewportHeight`].
976    fn vh_pct(self) -> Length;
977
978    /// Factor of the smallest of the nearest viewport's dimensions.
979    ///
980    /// Returns [`Length::ViewportMin`].
981    fn vmin(self) -> Length;
982
983    /// Percentage of the smallest of the nearest viewport's dimensions.
984    ///
985    /// Returns [`Length::ViewportMin`].
986    fn vmin_pct(self) -> Length;
987
988    /// Factor of the largest of the nearest viewport's dimensions.
989    ///
990    /// Returns [`Length::ViewportMax`].
991    fn vmax(self) -> Length;
992
993    /// Percentage of the largest of the nearest viewport's dimensions.
994    ///
995    /// Returns [`Length::ViewportMax`].
996    fn vmax_pct(self) -> Length;
997
998    /// Factor of the leftover layout space.
999    ///
1000    /// Note that this unit must be supported by the parent panel widget and property, otherwise it evaluates
1001    /// like a single item having all the available fill space as leftover space.
1002    ///
1003    /// Returns [`Length::Leftover`].
1004    fn lft(self) -> Length;
1005}
