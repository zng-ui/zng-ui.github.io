8/// Register a `FnOnce(&ProcessStartArgs)` closure to be called on [`init!`].
9///
10/// Components that spawn special process instances implemented on the same executable
11/// can use this macro to inject their own "main" without needing to ask the user to plug an init
12/// function on the executable main. The component can spawn an instance of the current executable
13/// with marker environment variables that identify the component's process.
14///
15/// [`init!`]: crate::init!
16///
17/// # Examples
18///
19/// The example below declares a "main" for a foo component and a function that spawns it.
20///
21/// ```
22/// zng_env::on_process_start!(|args| {
23///     if args.yield_count == 0 { return args.yield_once(); }
24///
25///     if std::env::var("FOO_MARKER").is_ok() {
26///         println!("Spawned as foo!");
27///         zng_env::exit(0);
28///     }
29/// });
30///
31/// fn main() {
32///     zng_env::init!(); // foo_main OR
33///     // normal main
34/// }
35///
36/// pub fn spawn_foo() -> std::io::Result<()> {
37///     std::process::Command::new(std::env::current_exe()?).env("FOO_MARKER", "").spawn()?;
38///     Ok(())
39/// }
40/// ```
41///
42/// Note that the handler yields once, this gives a chance for all handlers to run first before the handler is called again
43/// and takes over the process. It is good practice to yield at least once to ensure handlers that are supported to affect all
44/// processes actually init, as an example, the trace recorder may never start for the process if it does not yield.
45///
46/// Also note the use of custom [`exit`], it is important to call it to collaborate with [`on_process_exit`] handlers.
47///
48/// # App Context
49///
50/// This event happens on the executable process context, before any `APP` context starts, you can use
51/// `zng::app::on_app_start` here to register a handler to be called in the app context, if and when it starts.
52///
53/// # Web Assembly
54///
55/// Crates that declare `on_process_start` must have the [`wasm_bindgen`] dependency to compile for the `wasm32` target.
56///
57/// In `Cargo.toml` add this dependency:
58///
59/// ```toml
60/// [target.'cfg(target_arch = "wasm32")'.dependencies]
61/// wasm-bindgen = "*"
62/// ```
63///
64/// Try to match the version used by `zng-env`.
65///
66/// # Linker Optimizer Issues
67///
68/// The macOS system linker can "optimize" away crates that are only referenced via this macro, that is, a crate dependency
69/// that is not otherwise directly addressed by code. To workaround this issue you can add a bogus reference to the crate code, something
70/// that is not trivial to optimize away. Unfortunately this code must be added on the dependent crate, or on an intermediary dependency,
71/// if your crate is at risk of being used this way please document this issue.
72///
73/// See [`zng#437`] for an example of how to fix this issue.
74///
75/// [`wasm_bindgen`]: https://crates.io/crates/wasm-bindgen
76/// [`zng#437`]: https://github.com/zng-ui/zng/pull/437
77#[macro_export]
78macro_rules! on_process_start {
79    ($closure:expr) => {
80        $crate::__on_process_start! {$closure}
81    };
82}

162fn process_init_impl(handlers: &[fn(&ProcessStartArgs)]) -> MainExitHandler {
163    let process_state = std::mem::replace(
164        &mut *zng_unique_id::hot_static_ref!(PROCESS_LIFETIME_STATE).lock(),
165        ProcessLifetimeState::Inited,
166    );
167    assert_eq!(process_state, ProcessLifetimeState::BeforeInit, "init!() already called");
168
169    let mut yielded = vec![];
170    let mut next_handlers_count = handlers.len();
171    for h in handlers {
172        next_handlers_count -= 1;
173        let args = ProcessStartArgs {
174            next_handlers_count,
175            yield_count: 0,
176            yield_requested: AtomicBool::new(false),
177        };
178        h(&args);
179        if args.yield_requested.load(Ordering::Relaxed) {
180            yielded.push(h);
181            next_handlers_count += 1;
182        }
183    }
184
185    let mut yield_count = 0;
186    while !yielded.is_empty() {
187        yield_count += 1;
188        if yield_count > ProcessStartArgs::MAX_YIELD_COUNT {
189            eprintln!("start handlers requested `yield_start` more them 32 times");
190            break;
191        }
192
193        next_handlers_count = yielded.len();
194        for h in mem::take(&mut yielded) {
195            next_handlers_count -= 1;
196            let args = ProcessStartArgs {
197                next_handlers_count,
198                yield_count,
199                yield_requested: AtomicBool::new(false),
200            };
201            h(&args);
202            if args.yield_requested.load(Ordering::Relaxed) {
203                yielded.push(h);
204                next_handlers_count += 1;
205            }
206        }
207    }
208    MainExitHandler
209}

212pub(crate) fn process_init() -> impl Drop {
213    std::panic::set_hook(Box::new(console_error_panic_hook::hook));
214
215    let window = web_sys::window().expect("cannot 'init!', no window object");
216    let module = js_sys::Reflect::get(&window, &"__zng_env_init_module".into())
217        .expect("cannot 'init!', missing module in 'window.__zng_env_init_module'");
218
219    if module == wasm_bindgen::JsValue::undefined() || module == wasm_bindgen::JsValue::null() {
220        panic!("cannot 'init!', missing module in 'window.__zng_env_init_module'");
221    }
222
223    let module: js_sys::Object = module.into();
224
225    for entry in js_sys::Object::entries(&module) {
226        let entry: js_sys::Array = entry.into();
227        let ident = entry.get(0).as_string().expect("expected ident at entry[0]");
228
229        if ident.starts_with("__zng_env_start_") {
230            let func: js_sys::Function = entry.get(1).into();
231            if let Err(e) = func.call0(&wasm_bindgen::JsValue::NULL) {
232                panic!("'init!' function error, {e:?}");
233            }
234        }
235    }
236
237    process_init_impl(&WASM_INIT.with_borrow_mut(std::mem::take))
238}

240/// Arguments for [`on_process_start`] handlers.
241///
242/// Empty in this release.
243pub struct ProcessStartArgs {
244    /// Number of start handlers yet to run.
245    pub next_handlers_count: usize,
246
247    /// Number of times this handler has yielded.
248    ///
249    /// If this exceeds 32 times the handler is ignored.
250    pub yield_count: u16,
251
252    yield_requested: AtomicBool,
253}
254impl ProcessStartArgs {
255    /// Yield requests after this are ignored.
256    pub const MAX_YIELD_COUNT: u16 = 32;
257
258    /// Let other process start handlers run first.
259    ///
260    /// The handler must call this if it takes over the process and it cannot determinate if it should from the environment.
261    pub fn yield_once(&self) {
262        self.yield_requested.store(true, Ordering::Relaxed);
263    }
264}

279/// Terminates the current process with the specified exit code.
280///
281/// This function must be used instead of `std::process::exit` as it runs the [`on_process_exit`].
282pub fn exit(code: i32) -> ! {
283    run_exit_handlers(code);
284    std::process::exit(code)
285}
286
287fn run_exit_handlers(code: i32) {
288    *zng_unique_id::hot_static_ref!(PROCESS_LIFETIME_STATE).lock() = ProcessLifetimeState::Exiting;
289
290    let on_exit = mem::take(&mut *zng_unique_id::hot_static_ref!(ON_PROCESS_EXIT).lock());
291    let args = ProcessExitArgs { code };
292    for h in on_exit {
293        h(&args);
294    }
295}

304/// Register a `handler` to run once when the current process exits.
305///
306/// Note that the handler is only called if the process is terminated by [`exit`], or by the executable main
307/// function returning if [`init!`] is called on it.
308///
309/// [`init!`]: crate::init!
310pub fn on_process_exit(handler: impl FnOnce(&ProcessExitArgs) + Send + 'static) {
311    zng_unique_id::hot_static_ref!(ON_PROCESS_EXIT).lock().push(Box::new(handler))
312}

314/// Defines the state of the current process instance.
315///
316/// Use [`process_lifetime_state()`] to get.
317#[derive(Debug, Clone, Copy, PartialEq, Eq)]
318pub enum ProcessLifetimeState {
319    /// Init not called yet.
320    BeforeInit,
321    /// Init called and the function where it is called has not returned yet.
322    Inited,
323    /// Init called and the function where it is called is returning.
324    Exiting,
325}

327zng_unique_id::hot_static! {
328    static PROCESS_LIFETIME_STATE: Mutex<ProcessLifetimeState> = Mutex::new(ProcessLifetimeState::BeforeInit);
329}
330zng_unique_id::hot_static! {
331    static PROCESS_NAME: Mutex<Txt> = Mutex::new(Txt::from_static(""));
332}

334/// Get the state of the current process instance.
335pub fn process_lifetime_state() -> ProcessLifetimeState {

337}

339/// Gets a process runtime name.
340///
341/// The primary use of this name is to identify the process in logs, see [`set_process_name`] for details about the logged name.
342/// On set or init the name is logged as an info message "pid: {pid}, name: {name}".
343///
344/// # Common Names
345///
346/// All Zng provided process handlers name the process.
347///
348/// * `"app-process"` - Set by `APP` if no other name was set before the app starts building.
349/// * `"view-process"` - Set by the view-process implementer when running in multi process mode.
350/// * `"crash-handler-process"` - Set by the crash-handler when running with crash handling.
351/// * `"crash-dialog-process"` - Set by the crash-handler on the crash dialog process.
352/// * `"worker-process ({worker_name}, {pid})"` - Set by task worker processes if no name was set before the task runner server starts.
353pub fn process_name() -> Txt {
354    zng_unique_id::hot_static_ref!(PROCESS_NAME).lock().clone()
355}
356
357/// Changes the process runtime name.
358///
359/// This sets [`process_name`] and traces an info message "pid: {pid}, name: {name}". If the same PID is named multiple times
360/// the last name should be used when presenting the process in trace viewers.
361///
362/// The process name ideally should be set only by the [`on_process_start!`] "process takeover" handlers. You can use [`init_process_name`]
363/// to only set the name if it has not been set yet.
364pub fn set_process_name(name: impl Into<Txt>) {
365    set_process_name_impl(name.into(), true);
366}
367
368/// Set the process runtime name if it has not been named yet.
369///
370/// See [`set_process_name`] for more details.
371///
372/// Returns `true` if the name was set.
373pub fn init_process_name(name: impl Into<Txt>) -> bool {
374    set_process_name_impl(name.into(), false)
375}
376
377fn set_process_name_impl(new_name: Txt, replace: bool) -> bool {
378    let mut name = zng_unique_id::hot_static_ref!(PROCESS_NAME).lock();
379    if replace || name.is_empty() {
380        *name = new_name;
381        drop(name);
382        tracing::info!("pid: {}, name: {}", std::process::id(), process_name());
383        true
384    } else {
385        false
386    }
387}

389/// Panics with an standard message if `zng::env::init!()` was not called or was not called correctly.
390pub fn assert_inited() {
391    match process_lifetime_state() {
392        ProcessLifetimeState::BeforeInit => panic!("env not inited, please call `zng::env::init!()` in main"),
393        ProcessLifetimeState::Inited => {}
394        ProcessLifetimeState::Exiting => {
395            panic!("env not inited correctly, please call `zng::env::init!()` at the beginning of the actual main function")
396        }
397    }
