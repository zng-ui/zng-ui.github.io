218    /// Iterator that parses ANSI escape codes.
219    ///
220    /// This is the pull style parser used internally by the [`AnsiText!`] widget.
221    ///
222    /// [`AnsiText!`]: struct@crate::AnsiText
223    pub struct AnsiTextParser<'a> {
224        source: &'a str,
225        /// Current style.
226        pub style: AnsiStyle,
227    }
228    impl<'a> AnsiTextParser<'a> {
229        /// New parsing iterator.
230        pub fn new(source: &'a str) -> Self {
231            Self {
232                source,
233                style: AnsiStyle::default(),
234            }
235        }
236    }
237    impl<'a> Iterator for AnsiTextParser<'a> {
238        type Item = AnsiTxt<'a>;
239
240        fn next(&mut self) -> Option<Self::Item> {
241            const CSI: &str = "\x1b[";
242
243            fn is_esc_end(byte: u8) -> bool {
244                (0x40..=0x7e).contains(&byte)
245            }
246
247            loop {
248                if self.source.is_empty() {
249                    return None;
250                } else if let Some(source) = self.source.strip_prefix(CSI) {
251                    let mut esc_end = 0;
252                    while esc_end < source.len() && !is_esc_end(source.as_bytes()[esc_end]) {
253                        esc_end += 1;
254                    }
255                    esc_end += 1;
256
257                    let (esc, source) = source.split_at(esc_end);
258
259                    let esc = &esc[..(esc.len() - 1)];
260                    self.style.set(esc);
261
262                    self.source = source;
263                    continue;
264                } else if let Some(i) = self.source.find(CSI) {
265                    let (txt, source) = self.source.split_at(i);
266                    self.source = source;
267                    return Some(AnsiTxt {
268                        txt,
269                        style: self.style.clone(),
270                    });
271                } else {
272                    return Some(AnsiTxt {
273                        txt: std::mem::take(&mut self.source),
274                        style: self.style.clone(),
275                    });
276                }
277            }
278        }
279    }

281    impl AnsiStyle {
282        fn set(&mut self, esc_codes: &str) {
283            let mut esc_codes = esc_codes.split(';');
284            while let Some(code) = esc_codes.next() {
285                match code {
286                    "0" => *self = Self::default(),
287                    "1" => self.weight = AnsiWeight::Bold,
288                    "2" => self.weight = AnsiWeight::Faint,
289                    "3" => self.italic = true,
290                    "4" => self.underline = true,
291                    "5" => self.blink = true,
292                    "7" => self.invert_color = true,
293                    "8" => self.hidden = true,
294                    "9" => self.strikethrough = true,
295                    "22" => self.weight = AnsiWeight::Normal,
296                    "23" => self.italic = false,
297                    "24" => self.underline = false,
298                    "25" => self.blink = false,
299                    "27" => self.invert_color = false,
300                    "28" => self.hidden = false,
301                    "29" => self.strikethrough = false,
302                    "30" => self.color = AnsiColor::Black,
303                    "31" => self.color = AnsiColor::Red,
304                    "32" => self.color = AnsiColor::Green,
305                    "33" => self.color = AnsiColor::Yellow,
306                    "34" => self.color = AnsiColor::Blue,
307                    "35" => self.color = AnsiColor::Magenta,
308                    "36" => self.color = AnsiColor::Cyan,
309                    "37" => self.color = AnsiColor::White,
310                    "40" => self.color = AnsiColor::Black,
311                    "41" => self.color = AnsiColor::Red,
312                    "42" => self.color = AnsiColor::Green,
313                    "43" => self.color = AnsiColor::Yellow,
314                    "44" => self.color = AnsiColor::Blue,
315                    "45" => self.color = AnsiColor::Magenta,
316                    "46" => self.color = AnsiColor::Cyan,
317                    "47" => self.color = AnsiColor::White,
318                    "90" => self.color = AnsiColor::BrightBlack,
319                    "91" => self.color = AnsiColor::BrightRed,
320                    "92" => self.color = AnsiColor::BrightGreen,
321                    "93" => self.color = AnsiColor::BrightYellow,
322                    "94" => self.color = AnsiColor::BrightBlue,
323                    "95" => self.color = AnsiColor::BrightMagenta,
324                    "96" => self.color = AnsiColor::BrightCyan,
325                    "97" => self.color = AnsiColor::BrightWhite,
326                    "100" => self.background_color = AnsiColor::BrightBlack,
327                    "101" => self.background_color = AnsiColor::BrightRed,
328                    "102" => self.background_color = AnsiColor::BrightGreen,
329                    "103" => self.background_color = AnsiColor::BrightYellow,
330                    "104" => self.background_color = AnsiColor::BrightBlue,
331                    "105" => self.background_color = AnsiColor::BrightMagenta,
332                    "106" => self.background_color = AnsiColor::BrightCyan,
333                    "107" => self.background_color = AnsiColor::BrightWhite,
334                    "38" | "48" => {
335                        let target = if code == "38" {
336                            &mut self.color
337                        } else {
338                            &mut self.background_color
339                        };
340                        match esc_codes.next() {
341                            Some("5") => {
342                                let c = esc_codes.next().and_then(|c| c.parse().ok()).unwrap_or(0);
343                                *target = AnsiColor::Ansi256(c)
344                            }
345                            Some("2") => {
346                                let r = esc_codes.next().and_then(|c| c.parse().ok()).unwrap_or(0);
347                                let g = esc_codes.next().and_then(|c| c.parse().ok()).unwrap_or(0);
348                                let b = esc_codes.next().and_then(|c| c.parse().ok()).unwrap_or(0);
349
350                                *target = AnsiColor::TrueColor(r, g, b);
351                            }
352                            _ => {}
353                        }
354                    }
355                    _ => (),
356                }
357            }
358        }
359    }

476    /// Default [`TEXT_FN_VAR`].
477    ///
478    /// This view is configured by contextual variables like [`BLINK_INTERVAL_VAR`] and all text variables that are
479    /// not overridden by the ANSI style, like the font.
480    ///
481    /// Returns a `Text!` with the text and style.
482    pub fn default_text_fn(args: TextFnArgs) -> UiNode {
483        let mut text = Text::widget_new();
484
485        widget_set! {
486            &mut text;
487            txt = args.txt;
488        }
489
490        if args.style.background_color != AnsiColor::Black {
491            widget_set! {
492                &mut text;
493                background_color = args.style.background_color;
494            }
495        }
496        if args.style.color != AnsiColor::White {
497            widget_set! {
498                &mut text;
499                font_color = args.style.color;
500            }
501        }
502
503        if args.style.weight != AnsiWeight::Normal {
504            widget_set! {
505                &mut text;
506                font_weight = args.style.weight;
507            }
508        }
509        if args.style.italic {
510            widget_set! {
511                &mut text;
512                font_style = FontStyle::Italic;
513            }
514        }
515
516        if args.style.underline {
517            widget_set! {
518                &mut text;
519                underline = 1, LineStyle::Solid;
520            }
521        }
522        if args.style.strikethrough {
523            widget_set! {
524                &mut text;
525                strikethrough = 1, LineStyle::Solid;
526            }
527        }
528
529        if args.style.invert_color {
530            widget_set! {
531                &mut text;
532                invert_color = true;
533            }
534        }
535
536        if args.style.hidden {
537            widget_set! {
538                &mut text;
539                visibility = Visibility::Hidden;
540            }
541        }
542        if args.style.blink && !args.style.hidden {
543            let opacity = var(1.fct());
544
545            let interval = BLINK_INTERVAL_VAR.get();
546            if interval != Duration::ZERO && interval != Duration::MAX {
547                opacity.step_oci(0.fct(), interval).perm();
548
549                widget_set! {
550                    &mut text;
551                    opacity;
552                }
553            }
554        }
555
556        text.widget_build()
557    }

559    /// Default [`LINE_FN_VAR`].
560    ///
561    /// Returns a `Wrap!` for text with multiple segments, or returns the single segment, or an empty text.
562    pub fn default_line_fn(mut args: LineFnArgs) -> UiNode {
563        if args.text.is_empty() {
564            Text!("")
565        } else if args.text.len() == 1 {
566            args.text.remove(0)
567        } else {
568            Stack! {
569                rich_text = true;
570                direction = StackDirection::start_to_end();
571                children = args.text;
572            }
573        }
574    }

576    /// Default [`PAGE_FN_VAR`].
577    ///
578    /// Returns a `Stack!` for multiple lines, or return the single line, or a nil node.
579    pub fn default_page_fn(mut args: PageFnArgs) -> UiNode {
580        use crate::prelude::*;
581
582        if args.lines.is_empty() {
583            UiNode::nil()
584        } else if args.lines.len() == 1 {
585            args.lines.remove(0)
586        } else {
587            let len = args.lines.len();
588            Stack! {
589                rich_text = true;
590                direction = StackDirection::top_to_bottom();
591                children = args.lines;
592                lazy = LazyMode::lazy_vertical(wgt_fn!(|_| {
593                    let height_sample = zng_wgt_text::node::line_placeholder(50);
594                    zng_wgt_stack::lazy_sample(len, StackDirection::top_to_bottom(), 0, height_sample)
595                }));
596            }
597        }
598    }

672fn generate_ansi(txt: &Var<Txt>) -> UiNode {
673    use ansi_fn::*;
674    use std::mem;
675
676    txt.with(|txt| {
677        let text_fn = TEXT_FN_VAR.get();
678        let line_fn = LINE_FN_VAR.get();
679        let page_fn = PAGE_FN_VAR.get();
680        let panel_fn = PANEL_FN_VAR.get();
681        let lines_per_page = LINES_PER_PAGE_VAR.get() as usize;
682
683        let mut pages = Vec::with_capacity(4);
684        let mut lines = Vec::with_capacity(50);
685
686        for (i, line) in txt.lines().enumerate() {
687            let text = ansi_parse::AnsiTextParser::new(line)
688                .filter_map(|txt| {
689                    text_fn.call_checked(TextFnArgs {
690                        txt: txt.txt.to_txt(),
691                        style: txt.style,
692                    })
693                })
694                .collect();
695
696            lines.push(line_fn(LineFnArgs {
697                index: i as u32,
698                page_index: lines.len() as u32,
699                text,
700            }));
701
702            if lines.len() == lines_per_page {
703                let lines = mem::replace(&mut lines, Vec::with_capacity(50));
704                pages.push(page_fn(PageFnArgs {
705                    index: pages.len() as u32,
706                    lines: lines.into(),
707                }));
708            }
709        }
710
711        if !lines.is_empty() {
712            pages.push(page_fn(PageFnArgs {
713                index: pages.len() as u32,
714                lines: lines.into(),
715            }));
716        }
717
718        panel_fn(PanelFnArgs { pages: pages.into() })
719    })
720}

722/// Implements the ANSI parsing and view generation, configured by contextual properties.
723pub fn ansi_node(txt: impl IntoVar<Txt>) -> UiNode {
724    let txt = txt.into_var();
725    match_node(UiNode::nil(), move |c, op| match op {
726        UiNodeOp::Init => {
727            WIDGET
728                .sub_var(&txt)
729                .sub_var(&TEXT_FN_VAR)
730                .sub_var(&LINE_FN_VAR)
731                .sub_var(&PAGE_FN_VAR)
732                .sub_var(&PANEL_FN_VAR)
733                .sub_var(&LINES_PER_PAGE_VAR)
734                .sub_var(&BLINK_INTERVAL_VAR);
735
736            *c.node() = generate_ansi(&txt);
737        }
738        UiNodeOp::Deinit => {
739            c.deinit();
740            *c.node() = UiNode::nil();
741        }
742        UiNodeOp::Update { .. } => {
743            use ansi_fn::*;
744
745            if txt.is_new()
746                || TEXT_FN_VAR.is_new()
747                || LINE_FN_VAR.is_new()
748                || PAGE_FN_VAR.is_new()
749                || PANEL_FN_VAR.is_new()
750                || LINES_PER_PAGE_VAR.is_new()
751                || BLINK_INTERVAL_VAR.is_new()
752            {
753                c.node().deinit();
754                *c.node() = generate_ansi(&txt);
755                c.node().init();
756                WIDGET.update_info().layout().render();
757            }
758        }
759        _ => {}
760    })
761}
