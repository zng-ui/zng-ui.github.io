31    pub fn notification_dialog(&mut self, app_sender: &AppEventSender, id: DialogId, dialog: Notification) {
32        let mut n = notify_rust::Notification::new();
33        n.summary(&dialog.title).body(&dialog.message);
34        if let Some(t) = dialog.timeout {
35            n.timeout(t);
36        }
37        n.appname(&zng_env::about().app);
38
39        for a in &dialog.actions {
40            n.action(&a.id, &a.label);
41        }
42
43        // we need to use a blocking thread for each notification
44        if Arc::strong_count(&self.running_count) >= 16 {
45            let _ = app_sender.send(AppEvent::Notify(Event::NotificationResponse(
46                id,
47                NotificationResponse::Error("reached limit of active notifications".into()),
48            )));
49            return;
50        }
51
52        let app_sender = app_sender.clone();
53        let actions: HashSet<_> = dialog.actions.into_iter().map(|a| a.id).collect();
54        let count = self.running_count.clone();
55        let _ = std::thread::Builder::new()
56            .name("notify_rust".to_owned())
57            .stack_size(256 * 1024)
58            .spawn(move || {
59                let _count = count;
60                match n.show() {
61                    Ok(handle) => {
62                        let mut r = Txt::from_static("");
63                        handle.wait_for_action(|action| {
64                            r = Txt::from_str(action);
65                        });
66                        let r = if actions.contains(&r) {
67                            NotificationResponse::Action(r)
68                        } else {
69                            NotificationResponse::Dismissed
70                        };
71                        let _ = app_sender.send(AppEvent::Notify(Event::NotificationResponse(id, r)));
72                        // self.handles.push((id, handle));
73                    }
74                    Err(e) => {
75                        use zng_txt::ToTxt as _;
76                        use zng_view_api::Event;
77
78                        use crate::AppEvent;
79
80                        tracing::error!("failed to show notification, {e}");
81                        let _ = app_sender.send(AppEvent::Notify(Event::NotificationResponse(
82                            id,
83                            NotificationResponse::Error(e.to_txt()),
84                        )));
85                    }
86                }
87            });
88    }
