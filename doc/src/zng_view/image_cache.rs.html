98        &mut self,

103            downscale,

113        let resizer = self.resizer.clone();
114        rayon::spawn(move || {

118                resizer,

122                data,
123                max_decoded_len,
124                downscale,

127                parent,

134    pub fn add_pro(
135        &mut self,
136        ImageRequest {
137            format,
138            mut data,
139            max_decoded_len,
140            downscale,
141            mask,
142            entries,
143            parent,
144            ..
145        }: ImageRequest<IpcReceiver<IpcBytes>>,
146    ) -> ImageId {
147        let id = self.image_id_gen.lock().incr();
148        let id_gen = self.image_id_gen.clone();
149        let app_sender = self.app_sender.clone();
150        let resizer = self.resizer.clone();
151        rayon::spawn(move || {
152            // image crate does not implement progressive decoding, just receive all payloads and continue as `add` for now
153
154            let mut notified_header = false;
155
156            let mut all_data = match data.recv_blocking() {
157                Ok(f) => f,
158                Err(e) => {
159                    let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError {
160                        image: id,
161                        error: formatx!("no image data, {e}"),
162                    }));
163                    return;
164                }
165            };
166            if let Ok(n) = data.recv_blocking() {
167                // try parse header early at least
168                #[cfg(feature = "image_any")]
169                if let ImageDataFormat::FileExtension(_) | ImageDataFormat::MimeType(_) | ImageDataFormat::Unknown = &format
170                    && let Ok((fmt, entries)) = Self::decode_container(&format, &all_data[..])
171                    && entries.first() == Some(&(0, ImageEntryKind::Page))
172                    && let Ok(h) = Self::decode_metadata(&all_data[..], fmt, 0)
173                {
174                    let mut size = h.size;
175                    let decoded_len = size.width.0 as u64 * size.height.0 as u64 * 4;
176                    if decoded_len > max_decoded_len {
177                        let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError {
178                            image: id,
179                            error: formatx!(
180                                "image {size:?} needs to allocate {decoded_len} bytes, but max allowed size is {max_decoded_len} bytes",
181                            ),
182                        }));
183                        return;
184                    } else {
185                        // notify metadata already
186
187                        let (d_size, _) = downscale_sizes(downscale.as_ref(), h.size, &[]);
188                        size = d_size.unwrap_or(h.size);
189                        let og_color_size = image_color_type_to_vp(h.og_color_type);
190                        let mut meta = ImageMetadata::new(id, size, mask.is_some(), og_color_size.clone());
191                        meta.density = h.density;
192
193                        let _ = app_sender.send(AppEvent::Notify(Event::ImageMetadataDecoded(meta)));
194                        notified_header = true;
195                    }
196                }
197
198                let mut w = IpcBytes::new_writer_blocking();
199                let try_result = (|| -> std::io::Result<IpcBytes> {
200                    use std::io::Write as _;
201                    w.write_all(&all_data[..])?;
202                    w.write_all(&n[..])?;
203                    while let Ok(n) = data.recv_blocking() {
204                        w.write_all(&n[..])?;
205                    }
206                    w.finish()
207                })();
208                match try_result {
209                    Ok(d) => all_data = d,
210                    Err(e) => {
211                        let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError {
212                            image: id,
213                            error: formatx!("cannot receive image data, {e}"),
214                        }));
215                        return;
216                    }
217                }
218            }
219
220            Self::add_impl(
221                id_gen,
222                app_sender,
223                resizer,
224                notified_header,
225                id,
226                format,
227                all_data,
228                max_decoded_len,
229                downscale,
230                mask,
231                entries,
232                parent,
233            );
234        });
235        id
236    }

238    #[allow(clippy::too_many_arguments)]
239    fn add_impl(
240        id_gen: Arc<Mutex<ImageId>>,
241        app_sender: AppEventSender,
242        resizer: Arc<ResizerCache>,
243        notified_meta: bool,
244
245        id: ImageId,
246        format: ImageDataFormat,
247        data: IpcBytes,
248        max_decoded_len: u64,
249        downscale: Option<ImageDownscaleMode>,
250        mask: Option<ImageMaskMode>,
251        entries: ImageEntriesMode,
252        parent: Option<ImageEntryMetadata>,
253    ) {
254        let data_ref = &data[..];
255        macro_rules! error {
256            ($($tt:tt)*) => {{
257                let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError { image: id, error: formatx!($($tt)*) }));
258            }};
259        }
260        macro_rules! decoded {
261            ($r:tt, $og_color_type:expr, $return_data:expr) => {{
262                let (pixels, size, density, is_opaque, is_mask) = $r;
263                let mut meta = ImageMetadata::new(id, size, is_mask, $og_color_type);
264                meta.density = density;
265                meta.parent = parent;
266                if !notified_meta {
267                    let _ = app_sender.send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())));
268                }
269                let decoded = ImageDecoded::new(meta, pixels, is_opaque);
270                let mut out = if $return_data { Some(decoded.clone()) } else { None };
271                if app_sender.send(AppEvent::ImageCanRender(decoded)).is_err() {
272                    out = None;
273                }
274                out
275            }};
276        }
277
278        match format {
279            ImageDataFormat::Bgra8 {
280                size,
281                density,
282                original_color_type,
283            } => {
284                let downscale_sizes = self::downscale_sizes(downscale.as_ref(), size, &[]);
285
286                let expected_len = size.width.0 as usize * size.height.0 as usize * 4;
287                if data_ref.len() != expected_len {
288                    return error!(
289                        "pixels.len() is not width * height * 4, expected {expected_len}, found {}",
290                        data_ref.len()
291                    );
292                }
293
294                if let Some(mask) = mask {
295                    match Self::convert_bgra8_to_mask(size, data_ref, mask, density, downscale_sizes.0, &resizer) {
296                        Ok(r) => {
297                            if let Some(d) = decoded!(r, original_color_type, !downscale_sizes.1.is_empty()) {
298                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
299                            }
300                        }
301                        Err(e) => error!("{e}"),
302                    }
303                } else {
304                    match Self::downscale_decoded(mask, downscale_sizes.0, &resizer, size, data_ref) {
305                        Ok(Some((size, data_mut))) => match data_mut.finish_blocking() {
306                            Ok(data) => {
307                                let is_opaque = data_ref.chunks_exact(4).all(|c| c[3] == 255);
308                                if let Some(d) = decoded!(
309                                    (data, size, None, is_opaque, false),
310                                    original_color_type,
311                                    !downscale_sizes.1.is_empty()
312                                ) {
313                                    Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
314                                }
315                            }
316                            Err(e) => error!("{e}"),
317                        },
318                        Ok(None) => {
319                            let is_opaque = data_ref.chunks_exact(4).all(|c| c[3] == 255);
320                            if let Some(d) = decoded!(
321                                (data, size, None, is_opaque, false),
322                                original_color_type,
323                                !downscale_sizes.1.is_empty()
324                            ) {
325                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
326                            }
327                        }
328                        Err(e) => error!("{e}"),
329                    }
330                }
331            }
332            ImageDataFormat::A8 { size } => {
333                let downscale_sizes = self::downscale_sizes(downscale.as_ref(), size, &[]);
334
335                let expected_len = size.width.0 as usize * size.height.0 as usize;
336                if data.len() != expected_len {
337                    return error!("pixels.len() is not width * height, expected {expected_len}, found {}", data.len());
338                }
339
340                if mask.is_none() {
341                    match Self::convert_a8_to_bgra8(size, &data, None, downscale_sizes.0, &resizer) {
342                        Ok(r) => {
343                            if let Some(d) = decoded!(r, ColorType::A8, downscale_sizes.1.is_empty()) {
344                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
345                            }
346                        }
347                        Err(e) => error!("{e}"),
348                    }
349                } else {
350                    match Self::downscale_decoded(mask, downscale_sizes.0, &resizer, size, &data) {
351                        Ok(Some((size, data_mut))) => match data_mut.finish_blocking() {
352                            Ok(data) => {
353                                let is_opaque = data.iter().all(|&c| c == 255);
354                                if let Some(d) = decoded!((data, size, None, is_opaque, true), ColorType::A8, !downscale_sizes.1.is_empty())
355                                {
356                                    Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
357                                }
358                            }
359                            Err(e) => error!("{e}"),
360                        },
361                        Ok(None) => {
362                            let is_opaque = data.iter().all(|&c| c == 255);
363                            if let Some(d) = decoded!((data, size, None, is_opaque, true), ColorType::A8, !downscale_sizes.1.is_empty()) {
364                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
365                            }
366                        }
367                        Err(e) => error!("{e}"),
368                    }
369                }
370            }
371            // needs decoding
372            #[cfg(not(feature = "image_any"))]
373            fmt => {
374                let _ = (max_decoded_len, downscale);
375                return error!("no decoder for {fmt:?}");
376            }
377            #[cfg(feature = "image_any")]
378            fmt => {
379                let (fmt, entries_kind) = match Self::decode_container(&fmt, data_ref) {
380                    Ok(r) => r,
381                    Err(e) => return error!("{e}"),
382                };
383                if entries_kind.is_empty() {
384                    return error!("empty container");
385                }
386
387                let mut headers = Vec::with_capacity(entries_kind.len());
388                for (i, kind) in entries_kind {
389                    let h = match Self::decode_metadata(data_ref, fmt, i) {
390                        Ok(h) => h,
391                        Err(e) => return error!("{e}"),
392                    };
393                    headers.push((i, h, kind));
394                }
395                headers.retain(|h| {
396                    let decoded_len = h.1.size.width.0 as u64 * h.1.size.height.0 as u64 * 4;
397                    decoded_len <= max_decoded_len
398                });
399                if headers.is_empty() {
400                    return error!("cannot decode image within the allowed {max_decoded_len} bytes");
401                }
402                let mut page_count = 0;
403                headers.retain(|h| match &h.2 {
404                    ImageEntryKind::Page => {
405                        // always includes the first page
406                        let retain = page_count == 0 || entries.contains(ImageEntriesMode::PAGES);
407                        page_count += 1;
408                        retain
409                    }
410                    ImageEntryKind::Reduced { .. } => {
411                        // reduced requested AND including all pages or is reduced for first page
412                        entries.contains(ImageEntriesMode::REDUCED) && (entries.contains(ImageEntriesMode::PAGES) || page_count == 1)
413                    }
414                    ImageEntryKind::Other { .. } => entries.contains(ImageEntriesMode::OTHER),
415                    _ => unreachable!(),
416                });
417                if headers.is_empty() {
418                    return error!("image container has no page entries");
419                }
420
421                // group entries by page
422                let mut pages: Vec<(_, Vec<_>)> = vec![];
423                let mut others = vec![];
424                for entry in headers {
425                    match &entry.2 {
426                        ImageEntryKind::Page => {
427                            pages.push((entry, vec![]));
428                        }
429                        ImageEntryKind::Reduced { .. } => {
430                            if let Some(p) = pages.last_mut() {
431                                p.1.push(entry);
432                            } else if entries.contains(ImageEntriesMode::OTHER) {
433                                others.push(entry);
434                            }
435                        }
436                        ImageEntryKind::Other { .. } => {
437                            others.push(entry);
438                        }
439                        _ => unreachable!(),
440                    }
441                }
442
443                // collect work, so that metadata events can be send first, before decode/downscale starts
444                enum Task {
445                    Decode {
446                        entry_index: usize,
447                        entry_header: ImageHeader,
448                        id: ImageId,
449                        parent: Option<ImageEntryMetadata>,
450                        notify_meta: bool,
451                        downscale: Option<PxSize>,
452                    },
453                    SynthReduced {
454                        // source: previous task pixels
455                        size: PxSize,
456                        id: ImageId,
457                        parent: ImageEntryMetadata,
458                    },
459                }
460                let mut tasks = vec![];
461                let mut root_entries = 0;
462                for (page, entries) in pages.into_iter() {
463                    let page_size = page.1.size;
464                    let entry_sizes: Vec<_> = entries.iter().map(|e| e.1.size).collect();
465                    let downscale_sizes = self::downscale_sizes(downscale.as_ref(), page_size, &entry_sizes);
466
467                    let notify_meta = !tasks.is_empty() || !notified_meta;
468                    let parent = if tasks.is_empty() {
469                        parent.clone()
470                    } else {
471                        root_entries += 1;
472                        Some(ImageEntryMetadata::new(id, root_entries, page.2))
473                    };
474                    let id = if tasks.is_empty() { id } else { id_gen.lock().incr() };
475                    let mut page_entries = 0;
476                    let page_entries = if tasks.is_empty() { &mut root_entries } else { &mut page_entries };
477
478                    tasks.push(Task::Decode {
479                        entry_index: page.0,
480                        entry_header: page.1,
481                        id,
482                        parent,
483                        notify_meta,
484                        downscale: downscale_sizes.0,
485                    });
486
487                    // downscale are generated from the nearest larger entry, or previous downscale
488                    let mut entries_downscale = vec![vec![]; entries.len()];
489                    for synth in downscale_sizes.1 {
490                        let mut best_i = usize::MAX;
491                        let mut best_dist = Px::MAX;
492                        let page_dist = page_size.width - synth.width;
493                        if page_dist > Px(0) {
494                            best_i = entries.len();
495                            best_dist = page_dist;
496                        }
497                        for (i, entry) in entry_sizes.iter().enumerate() {
498                            let entry_dist = entry.width - synth.width;
499                            if entry_dist > Px(0) && entry_dist < best_dist {
500                                best_i = i;
501                                best_dist = entry_dist;
502                            }
503                        }
504
505                        if let Some(entry) = entries_downscale.get_mut(best_i) {
506                            entry.push(synth);
507                        } else {
508                            debug_assert_ne!(best_i, usize::MAX, "downscale_sizes did not filter correctly");
509                            *page_entries += 1;
510                            tasks.push(Task::SynthReduced {
511                                size: synth,
512                                id: id_gen.lock().incr(),
513                                parent: ImageEntryMetadata::new(id, *page_entries, ImageEntryKind::Reduced { synthetic: true }),
514                            });
515                        }
516                    }
517
518                    for (entry, downscale) in entries.into_iter().zip(entries_downscale) {
519                        *page_entries += 1;
520                        tasks.push(Task::Decode {
521                            entry_index: entry.0,
522                            entry_header: entry.1,
523                            id: id_gen.lock().incr(),
524                            parent: Some(ImageEntryMetadata::new(id, *page_entries, entry.2)),
525                            notify_meta,
526                            downscale: None,
527                        });
528                        for synth in downscale {
529                            *page_entries += 1;
530                            tasks.push(Task::SynthReduced {
531                                size: synth,
532                                id: id_gen.lock().incr(),
533                                parent: ImageEntryMetadata::new(id, *page_entries, ImageEntryKind::Reduced { synthetic: true }),
534                            });
535                        }
536                    }
537                }
538
539                // notify all metadata ahead of decode/downscale
540                let mut tasks_meta = vec![];
541                for task in &tasks {
542                    match task {
543                        Task::Decode {
544                            entry_header,
545                            id,
546                            parent,
547                            notify_meta,
548                            downscale,
549                            ..
550                        } => {
551                            let size = downscale.unwrap_or(entry_header.size);
552                            let og_color_type = image_color_type_to_vp(entry_header.og_color_type);
553                            let mut meta = ImageMetadata::new(*id, size, mask.is_some(), og_color_type);
554                            meta.density = entry_header.density;
555                            meta.parent = parent.clone();
556
557                            if *notify_meta
558                                && app_sender
559                                    .send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())))
560                                    .is_err()
561                            {
562                                return;
563                            }
564
565                            tasks_meta.push(meta);
566                        }
567                        Task::SynthReduced { size, id, parent } => {
568                            let mut meta = tasks_meta.last().cloned().unwrap();
569                            meta.id = *id;
570                            meta.size = *size;
571                            meta.parent = Some(parent.clone());
572
573                            if app_sender
574                                .send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())))
575                                .is_err()
576                            {
577                                return;
578                            }
579
580                            tasks_meta.push(meta);
581                        }
582                    }
583                }
584                let mut others_meta = vec![];
585                for entry in others {
586                    let og_color_type = image_color_type_to_vp(entry.1.og_color_type);
587                    let mut meta = ImageMetadata::new(id_gen.lock().incr(), entry.1.size, mask.is_some(), og_color_type);
588                    meta.density = entry.1.density;
589                    root_entries += 1;
590                    let kind = match entry.2 {
591                        ImageEntryKind::Reduced { .. } => ImageEntryKind::Other {
592                            kind: formatx!("unlinked reduced"),
593                        },
594                        ImageEntryKind::Page => unreachable!(),
595                        other => other,
596                    };
597                    meta.parent = Some(ImageEntryMetadata::new(id, root_entries, kind));
598
599                    if app_sender
600                        .send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())))
601                        .is_err()
602                    {
603                        return;
604                    }
605                    others_meta.push(meta);
606                }
607
608                let data = &data[..];
609                let mut prev_size = PxSize::zero();
610                let mut prev_pixels = IpcBytes::default();
611                let mut prev_is_opaque = false;
612                for (task, mut meta) in tasks.into_iter().zip(tasks_meta) {
613                    match task {
614                        Task::Decode {
615                            entry_index,
616                            entry_header,
617                            downscale,
618                            ..
619                        } => match Self::decode_image(data, fmt, entry_index) {
620                            Ok(img) => match Self::convert_decoded(
621                                img,
622                                mask,
623                                entry_header.density,
624                                entry_header.icc_profile,
625                                downscale,
626                                entry_header.orientation,
627                                &resizer,
628                            ) {
629                                Ok((pixels, size, density, is_opaque, is_mask)) => {
630                                    meta.size = size;
631                                    meta.density = density;
632                                    meta.is_mask = is_mask;
633                                    let decoded = ImageDecoded::new(meta, pixels.clone(), is_opaque);
634                                    if app_sender.send(AppEvent::ImageCanRender(decoded)).is_err() {
635                                        return;
636                                    }
637                                    prev_pixels = pixels;
638                                    prev_size = size;
639                                    prev_is_opaque = is_opaque;
640                                }
641                                Err(e) => {
642                                    return error!("{e}");
643                                }
644                            },
645                            Err(e) => return error!("{e}"),
646                        },
647                        Task::SynthReduced { size, .. } => {
648                            match Self::downscale_decoded(mask, Some(size), &resizer, prev_size, &prev_pixels) {
649                                Ok(r) => {
650                                    let (size, pixels_mut) = r.unwrap();
651                                    match pixels_mut.finish_blocking() {
652                                        Ok(pixels) => {
653                                            meta.size = size;
654                                            let decoded = ImageDecoded::new(meta, pixels.clone(), prev_is_opaque);
655                                            if app_sender.send(AppEvent::ImageCanRender(decoded)).is_err() {
656                                                return;
657                                            }
658                                            prev_pixels = pixels;
659                                            prev_size = size;
660                                        }
661                                        Err(e) => return error!("{e}"),
662                                    }
663                                }
664                                Err(e) => return error!("{e}"),
665                            }
666                        }
667                    }
668                }
669            }
670        }
671    }
672    fn add_downscale_entries(
673        id_gen: Arc<Mutex<ImageId>>,
674        app_sender: AppEventSender,
675        resizer: Arc<ResizerCache>,
676        source: ImageDecoded,
677        entries: Vec<PxSize>,
678    ) {
679        if entries.is_empty() {
680            return;
681        }
682        let mut metas = Vec::with_capacity(entries.len());
683        for (i, entry) in entries.iter().enumerate() {
684            let id = id_gen.lock().incr();
685            let mut meta = source.meta.clone();
686            meta.id = id;
687            meta.size = *entry;
688            meta.parent = Some(ImageEntryMetadata::new(
689                source.meta.id,
690                i,
691                ImageEntryKind::Reduced { synthetic: true },
692            ));
693
694            metas.push(meta.clone());
695            if app_sender.send(AppEvent::Notify(Event::ImageMetadataDecoded(meta))).is_err() {
696                return;
697            }
698        }
699
700        let mut source = source;
701        for meta in metas {
702            let size = meta.size;
703            match Self::add_downscale_entry(&app_sender, &resizer, &source, meta) {
704                Some(downscaled) => {
705                    source.meta.size = size;
706                    source.pixels = downscaled;
707                }
708                None => return,
709            }
710        }
711    }
712    fn add_downscale_entry(
713        app_sender: &AppEventSender,
714        resizer: &ResizerCache,
715        source: &ImageDecoded,
716        entry: ImageMetadata,
717    ) -> Option<IpcBytes> {
718        use fast_image_resize as fr;
719
720        let px_type = if entry.is_mask { fr::PixelType::U8x4 } else { fr::PixelType::U8 };
721        let source_img = fr::images::ImageRef::new(
722            source.meta.size.width.0 as _,
723            source.meta.size.height.0 as _,
724            &source.pixels,
725            px_type,
726        )
727        .unwrap();
728        let mut dest_buf = IpcBytes::new_mut_blocking(entry.size.width.0 as usize * entry.size.height.0 as usize * px_type.size()).ok()?;
729        let mut dest_img =
730            fr::images::Image::from_slice_u8(entry.size.width.0 as _, entry.size.height.0 as _, &mut dest_buf[..], px_type).unwrap();
731
732        let mut resize_opt = fr::ResizeOptions::new();
733        // is already pre multiplied
734        resize_opt.mul_div_alpha = false;
735        // default, best quality
736        resize_opt.algorithm = fr::ResizeAlg::Convolution(fr::FilterType::Lanczos3);
737        // try to reuse cache
738        match resizer.try_lock() {
739            Some(mut r) => r.resize(&source_img, &mut dest_img, Some(&resize_opt)),
740            None => fr::Resizer::new().resize(&source_img, &mut dest_img, Some(&resize_opt)),
741        }
742        .unwrap();
743
744        let pixels = dest_buf.finish_blocking().ok()?;
745
746        app_sender
747            .send(AppEvent::Notify(Event::ImageDecoded(ImageDecoded::new(
748                entry,
749                pixels.clone(),
750                source.is_opaque,
751            ))))
752            .ok()?;
753
754        Some(pixels)
755    }

866impl Image {
867    pub fn descriptor(&self) -> ImageDescriptor {
868        match &*self.0 {
869            ImageData::RawData {
870                size, is_opaque, range, ..
871            } => {
872                // no Webrender mipmaps here, thats only for the GPU,
873                // it does not help with performance rendering gigapixel images scaled to fit
874                let mut flags = webrender::api::ImageDescriptorFlags::empty();
875                if *is_opaque {
876                    flags |= webrender::api::ImageDescriptorFlags::IS_OPAQUE;
877                }
878                let is_mask = size.width.0 as usize * size.height.0 as usize == range.len();
879                ImageDescriptor {
880                    format: if is_mask {
881                        webrender::api::ImageFormat::R8
882                    } else {
883                        webrender::api::ImageFormat::BGRA8
884                    },
885                    size: size.cast().cast_unit(),
886                    stride: None,
887                    offset: 0,
888                    flags,
889                }
890            }
891            ImageData::NativeTexture { .. } => unreachable!(),
892        }
893    }

936    pub fn overflows_wr(&self) -> bool {
937        self.pixels().len() > Self::MAX_LEN
938    }

940    /// Returns the image split in "stripes" that fit the Webrender buffer length constraints.
941    ///
942    /// If the image cannot be split into stripes returns an empty list. This only happens if the image width is absurdly wide.
943    pub fn wr_stripes(&self) -> Box<[Image]> {
944        if !self.overflows_wr() {
945            return Box::new([self.clone()]);
946        }
947
948        match &*self.0 {
949            ImageData::RawData {
950                size,
951                pixels,
952                is_opaque,
953                density,
954                range,
955                stripes,
956                ..
957            } => {
958                debug_assert_eq!(range.len(), pixels.len());
959
960                let mut stripes = stripes.lock();
961                if stripes.is_empty() {
962                    *stripes = self.generate_stripes(*size, pixels, *is_opaque, *density);
963                }
964                (*stripes).clone()
965            }
966            _ => unreachable!(),
967        }
968    }
969    const MAX_LEN: usize = i32::MAX as usize;
970    fn generate_stripes(&self, full_size: PxSize, pixels: &IpcBytes, is_opaque: bool, density: Option<PxDensity2d>) -> Box<[Image]> {
971        let w = full_size.width.0 as usize * 4;
972        if w > Self::MAX_LEN {
973            tracing::error!("renderer does not support images with width * 4 > {}", Self::MAX_LEN);
974            return Box::new([]);
975        }
976
977        // find proportional split that fits, to avoid having the last stripe be to thin
978        let full_height = full_size.height.0 as usize;
979        let mut stripe_height = full_height / 2;
980        while w * stripe_height > Self::MAX_LEN {
981            stripe_height /= 2;
982        }
983        let stripe_len = w * stripe_height;
984        let stripes_len = full_height.div_ceil(stripe_height);
985        let stripe_height = Px(stripe_height as _);
986
987        let mut stripes = Vec::with_capacity(stripes_len);
988
989        for i in 0..stripes_len {
990            let y = stripe_height * Px(i as _);
991            let mut size = full_size;
992            size.height = stripe_height.min(full_size.height - y);
993
994            let offset = stripe_len * i;
995            let range = offset..((offset + stripe_len).min(pixels.len()));
996
997            let stripe = Image(Arc::new(ImageData::RawData {
998                size,
999                pixels: pixels.clone(),
1000                is_opaque,
1001                density,
1002                range,
1003                // always empty
1004                stripes: Mutex::new(Box::new([])),
1005            }));
1006
1007            stripes.push(stripe);
1008        }
1009
1010        stripes.into_boxed_slice()
1011    }
