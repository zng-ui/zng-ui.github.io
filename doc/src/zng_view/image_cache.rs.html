129        &mut self,
130        ImageRequest {
131            format,
132            data,
133            max_decoded_len,
134            downscale,
135            mask,
136            entries,
137            parent,
138            ..
139        }: ImageRequest<IpcBytes>,
140    ) -> ImageId {
141        let id = self.image_id_gen.lock().incr();
142        let id_gen = self.image_id_gen.clone();
143        let app_sender = self.app_sender.clone();

167                downscale,
168                mask,
169                entries,
170                parent,
171            );
172        });

177    pub fn add_pro(
178        &mut self,
179        ImageRequest {
180            format,
181            mut data,
182            max_decoded_len,
183            downscale,
184            mask,
185            entries,
186            parent,
187            ..
188        }: ImageRequest<IpcReceiver<IpcBytes>>,
189    ) -> ImageId {
190        let id = self.image_id_gen.lock().incr();
191        let id_gen = self.image_id_gen.clone();
192        let app_sender = self.app_sender.clone();
193        let resizer = self.resizer.clone();
194        #[cfg(feature = "image_cur")]
195        let image_cur_ext_id = self.image_cur_ext_id;
196        #[cfg(feature = "image_meta_exif")]
197        let exif_ext_id = self.exif_ext_id;
198        #[cfg(feature = "image_meta_icc")]
199        let icc_ext_id = self.icc_ext_id;
200        rayon::spawn(move || {
201            // image crate does not implement progressive decoding, just receive all payloads and continue as `add` for now
202
203            let mut notified_header = false;
204
205            let mut all_data = match data.recv_blocking() {
206                Ok(f) => f,
207                Err(e) => {
208                    let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError {
209                        image: id,
210                        error: formatx!("no image data, {e}"),
211                    }));
212                    return;
213                }
214            };
215            if let Ok(n) = data.recv_blocking() {
216                // try parse header early at least
217                #[cfg(feature = "image_any")]
218                if let ImageDataFormat::FileExtension(_) | ImageDataFormat::MimeType(_) | ImageDataFormat::Unknown = &format
219                    && let Ok((fmt, entries)) = Self::decode_container(&format, &all_data[..])
220                    && entries.first() == Some(&(0, ImageEntryKind::Page))
221                    && let Ok(h) = Self::decode_metadata(&all_data[..], fmt, 0)
222                {
223                    let mut size = h.size;
224                    let decoded_len = size.width.0 as u64 * size.height.0 as u64 * 4;
225                    if decoded_len > max_decoded_len {
226                        let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError {
227                            image: id,
228                            error: formatx!(
229                                "image {size:?} needs to allocate {decoded_len} bytes, but max allowed size is {max_decoded_len} bytes",
230                            ),
231                        }));
232                        return;
233                    } else {
234                        // notify metadata already
235
236                        let (d_size, _) = downscale_sizes(downscale.as_ref(), h.size, &[]);
237                        size = d_size.unwrap_or(h.size);
238                        let og_color_size = image_color_type_to_vp(h.og_color_type);
239                        let mut meta = ImageMetadata::new(id, size, mask.is_some(), og_color_size.clone());
240                        meta.density = h.density;
241                        meta.format_name = h.format_name;
242                        #[cfg(feature = "image_cur")]
243                        downscale_hotspot(image_cur_ext_id, size, &mut meta, h.size, h.cur_hotspot);
244                        #[cfg(feature = "image_meta_exif")]
245                        if let Some(exif) = &h.exif
246                            && !exif.is_empty()
247                        {
248                            meta.extensions
249                                .push((exif_ext_id, zng_view_api::api_extension::ApiExtensionPayload(exif.clone())));
250                        }
251                        #[cfg(feature = "image_meta_icc")]
252                        if let Some(icc) = &h.icc_profile
253                            && let Ok(icc) = icc.icc()
254                        {
255                            meta.extensions
256                                .push((icc_ext_id, zng_view_api::api_extension::ApiExtensionPayload(icc)));
257                        }
258
259                        let _ = app_sender.send(AppEvent::Notify(Event::ImageMetadataDecoded(meta)));
260                        notified_header = true;
261                    }
262                }
263
264                let mut w = IpcBytes::new_writer_blocking();
265                let try_result = (|| -> std::io::Result<IpcBytes> {
266                    use std::io::Write as _;
267                    w.write_all(&all_data[..])?;
268                    w.write_all(&n[..])?;
269                    while let Ok(n) = data.recv_blocking() {
270                        w.write_all(&n[..])?;
271                    }
272                    w.finish()
273                })();
274                match try_result {
275                    Ok(d) => all_data = d,
276                    Err(e) => {
277                        let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError {
278                            image: id,
279                            error: formatx!("cannot receive image data, {e}"),
280                        }));
281                        return;
282                    }
283                }
284            }
285
286            Self::add_impl(
287                id_gen,
288                app_sender,
289                resizer,
290                notified_header,
291                #[cfg(feature = "image_cur")]
292                image_cur_ext_id,
293                #[cfg(feature = "image_meta_exif")]
294                exif_ext_id,
295                #[cfg(feature = "image_meta_icc")]
296                icc_ext_id,
297                id,
298                format,
299                all_data,
300                max_decoded_len,
301                downscale,
302                mask,
303                entries,
304                parent,
305            );
306        });
307        id
308    }

310    #[allow(clippy::too_many_arguments)]
311    fn add_impl(
312        id_gen: Arc<Mutex<ImageId>>,
313        app_sender: AppEventSender,
314        resizer: Arc<ResizerCache>,
315        notified_meta: bool,
316        #[cfg(feature = "image_cur")] image_cur_ext_id: zng_view_api::api_extension::ApiExtensionId,
317        #[cfg(feature = "image_meta_exif")] exif_ext_id: zng_view_api::api_extension::ApiExtensionId,
318        #[cfg(feature = "image_meta_icc")] icc_ext_id: zng_view_api::api_extension::ApiExtensionId,
319
320        id: ImageId,
321        format: ImageDataFormat,
322        data: IpcBytes,
323        max_decoded_len: u64,
324        downscale: Option<ImageDownscaleMode>,
325        mask: Option<ImageMaskMode>,
326        entries: ImageEntriesMode,
327        parent: Option<ImageEntryMetadata>,
328    ) {
329        let data_ref = &data[..];
330        macro_rules! error {
331            ($($tt:tt)*) => {{
332                let _ = app_sender.send(AppEvent::Notify(Event::ImageDecodeError { image: id, error: formatx!($($tt)*) }));
333            }};
334        }
335        macro_rules! decoded {
336            ($r:tt, $og_color_type:expr, $return_data:expr) => {{
337                let (pixels, size, density, is_opaque, is_mask) = $r;
338                let mut meta = ImageMetadata::new(id, size, is_mask, $og_color_type);
339                meta.density = density;
340                meta.parent = parent;
341                meta.format_name = meta.original_color_type.name.clone();
342                if !notified_meta {
343                    let _ = app_sender.send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())));
344                }
345                let decoded = ImageDecoded::new(meta, pixels, is_opaque);
346                let mut out = if $return_data { Some(decoded.clone()) } else { None };
347                if app_sender.send(AppEvent::ImageCanRender(decoded)).is_err() {
348                    out = None;
349                }
350                out
351            }};
352        }
353
354        match format {
355            ImageDataFormat::Bgra8 {
356                size,
357                density,
358                original_color_type,
359            } => {
360                let downscale_sizes = self::downscale_sizes(downscale.as_ref(), size, &[]);
361
362                let expected_len = size.width.0 as usize * size.height.0 as usize * 4;
363                if data_ref.len() != expected_len {
364                    return error!(
365                        "pixels.len() is not width * height * 4, expected {expected_len}, found {}",
366                        data_ref.len()
367                    );
368                }
369
370                if let Some(mask) = mask {
371                    match Self::convert_bgra8_to_mask(size, data_ref, mask, density, downscale_sizes.0, &resizer) {
372                        Ok(r) => {
373                            if let Some(d) = decoded!(r, original_color_type, !downscale_sizes.1.is_empty()) {
374                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
375                            }
376                        }
377                        Err(e) => error!("{e}"),
378                    }
379                } else {
380                    match Self::downscale_decoded(mask, downscale_sizes.0, &resizer, size, data_ref) {
381                        Ok(Some((size, data_mut))) => match data_mut.finish_blocking() {
382                            Ok(data) => {
383                                let is_opaque = data_ref.chunks_exact(4).all(|c| c[3] == 255);
384                                if let Some(d) = decoded!(
385                                    (data, size, None, is_opaque, false),
386                                    original_color_type,
387                                    !downscale_sizes.1.is_empty()
388                                ) {
389                                    Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
390                                }
391                            }
392                            Err(e) => error!("{e}"),
393                        },
394                        Ok(None) => {
395                            let is_opaque = data_ref.chunks_exact(4).all(|c| c[3] == 255);
396                            if let Some(d) = decoded!(
397                                (data, size, None, is_opaque, false),
398                                original_color_type,
399                                !downscale_sizes.1.is_empty()
400                            ) {
401                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
402                            }
403                        }
404                        Err(e) => error!("{e}"),
405                    }
406                }
407            }
408            ImageDataFormat::A8 { size } => {
409                let downscale_sizes = self::downscale_sizes(downscale.as_ref(), size, &[]);
410
411                let expected_len = size.width.0 as usize * size.height.0 as usize;
412                if data.len() != expected_len {
413                    return error!("pixels.len() is not width * height, expected {expected_len}, found {}", data.len());
414                }
415
416                if mask.is_none() {
417                    match Self::convert_a8_to_bgra8(size, &data, None, downscale_sizes.0, &resizer) {
418                        Ok(r) => {
419                            if let Some(d) = decoded!(r, ColorType::A8, downscale_sizes.1.is_empty()) {
420                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
421                            }
422                        }
423                        Err(e) => error!("{e}"),
424                    }
425                } else {
426                    match Self::downscale_decoded(mask, downscale_sizes.0, &resizer, size, &data) {
427                        Ok(Some((size, data_mut))) => match data_mut.finish_blocking() {
428                            Ok(data) => {
429                                let is_opaque = data.iter().all(|&c| c == 255);
430                                if let Some(d) = decoded!((data, size, None, is_opaque, true), ColorType::A8, !downscale_sizes.1.is_empty())
431                                {
432                                    Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
433                                }
434                            }
435                            Err(e) => error!("{e}"),
436                        },
437                        Ok(None) => {
438                            let is_opaque = data.iter().all(|&c| c == 255);
439                            if let Some(d) = decoded!((data, size, None, is_opaque, true), ColorType::A8, !downscale_sizes.1.is_empty()) {
440                                Self::add_downscale_entries(id_gen, app_sender, resizer, d, downscale_sizes.1);
441                            }
442                        }
443                        Err(e) => error!("{e}"),
444                    }
445                }
446            }
447            // needs decoding
448            #[cfg(not(feature = "image_any"))]
449            fmt => {
450                let _ = (max_decoded_len, downscale);
451                return error!("no decoder for {fmt:?}");
452            }
453            #[cfg(feature = "image_any")]
454            fmt => {
455                let (fmt, entries_kind) = match Self::decode_container(&fmt, data_ref) {
456                    Ok(r) => r,
457                    Err(e) => return error!("{e}"),
458                };
459                if entries_kind.is_empty() {
460                    return error!("empty container");
461                }
462
463                let mut headers = Vec::with_capacity(entries_kind.len());
464                for (i, kind) in entries_kind {
465                    let h = match Self::decode_metadata(data_ref, fmt, i) {
466                        Ok(h) => h,
467                        Err(e) => return error!("{e}"),
468                    };
469                    headers.push((i, h, kind));
470                }
471                headers.retain(|h| {
472                    let decoded_len = h.1.size.width.0 as u64 * h.1.size.height.0 as u64 * 4;
473                    decoded_len <= max_decoded_len
474                });
475                if headers.is_empty() {
476                    return error!("cannot decode image within the allowed {max_decoded_len} bytes");
477                }
478                let mut page_count = 0;
479                headers.retain(|h| match &h.2 {
480                    ImageEntryKind::Page => {
481                        // always includes the first page
482                        let retain = page_count == 0 || entries.contains(ImageEntriesMode::PAGES);
483                        page_count += 1;
484                        retain
485                    }
486                    ImageEntryKind::Reduced { .. } => {
487                        // reduced requested AND including all pages or is reduced for first page
488                        entries.contains(ImageEntriesMode::REDUCED) && (entries.contains(ImageEntriesMode::PAGES) || page_count == 1)
489                    }
490                    ImageEntryKind::Other { .. } => entries.contains(ImageEntriesMode::OTHER),
491                    _ => unreachable!(),
492                });
493                if headers.is_empty() {
494                    return error!("image container has no page entries");
495                }
496
497                // group entries by page
498                let mut pages: Vec<(_, Vec<_>)> = vec![];
499                let mut others = vec![];
500                for entry in headers {
501                    match &entry.2 {
502                        ImageEntryKind::Page => {
503                            pages.push((entry, vec![]));
504                        }
505                        ImageEntryKind::Reduced { .. } => {
506                            if let Some(p) = pages.last_mut() {
507                                p.1.push(entry);
508                            } else if entries.contains(ImageEntriesMode::OTHER) {
509                                others.push(entry);
510                            }
511                        }
512                        ImageEntryKind::Other { .. } => {
513                            others.push(entry);
514                        }
515                        _ => unreachable!(),
516                    }
517                }
518
519                // collect work, so that metadata events can be send first, before decode/downscale starts
520                enum Task {
521                    Decode {
522                        entry_index: usize,
523                        entry_header: ImageHeader,
524                        id: ImageId,
525                        parent: Option<ImageEntryMetadata>,
526                        notify_meta: bool,
527                        downscale: Option<PxSize>,
528                    },
529                    SynthReduced {
530                        // source: previous task pixels
531                        size: PxSize,
532                        #[cfg(feature = "image_cur")]
533                        page_cur: (PxSize, Option<PxPoint>), // (page_size, pager_cur_hotspot)
534                        id: ImageId,
535                        parent: ImageEntryMetadata,
536                    },
537                }
538                let mut tasks = vec![];
539                let mut root_entries = 0;
540                for (page, entries) in pages.into_iter() {
541                    let page_size = page.1.size;
542                    let entry_sizes: Vec<_> = entries.iter().map(|e| e.1.size).collect();
543                    let downscale_sizes = self::downscale_sizes(downscale.as_ref(), page_size, &entry_sizes);
544
545                    let notify_meta = !tasks.is_empty() || !notified_meta;
546                    let parent = if tasks.is_empty() {
547                        parent.clone()
548                    } else {
549                        root_entries += 1;
550                        Some(ImageEntryMetadata::new(id, root_entries, page.2))
551                    };
552                    let id = if tasks.is_empty() { id } else { id_gen.lock().incr() };
553                    let mut page_entries = 0;
554                    let page_entries = if tasks.is_empty() { &mut root_entries } else { &mut page_entries };
555                    let page_cur_hotspot = page.1.cur_hotspot;
556
557                    tasks.push(Task::Decode {
558                        entry_index: page.0,
559                        entry_header: page.1,
560                        id,
561                        parent,
562                        notify_meta,
563                        downscale: downscale_sizes.0,
564                    });
565
566                    // downscale are generated from the nearest larger entry, or previous downscale
567                    let mut entries_downscale = vec![vec![]; entries.len()];
568                    for synth in downscale_sizes.1 {
569                        let mut best_i = usize::MAX;
570                        let mut best_dist = Px::MAX;
571                        let page_dist = page_size.width - synth.width;
572                        if page_dist > Px(0) {
573                            best_i = entries.len();
574                            best_dist = page_dist;
575                        }
576                        for (i, entry) in entry_sizes.iter().enumerate() {
577                            let entry_dist = entry.width - synth.width;
578                            if entry_dist > Px(0) && entry_dist < best_dist {
579                                best_i = i;
580                                best_dist = entry_dist;
581                            }
582                        }
583
584                        if let Some(entry) = entries_downscale.get_mut(best_i) {
585                            entry.push(synth);
586                        } else {
587                            debug_assert_ne!(best_i, usize::MAX, "downscale_sizes did not filter correctly");
588                            *page_entries += 1;
589                            tasks.push(Task::SynthReduced {
590                                size: synth,
591                                #[cfg(feature = "image_cur")]
592                                page_cur: (page_size, page_cur_hotspot),
593                                id: id_gen.lock().incr(),
594                                parent: ImageEntryMetadata::new(id, *page_entries, ImageEntryKind::Reduced { synthetic: true }),
595                            });
596                        }
597                    }
598
599                    for (entry, downscale) in entries.into_iter().zip(entries_downscale) {
600                        *page_entries += 1;
601                        tasks.push(Task::Decode {
602                            entry_index: entry.0,
603                            entry_header: entry.1,
604                            id: id_gen.lock().incr(),
605                            parent: Some(ImageEntryMetadata::new(id, *page_entries, entry.2)),
606                            notify_meta,
607                            downscale: None,
608                        });
609                        for synth in downscale {
610                            *page_entries += 1;
611                            tasks.push(Task::SynthReduced {
612                                size: synth,
613                                #[cfg(feature = "image_cur")]
614                                page_cur: (page_size, page_cur_hotspot),
615                                id: id_gen.lock().incr(),
616                                parent: ImageEntryMetadata::new(id, *page_entries, ImageEntryKind::Reduced { synthetic: true }),
617                            });
618                        }
619                    }
620                }
621
622                // notify all metadata ahead of decode/downscale
623                let mut tasks_meta = vec![];
624                for task in &tasks {
625                    match task {
626                        Task::Decode {
627                            entry_header,
628                            id,
629                            parent,
630                            notify_meta,
631                            downscale,
632                            ..
633                        } => {
634                            let size = downscale.unwrap_or(entry_header.size);
635                            let og_color_type = image_color_type_to_vp(entry_header.og_color_type);
636                            let mut meta = ImageMetadata::new(*id, size, mask.is_some(), og_color_type);
637                            meta.density = entry_header.density;
638                            meta.parent = parent.clone();
639                            meta.format_name = entry_header.format_name.clone();
640                            #[cfg(feature = "image_cur")]
641                            downscale_hotspot(image_cur_ext_id, size, &mut meta, entry_header.size, entry_header.cur_hotspot);
642                            #[cfg(feature = "image_meta_exif")]
643                            if let Some(exif) = &entry_header.exif
644                                && !exif.is_empty()
645                            {
646                                meta.extensions
647                                    .push((exif_ext_id, zng_view_api::api_extension::ApiExtensionPayload(exif.clone())));
648                            }
649                            #[cfg(feature = "image_meta_icc")]
650                            if let Some(icc) = &entry_header.icc_profile
651                                && let Ok(icc) = icc.icc()
652                            {
653                                meta.extensions
654                                    .push((icc_ext_id, zng_view_api::api_extension::ApiExtensionPayload(icc)));
655                            }
656
657                            if *notify_meta
658                                && app_sender
659                                    .send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())))
660                                    .is_err()
661                            {
662                                return;
663                            }
664
665                            tasks_meta.push(meta);
666                        }
667                        Task::SynthReduced {
668                            size,
669                            id,
670                            parent,
671                            #[cfg(feature = "image_cur")]
672                            page_cur,
673                        } => {
674                            let mut meta = tasks_meta.last().cloned().unwrap();
675                            meta.id = *id;
676                            meta.size = *size;
677                            meta.parent = Some(parent.clone());
678                            #[cfg(feature = "image_cur")]
679                            downscale_hotspot(image_cur_ext_id, *size, &mut meta, page_cur.0, page_cur.1);
680
681                            if app_sender
682                                .send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())))
683                                .is_err()
684                            {
685                                return;
686                            }
687
688                            tasks_meta.push(meta);
689                        }
690                    }
691                }
692                let mut others_meta = vec![];
693                for entry in others {
694                    let og_color_type = image_color_type_to_vp(entry.1.og_color_type);
695                    let mut meta = ImageMetadata::new(id_gen.lock().incr(), entry.1.size, mask.is_some(), og_color_type);
696                    meta.density = entry.1.density;
697                    meta.format_name = entry.1.format_name.clone();
698                    root_entries += 1;
699                    let kind = match entry.2 {
700                        ImageEntryKind::Reduced { .. } => ImageEntryKind::Other {
701                            kind: formatx!("unlinked reduced"),
702                        },
703                        ImageEntryKind::Page => unreachable!(),
704                        other => other,
705                    };
706                    meta.parent = Some(ImageEntryMetadata::new(id, root_entries, kind));
707
708                    if app_sender
709                        .send(AppEvent::Notify(Event::ImageMetadataDecoded(meta.clone())))
710                        .is_err()
711                    {
712                        return;
713                    }
714                    others_meta.push(meta);
715                }
716
717                let data = &data[..];
718                let mut prev_size = PxSize::zero();
719                let mut prev_pixels = IpcBytes::default();
720                let mut prev_is_opaque = false;
721                for (task, mut meta) in tasks.into_iter().zip(tasks_meta) {
722                    match task {
723                        Task::Decode {
724                            entry_index,
725                            entry_header,
726                            downscale,
727                            ..
728                        } => match Self::decode_image(data, fmt, entry_index) {
729                            Ok(img) => match Self::convert_decoded(
730                                img,
731                                mask,
732                                entry_header.density,
733                                entry_header.icc_profile.as_ref(),
734                                downscale,
735                                entry_header.orientation,
736                                &resizer,
737                            ) {
738                                Ok((pixels, size, density, is_opaque, is_mask)) => {
739                                    meta.size = size;
740                                    meta.density = density;
741                                    meta.format_name = entry_header.format_name.clone();
742                                    meta.is_mask = is_mask;
743                                    #[cfg(feature = "image_meta_exif")]
744                                    if let Some(exif) = entry_header.exif
745                                        && !exif.is_empty()
746                                    {
747                                        meta.extensions
748                                            .push((exif_ext_id, zng_view_api::api_extension::ApiExtensionPayload(exif)));
749                                    }
750                                    #[cfg(feature = "image_meta_icc")]
751                                    if let Some(icc) = &entry_header.icc_profile
752                                        && let Ok(icc) = icc.icc()
753                                    {
754                                        meta.extensions
755                                            .push((icc_ext_id, zng_view_api::api_extension::ApiExtensionPayload(icc)));
756                                    }
757
758                                    #[cfg(feature = "image_cur")]
759                                    downscale_hotspot(image_cur_ext_id, size, &mut meta, entry_header.size, entry_header.cur_hotspot);
760
761                                    let decoded = ImageDecoded::new(meta, pixels.clone(), is_opaque);
762                                    if app_sender.send(AppEvent::ImageCanRender(decoded)).is_err() {
763                                        return;
764                                    }
765                                    prev_pixels = pixels;
766                                    prev_size = size;
767                                    prev_is_opaque = is_opaque;
768                                }
769                                Err(e) => {
770                                    return error!("{e}");
771                                }
772                            },
773                            Err(e) => return error!("{e}"),
774                        },
775                        Task::SynthReduced {
776                            size,
777                            #[cfg(feature = "image_cur")]
778                            page_cur,
779                            ..
780                        } => match Self::downscale_decoded(mask, Some(size), &resizer, prev_size, &prev_pixels) {
781                            Ok(r) => {
782                                let (size, pixels_mut) = r.unwrap();
783                                match pixels_mut.finish_blocking() {
784                                    Ok(pixels) => {
785                                        meta.size = size;
786                                        #[cfg(feature = "image_cur")]
787                                        downscale_hotspot(image_cur_ext_id, size, &mut meta, page_cur.0, page_cur.1);
788                                        let decoded = ImageDecoded::new(meta, pixels.clone(), prev_is_opaque);
789                                        if app_sender.send(AppEvent::ImageCanRender(decoded)).is_err() {
790                                            return;
791                                        }
792                                        prev_pixels = pixels;
793                                        prev_size = size;
794                                    }
795                                    Err(e) => return error!("{e}"),
796                                }
797                            }
798                            Err(e) => return error!("{e}"),
799                        },
800                    }
801                }
802            }
803        }
804    }
805    fn add_downscale_entries(
806        id_gen: Arc<Mutex<ImageId>>,
807        app_sender: AppEventSender,
808        resizer: Arc<ResizerCache>,
809        source: ImageDecoded,
810        entries: Vec<PxSize>,
811    ) {
812        if entries.is_empty() {
813            return;
814        }
815        let mut metas = Vec::with_capacity(entries.len());
816        for (i, entry) in entries.iter().enumerate() {
817            let id = id_gen.lock().incr();
818            let mut meta = source.meta.clone();
819            meta.id = id;
820            meta.size = *entry;
821            meta.parent = Some(ImageEntryMetadata::new(
822                source.meta.id,
823                i,
824                ImageEntryKind::Reduced { synthetic: true },
825            ));
826
827            metas.push(meta.clone());
828            if app_sender.send(AppEvent::Notify(Event::ImageMetadataDecoded(meta))).is_err() {
829                return;
830            }
831        }
832
833        let mut source = source;
834        for meta in metas {
835            let size = meta.size;
836            match Self::add_downscale_entry(&app_sender, &resizer, &source, meta) {
837                Some(downscaled) => {
838                    source.meta.size = size;
839                    source.pixels = downscaled;
840                }
841                None => return,
842            }
843        }
844    }
845    fn add_downscale_entry(
846        app_sender: &AppEventSender,
847        resizer: &ResizerCache,
848        source: &ImageDecoded,
849        entry: ImageMetadata,
850    ) -> Option<IpcBytes> {
851        use fast_image_resize as fr;
852
853        let px_type = if entry.is_mask { fr::PixelType::U8x4 } else { fr::PixelType::U8 };
854        let source_img = fr::images::ImageRef::new(
855            source.meta.size.width.0 as _,
856            source.meta.size.height.0 as _,
857            &source.pixels,
858            px_type,
859        )
860        .unwrap();
861        let mut dest_buf = IpcBytes::new_mut_blocking(entry.size.width.0 as usize * entry.size.height.0 as usize * px_type.size()).ok()?;
862        let mut dest_img =
863            fr::images::Image::from_slice_u8(entry.size.width.0 as _, entry.size.height.0 as _, &mut dest_buf[..], px_type).unwrap();
864
865        let mut resize_opt = fr::ResizeOptions::new();
866        // is already pre multiplied
867        resize_opt.mul_div_alpha = false;
868        // default, best quality
869        resize_opt.algorithm = fr::ResizeAlg::Convolution(fr::FilterType::Lanczos3);
870        // try to reuse cache
871        match resizer.try_lock() {
872            Some(mut r) => r.resize(&source_img, &mut dest_img, Some(&resize_opt)),
873            None => fr::Resizer::new().resize(&source_img, &mut dest_img, Some(&resize_opt)),
874        }
875        .unwrap();
876
877        let pixels = dest_buf.finish_blocking().ok()?;
878
879        app_sender
880            .send(AppEvent::Notify(Event::ImageDecoded(ImageDecoded::new(
881                entry,
882                pixels.clone(),
883                source.is_opaque,
884            ))))
885            .ok()?;
886
887        Some(pixels)
888    }

926#[cfg(feature = "image_cur")]
927fn downscale_hotspot(
928    image_cur_ext_id: zng_view_api::api_extension::ApiExtensionId,
929    downscaled_size: PxSize,
930    meta: &mut ImageMetadata,
931    full_size: PxSize,
932    cur_hotspot: Option<PxPoint>,
933) {
934    if let Some(mut hotspot) = cur_hotspot {
935        hotspot.x *= downscaled_size.width.0 as f32 / full_size.width.0 as f32;
936        hotspot.y *= downscaled_size.height.0 as f32 / full_size.height.0 as f32;
937
938        meta.extensions.push((
939            image_cur_ext_id,
940            zng_view_api::api_extension::ApiExtensionPayload::serialize(&hotspot).unwrap(),
941        ));
942    }

1091    /// If this is `true` needs to replace with `wr_stripes`
1092    pub fn overflows_wr(&self) -> bool {

1094    }

1096    /// Returns the image split in "stripes" that fit the Webrender buffer length constraints.
1097    ///
1098    /// If the image cannot be split into stripes returns an empty list. This only happens if the image width is absurdly wide.
1099    pub fn wr_stripes(&self) -> Box<[Image]> {
1100        if !self.overflows_wr() {
1101            return Box::new([self.clone()]);
1102        }
1103
1104        match &*self.0 {
1105            ImageData::RawData {
1106                size,
1107                pixels,
1108                is_opaque,
1109                density,
1110                range,
1111                stripes,
1112                ..
1113            } => {
1114                debug_assert_eq!(range.len(), pixels.len());
1115
1116                let mut stripes = stripes.lock();
1117                if stripes.is_empty() {
1118                    *stripes = self.generate_stripes(*size, pixels, *is_opaque, *density);
1119                }
1120                (*stripes).clone()
1121            }
1122            _ => unreachable!(),
1123        }
1124    }
1125    const MAX_LEN: usize = i32::MAX as usize;
1126    fn generate_stripes(&self, full_size: PxSize, pixels: &IpcBytes, is_opaque: bool, density: Option<PxDensity2d>) -> Box<[Image]> {
1127        let w = full_size.width.0 as usize * 4;
1128        if w > Self::MAX_LEN {
1129            tracing::error!("renderer does not support images with width * 4 > {}", Self::MAX_LEN);
1130            return Box::new([]);
1131        }
1132
1133        // find proportional split that fits, to avoid having the last stripe be to thin
1134        let full_height = full_size.height.0 as usize;
1135        let mut stripe_height = full_height / 2;
1136        while w * stripe_height > Self::MAX_LEN {
1137            stripe_height /= 2;
1138        }
1139        let stripe_len = w * stripe_height;
1140        let stripes_len = full_height.div_ceil(stripe_height);
1141        let stripe_height = Px(stripe_height as _);
1142
1143        let mut stripes = Vec::with_capacity(stripes_len);
1144
1145        for i in 0..stripes_len {
1146            let y = stripe_height * Px(i as _);
1147            let mut size = full_size;
1148            size.height = stripe_height.min(full_size.height - y);
1149
1150            let offset = stripe_len * i;
1151            let range = offset..((offset + stripe_len).min(pixels.len()));
1152
1153            let stripe = Image(Arc::new(ImageData::RawData {
1154                size,
1155                pixels: pixels.clone(),
1156                is_opaque,
1157                density,
1158                range,
1159                // always empty
1160                stripes: Mutex::new(Box::new([])),
1161            }));
1162
1163            stripes.push(stripe);
1164        }
1165
1166        stripes.into_boxed_slice()
1167    }
