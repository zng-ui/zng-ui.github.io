44#[allow(clippy::large_enum_variant)] // enum is temporary
45enum GlWindowCreation {
46    /// Windows requires this.
47    Before(winit::window::Window),
48    /// Other platforms don't. X11 requires this because it needs to set the XVisualID.
49    After(winit::window::WindowAttributes),
50}
51fn winit_create_window(winit_loop: &ActiveEventLoop, window: &winit::window::WindowAttributes) -> winit::window::Window {
52    let mut retries = 0;
53    loop {
54        match winit_loop.create_window(window.clone()) {
55            Ok(w) => break w,
56            Err(e) => {
57                // Some platforms work after a retry
58                // X11: After a GLXBadWindow
59                retries += 1;
60                if retries == 10 {
61                    panic!("cannot create winit window, {e}")
62                } else if retries > 1 {
63                    tracing::error!("cannot create winit window (retry={retries}), {e}");
64                    thread::sleep(std::time::Duration::from_millis(retries * 100));
65                }
66            }
67        }
68    }
69}

72    /// New window context.
73    pub(crate) fn create_headed(
74        &mut self,
75        id: WindowId,
76        window: winit::window::WindowAttributes,
77        winit_loop: &ActiveEventLoop,
78        render_mode: RenderMode,
79        sender: &AppEventSender,
80        prefer_egl: bool,
81    ) -> (winit::window::Window, GlContext) {
82        let mut errors = vec![];
83
84        #[cfg(not(feature = "hardware"))]
85        let _ = prefer_egl;
86
87        for config in TryConfig::iter(render_mode) {
88            if self.unsupported_headed.contains(&config) {
89                errors.push((config, "previous attempt failed, not supported".into()));
90                continue;
91            }
92
93            let window = if cfg!(windows) || matches!(config.mode, RenderMode::Software) {
94                GlWindowCreation::Before(winit_create_window(winit_loop, &window))
95            } else {
96                GlWindowCreation::After(window.clone())
97            };
98
99            let r = util::catch_suppress(std::panic::AssertUnwindSafe(|| match config.mode {
100                #[cfg(feature = "hardware")]
101                RenderMode::Dedicated => self.create_headed_glutin(winit_loop, id, window, config.hardware_acceleration, prefer_egl),
102                #[cfg(feature = "hardware")]
103                RenderMode::Integrated => self.create_headed_glutin(winit_loop, id, window, Some(false), prefer_egl),
104                RenderMode::Software => self.create_headed_swgl(winit_loop, id, window),
105                _ => self.create_headed_swgl(winit_loop, id, window),
106            }));
107
108            let error = match r {
109                Ok(Ok(r)) => {
110                    let actual_mode = r.1.render_mode();
111                    if render_mode != actual_mode {
112                        tracing::warn!("render mode `{render_mode:?}` is not available, will use `{actual_mode:?}`");
113                    }
114                    return r;
115                }
116                Ok(Err(e)) => e,
117                Err(panic) => {
118                    let component = match config.mode {
119                        RenderMode::Dedicated => "glutin (headed, dedicated)",
120                        RenderMode::Integrated => "glutin (headed, integrated)",
121                        RenderMode::Software => "swgl (headed)",
122                        _ => "swgl (headed)",
123                    };
124                    let _ = sender.send(AppEvent::Notify(zng_view_api::Event::RecoveredFromComponentPanic {
125                        component: component.into(),
126                        recover: "will try other modes".into(),
127                        panic: panic.to_txt(),
128                    }));
129                    panic.msg.into()
130                }
131            };
132
133            tracing::error!("[{}] {}", config.name(), error);
134            errors.push((config, error));
135
136            self.unsupported_headed.insert(config);
137        }
138
139        let mut msg = "failed to create headed open-gl context:\n".to_owned();
140        for (config, error) in errors {
141            use std::fmt::Write;
142            writeln!(&mut msg, "  {:?}: {}", config.name(), error).unwrap();
143        }
144
145        panic!("{msg}")
146    }

218    fn create_headed_glutin(
219        &mut self,
220        event_loop: &ActiveEventLoop,
221        id: WindowId,
222        window: GlWindowCreation,
223        hardware: Option<bool>,
224        prefer_egl: bool,
225    ) -> Result<(winit::window::Window, GlContext), Box<dyn Error>> {
226        #[cfg(windows)]
227        let display_pref = {
228            let handle = Some(match &window {
229                GlWindowCreation::Before(w) => w.window_handle().unwrap().as_raw(),
230                GlWindowCreation::After(_) => unreachable!(),
231            });
232            if prefer_egl {
233                DisplayApiPreference::EglThenWgl(handle)
234            } else {
235                DisplayApiPreference::WglThenEgl(handle)
236            }
237        };
238
239        #[cfg(any(
240            target_os = "linux",
241            target_os = "dragonfly",
242            target_os = "freebsd",
243            target_os = "openbsd",
244            target_os = "netbsd",
245        ))]
246        let display_pref = {
247            let handle = Box::new(winit::platform::x11::register_xlib_error_hook);
248            if prefer_egl {
249                DisplayApiPreference::EglThenGlx(handle)
250            } else {
251                DisplayApiPreference::GlxThenEgl(handle)
252            }
253        };
254
255        #[cfg(target_os = "android")]
256        let display_pref = DisplayApiPreference::Egl;
257
258        #[cfg(target_os = "macos")]
259        let display_pref = DisplayApiPreference::Cgl;
260
261        let _ = prefer_egl;
262
263        let display_handle = match &window {
264            GlWindowCreation::Before(w) => w.display_handle().unwrap().as_raw(),
265            GlWindowCreation::After(_) => event_loop.display_handle().unwrap().as_raw(),
266        };
267
268        // SAFETY: we are trusting the `raw_display_handle` from winit here.
269        let display = unsafe { Display::new(display_handle, display_pref) }?;
270
271        let mut template = ConfigTemplateBuilder::new()
272            .with_alpha_size(8)
273            .with_transparency(cfg!(not(target_os = "android")))
274            .with_surface_type(ConfigSurfaceTypes::WINDOW)
275            .prefer_hardware_accelerated(hardware);
276        if let GlWindowCreation::Before(w) = &window {
277            template = template.compatible_with_native_window(w.window_handle().unwrap().as_raw());
278        }
279        let template = template.build();
280
281        // SAFETY: we are holding the `window` reference.
282        let config = unsafe { display.find_configs(template)?.next().ok_or("no display config") }?;
283
284        let window = match window {
285            GlWindowCreation::Before(w) => w,
286            GlWindowCreation::After(w) => {
287                #[cfg(any(
288                    target_os = "linux",
289                    target_os = "dragonfly",
290                    target_os = "freebsd",
291                    target_os = "openbsd",
292                    target_os = "netbsd"
293                ))]
294                let w = {
295                    use glutin::platform::x11::X11GlConfigExt as _;
296                    use winit::platform::x11::WindowAttributesExtX11 as _;
297
298                    if let Some(id) = config.x11_visual() {
299                        w.with_x11_visual(id.visual_id() as _)
300                    } else {
301                        w
302                    }
303                };
304                winit_create_window(event_loop, &w)
305            }
306        };
307
308        let window_handle = window.window_handle().unwrap().as_raw();
309
310        let size = window.inner_size();
311        let attrs = SurfaceAttributesBuilder::<WindowSurface>::new().build(
312            window_handle,
313            NonZeroU32::new(size.width).unwrap(),
314            NonZeroU32::new(size.height).unwrap(),
315        );
316
317        // SAFETY: the window handle is valid.
318        let surface = unsafe { display.create_window_surface(&config, &attrs)? };
319
320        let context_attributes = ContextAttributesBuilder::new().build(Some(window_handle));
321        // SAFETY: the window handle is valid.
322        let context = unsafe { display.create_context(&config, &context_attributes)? };
323
324        self.current.set(Some(id));
325        let context = context.make_current(&surface)?;
326
327        let gl_api = config.api();
328        let gl = if gl_api.contains(Api::OPENGL) {
329            // SAFETY: function pointers are directly from safe glutin here.
330            unsafe {
331                gl::GlFns::load_with(|symbol| {
332                    let symbol = CString::new(symbol).unwrap();
333                    display.get_proc_address(symbol.as_c_str())
334                })
335            }
336        } else if gl_api.contains(Api::GLES3) {
337            // SAFETY: function pointers are directly from safe glutin here.
338            unsafe {
339                gl::GlesFns::load_with(|symbol| {
340                    let symbol = CString::new(symbol).unwrap();
341                    display.get_proc_address(symbol.as_c_str())
342                })
343            }
344        } else {
345            return Err("no OpenGL or GLES3 available".into());
346        };
347
348        check_wr_gl_version(&*gl)?;
349
350        #[cfg(debug_assertions)]
351        let gl = gl::ErrorCheckingGl::wrap(gl.clone());
352
353        let mut context = GlContext {
354            id,
355            current: self.current.clone(),
356            backend: GlBackend::Glutin {
357                context,
358                surface,
359                headless: None,
360            },
361            gl,
362
363            render_mode: if hardware == Some(false) {
364                RenderMode::Integrated
365            } else {
366                RenderMode::Dedicated
367            },
368        };
369
370        context.resize(size);
371
372        Ok((window, context))
373    }

375    fn create_headed_swgl(
376        &mut self,
377        event_loop: &ActiveEventLoop,
378        id: WindowId,
379        window: GlWindowCreation,
380    ) -> Result<(winit::window::Window, GlContext), Box<dyn Error>> {
381        #[cfg(not(feature = "software"))]
382        {
383            let _ = (id, window, event_loop);
384            return Err("zng-view not build with \"software\" backend".into());
385        }
386
387        #[cfg(target_os = "android")]
388        {
389            let _ = (id, window, event_loop);
390            return Err("software blit not implemented for Android".into());
391        }
392
393        #[cfg(all(feature = "software", not(target_os = "android")))]
394        {
395            let window = match window {
396                GlWindowCreation::Before(w) => w,
397                GlWindowCreation::After(w) => event_loop.create_window(w)?,
398            };
399
400            // SAFETY: softbuffer context is managed like gl context, it is dropped before the window is dropped.
401            let static_window_ref = unsafe { mem::transmute::<&winit::window::Window, &'static winit::window::Window>(&window) };
402            let blit_context = softbuffer::Context::new(static_window_ref)?;
403            let blit_surface = softbuffer::Surface::new(&blit_context, static_window_ref)?;
404
405            let context = swgl::Context::create();
406            let gl = Rc::new(context);
407
408            // create_headed_glutin returns as current.
409            self.current.set(Some(id));
410            context.make_current();
411
412            let context = GlContext {
413                id,
414                current: self.current.clone(),
415                backend: GlBackend::Swgl {
416                    context,
417                    blit: Some((blit_context, blit_surface)),
418                },
419                gl,
420                render_mode: RenderMode::Software,
421            };
422            Ok((window, context))
423        }
424    }

581#[allow(clippy::large_enum_variant)] // glutin is the largest, but also most common
582enum GlBackend {
583    #[cfg(feature = "hardware")]
584    Glutin {
585        headless: Option<GlutinHeadless>,
586        context: PossiblyCurrentContext,
587        surface: Surface<WindowSurface>,
588    },
589
590    #[cfg(feature = "software")]
591    Swgl {
592        context: swgl::Context,
593        // is None for headless.
594        #[cfg(not(target_os = "android"))]
595        blit: Option<(
596            softbuffer::Context<&'static winit::window::Window>,
597            softbuffer::Surface<&'static winit::window::Window, &'static winit::window::Window>,
598        )>,
599        #[cfg(target_os = "android")]
600        blit: Option<((), ())>,
601    },
602
603    Dropped,
604}

606/// OpenGL context managed by [`GlContextManager`].

650    pub(crate) fn resize(&mut self, size: PhysicalSize<u32>) {
651        assert!(self.is_current());
652
653        match &mut self.backend {
654            #[cfg(feature = "hardware")]
655            GlBackend::Glutin {
656                context,
657                surface,
658                headless,
659            } => {
660                if let Some(h) = headless {
661                    h.resize(&self.gl, size.width as _, size.height as _);
662                } else {
663                    let width = NonZeroU32::new(size.width.max(1)).unwrap();
664                    let height = NonZeroU32::new(size.height.max(1)).unwrap();
665                    surface.resize(context, width, height);
666                }
667            }
668            #[cfg(feature = "software")]
669            GlBackend::Swgl { context, blit } => {
670                // NULL means SWGL manages the buffer, it also retains the buffer if the size did not change.
671                let w = size.width.max(1);
672                let h = size.height.max(1);
673                context.init_default_framebuffer(0, 0, w as i32, h as i32, 0, std::ptr::null_mut());
674
675                #[cfg(not(target_os = "android"))]
676                if let Some((_, surface)) = blit {
677                    surface.resize(NonZeroU32::new(w).unwrap(), NonZeroU32::new(h).unwrap()).unwrap();
678                }
679
680                #[cfg(target_os = "android")]
681                let _ = blit;
682            }
683            GlBackend::Dropped => unreachable!(),
684        }
685    }
686
687    pub(crate) fn make_current(&mut self) {
688        let id = Some(self.id);
689        if self.current.get() != id {
690            self.current.set(id);
691
692            match &self.backend {
693                #[cfg(feature = "hardware")]
694                GlBackend::Glutin { context, surface, .. } => context.make_current(surface).unwrap(),
695                #[cfg(feature = "software")]
696                GlBackend::Swgl { context, .. } => context.make_current(),
697                GlBackend::Dropped => unreachable!(),
698            }
699        }
700    }
701
702    pub(crate) fn swap_buffers(&mut self) {
703        assert!(self.is_current());
704
705        match &mut self.backend {
706            #[cfg(feature = "hardware")]
707            GlBackend::Glutin {
708                context,
709                surface,
710                headless,
711            } => {
712                if headless.is_none() {
713                    surface.swap_buffers(context).unwrap()
714                }
715            }
716            #[cfg(feature = "software")]
717            GlBackend::Swgl { context, blit } => {
718                #[cfg(target_os = "android")]
719                let _ = (context, blit);
720
721                #[cfg(not(target_os = "android"))]
722                if let Some((_, blit_surface)) = blit {
723                    gl::Gl::finish(context);
724                    let (data_ptr, w, h, stride) = context.get_color_buffer(0, true);
725
726                    if w == 0 || h == 0 {
727                        return;
728                    }
729
730                    // SAFETY: we trust SWGL
731                    assert!(stride == w * 4);
732                    let frame = unsafe { std::slice::from_raw_parts(data_ptr as *const u8, w as usize * h as usize * 4) };
733                    // bgra, max_y=0
734                    let frame = frame.chunks_exact(stride as _).rev().flat_map(|row| row.chunks_exact(4));
735                    let mut buffer = blit_surface.buffer_mut().unwrap();
736                    for (argb, bgra) in buffer.iter_mut().zip(frame) {
737                        let blue = bgra[0] as u32;
738                        let green = bgra[1] as u32;
739                        let red = bgra[2] as u32;
740                        let alpha = bgra[3] as u32;
741                        *argb = blue | (green << 8) | (red << 16) | (alpha << 24);
742                    }
743
744                    buffer.present().unwrap();
745                }
746            }
747            GlBackend::Dropped => unreachable!(),
748        }
749    }
750}
751impl Drop for GlContext {
752    fn drop(&mut self) {
753        self.make_current();
754
755        match mem::replace(&mut self.backend, GlBackend::Dropped) {
756            #[cfg(feature = "hardware")]
757            GlBackend::Glutin { headless, .. } => {
758                if let Some(h) = headless {
759                    let _ = h.hidden_window;
760
761                    h.destroy(&self.gl);
762                }
763            }
764            #[cfg(feature = "software")]
765            GlBackend::Swgl { context, .. } => context.destroy(),
766            GlBackend::Dropped => unreachable!(),
767        }
768    }
769}

771/// Warmup the OpenGL driver in a throwaway thread, some NVIDIA drivers have a slow startup (500ms~),
772/// hopefully this loads it in parallel while the app is starting up so we don't block creating the first window.
773#[cfg(all(windows, feature = "hardware"))]
774pub(crate) fn warmup() {
775    // idea copied from here:
776    // https://hero.handmade.network/forums/code-discussion/t/2503-day_235_opengl%2527s_pixel_format_takes_a_long_time#13029
777
778    use windows_sys::Win32::Graphics::{
779        Gdi::*,
780        OpenGL::{self},
781    };
782
783    let _ = std::thread::Builder::new()
784        .name("warmup".to_owned())
785        .stack_size(3 * 64 * 1024)
786        .spawn(|| unsafe {
787            let _span = tracing::trace_span!("open-gl-init").entered();
788            let hdc = GetDC(std::ptr::null_mut());
789            let _ = OpenGL::DescribePixelFormat(hdc, 0, 0, std::ptr::null_mut());
790            ReleaseDC(std::ptr::null_mut(), hdc);
791        });
792}

794#[cfg(not(all(windows, feature = "hardware")))]
795pub(crate) fn warmup() {}

797// check if equal or newer then 3.1
798#[cfg(feature = "hardware")]
799fn check_wr_gl_version(gl: &dyn gl::Gl) -> Result<(), String> {
800    let mut version = [0; 2];
801    let is_2_or_1;
802    // SAFETY: get_integer_v API available in all impls
803    unsafe {
804        gl.get_integer_v(gl::MAJOR_VERSION, &mut version[..1]);
805        is_2_or_1 = gl.get_error() == gl::INVALID_ENUM; // MAJOR_VERSION is only 3.0 and above
806        gl.get_integer_v(gl::MINOR_VERSION, &mut version[1..]);
807    };
808
809    if !is_2_or_1 && version[0] >= 3 {
810        let min_minor = match gl.get_type() {
811            gl::GlType::Gl => 1,
812            gl::GlType::Gles => 0,
813        };
814        if version[1] >= min_minor {
815            return Ok(());
816        }
817    }
818    Err(format!(
819        "webrender requires OpenGL >=3.1 or OpenGL ES >=3.0, found {}",
820        gl.get_string(gl::VERSION)
821    ))
822}

824/// Glutin, SWGL config to attempt.
825#[derive(Debug, PartialEq, Eq, Clone, Copy, Hash)]
826struct TryConfig {
827    mode: RenderMode,
828    hardware_acceleration: Option<bool>,
829}
830impl TryConfig {
831    fn iter(mode: RenderMode) -> impl Iterator<Item = TryConfig> {
832        let mut configs = Vec::with_capacity(4);
833        let mut try_hardware_none = false;
834        for mode in mode.with_fallbacks() {
835            match mode {
836                RenderMode::Dedicated => {
837                    configs.push(TryConfig {
838                        mode,
839                        hardware_acceleration: Some(true),
840                    });
841                    try_hardware_none = true;
842                }
843                RenderMode::Integrated => configs.push(TryConfig {
844                    mode,
845                    hardware_acceleration: Some(false),
846                }),
847                RenderMode::Software => {
848                    if mem::take(&mut try_hardware_none) {
849                        // some dedicated hardware end-up classified as generic integrated for some reason,
850                        // so we try `None`, after `Some(false)` and before `Software`.
851                        configs.push(TryConfig {
852                            mode: RenderMode::Dedicated,
853                            hardware_acceleration: None,
854                        });
855                    }
856                    configs.push(TryConfig {
857                        mode,
858                        hardware_acceleration: Some(false),
859                    });
860                }
861                _ => {}
862            }
863        }
864        configs.into_iter()
865    }
866
867    pub fn name(&self) -> &str {
868        match self.hardware_acceleration {
869            Some(true) => "Dedicated",
870            Some(false) => match self.mode {
871                RenderMode::Integrated => "Integrated",
872                RenderMode::Software => "Software",
873                RenderMode::Dedicated => unreachable!(),
874                _ => unreachable!(),
875            },
876            None => "Dedicated (generic)",
877        }
878    }
879}

881#[cfg(feature = "hardware")]
882struct GlutinHeadless {
883    hidden_window: winit::window::Window,
884
885    // actual surface.
886    rbos: [u32; 2],
887    fbo: u32,
888}
889#[cfg(feature = "hardware")]
890impl GlutinHeadless {
891    fn new(gl: &Rc<dyn gl::Gl>, hidden_window: winit::window::Window) -> Self {
892        // create a surface for Webrender:
893
894        let rbos = gl.gen_renderbuffers(2);
895
896        let rbos = [rbos[0], rbos[1]];
897        let fbo = gl.gen_framebuffers(1)[0];
898
899        gl.bind_renderbuffer(gl::RENDERBUFFER, rbos[0]);
900        gl.renderbuffer_storage(gl::RENDERBUFFER, gl::RGBA8, 1, 1);
901
902        gl.bind_renderbuffer(gl::RENDERBUFFER, rbos[1]);
903        gl.renderbuffer_storage(gl::RENDERBUFFER, gl::DEPTH24_STENCIL8, 1, 1);
904
905        gl.viewport(0, 0, 1, 1);
906
907        gl.bind_framebuffer(gl::FRAMEBUFFER, fbo);
908        gl.framebuffer_renderbuffer(gl::FRAMEBUFFER, gl::COLOR_ATTACHMENT0, gl::RENDERBUFFER, rbos[0]);
909        gl.framebuffer_renderbuffer(gl::FRAMEBUFFER, gl::DEPTH_STENCIL_ATTACHMENT, gl::RENDERBUFFER, rbos[1]);
910
911        GlutinHeadless { hidden_window, rbos, fbo }
912    }
913
914    fn resize(&self, gl: &Rc<dyn gl::Gl>, width: i32, height: i32) {
915        gl.bind_renderbuffer(gl::RENDERBUFFER, self.rbos[0]);
916        gl.renderbuffer_storage(gl::RENDERBUFFER, gl::RGBA8, width, height);
917
918        gl.bind_renderbuffer(gl::RENDERBUFFER, self.rbos[1]);
919        gl.renderbuffer_storage(gl::RENDERBUFFER, gl::DEPTH24_STENCIL8, width, height);
920
921        gl.viewport(0, 0, width, height);
922    }
923
924    fn destroy(self, gl: &Rc<dyn gl::Gl>) {
925        gl.delete_framebuffers(&[self.fbo]);
926        gl.delete_renderbuffers(&self.rbos);
927    }
928}
