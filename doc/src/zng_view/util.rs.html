27/// Sets a window subclass that calls a raw event handler.
28///
29/// Use this to receive Windows OS events not covered in [`raw_events`].
30///
31/// Returns if adding a subclass handler succeeded.
32///
33/// # Handler
34///
35/// The handler inputs are the first 4 arguments of a [`SUBCLASSPROC`].
36/// You can use closure capture to include extra data.
37///
38/// The handler must return `Some(LRESULT)` to stop the propagation of a specific message.
39///
40/// The handler is dropped after it receives the `WM_DESTROY` message.
41///
42/// # Panics
43///
44/// Panics in headless mode.
45///
46/// [`raw_events`]: crate::app::raw_events
47/// [`SUBCLASSPROC`]: https://docs.microsoft.com/en-us/windows/win32/api/commctrl/nc-commctrl-subclassproc
48#[cfg(windows)]
49pub fn set_raw_windows_event_handler<H>(hwnd: windows_sys::Win32::Foundation::HWND, subclass_id: usize, handler: H) -> bool
50where
51    H: FnMut(
52            windows_sys::Win32::Foundation::HWND,
53            u32,
54            windows_sys::Win32::Foundation::WPARAM,
55            windows_sys::Win32::Foundation::LPARAM,
56        ) -> Option<windows_sys::Win32::Foundation::LRESULT>
57        + 'static,
58{
59    let data = Box::new(handler);
60    unsafe {
61        windows_sys::Win32::UI::Shell::SetWindowSubclass(hwnd, Some(subclass_raw_event_proc::<H>), subclass_id, Box::into_raw(data) as _)
62            != 0
63    }
64}

66#[cfg(windows)]
67unsafe extern "system" fn subclass_raw_event_proc<H>(
68    hwnd: windows_sys::Win32::Foundation::HWND,
69    msg: u32,
70    wparam: windows_sys::Win32::Foundation::WPARAM,
71    lparam: windows_sys::Win32::Foundation::LPARAM,
72    _id: usize,
73    data: usize,
74) -> windows_sys::Win32::Foundation::LRESULT
75where
76    H: FnMut(
77            windows_sys::Win32::Foundation::HWND,
78            u32,
79            windows_sys::Win32::Foundation::WPARAM,
80            windows_sys::Win32::Foundation::LPARAM,
81        ) -> Option<windows_sys::Win32::Foundation::LRESULT>
82        + 'static,
83{
84    use windows_sys::Win32::UI::WindowsAndMessaging::WM_DESTROY;
85    match msg {
86        WM_DESTROY => {
87            // last call and cleanup.
88            let mut handler = unsafe { Box::from_raw(data as *mut H) };
89            handler(hwnd, msg, wparam, lparam).unwrap_or_default()
90        }
91
92        msg => {
93            let handler = unsafe { &mut *(data as *mut H) };
94            if let Some(r) = handler(hwnd, msg, wparam, lparam) {
95                r
96            } else {
97                unsafe { windows_sys::Win32::UI::Shell::DefSubclassProc(hwnd, msg, wparam, lparam) }
98            }
99        }
100    }
101}

939/// If `true` our custom panic hook must not log anything.
940#[cfg(ipc)]
941pub(crate) fn suppress_panic() -> bool {
942    SUPPRESS.get()
943}
944#[cfg(ipc)]
945pub(crate) fn set_suppressed_panic(panic: SuppressedPanic) {
946    SUPPRESSED_PANIC.set(Some(panic));

949#[derive(Debug)]
950pub(crate) struct SuppressedPanic {
951    pub thread: Txt,
952    pub msg: Txt,
953    pub file: Txt,
954    pub line: u32,
955    pub column: u32,
956    pub backtrace: Backtrace,
957}
958impl SuppressedPanic {
959    #[cfg(ipc)]
960    pub fn from_hook(info: &std::panic::PanicHookInfo, backtrace: Backtrace) -> Self {
961        let current_thread = std::thread::current();
962        let thread = current_thread.name().unwrap_or("<unnamed>");
963        let msg = Self::payload(info.payload());
964
965        let (file, line, column) = if let Some(l) = info.location() {
966            (l.file(), l.line(), l.column())
967        } else {
968            ("<unknown>", 0, 0)
969        };
970        Self {
971            thread: thread.to_txt(),
972            msg,
973            file: file.to_txt(),
974            line,
975            column,
976            backtrace,
977        }
978    }
979
980    pub fn from_catch(p: Box<dyn Any>) -> Self {
981        Self {
982            thread: Txt::from("<unknown>"),
983            msg: Self::payload(&*p),
984            file: Txt::from("<unknown>"),
985            line: 0,
986            column: 0,
987            backtrace: Backtrace::disabled(),
988        }
989    }
990
991    fn payload(p: &dyn Any) -> Txt {
992        match p.downcast_ref::<&'static str>() {
993            Some(s) => s,
994            None => match p.downcast_ref::<String>() {
995                Some(s) => &s[..],
996                None => "Box<dyn Any>",
997            },
998        }
999        .to_txt()
1000    }
1001}
1002impl std::error::Error for SuppressedPanic {}
1003impl fmt::Display for SuppressedPanic {
1004    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
1005        write!(
1006            f,
1007            "thread '{}' panicked at {}:{}:{}:\n{}\n{}",
1008            self.thread, self.file, self.line, self.column, self.msg, self.backtrace,
1009        )
1010    }
1011}

1013/// Like [`std::panic::catch_unwind`], but flags [`suppress_panic`] for our custom panic hook.
1014pub(crate) fn catch_suppress<T>(f: impl FnOnce() -> T + std::panic::UnwindSafe) -> Result<T, Box<SuppressedPanic>> {
1015    SUPPRESS.set(true);
1016    let _cleanup = RunOnDrop::new(|| SUPPRESS.set(false));
1017    std::panic::catch_unwind(f).map_err(|e| {
1018        SUPPRESSED_PANIC.with_borrow_mut(|p| match p.take() {
1019            Some(p) => Box::new(p),
1020            None => Box::new(SuppressedPanic::from_catch(e)),
1021        })
1022    })
1023}

1025struct RunOnDrop<F: FnOnce()>(Option<F>);
1026impl<F: FnOnce()> RunOnDrop<F> {
1027    pub fn new(clean: F) -> Self {
1028        RunOnDrop(Some(clean))
1029    }
1030}
1031impl<F: FnOnce()> Drop for RunOnDrop<F> {
1032    fn drop(&mut self) {
1033        if let Some(clean) = self.0.take() {

1250fn access_node_to_kit(
1251    node: zng_view_api::access::AccessNodeRef,
1252    output: &mut Vec<(accesskit::NodeId, accesskit::Node)>,
1253) -> accesskit::NodeId {
1254    let node_id = access_id_to_kit(node.id);
1255    let node_role = node.role.map(access_role_to_kit).unwrap_or(accesskit::Role::Unknown);
1256    let mut builder = accesskit::Node::new(node_role);
1257
1258    // add bounds and transform
1259    if !node.size.is_empty() {
1260        let mut bounds = accesskit::Rect::new(0.0, 0.0, node.size.width.0 as f64, node.size.height.0 as f64);
1261        if !node.transform.is_identity() {
1262            if node.children_count() == 0
1263                && let PxTransform::Offset(o) = node.transform
1264            {
1265                let (x, y) = o.cast().to_tuple();
1266                bounds = bounds.with_origin(accesskit::Point::new(x, y));
1267            } else {
1268                let t = node.transform.to_transform().to_2d();
1269                builder.set_transform(accesskit::Affine::new(t.cast().to_array()));
1270            }
1271        }
1272        builder.set_bounds(bounds);
1273    }
1274
1275    // add actions
1276    for cmd in &node.commands {
1277        use zng_view_api::access::AccessCmdName::*;
1278
1279        match cmd {
1280            Click => {
1281                builder.add_action(accesskit::Action::Click);
1282                builder.add_action(accesskit::Action::ShowContextMenu);
1283            }
1284            Focus => {
1285                builder.add_action(accesskit::Action::Focus);
1286                builder.add_action(accesskit::Action::Blur);
1287            }
1288            SetExpanded => {
1289                builder.add_action(accesskit::Action::Expand);
1290                builder.add_action(accesskit::Action::Collapse);
1291            }
1292            Increment => {
1293                builder.add_action(accesskit::Action::Increment);
1294            }
1295            SetToolTipVis => {
1296                builder.add_action(accesskit::Action::ShowTooltip);
1297                builder.add_action(accesskit::Action::HideTooltip);
1298            }
1299            Scroll => {
1300                builder.add_action(accesskit::Action::ScrollUp);
1301                builder.add_action(accesskit::Action::ScrollLeft);
1302                builder.add_action(accesskit::Action::ScrollDown);
1303                builder.add_action(accesskit::Action::ScrollRight);
1304                builder.add_action(accesskit::Action::ScrollIntoView);
1305                builder.add_action(accesskit::Action::ScrollToPoint);
1306            }
1307            ReplaceSelectedText => {
1308                builder.add_action(accesskit::Action::ReplaceSelectedText);
1309            }
1310            SelectText => {
1311                builder.add_action(accesskit::Action::SetTextSelection);
1312            }
1313            SetString => {
1314                builder.add_action(accesskit::Action::SetValue);
1315            }
1316            SetNumber => {
1317                builder.add_action(accesskit::Action::SetValue);
1318            }
1319            _ => {}
1320        }
1321    }
1322
1323    // add state
1324    for state in &node.state {
1325        use zng_view_api::access::{self, AccessState::*};
1326
1327        match state {
1328            AutoComplete(s) => {
1329                if *s == access::AutoComplete::BOTH {
1330                    builder.set_auto_complete(accesskit::AutoComplete::Both)
1331                } else if *s == access::AutoComplete::INLINE {
1332                    builder.set_auto_complete(accesskit::AutoComplete::Inline)
1333                } else if *s == access::AutoComplete::LIST {
1334                    builder.set_auto_complete(accesskit::AutoComplete::List)
1335                }
1336            }
1337            Checked(b) => builder.set_toggled(match b {
1338                Some(true) => accesskit::Toggled::True,
1339                Some(false) => accesskit::Toggled::False,
1340                None => accesskit::Toggled::Mixed,
1341            }),
1342            Current(kind) => match kind {
1343                access::CurrentKind::Page => builder.set_aria_current(accesskit::AriaCurrent::Page),
1344                access::CurrentKind::Step => builder.set_aria_current(accesskit::AriaCurrent::Step),
1345                access::CurrentKind::Location => builder.set_aria_current(accesskit::AriaCurrent::Location),
1346                access::CurrentKind::Date => builder.set_aria_current(accesskit::AriaCurrent::Date),
1347                access::CurrentKind::Time => builder.set_aria_current(accesskit::AriaCurrent::Time),
1348                access::CurrentKind::Item => builder.set_aria_current(accesskit::AriaCurrent::True),
1349                _ => {}
1350            },
1351            Disabled => builder.set_disabled(),
1352            ErrorMessage(id) => builder.set_error_message(access_id_to_kit(*id)),
1353            Expanded(b) => builder.set_expanded(*b),
1354            Popup(pop) => match pop {
1355                access::Popup::Menu => builder.set_has_popup(accesskit::HasPopup::Menu),
1356                access::Popup::ListBox => builder.set_has_popup(accesskit::HasPopup::Listbox),
1357                access::Popup::Tree => builder.set_has_popup(accesskit::HasPopup::Tree),
1358                access::Popup::Grid => builder.set_has_popup(accesskit::HasPopup::Grid),
1359                access::Popup::Dialog => builder.set_has_popup(accesskit::HasPopup::Dialog),
1360                _ => {}
1361            },
1362            Invalid(i) => {
1363                if i.contains(access::Invalid::SPELLING) {
1364                    builder.set_invalid(accesskit::Invalid::Spelling)
1365                } else if i.contains(access::Invalid::GRAMMAR) {
1366                    builder.set_invalid(accesskit::Invalid::Grammar)
1367                } else if i.contains(access::Invalid::ANY) {
1368                    builder.set_invalid(accesskit::Invalid::True)
1369                }
1370            }
1371            Label(s) => builder.set_label(s.clone().into_owned().into_boxed_str()),
1372            Level(n) => builder.set_level(n.get() as usize),
1373            Modal => builder.set_modal(),
1374            MultiSelectable => builder.set_multiselectable(),
1375            Orientation(o) => match o {
1376                access::Orientation::Horizontal => builder.set_orientation(accesskit::Orientation::Horizontal),
1377                access::Orientation::Vertical => builder.set_orientation(accesskit::Orientation::Vertical),
1378            },
1379            Placeholder(p) => builder.set_placeholder(p.clone().into_owned().into_boxed_str()),
1380            ReadOnly => builder.set_read_only(),
1381            Required => builder.set_required(),
1382            Selected => builder.set_selected(true),
1383            Sort(o) => match o {
1384                access::SortDirection::Ascending => builder.set_sort_direction(accesskit::SortDirection::Ascending),
1385                access::SortDirection::Descending => builder.set_sort_direction(accesskit::SortDirection::Descending),
1386            },
1387            ValueMax(m) => builder.set_max_numeric_value(*m),
1388            ValueMin(m) => builder.set_min_numeric_value(*m),
1389            Value(v) => builder.set_numeric_value(*v),
1390            ValueText(v) => builder.set_value(v.clone().into_owned().into_boxed_str()),
1391            Live { indicator, atomic, busy } => {
1392                builder.set_live(match indicator {
1393                    access::LiveIndicator::Assertive => accesskit::Live::Assertive,
1394                    access::LiveIndicator::OnlyFocused => accesskit::Live::Off,
1395                    access::LiveIndicator::Polite => accesskit::Live::Polite,
1396                    _ => accesskit::Live::Polite,
1397                });
1398                if *atomic {
1399                    builder.set_live_atomic();
1400                }
1401                if *busy {
1402                    builder.set_busy();
1403                }
1404            }
1405            ActiveDescendant(id) => builder.set_active_descendant(access_id_to_kit(*id)),
1406            ColCount(c) => builder.set_column_count(*c),
1407            ColIndex(i) => builder.set_column_index(*i),
1408            ColSpan(s) => builder.set_column_span(*s),
1409            Controls(ids) => builder.set_controls(ids.iter().copied().map(access_id_to_kit).collect::<Vec<_>>()),
1410            DescribedBy(ids) => builder.set_described_by(ids.iter().copied().map(access_id_to_kit).collect::<Vec<_>>()),
1411            Details(ids) => builder.set_details(ids.iter().copied().map(access_id_to_kit).collect::<Vec<_>>()),
1412            FlowTo(ids) => builder.set_flow_to(ids.iter().copied().map(access_id_to_kit).collect::<Vec<_>>()),
1413            LabelledBy(ids) => builder.set_labelled_by(ids.iter().copied().map(access_id_to_kit).collect::<Vec<_>>()),
1414            LabelledByChild => {
1415                let labelled_by: Vec<_> = if node.children.is_empty() {
1416                    node.children().map(|c| access_id_to_kit(c.id)).collect()
1417                } else {
1418                    node.children.iter().map(|id| access_id_to_kit(*id)).collect()
1419                };
1420                builder.set_labelled_by(labelled_by);
1421            }
1422            Owns(ids) => {
1423                for id in ids {
1424                    builder.push_child(access_id_to_kit(*id));
1425                }
1426            }
1427            ItemIndex(p) => builder.set_position_in_set(*p),
1428            RowCount(c) => builder.set_row_count(*c),
1429            RowIndex(i) => builder.set_row_index(*i),
1430            RowSpan(s) => builder.set_row_span(*s),
1431            ItemCount(s) => builder.set_size_of_set(*s),
1432            Lang(l) => builder.set_language(l.to_string()),
1433
1434            ScrollHorizontal(x) => {
1435                builder.set_scroll_x(*x as f64);
1436                builder.set_scroll_x_min(0.0);
1437                builder.set_scroll_x_max(1.0);
1438            }
1439            ScrollVertical(y) => {
1440                builder.set_scroll_y(*y as f64);
1441                builder.set_scroll_y_min(0.0);
1442                builder.set_scroll_y_max(1.0);
1443            }
1444            _ => {}
1445        }
1446    }
1447
1448    // add descendants
1449    if node.children.is_empty() {
1450        for child in node.children() {
1451            let child_id = access_node_to_kit(child, output);
1452            builder.push_child(child_id);
1453        }
1454    } else {
1455        for id in &node.children {
1456            builder.push_child(access_id_to_kit(*id));
1457        }
1458        for child in node.children() {
1459            let _ = access_node_to_kit(child, output);
1460        }
1461    }
1462
1463    let node = builder;
1464    output.push((node_id, node));
1465    node_id
1466}
