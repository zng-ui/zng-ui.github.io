3//!
4//! View-Process implementation.
5//!
6//! This implementation supports headed and headless apps in Windows, Linux and MacOS.
7//!
8//! # Usage
9//!
10//! First add this to your `Cargo.toml`:
11//!
12//! ```toml
13//! [dependencies]
14//! zng = "0.17.1"
15//! zng-view = "0.12.1"
16//! ```
17//!
18//! Then call `zng::env::init` before any other code in `main` to setup a view-process that uses
19//! the same app executable:
20//!
21//! ```
22//! # macro_rules! _demo {()=>{
23//! use zng::prelude::*;
24//!
25//! fn main() {
26//!     zng::env::init!();
27//!
28//!     APP.defaults().run_window(|ctx| unimplemented!())
29//! }
30//! # }}
31//! ```
32//!
33//! When the app is executed `run_window` gets called and internally starts the view-process.
34//! The current executable is started this time configured to be a view-process, `init` detects this and highjacks the process
35//! **never returning**.
36//!
37//! # Software Backend
38//!
39//! The `webrender/swgl` software renderer can be used as fallback when no native OpenGL 3.2 driver is available, to build it
40//! the feature `"software"` must be enabled (it is by default) and on Windows MSVC the `clang-cl` dependency must be installed and
41//! associated with the `CC` and `CXX` environment variables, if requirements are not met a warning is emitted and the build fails.
42//!
43//! To install dependencies on Windows:
44//!
45//! * Install LLVM (<https://releases.llvm.org/>) and add it to the `PATH` variable:
46//! ```bat
47//! setx PATH %PATH%;C:\Program Files\LLVM\bin
48//! ```
49//! * Associate `CC` and `CXX` with `clang-cl`:
50//! ```bat
51//! setx CC clang-cl
52//! setx CXX clang-cl
53//! ```
54//! Note that you may need to reopen the terminal for the environment variables to be available (setx always requires this).
55//!
56//! # Pre-built
57//!
58//! There is a pre-built release of this crate, [`zng-view-prebuilt`], it works as a drop-in replacement
59// that dynamically links with a pre-built library, for Windows, Linux and MacOS.
60//!
61//! In the `Cargo.toml` file:
62//!
63//! ```toml
64//! zng-view-prebuilt = "0.1"
65//! ```
66//!
67//! The pre-built crate includes the `"software"` and `"ipc"` features, in fact `ipc` is required, even for running on the same process,
68//! you can also configure where the pre-build library is installed, see the [`zng-view-prebuilt`] documentation for details.
69//!
70//! The pre-build crate does not support [`extensions`].
71//!
72//! # API Extensions
73//!
74//! This implementation of the view API provides these extensions:
75//!
76//! * `"zng-view.webrender_debug"`: `{ flags: DebugFlags, profiler_ui: String }`, sets Webrender debug flags.
77//!     - The `zng-wgt-webrender-debug` crate implements a property that uses this extension.
78//! * `"zng-view.prefer_angle": bool`, on Windows, prefer ANGLE(EGL) over WGL if the `libEGL.dll` and `libGLESv2.dll`
79//!    libraries can by dynamically loaded. The `extend-view` example demonstrates this extension.
80//!
81//! You can also inject your own extensions, see the [`extensions`] module for more details.
82//!
83//! [`zng-view-prebuilt`]: https://crates.io/crates/zng-view-prebuilt/
84//!

181/// Runs the view-process server.
182///
183/// Note that this only needs to be called if the view-process is not built on the same executable, if
184/// it is you only need to call [`zng_env::init!`] at the beginning of the executable main.
185///
186/// You can also disable start on init by setting the `ZNG_VIEW_NO_INIT_START` environment variable. In this
187/// case you must manually call this function.
188#[cfg(ipc)]
189pub fn view_process_main() {
190    let config = match ViewConfig::from_env() {
191        Some(c) => c,
192        None => return,
193    };
194
195    zng_env::set_process_name("view-process");
196
197    std::panic::set_hook(Box::new(init_abort));
198    config.assert_version(false);
199    let c = ipc::connect_view_process(config.server_name).expect("failed to connect to app-process");
200
201    let mut ext = ViewExtensions::new();
202    for e in extensions::VIEW_EXTENSIONS {
203        e(&mut ext);
204    }
205
206    if config.headless {
207        App::run_headless(c, ext);
208    } else {
209        App::run_headed(c, ext);
210    }
211
212    zng_env::exit(0)
213}

230/// Runs the view-process server in the current process and calls `run_app` to also
231/// run the app in the current process. Note that `run_app` will be called in a different thread.
232///
233/// In this mode the app only uses a single process, reducing the memory footprint, but it is also not
234/// resilient to video driver crashes, the view server **does not** respawn in this mode.
235///
236/// # Panics
237///
238/// Panics if not called in the main thread, this is a requirement of some operating systems.
239///
240/// ## Background Panics Warning
241///
242/// Note that `webrender` can freeze due to panics in worker threads without propagating
243/// the panics to the main thread, this causes the app to stop responding while still receiving
244/// event signals, causing the operating system to not detect that the app is frozen. It is recommended
245/// that you build with `panic=abort` or use [`std::panic::set_hook`] to detect these background panics.
246///
247/// # Android
248///
249/// In Android builds `android::init_android_app` must be called before this function, otherwise it will panic.
250pub fn run_same_process(run_app: impl FnOnce() + Send + 'static) {
251    run_same_process_extended(run_app, ViewExtensions::new)
252}

254/// Like [`run_same_process`] but with custom API extensions.
255///
256/// Note that any linked [`view_process_extension!`] extensions are also run, after `ext`.
257pub fn run_same_process_extended(run_app: impl FnOnce() + Send + 'static, ext: fn() -> ViewExtensions) {
258    let app_thread = thread::Builder::new()
259        .name("app".to_owned())
260        .spawn(move || {
261            // SAFETY: we exit the process in case of panic.
262            if let Err(e) = std::panic::catch_unwind(std::panic::AssertUnwindSafe(run_app)) {
263                thread::spawn(|| {
264                    // Sometimes the channel does not disconnect on panic,
265                    // observed this issue on a panic in `AppExtension::init`.
266                    //
267                    // This workaround ensures that don't become a zombie process.
268                    thread::sleep(std::time::Duration::from_secs(5));
269                    eprintln!("run_same_process did not exit after 5s of a fatal panic, exiting now");
270                    zng_env::exit(101);
271                });
272                // Propagate panic in case the normal disconnect/shutdown handler works.
273                std::panic::resume_unwind(e);
274            }
275        })
276        .unwrap();
277
278    let config = ViewConfig::wait_same_process();
279    config.assert_version(true);
280
281    let c = ipc::connect_view_process(config.server_name).expect("failed to connect to app in same process");
282
283    let mut ext = ext();
284    for e in extensions::VIEW_EXTENSIONS {
285        e(&mut ext);
286    }
287
288    if config.headless {
289        App::run_headless(c, ext);
290    } else {
291        App::run_headed(c, ext);
292    }
293
294    if let Err(p) = app_thread.join() {
295        std::panic::resume_unwind(p);
296    }
297}

322#[cfg(ipc)]
323fn panic_hook(info: &std::panic::PanicHookInfo, details: &str) {
324    // see `default_hook` in https://doc.rust-lang.org/src/std/panicking.rs.html#182
325
326    let panic = util::SuppressedPanic::from_hook(info, std::backtrace::Backtrace::force_capture());
327
328    if crate::util::suppress_panic() {
329        crate::util::set_suppressed_panic(panic);
330    } else {
331        eprintln!("{panic}\n{details}");
332        zng_env::exit(101) // Rust panic exit code.
333    }

406impl winit::application::ApplicationHandler<AppEvent> for App {
407    fn resumed(&mut self, winit_loop: &ActiveEventLoop) {
408        if let AppState::Suspended = self.app_state {
409            let mut winit_loop_guard = self.winit_loop.set(winit_loop);
410
411            self.exts.resumed();
412            self.generation = self.generation.next();
413
414            self.init(self.generation.next(), true, self.headless);
415
416            winit_loop_guard.unset(&mut self.winit_loop);
417        } else {
418            self.exts.init(&self.app_sender);
419        }
420        self.app_state = AppState::Resumed;
421
422        self.update_memory_watcher(winit_loop);
423    }
424
425    fn window_event(&mut self, winit_loop: &ActiveEventLoop, window_id: winit::window::WindowId, event: WindowEvent) {
426        let i = if let Some((i, _)) = self.windows.iter_mut().enumerate().find(|(_, w)| w.window_id() == window_id) {
427            i
428        } else {
429            return;
430        };
431
432        let _s = tracing::trace_span!("on_window_event", ?event).entered();
433
434        let mut winit_loop_guard = self.winit_loop.set(winit_loop);
435
436        self.windows[i].on_window_event(&event);
437
438        let id = self.windows[i].id();
439        let scale_factor = self.windows[i].scale_factor();
440
441        // Linux dialog is not actually modal, the parent window can continue to receive interaction events,
442        // this macro return early when a modal dialog is open in Linux.
443        #[cfg(any(
444            target_os = "linux",
445            target_os = "dragonfly",
446            target_os = "freebsd",
447            target_os = "netbsd",
448            target_os = "openbsd"
449        ))]
450        let modal_dialog_active = self.windows[i].modal_dialog_active();
451        #[cfg(any(
452            target_os = "linux",
453            target_os = "dragonfly",
454            target_os = "freebsd",
455            target_os = "netbsd",
456            target_os = "openbsd"
457        ))]
458        macro_rules! linux_modal_dialog_bail {
459            () => {
460                if modal_dialog_active {
461                    winit_loop_guard.unset(&mut self.winit_loop);
462                    return;
463                }
464            };
465        }
466        #[cfg(not(any(
467            target_os = "linux",
468            target_os = "dragonfly",
469            target_os = "freebsd",
470            target_os = "netbsd",
471            target_os = "openbsd"
472        )))]
473        macro_rules! linux_modal_dialog_bail {
474            () => {};
475        }
476
477        match event {
478            WindowEvent::RedrawRequested => self.windows[i].redraw(),
479            WindowEvent::Resized(_) => {
480                let size = if let Some(size) = self.windows[i].resized() {
481                    size
482                } else {
483                    winit_loop_guard.unset(&mut self.winit_loop);
484                    return;
485                };
486
487                // give the app 300ms to send a new frame, this is the collaborative way to
488                // resize, it should reduce the changes of the user seeing the clear color.
489
490                let deadline = Instant::now() + Duration::from_millis(300);
491
492                // await already pending frames.
493                if self.windows[i].is_rendering_frame() {
494                    tracing::debug!("resize requested while still rendering");
495
496                    // forward requests until webrender finishes or timeout.
497                    while let Ok(req) = self.request_recv.recv_deadline(deadline) {
498                        match req {
499                            RequestEvent::Request(req) => {
500                                let rsp = self.respond(req);
501                                if rsp.must_be_send() {
502                                    let _ = self.response_sender.send(rsp);
503                                }
504                            }
505                            RequestEvent::FrameReady(id, msg) => {
506                                self.on_frame_ready(id, msg);
507                                if id == self.windows[i].id() {
508                                    break;
509                                }
510                            }
511                        }
512                    }
513                }
514
515                if let Some(state) = self.windows[i].state_change() {
516                    self.notify(Event::WindowChanged(WindowChanged::state_changed(id, state, EventCause::System)));
517                }
518
519                if let Some(handle) = self.windows[i].monitor_change() {
520                    let m_id = self.monitor_handle_to_id(&handle);
521
522                    self.notify(Event::WindowChanged(WindowChanged::monitor_changed(id, m_id, EventCause::System)));
523                }
524
525                let wait_id = Some(self.resize_frame_wait_id_gen.incr());
526
527                // send event, the app code should send a frame in the new size as soon as possible.
528                self.notify(Event::WindowChanged(WindowChanged::resized(id, size, EventCause::System, wait_id)));
529
530                self.flush_coalesced();
531
532                // "modal" loop, breaks in 300ms or when a frame is received.
533                let mut received_frame = false;
534                loop {
535                    match self.request_recv.recv_deadline(deadline) {
536                        Ok(req) => {
537                            match req {
538                                RequestEvent::Request(req) => {
539                                    received_frame = req.is_frame(id, wait_id);
540                                    if received_frame || req.affects_window_rect(id) {
541                                        // received new frame
542                                        let rsp = self.respond(req);
543                                        if rsp.must_be_send() {
544                                            let _ = self.response_sender.send(rsp);
545                                        }
546                                        break;
547                                    } else {
548                                        // received some other request, forward it.
549                                        let rsp = self.respond(req);
550                                        if rsp.must_be_send() {
551                                            let _ = self.response_sender.send(rsp);
552                                        }
553                                    }
554                                }
555                                RequestEvent::FrameReady(id, msg) => self.on_frame_ready(id, msg),
556                            }
557                        }
558
559                        Err(flume::RecvTimeoutError::Timeout) => {
560                            // did not receive a new frame in time.
561                            break;
562                        }
563                        Err(flume::RecvTimeoutError::Disconnected) => {
564                            winit_loop_guard.unset(&mut self.winit_loop);
565                            unreachable!()
566                        }
567                    }
568                }
569
570                // if we are still within 300ms, await webrender.
571                if received_frame && deadline > Instant::now() {
572                    // forward requests until webrender finishes or timeout.
573                    while let Ok(req) = self.request_recv.recv_deadline(deadline) {
574                        match req {
575                            RequestEvent::Request(req) => {
576                                let rsp = self.respond(req);
577                                if rsp.must_be_send() {
578                                    let _ = self.response_sender.send(rsp);
579                                }
580                            }
581                            RequestEvent::FrameReady(id, msg) => {
582                                self.on_frame_ready(id, msg);
583                                if id == self.windows[i].id() {
584                                    break;
585                                }
586                            }
587                        }
588                    }
589                }
590            }
591            WindowEvent::Moved(_) => {
592                let (global_position, position) = if let Some(p) = self.windows[i].moved() {
593                    p
594                } else {
595                    winit_loop_guard.unset(&mut self.winit_loop);
596                    return;
597                };
598
599                if let Some(state) = self.windows[i].state_change() {
600                    self.notify(Event::WindowChanged(WindowChanged::state_changed(id, state, EventCause::System)));
601                }
602
603                self.notify(Event::WindowChanged(WindowChanged::moved(
604                    id,
605                    global_position,
606                    position,
607                    EventCause::System,
608                )));
609
610                if let Some(handle) = self.windows[i].monitor_change() {
611                    let m_id = self.monitor_handle_to_id(&handle);
612
613                    self.notify(Event::WindowChanged(WindowChanged::monitor_changed(id, m_id, EventCause::System)));
614                }
615            }
616            WindowEvent::CloseRequested => {
617                linux_modal_dialog_bail!();
618                self.notify(Event::WindowCloseRequested(id))
619            }
620            WindowEvent::Destroyed => {
621                self.windows.remove(i);
622                self.notify(Event::WindowClosed(id));
623            }
624            WindowEvent::HoveredFile(file) => {
625                linux_modal_dialog_bail!();
626
627                // winit does not provide mouse move events during drag/drop,
628                // so we enable device events to get mouse move, and use native APIs to get
629                // the cursor position on the window.
630                if self.device_events_filter.input.is_empty() {
631                    winit_loop.listen_device_events(winit::event_loop::DeviceEvents::Always);
632                }
633                self.drag_drop_hovered = Some((id, DipPoint::splat(Dip::new(-1000))));
634                self.notify(Event::DragHovered {
635                    window: id,
636                    data: vec![DragDropData::Path(file)],
637                    allowed: DragDropEffect::all(),
638                });
639            }
640            WindowEvent::DroppedFile(file) => {
641                linux_modal_dialog_bail!();
642
643                if self.device_events_filter.input.is_empty() {
644                    winit_loop.listen_device_events(winit::event_loop::DeviceEvents::Never);
645                }
646
647                let mut delay_to_next_move = true;
648
649                // some systems (x11) don't receive even device mouse move
650                if let Some(position) = self.windows[i].drag_drop_cursor_pos() {
651                    self.notify(Event::DragMoved {
652                        window: id,
653                        coalesced_pos: vec![],
654                        position,
655                    });
656                    delay_to_next_move = false;
657                } else if let Some((_, pos)) = self.drag_drop_hovered {
658                    delay_to_next_move = pos.x < Dip::new(0);
659                }
660
661                if delay_to_next_move {
662                    self.drag_drop_next_move = Some((Instant::now(), file));
663                } else {
664                    self.notify(Event::DragDropped {
665                        window: id,
666                        data: vec![DragDropData::Path(file)],
667                        allowed: DragDropEffect::all(),
668                        drop_id: DragDropId(0),
669                    });
670                }
671            }
672            WindowEvent::HoveredFileCancelled => {
673                linux_modal_dialog_bail!();
674
675                self.drag_drop_hovered = None;
676                if self.device_events_filter.input.is_empty() {
677                    winit_loop.listen_device_events(winit::event_loop::DeviceEvents::Never);
678                }
679
680                if self.drag_drop_next_move.is_none() {
681                    // x11 sends a cancelled after drop
682                    self.notify(Event::DragCancelled { window: id });
683                }
684            }
685            WindowEvent::Focused(mut focused) => {
686                if self.windows[i].focused_changed(&mut focused) {
687                    if focused {
688                        self.notify(Event::FocusChanged { prev: None, new: Some(id) });
689
690                        // some platforms (Wayland) don't change size on minimize/restore, so we check here too.
691                        if let Some(state) = self.windows[i].state_change() {
692                            self.notify(Event::WindowChanged(WindowChanged::state_changed(id, state, EventCause::System)));
693                        }
694                    } else {
695                        self.pending_modifiers_focus_clear = true;
696                        self.notify(Event::FocusChanged { prev: Some(id), new: None });
697                    }
698                }
699            }
700            WindowEvent::KeyboardInput {
701                device_id,
702                event,
703                is_synthetic,
704            } => {
705                linux_modal_dialog_bail!();
706
707                if !is_synthetic && self.windows[i].is_focused() {
708                    // see the Window::focus comments.
709                    #[cfg(windows)]
710                    if self.skip_ralt
711                        && let winit::keyboard::PhysicalKey::Code(winit::keyboard::KeyCode::AltRight) = event.physical_key
712                    {
713                        winit_loop_guard.unset(&mut self.winit_loop);
714                        return;
715                    }
716
717                    let state = util::element_state_to_key_state(event.state);
718                    #[cfg(not(target_os = "android"))]
719                    let key = util::winit_key_to_key(event.key_without_modifiers());
720                    let key_modified = util::winit_key_to_key(event.logical_key);
721                    #[cfg(target_os = "android")]
722                    let key = key_modified.clone();
723                    let key_code = util::winit_physical_key_to_key_code(event.physical_key);
724                    let key_location = util::winit_key_location_to_zng(event.location);
725                    let d_id = self.input_device_id(device_id, InputDeviceCapability::KEY);
726
727                    let mut send_event = true;
728
729                    if key.is_modifier() {
730                        match state {
731                            KeyState::Pressed => {
732                                send_event = self
733                                    .pressed_modifiers
734                                    .insert((key.clone(), key_location), (d_id, key_code))
735                                    .is_none();
736                            }
737                            KeyState::Released => send_event = self.pressed_modifiers.remove(&(key.clone(), key_location)).is_some(),
738                        }
739                    }
740
741                    if send_event {
742                        self.notify(Event::KeyboardInput {
743                            window: id,
744                            device: d_id,
745                            key_code,
746                            key_location,
747                            state,
748                            text: match event.text {
749                                Some(s) => Txt::from_str(s.as_str()),
750                                #[cfg(target_os = "android")]
751                                None => match (state, &key) {
752                                    (KeyState::Pressed, Key::Char(c)) => Txt::from(*c),
753                                    (KeyState::Pressed, Key::Str(s)) => s.clone(),
754                                    _ => Txt::default(),
755                                },
756                                #[cfg(not(target_os = "android"))]
757                                None => Txt::default(),
758                            },
759                            key,
760                            key_modified,
761                        });
762                    }
763                }
764            }
765            WindowEvent::ModifiersChanged(m) => {
766                linux_modal_dialog_bail!();
767                if self.windows[i].is_focused() {
768                    self.pending_modifiers_update = Some(m.state());
769                }
770            }
771            WindowEvent::CursorMoved { device_id, position, .. } => {
772                linux_modal_dialog_bail!();
773
774                let px_p = position.to_px();
775                let p = px_p.to_dip(scale_factor);
776                let d_id = self.input_device_id(device_id, InputDeviceCapability::POINTER_MOTION);
777
778                let mut is_after_cursor_enter = false;
779                if let Some(i) = self.cursor_entered_expect_move.iter().position(|&w| w == id) {
780                    self.cursor_entered_expect_move.remove(i);
781                    is_after_cursor_enter = true;
782                }
783
784                if self.windows[i].cursor_moved(p, d_id) || is_after_cursor_enter {
785                    self.notify(Event::MouseMoved {
786                        window: id,
787                        device: d_id,
788                        coalesced_pos: vec![],
789                        position: p,
790                    });
791                }
792
793                if let Some((drop_moment, file)) = self.drag_drop_next_move.take()
794                    && drop_moment.elapsed() < Duration::from_millis(300)
795                {
796                    let window_id = self.windows[i].id();
797                    self.notify(Event::DragMoved {
798                        window: window_id,
799                        coalesced_pos: vec![],
800                        position: p,
801                    });
802                    self.notify(Event::DragDropped {
803                        window: window_id,
804                        data: vec![DragDropData::Path(file)],
805                        allowed: DragDropEffect::all(),
806                        drop_id: DragDropId(0),
807                    });
808                }
809            }
810            WindowEvent::CursorEntered { device_id } => {
811                linux_modal_dialog_bail!();
812                if self.windows[i].cursor_entered() {
813                    let d_id = self.input_device_id(device_id, InputDeviceCapability::POINTER_MOTION);
814                    self.notify(Event::MouseEntered { window: id, device: d_id });
815                    self.cursor_entered_expect_move.push(id);
816                }
817            }
818            WindowEvent::CursorLeft { device_id } => {
819                linux_modal_dialog_bail!();
820                if self.windows[i].cursor_left() {
821                    let d_id = self.input_device_id(device_id, InputDeviceCapability::POINTER_MOTION);
822                    self.notify(Event::MouseLeft { window: id, device: d_id });
823
824                    // unlikely but possible?
825                    if let Some(i) = self.cursor_entered_expect_move.iter().position(|&w| w == id) {
826                        self.cursor_entered_expect_move.remove(i);
827                    }
828                }
829            }
830            WindowEvent::MouseWheel {
831                device_id, delta, phase, ..
832            } => {
833                linux_modal_dialog_bail!();
834                let d_id = self.input_device_id(device_id, InputDeviceCapability::SCROLL_MOTION);
835                self.notify(Event::MouseWheel {
836                    window: id,
837                    device: d_id,
838                    delta: util::winit_mouse_wheel_delta_to_zng(delta),
839                    phase: util::winit_touch_phase_to_zng(phase),
840                });
841            }
842            WindowEvent::MouseInput {
843                device_id, state, button, ..
844            } => {
845                linux_modal_dialog_bail!();
846                let d_id = self.input_device_id(device_id, InputDeviceCapability::BUTTON);
847                self.notify(Event::MouseInput {
848                    window: id,
849                    device: d_id,
850                    state: util::element_state_to_button_state(state),
851                    button: util::winit_mouse_button_to_zng(button),
852                });
853            }
854            WindowEvent::TouchpadPressure {
855                device_id,
856                pressure,
857                stage,
858            } => {
859                linux_modal_dialog_bail!();
860                let d_id = self.input_device_id(device_id, InputDeviceCapability::empty());
861                self.notify(Event::TouchpadPressure {
862                    window: id,
863                    device: d_id,
864                    pressure,
865                    stage,
866                });
867            }
868            WindowEvent::AxisMotion { device_id, axis, value } => {
869                linux_modal_dialog_bail!();
870                let d_id = self.input_device_id(device_id, InputDeviceCapability::AXIS_MOTION);
871                self.notify(Event::AxisMotion {
872                    window: id,
873                    device: d_id,
874                    axis: AxisId(axis),
875                    value,
876                });
877            }
878            WindowEvent::Touch(t) => {
879                let d_id = self.input_device_id(t.device_id, InputDeviceCapability::empty());
880                let position = t.location.to_px().to_dip(scale_factor);
881
882                let notify = match t.phase {
883                    winit::event::TouchPhase::Moved => self.windows[i].touch_moved(position, d_id, t.id),
884                    winit::event::TouchPhase::Started => true,
885                    winit::event::TouchPhase::Ended | winit::event::TouchPhase::Cancelled => {
886                        self.windows[i].touch_end(d_id, t.id);
887                        true
888                    }
889                };
890
891                if notify {
892                    self.notify(Event::Touch {
893                        window: id,
894                        device: d_id,
895                        touches: vec![TouchUpdate::new(
896                            TouchId(t.id),
897                            util::winit_touch_phase_to_zng(t.phase),
898                            position,
899                            t.force.map(util::winit_force_to_zng),
900                        )],
901                    });
902                }
903            }
904            WindowEvent::ScaleFactorChanged { scale_factor, .. } => {
905                let monitor;
906                let mut is_monitor_change = false;
907
908                if let Some(new_monitor) = self.windows[i].monitor_change() {
909                    monitor = Some(new_monitor);
910                    is_monitor_change = true;
911                } else {
912                    monitor = self.windows[i].monitor();
913                }
914
915                let monitor = if let Some(handle) = monitor {
916                    self.monitor_handle_to_id(&handle)
917                } else {
918                    MonitorId::INVALID
919                };
920
921                if is_monitor_change {
922                    self.notify(Event::WindowChanged(WindowChanged::monitor_changed(
923                        id,
924                        monitor,
925                        EventCause::System,
926                    )));
927                }
928                self.notify(Event::ScaleFactorChanged {
929                    monitor,
930                    windows: vec![id],
931                    scale_factor: scale_factor as f32,
932                });
933
934                if let Some(size) = self.windows[i].resized() {
935                    self.notify(Event::WindowChanged(WindowChanged::resized(id, size, EventCause::System, None)));
936                }
937            }
938            WindowEvent::Ime(ime) => {
939                linux_modal_dialog_bail!();
940
941                match ime {
942                    winit::event::Ime::Preedit(s, c) => {
943                        let caret = c.unwrap_or((s.len(), s.len()));
944                        let ime = Ime::Preview(s.into(), caret);
945                        self.notify(Event::Ime { window: id, ime });
946                    }
947                    winit::event::Ime::Commit(s) => {
948                        let ime = Ime::Commit(s.into());
949                        self.notify(Event::Ime { window: id, ime });
950                    }
951                    winit::event::Ime::Enabled => {}
952                    winit::event::Ime::Disabled => {}
953                }
954            }
955            WindowEvent::ThemeChanged(_) => {}
956            WindowEvent::Occluded(_) => {}
957            WindowEvent::ActivationTokenDone { .. } => {}
958            WindowEvent::PinchGesture { .. } => {}
959            WindowEvent::RotationGesture { .. } => {}
960            WindowEvent::DoubleTapGesture { .. } => {}
961            WindowEvent::PanGesture { .. } => {}
962        }
963
964        winit_loop_guard.unset(&mut self.winit_loop);
965    }

976    fn user_event(&mut self, winit_loop: &ActiveEventLoop, ev: AppEvent) {
977        let mut winit_loop_guard = self.winit_loop.set(winit_loop);
978        match ev {
979            AppEvent::Request => {
980                while let Ok(req) = self.request_recv.try_recv() {
981                    match req {
982                        RequestEvent::Request(req) => {
983                            let rsp = self.respond(req);
984                            if rsp.must_be_send() && self.response_sender.send(rsp).is_err() {
985                                // lost connection to app-process
986                                self.exited = true;
987                                self.winit_loop.exit();
988                            }
989                        }
990                        RequestEvent::FrameReady(wid, msg) => self.on_frame_ready(wid, msg),
991                    }
992                }
993            }
994            AppEvent::Notify(ev) => self.notify(ev),
995            AppEvent::WinitFocused(window_id, focused) => self.window_event(winit_loop, window_id, WindowEvent::Focused(focused)),
996            AppEvent::RefreshMonitors => self.refresh_monitors(),
997            AppEvent::ParentProcessExited => {
998                self.exited = true;
999                self.winit_loop.exit();
1000            }
1001            AppEvent::ImageLoaded(data) => {
1002                self.image_cache.loaded(data);
1003            }
1004            AppEvent::MonitorPowerChanged => {
1005                // if a window opens in power-off it is blank until redraw.
1006                for w in &mut self.windows {
1007                    w.redraw();
1008                }
1009            }
1010            AppEvent::SetDeviceEventsFilter(filter) => {
1011                self.set_device_events_filter(filter, Some(winit_loop));
1012            }
1013        }
1014        winit_loop_guard.unset(&mut self.winit_loop);
1015    }

1017    fn device_event(&mut self, winit_loop: &ActiveEventLoop, device_id: winit::event::DeviceId, event: DeviceEvent) {
1018        let filter = self.device_events_filter.input;
1019
1020        if !filter.is_empty() {
1021            let _s = tracing::trace_span!("on_device_event", ?event);
1022
1023            let mut winit_loop_guard = self.winit_loop.set(winit_loop);
1024
1025            match &event {
1026                DeviceEvent::Added => {
1027                    let _ = self.input_device_id(device_id, InputDeviceCapability::empty());
1028                    // already notifies here
1029                }
1030                DeviceEvent::Removed => {
1031                    if let Some(i) = self.devices.iter().position(|(_, id, _)| *id == device_id) {
1032                        self.devices.remove(i);
1033                        self.notify_input_devices_changed();
1034                    }
1035                }
1036                DeviceEvent::MouseMotion { delta } => {
1037                    let cap = InputDeviceCapability::POINTER_MOTION;
1038                    if filter.contains(cap) {
1039                        let d_id = self.input_device_id(device_id, cap);
1040                        self.notify(Event::InputDeviceEvent {
1041                            device: d_id,
1042                            event: InputDeviceEvent::PointerMotion {
1043                                delta: euclid::vec2(delta.0, delta.1),
1044                            },
1045                        });
1046                    }
1047                }
1048                DeviceEvent::MouseWheel { delta } => {
1049                    let cap = InputDeviceCapability::SCROLL_MOTION;
1050                    if filter.contains(cap) {
1051                        let d_id = self.input_device_id(device_id, cap);
1052                        self.notify(Event::InputDeviceEvent {
1053                            device: d_id,
1054                            event: InputDeviceEvent::ScrollMotion {
1055                                delta: util::winit_mouse_wheel_delta_to_zng(*delta),
1056                            },
1057                        });
1058                    }
1059                }
1060                DeviceEvent::Motion { axis, value } => {
1061                    let cap = InputDeviceCapability::AXIS_MOTION;
1062                    if filter.contains(cap) {
1063                        let d_id = self.input_device_id(device_id, cap);
1064                        self.notify(Event::InputDeviceEvent {
1065                            device: d_id,
1066                            event: InputDeviceEvent::AxisMotion {
1067                                axis: AxisId(*axis),
1068                                value: *value,
1069                            },
1070                        });
1071                    }
1072                }
1073                DeviceEvent::Button { button, state } => {
1074                    let cap = InputDeviceCapability::BUTTON;
1075                    if filter.contains(cap) {
1076                        let d_id = self.input_device_id(device_id, cap);
1077                        self.notify(Event::InputDeviceEvent {
1078                            device: d_id,
1079                            event: InputDeviceEvent::Button {
1080                                button: ButtonId(*button),
1081                                state: util::element_state_to_button_state(*state),
1082                            },
1083                        });
1084                    }
1085                }
1086                DeviceEvent::Key(k) => {
1087                    let cap = InputDeviceCapability::KEY;
1088                    if filter.contains(cap) {
1089                        let d_id = self.input_device_id(device_id, cap);
1090                        self.notify(Event::InputDeviceEvent {
1091                            device: d_id,
1092                            event: InputDeviceEvent::Key {
1093                                key_code: util::winit_physical_key_to_key_code(k.physical_key),
1094                                state: util::element_state_to_key_state(k.state),
1095                            },
1096                        });
1097                    }
1098                }
1099            }
1100
1101            winit_loop_guard.unset(&mut self.winit_loop);
1102        }
1103
1104        if let Some((id, pos)) = &mut self.drag_drop_hovered
1105            && let DeviceEvent::MouseMotion { .. } = &event
1106            && let Some(win) = self.windows.iter().find(|w| w.id() == *id)
1107            && let Some(new_pos) = win.drag_drop_cursor_pos()
1108            && *pos != new_pos
1109        {
1110            *pos = new_pos;
1111            let event = Event::DragMoved {
1112                window: *id,
1113                coalesced_pos: vec![],
1114                position: *pos,
1115            };
1116            self.notify(event);
1117        }
1118    }

1255            fn resumed(&mut self, winit_loop: &ActiveEventLoop) {
1256                let mut winit_loop_guard = self.app.winit_loop.set(winit_loop);
1257
1258                self.app.resumed(winit_loop);
1259                self.app.start_receiving(self.request_receiver.take().unwrap());
1260
1261                'app_loop: while !self.app.exited {
1262                    match self.app_receiver.recv() {
1263                        Ok(app_ev) => match app_ev {
1264                            AppEvent::Request => {
1265                                while let Ok(request) = self.app.request_recv.try_recv() {
1266                                    match request {
1267                                        RequestEvent::Request(request) => {
1268                                            let response = self.app.respond(request);
1269                                            if response.must_be_send() && self.app.response_sender.send(response).is_err() {
1270                                                self.app.exited = true;
1271                                                break 'app_loop;
1272                                            }
1273                                        }
1274                                        RequestEvent::FrameReady(id, msg) => {
1275                                            let r = if let Some(s) = self.app.surfaces.iter_mut().find(|s| s.id() == id) {
1276                                                Some(s.on_frame_ready(msg, &mut self.app.image_cache))
1277                                            } else {

1295                            }

1314                        }
1315                    }
1316                }
1317
1318                self.app.winit_loop.exit();
1319

1321            }

1323            fn window_event(&mut self, _: &ActiveEventLoop, _: winit::window::WindowId, _: WindowEvent) {}

1568    pub(crate) fn notify(&mut self, event: Event) {
1569        let now = Instant::now();
1570        if let Some((mut coal, timestamp)) = self.coalescing_event.take() {
1571            let r = if now.saturating_duration_since(timestamp) >= Duration::from_millis(16) {
1572                Err(event)
1573            } else {
1574                coal.coalesce(event)
1575            };
1576            match r {
1577                Ok(()) => self.coalescing_event = Some((coal, timestamp)),
1578                Err(event) => match (&mut coal, event) {
1579                    (
1580                        Event::KeyboardInput {
1581                            window,
1582                            device,
1583                            state,
1584                            text,
1585                            ..
1586                        },
1587                        Event::KeyboardInput {
1588                            window: n_window,
1589                            device: n_device,
1590                            text: n_text,
1591                            ..
1592                        },
1593                    ) if !n_text.is_empty() && *window == n_window && *device == n_device && *state == KeyState::Pressed => {
1594                        // text after key-press
1595                        if text.is_empty() {
1596                            *text = n_text;
1597                        } else {
1598                            text.push_str(&n_text);
1599                        };
1600                        self.coalescing_event = Some((coal, now));
1601                    }
1602                    (_, event) => {
1603                        let mut error = self.event_sender.send(coal).is_err();
1604                        error |= self.event_sender.send(event).is_err();
1605
1606                        if error {
1607                            let _ = self.app_sender.send(AppEvent::ParentProcessExited);
1608                        }
1609                    }
1610                },
1611            }
1612        } else {
1613            self.coalescing_event = Some((event, now));
1614        }
1615
1616        if self.headless {
1617            self.flush_coalesced();
1618        }
1619    }

2455/// Message inserted in the request loop from the view-process.
2456///
2457/// These *events* are detached from [`AppEvent`] so that we can continue receiving requests while
2458/// the main loop is blocked in a resize operation.
2459#[allow(clippy::large_enum_variant)] // Request is the largest, but also most common
2460#[derive(Debug)]
2461enum RequestEvent {
2462    /// A request from the [`Api`].
2463    Request(Request),
2464    /// Webrender finished rendering a frame, ready for redraw.
2465    FrameReady(WindowId, FrameReadyMsg),
2466}
2467
2468#[derive(Debug)]
2469pub(crate) struct FrameReadyMsg {
2470    pub composite_needed: bool,
2471}

2473/// Abstraction over channel senders that can inject [`AppEvent`] in the app loop.
2474#[derive(Clone)]
2475pub(crate) enum AppEventSender {
2476    Headed(EventLoopProxy<AppEvent>, flume::Sender<RequestEvent>),
2477    Headless(flume::Sender<AppEvent>, flume::Sender<RequestEvent>),
2478}
2479impl AppEventSender {
2480    /// Send an event.
2481    fn send(&self, ev: AppEvent) -> Result<(), ipc::ViewChannelError> {
2482        match self {
2483            AppEventSender::Headed(p, _) => p.send_event(ev).map_err(|_| ipc::ViewChannelError::Disconnected),
2484            AppEventSender::Headless(p, _) => p.send(ev).map_err(|_| ipc::ViewChannelError::Disconnected),
2485        }
2486    }
2487
2488    /// Send a request.
2489    fn request(&self, req: Request) -> Result<(), ipc::ViewChannelError> {
2490        match self {
2491            AppEventSender::Headed(_, p) => p.send(RequestEvent::Request(req)).map_err(|_| ipc::ViewChannelError::Disconnected),
2492            AppEventSender::Headless(_, p) => p.send(RequestEvent::Request(req)).map_err(|_| ipc::ViewChannelError::Disconnected),
2493        }?;
2494        self.send(AppEvent::Request)
2495    }
2496
2497    /// Send a frame-ready.
2498    fn frame_ready(&self, window_id: WindowId, msg: FrameReadyMsg) -> Result<(), ipc::ViewChannelError> {
2499        match self {
2500            AppEventSender::Headed(_, p) => p
2501                .send(RequestEvent::FrameReady(window_id, msg))
2502                .map_err(|_| ipc::ViewChannelError::Disconnected),
2503            AppEventSender::Headless(_, p) => p
2504                .send(RequestEvent::FrameReady(window_id, msg))
2505                .map_err(|_| ipc::ViewChannelError::Disconnected),
2506        }?;
2507        self.send(AppEvent::Request)
2508    }
2509}
