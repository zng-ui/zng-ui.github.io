97impl IpcDynamicImage {
98    pub fn decode(buf: &[u8], format: ImageFormat, entry: usize) -> image::ImageResult<Self> {
99        match format {
100            #[cfg(feature = "image_ico")]
101            ImageFormat::Ico => return Self::decode_ico(buf, entry),
102            #[cfg(feature = "image_tiff")]
103            ImageFormat::Tiff => return Self::decode_tiff(buf, entry),
104            _ => {}
105        }
106
107        let buf = std::io::Cursor::new(buf);
108
109        let mut reader = image::ImageReader::new(buf);
110        reader.set_format(format);
111        reader.no_limits();
112
113        let decoder = reader.into_decoder()?;
114
115        let (w, h) = decoder.dimensions();
116        let color_type = decoder.color_type();
117
118        let mut buf = Self::alloc_buf(decoder.total_bytes())?;
119        decoder.read_image(&mut buf[..])?;
120
121        Self::from_decoded(buf, color_type, w, h)
122    }
123
124    #[cfg(feature = "image_ico")]
125    fn decode_ico(buf: &[u8], entry: usize) -> image::ImageResult<Self> {
126        let buf = std::io::Cursor::new(buf);
127
128        let icon = ico::IconDir::read(buf)?;
129
130        let entry = icon.entries()[entry].decode()?;
131        let (w, h) = (entry.width(), entry.height());
132
133        let buf = IpcBytesMut::from_vec_blocking(entry.into_rgba_data())?;
134
135        Self::from_decoded(buf, ColorType::Rgba8, w, h)
136    }
137
138    #[cfg(feature = "image_tiff")]
139    fn decode_tiff(buf: &[u8], entry: usize) -> image::ImageResult<Self> {
140        let buf = std::io::Cursor::new(buf);
141
142        let mut tiff = tiff::decoder::Decoder::new(buf).map_err(tiff_error)?;
143        tiff.seek_to_image(entry).map_err(tiff_error)?;
144
145        let (w, h) = tiff.dimensions().map_err(tiff_error)?;
146        let color_type = tiff_color_type(&mut tiff)?;
147
148        let total_bytes = tiff.image_buffer_layout().map_err(tiff_error)?.len;
149        let mut buf = Self::alloc_buf(total_bytes as u64)?;
150
151        tiff.read_image_bytes(&mut buf[..]).map_err(tiff_error)?;
152
153        Self::from_decoded(buf, color_type, w, h)
154    }
155
156    fn from_decoded(buf: IpcBytesMut, color_type: image::ColorType, w: u32, h: u32) -> image::ImageResult<Self> {
157        // copied from image-0.25.9\src\images\dynimage.rs
158        match color_type {
159            ColorType::Rgb8 => ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgb8),
160            ColorType::Rgba8 => ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgba8),
161            ColorType::L8 => ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageLuma8),
162            ColorType::La8 => ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageLumaA8),
163            ColorType::Rgb16 => ImageBuffer::from_raw(w, h, buf.cast()).map(IpcDynamicImage::ImageRgb16),
164            ColorType::Rgba16 => ImageBuffer::from_raw(w, h, buf.cast()).map(IpcDynamicImage::ImageRgba16),
165            ColorType::Rgb32F => ImageBuffer::from_raw(w, h, buf.cast()).map(IpcDynamicImage::ImageRgb32F),
166            ColorType::Rgba32F => ImageBuffer::from_raw(w, h, buf.cast()).map(IpcDynamicImage::ImageRgba32F),
167            ColorType::L16 => ImageBuffer::from_raw(w, h, buf.cast()).map(IpcDynamicImage::ImageLuma16),
168            ColorType::La16 => ImageBuffer::from_raw(w, h, buf.cast()).map(IpcDynamicImage::ImageLumaA16),
169            _ => unreachable!(),
170        }
171        .ok_or_else(|| ImageError::Parameter(ParameterError::from_kind(ParameterErrorKind::DimensionMismatch)))
172    }
173
174    fn alloc_buf(len: u64) -> image::ImageResult<IpcBytesMut> {
175        if len > usize::MAX as u64 {
176            return Err(ImageError::Limits(LimitError::from_kind(LimitErrorKind::InsufficientMemory)));
177        }
178
179        let buf = IpcBytes::new_mut_blocking(len as usize)?;
180        Ok(buf)
181    }
182
183    pub fn dimensions(&self) -> (u32, u32) {
184        dynamic_map!(*self, ref p, p.dimensions())
185    }
186}

188#[cfg(feature = "image_tiff")]
189fn tiff_error(e: tiff::TiffError) -> image::ImageError {
190    // https://docs.rs/image/0.25.9/src/image/codecs/tiff.rs.html#190
191    match e {
192        tiff::TiffError::IoError(err) => image::ImageError::IoError(err),
193        err @ (tiff::TiffError::FormatError(_) | tiff::TiffError::IntSizeError | tiff::TiffError::UsageError(_)) => {
194            image::ImageError::Decoding(DecodingError::new(ImageFormat::Tiff.into(), err))
195        }
196        tiff::TiffError::UnsupportedError(desc) => image::ImageError::Unsupported(UnsupportedError::from_format_and_kind(
197            ImageFormat::Tiff.into(),
198            UnsupportedErrorKind::GenericFeature(desc.to_string()),
199        )),
200        tiff::TiffError::LimitsExceeded => image::ImageError::Limits(LimitError::from_kind(LimitErrorKind::InsufficientMemory)),
201    }
202}
203#[cfg(feature = "image_tiff")]
204pub(crate) fn tiff_color_type<R: std::io::Read + std::io::Seek>(tiff: &mut tiff::decoder::Decoder<R>) -> image::ImageResult<ColorType> {
205    // https://docs.rs/image/0.25.9/src/image/codecs/tiff.rs.html#182
206    fn err_unknown_color_type(value: u8) -> ImageError {
207        ImageError::Unsupported(UnsupportedError::from_format_and_kind(
208            ImageFormat::Tiff.into(),
209            UnsupportedErrorKind::Color(ExtendedColorType::Unknown(value)),
210        ))
211    }
212    // https://docs.rs/image/0.25.9/src/image/codecs/tiff.rs.html#74
213    let color_type = match tiff.colortype().map_err(tiff_error)? {
214        tiff::ColorType::Gray(1) => ColorType::L8,
215        tiff::ColorType::Gray(8) => ColorType::L8,
216        tiff::ColorType::Gray(16) => ColorType::L16,
217        tiff::ColorType::GrayA(8) => ColorType::La8,
218        tiff::ColorType::GrayA(16) => ColorType::La16,
219        tiff::ColorType::RGB(8) => ColorType::Rgb8,
220        tiff::ColorType::RGB(16) => ColorType::Rgb16,
221        tiff::ColorType::RGBA(8) => ColorType::Rgba8,
222        tiff::ColorType::RGBA(16) => ColorType::Rgba16,
223        tiff::ColorType::CMYK(8) => ColorType::Rgb8,
224        tiff::ColorType::CMYK(16) => ColorType::Rgb16,
225        tiff::ColorType::RGB(32) => ColorType::Rgb32F,
226        tiff::ColorType::RGBA(32) => ColorType::Rgba32F,
227
228        tiff::ColorType::Palette(n) | tiff::ColorType::Gray(n) => return Err(err_unknown_color_type(n)),
229        tiff::ColorType::GrayA(n) => return Err(err_unknown_color_type(n.saturating_mul(2))),
230        tiff::ColorType::RGB(n) => return Err(err_unknown_color_type(n.saturating_mul(3))),
231        tiff::ColorType::YCbCr(n) => return Err(err_unknown_color_type(n.saturating_mul(3))),
232        tiff::ColorType::RGBA(n) | tiff::ColorType::CMYK(n) => return Err(err_unknown_color_type(n.saturating_mul(4))),
233        tiff::ColorType::Multiband { bit_depth, num_samples } => {
234            return Err(err_unknown_color_type(bit_depth.saturating_mul(num_samples.min(255) as u8)));
235        }
236        _ => return Err(err_unknown_color_type(0)),
237    };
238
239    Ok(color_type)
240}
241#[cfg(feature = "image_tiff")]
242pub(crate) fn tiff_orientation<R: std::io::Read + std::io::Seek>(
243    tiff: &mut tiff::decoder::Decoder<R>,
244) -> image::ImageResult<image::metadata::Orientation> {
245    // https://docs.rs/image/latest/src/image/codecs/tiff.rs.html#290
246    let orientation = tiff.find_tag(tiff::tags::Tag::Orientation).map_err(tiff_error)?;
247    let orientation = orientation.and_then(|v| image::metadata::Orientation::from_exif(v.into_u16().ok()?.min(255) as u8));
248    Ok(orientation.unwrap_or(image::metadata::Orientation::NoTransforms))
249}
250#[cfg(feature = "image_tiff")]
251pub(crate) fn tiff_density<R: std::io::Read + std::io::Seek>(tiff: &mut tiff::decoder::Decoder<R>) -> Option<zng_unit::PxDensity2d> {
252    use zng_unit::PxDensity2d;
253    use zng_unit::PxDensityUnits as _;
254
255    use tiff::{decoder::ifd::Value, tags::Tag};
256    let res_unit = tiff.get_tag(Tag::ResolutionUnit).ok().and_then(|t| t.into_u16().ok()).unwrap_or(2);
257    if let Ok(Value::Rational(x_num, x_denom)) = tiff.get_tag(Tag::XResolution)
258        && let Ok(Value::Rational(y_num, y_denom)) = tiff.get_tag(Tag::YResolution)
259    {
260        let x = x_num as f32 / x_denom as f32;
261        let y = y_num as f32 / y_denom as f32;
262        match res_unit {
263            // inches
264            2 => {
265                return Some(PxDensity2d::new(x.ppi(), y.ppi()));
266            }
267            // centimeters
268            3 => {
269                return Some(PxDensity2d::new(x.ppcm(), y.ppcm()));
270            }
271            _ => {}
272        }
273    }
274    None
275}

282#[cfg(feature = "image_jpeg")]
283pub(crate) fn jpeg_density(data: &[u8]) -> Option<zng_unit::PxDensity2d> {
284    use zng_unit::PxDensity2d;
285    use zng_unit::PxDensityUnits as _;
286
287    // `image` uses `zune-jpeg`, that decoder does not parse density correctly,
288    // so we do it manually here
289    fn parse_density(data: &[u8]) -> Option<(u8, u16, u16)> {
290        let mut i = 0;
291        while i + 4 < data.len() {
292            // APP0
293            if data[i] == 0xFF && data[i + 1] == 0xE0 {
294                let len = u16::from_be_bytes([data[i + 2], data[i + 3]]) as usize;
295                if i + 2 + len > data.len() {
296                    break;
297                }
298
299                // APP0 payload starts at i+4, identifier is 5 bytes: "JFIF\0"
300                let p = i + 4;
301                if &data[p..p + 5] == b"JFIF\0" && p + 14 <= data.len() {
302                    let unit = data[p + 7];
303                    let x = u16::from_be_bytes([data[p + 8], data[p + 9]]);
304                    let y = u16::from_be_bytes([data[p + 10], data[p + 11]]);
305                    return Some((unit, x, y));
306                }
307
308                i += 2 + len;
309            } else if data[i] == 0xFF && data[i + 1] == 0xDA {
310                // Start of Scan
311                break;
312            } else {
313                i += 1;
314            }
315        }
316        None
317    }
318    if let Some((unit, x, y)) = parse_density(data) {
319        match unit {
320            // inches
321            1 => {
322                return Some(PxDensity2d::new((x as f32).ppi(), (y as f32).ppi()));
323            }
324            // centimeters
325            2 => {
326                return Some(PxDensity2d::new((x as f32).ppcm(), (y as f32).ppcm()));
327            }
328            _ => {}
329        }
330    }
331    None
