20    /// Create frame_image for an `Api::frame_image` request.
21    pub fn frame_image(
22        &mut self,
23        gl: &dyn gleam::gl::Gl,
24        rect: PxRect,
25        window_id: WindowId,
26        frame_id: FrameId,
27        scale_factor: Factor,
28        mask: Option<ImageMaskMode>,
29    ) -> ImageId {
30        if frame_id == FrameId::INVALID {
31            let id = self.image_id_gen.incr();
32            let _ = self.app_sender.send(AppEvent::Notify(Event::ImageLoadError {
33                image: id,
34                error: formatx!("no frame rendered in window `{window_id:?}`"),
35            }));
36            let _ = self.app_sender.send(AppEvent::Notify(Event::FrameImageReady {
37                window: window_id,
38                frame: frame_id,
39                image: id,
40                selection: rect,
41            }));
42            return id;
43        }
44
45        let data = self.frame_image_data(gl, rect, scale_factor, mask);
46
47        let id = data.id;
48
49        let _ = self.app_sender.send(AppEvent::ImageLoaded(data));
50        let _ = self.app_sender.send(AppEvent::Notify(Event::FrameImageReady {
51            window: window_id,
52            frame: frame_id,
53            image: id,
54            selection: rect,
55        }));
56
57        id
58    }

60    /// Create frame_image for a capture request in the FrameRequest.
61    pub fn frame_image_data(
62        &mut self,
63        gl: &dyn gleam::gl::Gl,
64        rect: PxRect,
65        scale_factor: Factor,
66        mask: Option<ImageMaskMode>,
67    ) -> ImageLoadedData {
68        let data = self.frame_image_data_impl(gl, rect, scale_factor, mask);
69
70        self.images.insert(
71            data.id,
72            Image(Arc::new(ImageData::RawData {
73                size: data.size,
74                range: 0..data.pixels.len(),
75                pixels: data.pixels.clone(),
76                is_opaque: data.is_opaque,
77                density: data.density,
78                mipmap: Mutex::new(Box::new([])),
79                stripes: Mutex::new(Box::new([])),
80            })),
81        );
82
83        data
84    }

86    fn frame_image_data_impl(
87        &mut self,
88        gl: &dyn gleam::gl::Gl,
89        rect: PxRect,
90        scale_factor: Factor,
91        mask: Option<ImageMaskMode>,
92    ) -> ImageLoadedData {
93        let format = match gl.get_type() {
94            gleam::gl::GlType::Gl => gleam::gl::BGRA,
95            gleam::gl::GlType::Gles => gleam::gl::RGBA,
96        };
97        let pixels_flipped = gl.read_pixels(
98            rect.origin.x.0,
99            rect.origin.y.0,
100            rect.size.width.0,
101            rect.size.height.0,
102            format,
103            gleam::gl::UNSIGNED_BYTE,
104        );
105        let mut buf = IpcBytes::new_mut_blocking(pixels_flipped.len()).unwrap();
106        assert_eq!(rect.size.width.0 as usize * rect.size.height.0 as usize * 4, buf.len());
107        let stride = 4 * rect.size.width.0 as usize;
108        for (px, buf) in pixels_flipped.chunks_exact(stride).rev().zip(buf.chunks_exact_mut(stride)) {
109            buf.copy_from_slice(px);
110        }
111
112        if let Some(mask) = mask {
113            if format == gleam::gl::BGRA {
114                for bgra in buf.chunks_exact_mut(4) {
115                    bgra.swap(0, 3);
116                }
117            }
118            let density = 96.0 * scale_factor.0;
119            let density = Some(PxDensity2d::splat(density.ppi()));
120
121            let (pixels, size, density, is_opaque, is_mask) = Self::convert_decoded(
122                IpcDynamicImage::ImageRgba8(
123                    image::ImageBuffer::from_raw(rect.size.width.0 as u32, rect.size.height.0 as u32, buf).unwrap(),
124                ),
125                Some(mask),
126                density,
127                None,
128                None,
129                image::metadata::Orientation::NoTransforms,
130                &self.resizer,
131            )
132            .unwrap(); // frame size is not large enough to trigger an memmap that can fail
133
134            let id = self.add(ImageRequest::new(
135                ImageDataFormat::A8 { size },
136                pixels.clone(),
137                u64::MAX,
138                None,
139                Some(mask),
140            ));
141
142            ImageLoadedData::new(id, size, density, is_opaque, is_mask, pixels)
143        } else {
144            if format == gleam::gl::RGBA {
145                for rgba in buf.chunks_exact_mut(4) {
146                    rgba.swap(0, 3);
147                }
148            }
149
150            let is_opaque = buf.chunks_exact(4).all(|bgra| bgra[3] == 255);
151
152            let data = buf.finish_blocking().unwrap(); // frame size is not large enough to trigger an memmap that can fail
153            let density = 96.0 * scale_factor.0;
154            let density = Some(PxDensity2d::splat(density.ppi()));
155            let size = rect.size;
156
157            let id = self.add(ImageRequest::new(
158                ImageDataFormat::Bgra8 { size, density },
159                data.clone(),
160                u64::MAX,
161                None,
162                mask,
163            ));
164
165            ImageLoadedData::new(id, size, density, is_opaque, false, data)
166        }
167    }
