65macro_rules! dynamic_map(
66        ($dynimage: expr, $image: pat => $action: expr) => ({
67            use IpcDynamicImage::*;
68            match $dynimage {
69                ImageLuma8($image) => ImageLuma8($action),
70                ImageLumaA8($image) => ImageLumaA8($action),
71                ImageRgb8($image) => ImageRgb8($action),
72                ImageRgba8($image) => ImageRgba8($action),
73                ImageLuma16($image) => ImageLuma16($action),
74                ImageLumaA16($image) => ImageLumaA16($action),
75                ImageRgb16($image) => ImageRgb16($action),
76                ImageRgba16($image) => ImageRgba16($action),
77                ImageRgb32F($image) => ImageRgb32F($action),
78                ImageRgba32F($image) => ImageRgba32F($action),
79            }
80        });
81
82        ($dynimage: expr, $image:pat_param, $action: expr) => ({
83            use IpcDynamicImage::*;
84            match $dynimage {
85                ImageLuma8($image) => $action,
86                ImageLumaA8($image) => $action,
87                ImageRgb8($image) => $action,
88                ImageRgba8($image) => $action,
89                ImageLuma16($image) => $action,
90                ImageLumaA16($image) => $action,
91                ImageRgb16($image) => $action,
92                ImageRgba16($image) => $action,
93                ImageRgb32F($image) => $action,
94                ImageRgba32F($image) => $action,

99impl IpcDynamicImage {
100    pub fn decode<R: BufRead + Seek>(reader: ImageReader<R>) -> image::ImageResult<Self> {
101        let decoder = reader.into_decoder()?;
102        let (w, h) = decoder.dimensions();
103        let color_type = decoder.color_type();
104
105        // copied from image-0.25.9\src\images\dynimage.rs
106        match color_type {
107            ColorType::Rgb8 => {
108                let buf = decoder_to_ipc(decoder)?;
109                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgb8)
110            }
111
112            ColorType::Rgba8 => {
113                let buf = decoder_to_ipc(decoder)?;
114                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgba8)
115            }
116
117            ColorType::L8 => {
118                let buf = decoder_to_ipc(decoder)?;
119                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageLuma8)
120            }
121
122            ColorType::La8 => {
123                let buf = decoder_to_ipc(decoder)?;
124                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageLumaA8)
125            }
126
127            ColorType::Rgb16 => {
128                let buf = decoder_to_ipc(decoder)?.cast();
129                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgb16)
130            }
131
132            ColorType::Rgba16 => {
133                let buf = decoder_to_ipc(decoder)?.cast();
134                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgba16)
135            }
136
137            ColorType::Rgb32F => {
138                let buf = decoder_to_ipc(decoder)?.cast();
139                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgb32F)
140            }
141
142            ColorType::Rgba32F => {
143                let buf = decoder_to_ipc(decoder)?.cast();
144                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageRgba32F)
145            }
146
147            ColorType::L16 => {
148                let buf = decoder_to_ipc(decoder)?.cast();
149                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageLuma16)
150            }
151
152            ColorType::La16 => {
153                let buf = decoder_to_ipc(decoder)?.cast();
154                ImageBuffer::from_raw(w, h, buf).map(IpcDynamicImage::ImageLumaA16)
155            }
156            _ => unreachable!(),
157        }
158        .ok_or_else(|| ImageError::Parameter(ParameterError::from_kind(ParameterErrorKind::DimensionMismatch)))
159    }
160
161    pub fn dimensions(&self) -> (u32, u32) {
162        dynamic_map!(*self, ref p, p.dimensions())
163    }
164}

166fn decoder_to_ipc(decoder: impl image::ImageDecoder) -> ImageResult<IpcBytesMut> {
167    let total_bytes = decoder.total_bytes();
168    if total_bytes > usize::MAX as u64 {
169        return Err(ImageError::Limits(LimitError::from_kind(LimitErrorKind::InsufficientMemory)));
170    }
171
172    let mut buf = IpcBytes::new_mut_blocking(total_bytes as usize)?;
173    decoder.read_image(&mut buf[..])?;
174    Ok(buf)
175}
