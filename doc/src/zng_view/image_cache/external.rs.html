23/// Implements [`ExternalImageHandler`].
24///
25/// # Safety
26///
27/// This is only safe if use with [`ImageUseMap`].
28pub(crate) struct WrImageCache {
29    locked: Vec<Arc<ImageData>>,
30}
31impl WrImageCache {
32    pub fn new_boxed() -> Box<dyn ExternalImageHandler> {
33        Box::new(WrImageCache { locked: vec![] })
34    }
35}
36impl ExternalImageHandler for WrImageCache {
37    fn lock(&mut self, key: ExternalImageId, _channel_index: u8, _is_composited: bool) -> ExternalImage<'_> {
38        // SAFETY: this is safe because the Arc is kept alive in `ImageUseMap`.
39        let img = unsafe {
40            let ptr = key.0 as *const ImageData;
41            Arc::increment_strong_count(ptr);
42            Arc::<ImageData>::from_raw(ptr)
43        };
44
45        self.locked.push(img); // keep alive in case the image is removed mid-use
46
47        match &**self.locked.last().unwrap() {
48            ImageData::RawData { pixels, range, .. } => {
49                ExternalImage {
50                    uv: TexelRect::invalid(), // `RawData` does not use `uv`.
51                    source: ExternalImageSource::RawData(&pixels[range.clone()]),
52                }
53            }
54            ImageData::NativeTexture { uv, texture: id } => ExternalImage {
55                uv: *uv,
56                source: ExternalImageSource::NativeTexture(*id),
57            },
58        }
59    }
60
61    fn unlock(&mut self, key: ExternalImageId, _channel_index: u8) {
62        if let Some(i) = self.locked.iter().position(|d| ExternalImageId(Arc::as_ptr(d) as _) == key) {
63            self.locked.swap_remove(i);
64        } else {
65            debug_assert!(false);
66        }
67    }
68}

70impl Image {
71    fn external_id(&self) -> ExternalImageId {
72        ExternalImageId(Arc::as_ptr(&self.0) as u64)
73    }
74
75    fn data(&self) -> webrender::api::ImageData {
76        webrender::api::ImageData::External(ExternalImageData {
77            id: self.external_id(),
78            channel_index: 0,
79            image_type: ExternalImageType::Buffer,
80            normalized_uvs: false,
81        })
82    }

92/// Track and manage images used in a renderer.
93///
94/// The renderer must use [`WrImageCache`] as the external image source.
95pub(crate) struct ImageUseMap {
96    id_tex: FxHashMap<ExternalImageId, ImageUse>,
97    tex_id: FxHashMap<ImageTextureId, ExternalImageId>,
98    resizer_cache: Arc<ResizerCache>,
99}
100impl ImageUseMap {
101    pub fn new(resizer_cache: Arc<ResizerCache>) -> Self {
102        Self {
103            id_tex: FxHashMap::default(),
104            tex_id: FxHashMap::default(),
105            resizer_cache,
106        }
107    }
108    pub fn new_use(&mut self, image: &Image, document_id: DocumentId, api: &mut RenderApi) -> ImageTextureId {
109        let id = image.external_id();
110        match self.id_tex.entry(id) {
111            Entry::Occupied(e) => e.get().texture_id,
112            Entry::Vacant(e) => {
113                let key = api.generate_image_key();
114                let tex_id = ImageTextureId::from_raw(key.1);
115                e.insert(ImageUse {
116                    // keep the image Arc alive, we expect this in `WrImageCache`.
117                    image: image.clone(),
118                    texture_id: tex_id,
119                    stripes: Box::new([]),
120                    mipmap: vec![],
121                });
122                self.tex_id.insert(tex_id, id);
123
124                // register the top image, most common usage and it is cheap, but
125                // during `push_display_list_img` may register other derived images
126                // like gigapixel stripes or resized images.
127                let mut txn = webrender::Transaction::new();
128                txn.add_image(key, image.descriptor(), image.data(), None);
129                api.send_transaction(document_id, txn);
130
131                tex_id
132            }
133        }
134    }

136    pub fn update_use(
137        &mut self,
138        texture_id: ImageTextureId,
139        image: &Image,
140        dirty_rect: Option<PxRect>,
141        document_id: DocumentId,
142        api: &mut RenderApi,
143    ) -> bool {
144        if let Entry::Occupied(mut e) = self.tex_id.entry(texture_id) {
145            let id = image.external_id();
146            if *e.get() != id {
147                let prev_image = self.id_tex.get(&id).unwrap();
148                if prev_image.image.descriptor() != image.descriptor() {
149                    tracing::error!("cannot update image use {texture_id:?}, new image has different dimensions");
150                    return false;
151                }
152
153                let prev_id = e.insert(id);
154                let prev_image = self.id_tex.remove(&prev_id).unwrap();
155                let mut txn = webrender::Transaction::new();
156
157                // update only the straight forward usage
158                txn.update_image(
159                    ImageKey(api.get_namespace_id(), texture_id.get()),
160                    image.descriptor(),
161                    image.data(),
162                    &match dirty_rect {
163                        Some(r) => ImageDirtyRect::Partial(r.to_box2d().cast().cast_unit()),
164                        None => ImageDirtyRect::All,
165                    },
166                );
167
168                // remove derived usages
169                for stripe in prev_image.stripes {
170                    txn.delete_image(ImageKey(api.get_namespace_id(), stripe.get()));
171                }
172                for mip in prev_image.mipmap {
173                    txn.delete_image(ImageKey(api.get_namespace_id(), mip.texture_id.get()));
174                    for stripe in mip.stripes {
175                        txn.delete_image(ImageKey(api.get_namespace_id(), stripe.get()));
176                    }
177                }
178
179                self.id_tex.insert(
180                    id,
181                    ImageUse {
182                        image: image.clone(),
183                        texture_id,
184                        stripes: Box::new([]),
185                        mipmap: vec![],
186                    },
187                );
188                api.send_transaction(document_id, txn);
189            }
190
191            true
192        } else {
193            tracing::error!("cannot update image use, texture not found");
194            false
195        }
196    }

198    pub fn delete(&mut self, texture_id: ImageTextureId, document_id: DocumentId, api: &mut RenderApi) {
199        if let Some(id) = self.tex_id.remove(&texture_id) {
200            let image = self.id_tex.remove(&id).unwrap(); // remove but keep alive until the transaction is done.
201            let mut txn = webrender::Transaction::new();
202
203            txn.delete_image(ImageKey(api.get_namespace_id(), image.texture_id.get()));
204            for stripe in image.stripes {
205                txn.delete_image(ImageKey(api.get_namespace_id(), stripe.get()));
206            }
207            for mip in image.mipmap {
208                txn.delete_image(ImageKey(api.get_namespace_id(), mip.texture_id.get()));
209                for stripe in mip.stripes {
210                    txn.delete_image(ImageKey(api.get_namespace_id(), stripe.get()));
211                }
212            }
213
214            api.send_transaction(document_id, txn);
215        }
216    }

218    #[expect(clippy::too_many_arguments)]
219    pub fn push_display_list_img(
220        &mut self,
221        document_id: DocumentId,
222        api: &mut RenderApi,
223        wr_list: &mut wr::DisplayListBuilder,
224        sc: &mut SpaceAndClip,
225        cache: &DisplayListCache,
226        clip_rect: PxRect,
227        image_id: ImageTextureId,
228        image_size: PxSize,
229        rendering: ImageRendering,
230        tile_size: PxSize,
231        tile_spacing: PxSize,
232    ) {
233        let mut img = match self.tex_id.get(&image_id) {
234            Some(id) => self.id_tex.get_mut(id).unwrap(),
235            None => return,
236        };
237
238        // replace if high-quality downscaled, if available
239        let mip = img.image.mip(tile_size, &self.resizer_cache);
240        if img.image.external_id() != mip.external_id() {
241            if let Some(i) = img.mipmap.iter().position(|i| i.image.external_id() == mip.external_id()) {
242                img = &mut img.mipmap[i];
243            } else {
244                // register main texture ID for the downscaled replacement
245                let key = api.generate_image_key();
246                let mut txn = webrender::Transaction::new();
247                txn.add_image(key, mip.descriptor(), mip.data(), None);
248                api.send_transaction(document_id, txn);
249                img.mipmap.push(ImageUse {
250                    image: mip,
251                    texture_id: ImageTextureId::from_raw(key.1),
252                    stripes: Box::new([]),
253                    mipmap: vec![],
254                });
255                img = img.mipmap.last_mut().unwrap();
256            }
257        }
258
259        if tile_spacing.is_empty() && tile_size == image_size {
260            let full_size = img.image.size();
261
262            if !img.image.overflows_wr() {
263                // normal sized image
264                let bounds = clip_rect.to_wr();
265                let clip = sc.clip_chain_id(wr_list);
266                let props = wr::CommonItemProperties {
267                    clip_rect: bounds,
268                    clip_chain_id: clip,
269                    spatial_id: sc.spatial_id(),
270                    flags: sc.primitive_flags(),
271                };
272                wr_list.push_image(
273                    &props,
274                    PxRect::from_size(image_size).to_wr(),
275                    rendering.to_wr(),
276                    wr::AlphaType::PremultipliedAlpha,
277                    wr::ImageKey(cache.id_namespace(), img.texture_id.get()),
278                    wr::ColorF::WHITE,
279                );
280            } else {
281                if img.stripes.is_empty() {
282                    let stripes = img.image.wr_stripes();
283                    if stripes.is_empty() {
284                        // error returns empty
285                        return;
286                    }
287
288                    // register texture IDs for the stripes
289                    let mut stripe_ids = Vec::with_capacity(stripes.len());
290                    let mut txn = webrender::Transaction::new();
291                    for stripe in stripes {
292                        let key = api.generate_image_key();
293                        let tex_id = ImageTextureId::from_raw(key.1);
294                        stripe_ids.push(tex_id);
295                        txn.add_image(key, stripe.descriptor(), stripe.data(), None);
296                    }
297                    api.send_transaction(document_id, txn);
298                    img.stripes = stripe_ids.into_boxed_slice();
299                }
300
301                // gigantic image split into stripes
302                let scale_x = full_size.width.0 as f32 / image_size.width.0 as f32;
303                let scale_y = full_size.height.0 as f32 / image_size.height.0 as f32;
304
305                let mut scaled_y = Px(0);
306                for (stripe_id, img) in img.stripes.iter().zip(img.image.wr_stripes()) {
307                    let scaled_stripe = {
308                        let mut r = img.size();
309                        r.width /= scale_x;
310                        r.height /= scale_y;
311                        r
312                    };
313                    if scaled_y != Px(0) {
314                        let spatial_id = wr_list.push_reference_frame(
315                            PxPoint::zero().to_wr(),
316                            sc.spatial_id(),
317                            wr::TransformStyle::Flat,
318                            wr::PropertyBinding::Value(wr::units::LayoutTransform::translation(0.0, scaled_y.0 as f32, 0.0)),
319                            wr::ReferenceFrameKind::Transform {
320                                is_2d_scale_translation: true,
321                                should_snap: false,
322                                paired_with_perspective: false,
323                            },
324                            sc.next_view_process_frame_id().to_wr(),
325                        );
326                        sc.push_spatial(spatial_id);
327                    }
328                    let mut clip_rect = clip_rect;
329                    clip_rect.origin.y -= scaled_y;
330                    let clip = sc.clip_chain_id(wr_list);
331                    let props = wr::CommonItemProperties {
332                        clip_rect: clip_rect.to_wr(),
333                        clip_chain_id: clip,
334                        spatial_id: sc.spatial_id(),
335                        flags: sc.primitive_flags(),
336                    };
337                    wr_list.push_image(
338                        &props,
339                        PxRect::from_size(scaled_stripe).to_wr(),
340                        rendering.to_wr(),
341                        wr::AlphaType::PremultipliedAlpha,
342                        wr::ImageKey(cache.id_namespace(), stripe_id.get()),
343                        wr::ColorF::WHITE,
344                    );
345                    if scaled_y != Px(0) {
346                        wr_list.pop_reference_frame();
347                        sc.pop_spatial();
348                    }
349                    scaled_y += scaled_stripe.height;
350                }
351            }
352        } else {
353            if img.image.overflows_wr() {
354                tracing::error!("cannot tile or repeat image, too large");
355                return;
356            }
357
358            let bounds = clip_rect.to_wr();
359            let clip = sc.clip_chain_id(wr_list);
360            let props = wr::CommonItemProperties {
361                clip_rect: bounds,
362                clip_chain_id: clip,
363                spatial_id: sc.spatial_id(),
364                flags: sc.primitive_flags(),
365            };
366            wr_list.push_repeating_image(
367                &props,
368                PxRect::from_size(image_size).to_wr(),
369                tile_size.to_wr(),
370                tile_spacing.to_wr(),
371                rendering.to_wr(),
372                wr::AlphaType::PremultipliedAlpha,
373                wr::ImageKey(cache.id_namespace(), img.texture_id.get()),
374                wr::ColorF::WHITE,
375            );
376        }
377    }
